\chapter{Normalization: canonical kernels, trace functionals, and error ledgers}
\label{chap:08-normalization}

\epigraph{Normalization is the passage from a formula that computes to an object that is invariant.}{}

\section{Purpose and output of the normalization layer}
\label{sec:08.scope}

The localized trace formalism developed in Chapters~\ref{chap:01-introduction}--\ref{chap:07-main-results}
depends, at intermediate stages, on auxiliary choices: cutoff profiles, microlocal partitions,
and quantization conventions. The role of the present chapter is to \emph{factor out} these choices and
to isolate the invariants that remain stable under admissible modifications.

\medskip
\noindent
\textbf{Output.} We construct:
\[
\mathcal{T}_\Gamma[h] \quad \text{(canonical normalized trace functional)} ,
\qquad
\mathcal{E}_\Gamma[h] \quad \text{(canonical error ledger)} ,
\]
with the following property: for every admissible implementation of the localized trace
(raw trace, cutoff, microlocalization, quantization), the resulting trace value is
\emph{equal to} $\mathcal{T}_\Gamma[h]$ up to an \emph{explicit identity correction} and an error bounded by
$\mathcal{E}_\Gamma[h]$.

\paragraph{Standing framework.}
We work with the geometric/dynamical input $(X,\Gamma)$ and the operator-theoretic setup fixed earlier.
All test functions $h$ belong to the admissible class specified in the main trace formula
(typically Schwartz or Paley--Wiener type, depending on the global model).

\section{Test classes and the zero-mode removal}
\label{sec:08.test-class}

\subsection{Even/odd splitting}
\label{subsec:08.even-odd}

Let $h:\R\to\C$ be admissible. Define
\[
h^{\mathrm{ev}}(t)=\frac12\bigl(h(t)+h(-t)\bigr),\qquad
h^{\mathrm{odd}}(t)=\frac12\bigl(h(t)-h(-t)\bigr),
\]
so that $h=h^{\mathrm{ev}}+h^{\mathrm{odd}}$.
This splitting is intrinsic and will be used to control symmetry constraints in later chapters.

\subsection{Normalization map and admissible cutoffs}
\label{subsec:08.norm-map}

\begin{definition}[Admissible cutoff profiles]
\label{def:08.cutoff}
A function $\chi\in C_c^\infty(\R)$ is called an \emph{admissible cutoff} if $\chi\equiv 1$ on a neighborhood of $0$.
\end{definition}

\begin{definition}[Zero-mode removal]
\label{def:08.zero-mode}
Given an admissible cutoff $\chi$ and an admissible test $h$, define the normalized test
\[
h_\chi := h - h(0)\,\chi.
\]
\end{definition}

\begin{remark}[Why the zero-mode is removed]
\label{rem:08.why-subtract}
The term $h(0)\chi$ isolates the contribution of the identity sector (the ``localization of the diagonal'').
Removing it forces the trace functional to respond only to geometric/dynamical content.
Normalization is therefore the mechanism that prevents spurious dependence on the localization profile.
\end{remark}

\begin{definition}[Admissible pairs]
\label{def:08.admissible-pairs}
A pair $(h,\chi)$ is \emph{admissible} if $h$ is admissible in the sense of the main trace formula,
$\chi$ is an admissible cutoff (Definition~\ref{def:08.cutoff}),
and $h_\chi$ remains admissible in the same test class.
\end{definition}

\section{Canonical normalized trace: definition and cutoff invariance}
\label{sec:08.normalized-trace}

Let $\mathcal{L}_\Gamma[\cdot]$ denote the raw localized trace functional constructed in
Chapter~\ref{chap:07-main-results} (including identity/hyperbolic/elliptic terms and, when present,
continuous spectrum contributions).

\begin{definition}[Normalized trace functional associated to a cutoff]
\label{def:08.Tchi}
For an admissible pair $(h,\chi)$ define
\[
\mathcal{T}_\Gamma[h;\chi] := \mathcal{L}_\Gamma[h_\chi]
= \mathcal{L}_\Gamma\bigl[h - h(0)\chi\bigr].
\]
\end{definition}

\begin{lemma}[Cutoff change identity]
\label{lem:08.cutoff-change}
Let $(h,\chi_1)$ and $(h,\chi_2)$ be admissible. Then
\[
\mathcal{T}_\Gamma[h;\chi_1]-\mathcal{T}_\Gamma[h;\chi_2]
= -\,h(0)\,\mathcal{L}_\Gamma[\chi_1-\chi_2].
\]
\end{lemma}

\begin{proof}
By Definition~\ref{def:08.Tchi},
\[
\mathcal{T}_\Gamma[h;\chi_j]=\mathcal{L}_\Gamma[h-h(0)\chi_j].
\]
Subtract and use linearity of $\mathcal{L}_\Gamma$ on the admissible class.
\end{proof}

\begin{theorem}[Canonicalization modulo the identity sector]
\label{thm:08.canonical-mod-identity}
Assume the identity contribution to $\mathcal{L}_\Gamma[\cdot]$ is explicit (as provided by the trace formula in
Chapter~\ref{chap:07-main-results}). Then for any two admissible cutoffs $\chi_1,\chi_2$,
the difference $\mathcal{T}_\Gamma[h;\chi_1]-\mathcal{T}_\Gamma[h;\chi_2]$ is an explicit identity-term correction
depending only on $h(0)$ and on $\chi_1-\chi_2$.
\end{theorem}

\begin{proof}
Lemma~\ref{lem:08.cutoff-change} reduces the difference to $-h(0)\mathcal{L}_\Gamma[\chi_1-\chi_2]$.
Since $\chi_1-\chi_2$ vanishes near $0$, the localized decomposition confines the effect to the identity sector,
for which the trace formula gives an explicit expression.
\end{proof}

\begin{definition}[Canonical normalized trace as an equivalence class]
\label{def:08.canonical-class}
Define $\mathcal{T}_\Gamma[h]$ to be the class of $\mathcal{T}_\Gamma[h;\chi]$ modulo explicit identity corrections.
Equivalently, choose any admissible $\chi$ and set
\[
\mathcal{T}_\Gamma[h] := \mathcal{T}_\Gamma[h;\chi]
\quad\text{with the convention that all identity corrections are recorded explicitly.}
\]
\end{definition}

\section{Stability under quantization and microlocal refinements}
\label{sec:08.quantization-refinement}

\subsection{Quantization transfer}
\label{subsec:08.quantization}

Let $\Op$ denote a quantization convention used in the microlocal construction (Chapter~\ref{chap:05-microlocal}).
Different admissible conventions agree at principal level and differ by lower-order terms.

\begin{assumption}[Quantization stability bound]
\label{ass:08.quant-stability}
Let $\Op_1,\Op_2$ be admissible quantizations. For each admissible pair $(h,\chi)$ one has
\[
\mathcal{L}^{(1)}_\Gamma[h_\chi]-\mathcal{L}^{(2)}_\Gamma[h_\chi]
= \Err_{\mathrm{quant}}(h,\chi),
\]
and $\Err_{\mathrm{quant}}(h,\chi)$ is controlled by the microlocal remainder estimates from
Chapter~\ref{chap:05-microlocal}.
\end{assumption}

\begin{definition}[Quantization ledger]
\label{def:08.quant-ledger}
Let $\mathcal{E}^{\mathrm{quant}}_\Gamma[h;\chi]$ denote the sharp bound available for
$\abs{\Err_{\mathrm{quant}}(h,\chi)}$ in the asymptotic regime of interest.
\end{definition}

\subsection{Refinement invariance}
\label{subsec:08.refine}

Let $\{\Pi_j\}$ be an admissible microlocal partition as in Chapters~\ref{chap:04-projector}--\ref{chap:05-microlocal}.
A refinement $\{\widetilde{\Pi}_k\}$ should not change the trace beyond the established remainder terms.

\begin{assumption}[Refinement stability bound]
\label{ass:08.refine-stability}
If $\{\widetilde{\Pi}_k\}$ is an admissible refinement of $\{\Pi_j\}$, then for each admissible pair $(h,\chi)$,
\[
\mathcal{L}_\Gamma[h_\chi]_{\{\Pi_j\}} - \mathcal{L}_\Gamma[h_\chi]_{\{\widetilde{\Pi}_k\}}
= \Err_{\mathrm{ref}}(h,\chi),
\]
and $\Err_{\mathrm{ref}}(h,\chi)$ is controlled by the remainder bounds proved earlier.
\end{assumption}

\begin{definition}[Refinement ledger]
\label{def:08.ref-ledger}
Let $\mathcal{E}^{\mathrm{ref}}_\Gamma[h;\chi]$ denote the sharp bound available for
$\abs{\Err_{\mathrm{ref}}(h,\chi)}$.
\end{definition}

\section{The consolidated error ledger and the stability theorem}
\label{sec:08.consolidated-error}

\begin{definition}[Total normalization ledger]
\label{def:08.total-ledger}
For an admissible pair $(h,\chi)$ define
\[
\mathcal{E}_\Gamma[h;\chi]
:= \mathcal{E}^{\mathrm{quant}}_\Gamma[h;\chi]
 + \mathcal{E}^{\mathrm{ref}}_\Gamma[h;\chi]
 + \mathcal{E}^{\mathrm{tail}}_\Gamma[h;\chi]
 + \mathcal{E}^{\mathrm{cont}}_\Gamma[h;\chi],
\]
where $\mathcal{E}^{\mathrm{tail}}_\Gamma$ accounts for truncation/smoothing tails and
$\mathcal{E}^{\mathrm{cont}}_\Gamma$ accounts for continuous spectrum terms (when present),
as established in Chapter~\ref{chap:07-main-results}.
\end{definition}

\begin{theorem}[Normalization stability]
\label{thm:08.normalization-stability}
Fix an admissible implementation of the localized trace, i.e.\ choices of cutoff $\chi$, quantization $\Op$,
and microlocal partition, and define $\mathcal{T}_\Gamma[h;\chi]$ as in Definition~\ref{def:08.Tchi}.
For any other admissible implementation $(\chi',\Op',\mathrm{partition}')$ one has
\[
\mathcal{T}_\Gamma[h;\chi] - \mathcal{T}'_\Gamma[h;\chi']
= \Delta_{\mathrm{id}}(h;\chi,\chi') + \Err_{\mathrm{norm}}(h),
\]
where $\Delta_{\mathrm{id}}(h;\chi,\chi')$ is the explicit identity correction from
Theorem~\ref{thm:08.canonical-mod-identity} and
\[
\abs{\Err_{\mathrm{norm}}(h)}
\le \mathcal{E}_\Gamma[h;\chi] + \mathcal{E}'_\Gamma[h;\chi'].
\]
In particular, modulo the explicitly recorded identity correction, the normalized trace functional
$\mathcal{T}_\Gamma[h]$ is canonical up to a quantified remainder controlled by the ledger.
\end{theorem}

\begin{proof}
Combine:
(i) cutoff invariance modulo the identity correction (Lemma~\ref{lem:08.cutoff-change} and
Theorem~\ref{thm:08.canonical-mod-identity}),
(ii) quantization stability (Assumption~\ref{ass:08.quant-stability}),
(iii) refinement stability (Assumption~\ref{ass:08.refine-stability}),
and absorb the remaining auxiliary modifications into the tail/continuous-spectrum ledgers already established in
Chapter~\ref{chap:07-main-results}.
\end{proof}

\section{Corollaries used in subsequent chapters}
\label{sec:08.corollaries}

\begin{corollary}[Drop-in rule]
\label{cor:08.drop-in}
Any argument in Chapters~\ref{chap:09-local-weyl}--\ref{chap:10-applications} may be formulated solely in terms of
the canonical pair $\bigl(\mathcal{T}_\Gamma[h],\mathcal{E}_\Gamma[h]\bigr)$, without reference to a particular cutoff,
quantization, or microlocal cover, provided the explicit identity correction is recorded.
\end{corollary}

\begin{corollary}[Rescaled tests]
\label{cor:08.rescaling}
Let $h_T(t):=h(t/T)$ for $T\ge 1$ and assume $(h_T,\chi)$ is admissible in the regime of interest.
Then the map $T\mapsto \mathcal{T}_\Gamma[h_T]$ is stable under admissible auxiliary modifications,
with stability constants controlled by the ledger $\mathcal{E}_\Gamma[h_T]$ (up to explicit identity corrections).
\end{corollary}

\section{Bibliographic notes}
\label{sec:08.bib-notes}

The normalization mechanism implemented above follows the standard microlocal trace paradigm and the
Selberg/Arthur tradition; see, e.g., \cite{Hejhal1983,Zworski2012,DyatlovZworski2019}
for scattering and semiclassical technology and \cite{Buser1992,Iwaniec2002}
for spectral geometry and automorphic background.
