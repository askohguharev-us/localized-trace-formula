\chapter{Normalization: canonical trace functionals and stable error ledgers}
\label{chap:08-normalization}

\epigraph{Normalization is the act of turning a formula into an object.}{}

\section{Purpose and scope}
\label{sec:08.scope}

The goal of this chapter is to isolate a normalization layer which makes the localized trace formula
\emph{canonical} and \emph{portable}.
More precisely, we construct a normalized trace functional
\[
\mathcal{T}_\Gamma[h]
\]
together with an explicit error ledger
\[
\mathcal{E}_\Gamma[h],
\]
so that all subsequent arguments depend only on the normalized object and on a finite list of stable
error contributions, and not on auxiliary choices such as cutoffs, quantization conventions,
or microlocal partitions.

The normalization procedure is designed to be compatible with the framework fixed in
Chapters~\ref{chap:01-introduction}--\ref{chap:07-main-results}.
In particular, the geometric object $(X,\Gamma)$ and the associated Laplace-type operator $L$
(with its spectral resolution and the trace class regularization established earlier)
are fixed once and for all.
All admissible test functions $h$ belong to the class introduced in the preceding chapters
(e.g.\ Schwartz class or Paley--Wiener class, depending on the global setting).

The main outcome is the following stability statement: once normalization is imposed,
any admissible modification of the localization architecture changes the trace functional only by an
explicit identity-sector correction and a remainder controlled by the global error ledger.

\section{Admissible test functions and canonical subtraction}
\label{sec:08.test-class}

\subsection{Test class and even/odd splitting}
\label{subsec:08.even-odd}

Let $h:\R\to\C$ be an admissible test function in the sense of
Chapters~\ref{chap:02-preliminaries}--\ref{chap:07-main-results}.
We write the standard decomposition
\[
h^{\mathrm{ev}}(t)
=\frac12\bigl(h(t)+h(-t)\bigr),
\qquad
h^{\mathrm{odd}}(t)
=\frac12\bigl(h(t)-h(-t)\bigr),
\]
so that $h=h^{\mathrm{ev}}+h^{\mathrm{odd}}$.
In most applications only the even component contributes to the geometric trace terms, but we keep the
full class for generality and for later compatibility with twisted traces.

\subsection{Canonical subtraction of the zero mode}
\label{subsec:08.zero-mode}

The raw localized trace functional $\mathcal{L}_\Gamma[h]$ constructed in Chapter~\ref{chap:07-main-results}
contains an identity-sector contribution which depends on the behavior of $h$ at $0$.
In order to isolate the part of the trace sensitive to the geometry and dynamics, we introduce a canonical
subtraction.

\begin{definition}[Reference cutoff and normalization map]
\label{def:08.norm-map}
Fix a reference function $\chi\in C_c^\infty(\R)$ such that $\chi\equiv 1$ in a neighborhood of $0$.
Define the normalization map
\[
\mathsf{N}: h \longmapsto h^\sharp := h - h(0)\,\chi.
\]
\end{definition}

\begin{remark}
\label{rem:08.norm-remark}
The map $\mathsf{N}$ removes the contribution of the constant mode $h(0)$ in a controlled way.
This is the minimal normalization compatible with the trace class structure established earlier:
it modifies $h$ only by a compactly supported smooth term and preserves all admissibility properties
required for the localized trace formula.
\end{remark}

\begin{definition}[Admissible cutoff]
\label{def:08.adm-cutoff}
A function $\chi\in C_c^\infty(\R)$ is called an admissible cutoff if $\chi\equiv 1$ in a neighborhood of $0$.
\end{definition}

In what follows we keep one admissible cutoff $\chi$ fixed, but we shall prove that all normalized objects
are canonical modulo explicit identity-sector corrections.

\section{Normalized trace functional}
\label{sec:08.normalized-trace}

\subsection{Definition and first properties}
\label{subsec:08.def-first}

\begin{definition}[Normalized trace functional]
\label{def:08.norm-trace}
For an admissible test function $h$ define
\[
\mathcal{T}_\Gamma[h]
:=\mathcal{L}_\Gamma[h^\sharp]
=\mathcal{L}_\Gamma\bigl[h-h(0)\chi\bigr].
\]
\end{definition}

\begin{remark}[Immediate consequences]
\label{rem:08.immediate}
\leavevmode
\begin{enumerate}
  \item\label{it:08.immediate1}
  By construction, $\mathcal{T}_\Gamma[\chi]=0$.
  \item\label{it:08.immediate2}
  The functional $\mathcal{T}_\Gamma[\cdot]$ is linear on the admissible test class.
  \item\label{it:08.immediate3}
  The dependence on the auxiliary cutoff $\chi$ is confined to the identity-sector subtraction term.
\end{enumerate}
\end{remark}

\subsection{Cutoff change formula}
\label{subsec:08.cutoff-change}

We next quantify the dependence of the normalized trace on the choice of cutoff profile.

\begin{lemma}[Cutoff change formula]
\label{lem:08.cutoff-change}
Let $\chi_1,\chi_2$ be admissible cutoffs and let $h$ be an admissible test function.
Define the corresponding normalized traces
\[
\mathcal{T}^{(j)}_\Gamma[h]
:=\mathcal{L}_\Gamma\bigl[h-h(0)\chi_j\bigr],
\qquad j=1,2.
\]
Then
\[
\mathcal{T}^{(1)}_\Gamma[h]-\mathcal{T}^{(2)}_\Gamma[h]
=-h(0)\,\mathcal{L}_\Gamma[\chi_1-\chi_2].
\]
\end{lemma}

\begin{proof}
By linearity of $\mathcal{L}_\Gamma$,
\[
\mathcal{T}^{(1)}_\Gamma[h]-\mathcal{T}^{(2)}_\Gamma[h]
=
\mathcal{L}_\Gamma\bigl[h-h(0)\chi_1\bigr]
-
\mathcal{L}_\Gamma\bigl[h-h(0)\chi_2\bigr]
=
-h(0)\,\mathcal{L}_\Gamma[\chi_1-\chi_2].
\]
\end{proof}

The lemma shows that cutoff dependence is completely controlled by the raw trace applied to a compactly supported
smooth function.
In particular, if the identity contribution to $\mathcal{L}_\Gamma$ is explicitly known, then the cutoff change
is explicitly computable.

\begin{theorem}[Canonicality modulo identity-sector correction]
\label{thm:08.canonical-mod-identity}
Assume that the identity contribution in the localized trace formula of
Chapter~\ref{chap:07-main-results} is explicit.
Then for any admissible cutoffs $\chi_1,\chi_2$ and any admissible test $h$,
the difference
\[
\mathcal{T}^{(1)}_\Gamma[h]-\mathcal{T}^{(2)}_\Gamma[h]
\]
is an explicit identity-sector correction depending only on $h(0)$ and on the fixed identity term of the trace formula.
\end{theorem}

\begin{proof}
By Lemma~\ref{lem:08.cutoff-change},
\[
\mathcal{T}^{(1)}_\Gamma[h]-\mathcal{T}^{(2)}_\Gamma[h]
=-h(0)\,\mathcal{L}_\Gamma[\chi_1-\chi_2].
\]
Since $\chi_1-\chi_2$ is compactly supported and smooth and vanishes near $0$, its contribution is confined to the
identity sector of the localized trace decomposition established in Chapter~\ref{chap:07-main-results}.
Therefore the explicit identity term yields an explicit formula for the difference.
\end{proof}

\section{Normalization stability under admissible auxiliary changes}
\label{sec:08.stability}

The construction of $\mathcal{L}_\Gamma$ in Chapter~\ref{chap:07-main-results} depends on auxiliary choices:
a quantization convention, a microlocal partition, and (in the noncompact setting) a regularization scheme for the
continuous spectrum.
The purpose of this section is to show that the normalized trace functional $\mathcal{T}_\Gamma[h]$ is stable under
all such admissible modifications, at the precision level required in later chapters.

\subsection{Stability under quantization conventions}
\label{subsec:08.quantization}

Let $\Op$ denote the quantization scheme used to define the microlocal projectors and the localized kernels
(see Chapter~\ref{chap:05-microlocal}).
Different admissible quantizations differ only by lower-order operators.

\begin{assumption}[Quantization stability]
\label{ass:08.quant-stability}
Let $\Op_1$ and $\Op_2$ be two admissible quantization conventions.
Let $\mathcal{L}^{(1)}_\Gamma$ and $\mathcal{L}^{(2)}_\Gamma$ denote the corresponding raw localized traces.
Then for every admissible test function $g$ one has
\[
\mathcal{L}^{(1)}_\Gamma[g]-\mathcal{L}^{(2)}_\Gamma[g]
=
\Err_{\mathrm{quant}}(g),
\]
where $\Err_{\mathrm{quant}}(g)$ satisfies the bounds provided by the microlocal calculus of
Chapter~\ref{chap:05-microlocal}.
\end{assumption}

\begin{definition}[Quantization error contribution]
\label{def:08.quant-ledger}
For an admissible test function $h$ define
\[
\mathcal{E}^{\mathrm{quant}}_\Gamma[h]
:=
\sup_{\text{admissible }\Op_1,\Op_2}
\bigl|\Err_{\mathrm{quant}}(h^\sharp)\bigr|.
\]
\end{definition}

The precise bound depends on the asymptotic regime (e.g.\ the semiclassical parameter or the localization window),
and is taken from the established estimates in the earlier chapters.

\subsection{Stability under microlocal refinements}
\label{subsec:08.refinement}

Let $\{\Pi_j\}$ be a microlocal partition of unity used in the construction of the localized trace
(Chapter~\ref{chap:04-projector}--\ref{chap:05-microlocal}).
Refining the partition changes only the decomposition, not the operator itself, and introduces only controlled
remainder terms.

\begin{assumption}[Refinement stability]
\label{ass:08.refine-stability}
Let $\{\Pi_j\}$ and $\{\widetilde{\Pi}_k\}$ be two admissible microlocal partitions, with
$\{\widetilde{\Pi}_k\}$ a refinement of $\{\Pi_j\}$.
Let $\mathcal{L}_{\Gamma,\{\Pi_j\}}$ and $\mathcal{L}_{\Gamma,\{\widetilde{\Pi}_k\}}$ denote the corresponding raw traces.
Then for every admissible test function $g$,
\[
\mathcal{L}_{\Gamma,\{\Pi_j\}}[g]-\mathcal{L}_{\Gamma,\{\widetilde{\Pi}_k\}}[g]
=
\Err_{\mathrm{ref}}(g),
\]
where $\Err_{\mathrm{ref}}(g)$ is bounded by the remainder estimates of Chapter~\ref{chap:05-microlocal}.
\end{assumption}

\begin{definition}[Refinement error contribution]
\label{def:08.ref-ledger}
For an admissible test function $h$ define
\[
\mathcal{E}^{\mathrm{ref}}_\Gamma[h]
:=
\sup_{\text{admissible refinements}}
\bigl|\Err_{\mathrm{ref}}(h^\sharp)\bigr|.
\]
\end{definition}

\subsection{Tail and continuous-spectrum contributions}
\label{subsec:08.tail-cont}

The localized trace formula in Chapter~\ref{chap:07-main-results} may involve truncations in time/length
or smoothing at infinity, and, in noncompact settings, contributions from the continuous spectrum.

\begin{definition}[Tail and continuous-spectrum ledgers]
\label{def:08.tail-cont-ledgers}
Let $\mathcal{E}^{\mathrm{tail}}_\Gamma[h]$ denote the sharpest available bound on the contribution of asymptotic tails
in the localization procedure for the test function $h^\sharp$.
If a continuous spectrum term is present, let $\mathcal{E}^{\mathrm{cont}}_\Gamma[h]$ denote the corresponding bound.
Both quantities are taken from the estimates proved in Chapter~\ref{chap:07-main-results}.
\end{definition}

\section{Consolidated normalization ledger}
\label{sec:08.consolidated-ledger}

We now package all normalization-related remainders into a single object.

\begin{definition}[Total normalization ledger]
\label{def:08.total-ledger}
For an admissible test function $h$ define the consolidated normalization ledger
\[
\mathcal{E}_\Gamma[h]
:=
\mathcal{E}^{\mathrm{quant}}_\Gamma[h]
+
\mathcal{E}^{\mathrm{ref}}_\Gamma[h]
+
\mathcal{E}^{\mathrm{tail}}_\Gamma[h]
+
\mathcal{E}^{\mathrm{cont}}_\Gamma[h].
\]
\end{definition}

This quantity provides an explicit upper bound for the total instability introduced by admissible auxiliary changes.

\section{Normalization stability theorem}
\label{sec:08.norm-stability}

We now state the principal result of this chapter.

\begin{theorem}[Stability of the normalized trace]
\label{thm:08.normalization-stability}
Fix an admissible cutoff $\chi$ and define $\mathcal{T}_\Gamma[h]=\mathcal{L}_\Gamma[h-h(0)\chi]$.
Let $\mathcal{T}'_\Gamma[h]$ be the normalized trace obtained from any other admissible construction
(e.g.\ a different admissible cutoff $\chi'$, different admissible quantization, microlocal partition,
or continuous-spectrum regularization).
Then for every admissible test function $h$,
\[
\mathcal{T}_\Gamma[h]-\mathcal{T}'_\Gamma[h]
=
\Delta_{\mathrm{id}}(h;\chi,\chi')
+
\Err_{\mathrm{norm}}(h),
\]
where:
\begin{enumerate}
  \item $\Delta_{\mathrm{id}}(h;\chi,\chi')$ is an explicit identity-sector correction depending only on $h(0)$ and on the
  identity contribution of the trace formula;
  \item the remainder satisfies
  \[
  |\Err_{\mathrm{norm}}(h)|
  \le
  \mathcal{E}_\Gamma[h]
  +
  \mathcal{E}'_\Gamma[h],
  \]
  where $\mathcal{E}_\Gamma$ and $\mathcal{E}'_\Gamma$ are the corresponding consolidated ledgers for the two constructions.
\end{enumerate}
In particular, modulo the explicit identity correction, the normalized trace functional is canonical up to a quantified
error controlled by the total ledger.
\end{theorem}

\begin{proof}
The cutoff dependence is governed by Lemma~\ref{lem:08.cutoff-change} and yields the explicit identity correction
$\Delta_{\mathrm{id}}$ as in Theorem~\ref{thm:08.canonical-mod-identity}.
All remaining auxiliary changes are controlled by the stability assumptions
(Assumptions~\ref{ass:08.quant-stability} and~\ref{ass:08.refine-stability})
and by the tail/continuous-spectrum bounds incorporated into the ledgers.
Summing the corresponding bounds yields the stated estimate for $\Err_{\mathrm{norm}}(h)$.
\end{proof}

\section{Consequences for later chapters}
\label{sec:08.consequences}

We record two corollaries which will be used repeatedly in the sequel.

\begin{corollary}[Drop-in principle]
\label{cor:08.drop-in}
All statements in Chapters~\ref{chap:09-local-weyl}--\ref{chap:10-applications} depending on the localized trace formula
may be formulated solely in terms of the normalized functional $\mathcal{T}_\Gamma[h]$ and the consolidated ledger
$\mathcal{E}_\Gamma[h]$, without explicit reference to the auxiliary choices used to define the raw trace, provided the
identity-sector correction is treated explicitly.
\end{corollary}

\begin{corollary}[Stability under rescaling of tests]
\label{cor:08.rescaling}
Let $h_T(t):=h(t/T)$ with $T\ge 1$, and assume $h_T$ remains admissible for the range of $T$ under consideration.
Then the map
\[
T \longmapsto \mathcal{T}_\Gamma[h_T]
\]
is stable under admissible auxiliary changes, with stability constants controlled by the ledger
$\mathcal{E}_\Gamma[h_T]$ uniformly in $T$ (modulo the explicit identity-sector correction).
\end{corollary}

\section{Bibliographic remarks}
\label{sec:08.bib}

The normalization principle used here is standard in microlocal trace theory and in the Selberg trace formula
tradition.
For background on spectral geometry and the Selberg trace formula we refer to
\cite{Buser1992,Hejhal1983,Iwaniec2002}.
For semiclassical trace methods and scattering resonance technology see
\cite{Zworski2012,DyatlovZworski2019}.
