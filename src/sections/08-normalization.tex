\chapter{Normalization and error ledgers: canonical traces stable under auxiliary choices}
\label{chap:08-normalization}

\epigraph{Normalization is the act of upgrading a formula into a transportable invariant.}{}

\section{Purpose of the normalization layer}
\label{sec:08.purpose}

This chapter isolates the \emph{normalization layer} whose sole task is to make the localized trace mechanism
stable under admissible auxiliary changes:
\begin{enumerate}
  \item changes of cutoff profiles (localization near the identity / short-time sector),
  \item refinements of microlocal partitions,
  \item changes of quantization conventions (Weyl vs.\ Kohn--Nirenberg, semiclassical variants),
  \item inclusion/exclusion of continuous spectrum terms (when the global setting requires them).
\end{enumerate}

\paragraph{Output.}
For an admissible test $h$ and an admissible cutoff $\chi$, we define:
\[
\mathcal{T}_\Gamma[h;\chi] \quad\text{(canonical normalized trace)}
\qquad\text{and}\qquad
\mathcal{E}_\Gamma[h;\chi] \quad\text{(error ledger)}.
\]
The philosophy is rigid: \emph{any dependence on auxiliary choices must either (i) be eliminated by normalization,
or (ii) be recorded explicitly in the error ledger.}

\paragraph{Standing framework.}
All geometric and functional-analytic conventions are those fixed in
Chapters~\ref{chap:01-introduction}--\ref{chap:07-main-results}.

\section{Admissible tests and the zero-mode subtraction}
\label{sec:08.tests}

Let $h:\mathbb{R}\to\mathbb{C}$ be an admissible test function (as specified earlier: Schwartz/Paley--Wiener type,
depending on the global trace setting). We split
\[
h^{\mathrm{ev}}(t)=\frac12\bigl(h(t)+h(-t)\bigr),\qquad
h^{\mathrm{odd}}(t)=\frac12\bigl(h(t)-h(-t)\bigr),
\]
so that $h=h^{\mathrm{ev}}+h^{\mathrm{odd}}$.

\begin{definition}[Admissible cutoffs]
\label{def:08.cutoff}
A cutoff $\chi$ is admissible if $\chi\in C_c^\infty(\mathbb{R})$, $\chi\equiv 1$ on a neighborhood of $0$,
and its support is contained in a fixed short-time window:
\[
\supp(\chi)\subset (-t_0,t_0),
\]
where $t_0>0$ is chosen once and for all in accordance with the short-time locality regime of
Chapter~\ref{chap:07-main-results}.
\end{definition}

\begin{definition}[Normalization map]
\label{def:08.normmap}
Fix an admissible cutoff $\chi$. Define
\[
\mathsf{N}_\chi(h)\;:=\;h_\chi \;=\; h - h(0)\,\chi.
\]
\end{definition}

\begin{remark}[Why subtract $h(0)\chi$]
\label{rem:08.why}
The subtraction isolates the component of the trace that carries genuine geometric/dynamical information and removes
a non-transportable ``zero-mode'' sensitivity of the raw localized trace.
The entire point of this chapter is to prove that the remaining dependence on $\chi$ is either explicitly computable
(identity-sector correction) or absorbed in the ledger.
\end{remark}

\section{Canonical normalized trace functional}
\label{sec:08.canonical-trace}

Let $\mathcal{L}_\Gamma[\cdot]$ denote the \emph{raw} localized trace functional constructed in the main trace formula
(Chapter~\ref{chap:07-main-results}). It encodes the geometric input (identity, hyperbolic/elliptic, and possibly
continuous spectrum terms).

\begin{definition}[Canonical normalized trace]
\label{def:08.T}
For an admissible pair $(h,\chi)$ define
\[
\mathcal{T}_\Gamma[h;\chi] \;:=\; \mathcal{L}_\Gamma[h_\chi]
\;=\;\mathcal{L}_\Gamma\!\bigl[h-h(0)\chi\bigr].
\]
\end{definition}

\begin{remark}[Immediate properties]
\label{rem:08.props}
\leavevmode
\begin{enumerate}
  \item $\mathcal{T}_\Gamma[\chi;\chi]=0$ by construction.
  \item $\chi$ enters only through the extracted zero-mode term. The remaining sections quantify how and where this
  dependence can survive.
\end{enumerate}
\end{remark}

\section{Cutoff transport: what changes and what is invariant}
\label{sec:08.cutoff-transport}

\begin{lemma}[Cutoff change identity]
\label{lem:08.cutoff-change}
Let $\chi_1,\chi_2$ be admissible cutoffs. Then for any admissible $h$,
\[
\mathcal{T}_\Gamma[h;\chi_1]-\mathcal{T}_\Gamma[h;\chi_2]
= -h(0)\,\mathcal{L}_\Gamma[\chi_1-\chi_2].
\]
\end{lemma}

\begin{proof}
By Definition~\ref{def:08.T},
\[
\mathcal{T}_\Gamma[h;\chi_j]=\mathcal{L}_\Gamma[h-h(0)\chi_j].
\]
Subtract and use linearity of $\mathcal{L}_\Gamma$.
\end{proof}

\begin{theorem}[Canonicalization modulo a computable short-time correction]
\label{thm:08.canon-mod-short}
Assume the short-time/identity-sector expansion of $\mathcal{L}_\Gamma[\phi]$ is available for
$\phi\in C_c^\infty(\mathbb{R})$ supported in $(-t_0,t_0)$, as provided by the localized trace analysis in
Chapter~\ref{chap:07-main-results}. Then for any admissible cutoffs $\chi_1,\chi_2$,
\[
\mathcal{T}_\Gamma[h;\chi_1]\;=\;\mathcal{T}_\Gamma[h;\chi_2] \;+\; \Delta_{\mathrm{st}}(h;\chi_1,\chi_2),
\]
where the correction term
\[
\Delta_{\mathrm{st}}(h;\chi_1,\chi_2) \;:=\; -h(0)\,\mathcal{L}_\Gamma[\chi_1-\chi_2]
\]
is \emph{explicitly computable from the short-time sector} (and hence transportable across later chapters).
\end{theorem}

\begin{remark}
\label{rem:08.no-phantom-identity}
We do \emph{not} assert that $\chi_1-\chi_2$ contributes only to the identity term in complete generality.
What we use is the already-proved \emph{short-time locality regime}: for admissible supports contained in $(-t_0,t_0)$,
the trace decomposition identifies exactly which sectors can contribute and provides an explicit expansion for them.
\end{remark}

\section{Quantization and microlocal refinement: ledger formulation}
\label{sec:08.quant-refine}

\subsection{Quantization transport}
\label{subsec:08.quant}

Let $\Op$ denote the quantization convention fixed earlier (Weyl/Kohn--Nirenberg, semiclassical variants).

\begin{assumption}[Quantization stability at the normalized level]
\label{ass:08.quant-stability}
If $\Op_1$ and $\Op_2$ are two admissible quantizations used to build the localized kernels/projectors, then for all
admissible tests $h_\chi$,
\[
\mathcal{L}^{(1)}_\Gamma[h_\chi]-\mathcal{L}^{(2)}_\Gamma[h_\chi]
= \Err_{\mathrm{quant}}(h_\chi),
\]
where $\Err_{\mathrm{quant}}(h_\chi)$ is controlled by the symbol-transfer calculus of
Chapter~\ref{chap:05-microlocal}.
\end{assumption}

\begin{definition}[Quantization ledger]
\label{def:08.ledger-quant}
Let $\mathcal{E}^{\mathrm{quant}}_\Gamma[h;\chi]$ denote the sharpest bound available for
$\abs{\Err_{\mathrm{quant}}(h_\chi)}$ in the regime of interest.
\end{definition}

\subsection{Refinement transport}
\label{subsec:08.refine}

Let $\{\Pi_j\}$ be a microlocal partition used to localize the trace, and let $\{\widetilde{\Pi}_k\}$ be an admissible
refinement in the sense of Chapter~\ref{chap:05-microlocal}.

\begin{assumption}[Refinement stability at the normalized level]
\label{ass:08.ref-stability}
For all admissible tests $h_\chi$,
\[
\mathcal{L}_\Gamma[h_\chi]_{\{\Pi_j\}} - \mathcal{L}_\Gamma[h_\chi]_{\{\widetilde{\Pi}_k\}}
= \Err_{\mathrm{ref}}(h_\chi),
\]
with $\Err_{\mathrm{ref}}(h_\chi)$ controlled by the microlocal remainder bounds established earlier.
\end{assumption}

\begin{definition}[Refinement ledger]
\label{def:08.ledger-ref}
Let $\mathcal{E}^{\mathrm{ref}}_\Gamma[h;\chi]$ denote the sharpest bound available for
$\abs{\Err_{\mathrm{ref}}(h_\chi)}$.
\end{definition}

\section{Consolidated normalization ledger}
\label{sec:08.ledger-total}

\begin{definition}[Total normalization ledger]
\label{def:08.ledger-total}
For an admissible pair $(h,\chi)$ define
\[
\mathcal{E}_\Gamma[h;\chi]
:= \mathcal{E}^{\mathrm{quant}}_\Gamma[h;\chi]
 + \mathcal{E}^{\mathrm{ref}}_\Gamma[h;\chi]
 + \mathcal{E}^{\mathrm{tail}}_\Gamma[h;\chi]
 + \mathcal{E}^{\mathrm{cont}}_\Gamma[h;\chi],
\]
where:
\begin{itemize}
  \item $\mathcal{E}^{\mathrm{tail}}_\Gamma$ records truncation/smoothing tails inherent in the localization scheme,
  \item $\mathcal{E}^{\mathrm{cont}}_\Gamma$ records continuous-spectrum contributions (if present).
\end{itemize}
Each term is taken from the quantitative estimates already proved in
Chapters~\ref{chap:05-microlocal}--\ref{chap:07-main-results}.
\end{definition}

\begin{theorem}[Normalization transport theorem]
\label{thm:08.transport}
Let $(\chi,\Op,\{\Pi\})$ and $(\chi',\Op',\{\Pi'\})$ be two admissible sets of auxiliary choices, and define the
corresponding normalized traces $\mathcal{T}_\Gamma$ and $\mathcal{T}'_\Gamma$. Then
\[
\mathcal{T}_\Gamma[h;\chi] - \mathcal{T}'_\Gamma[h;\chi']
= \Delta_{\mathrm{st}}(h;\chi,\chi') + \Err_{\mathrm{norm}}(h),
\]
where $\Delta_{\mathrm{st}}(h;\chi,\chi')$ is the explicit short-time correction of
Theorem~\ref{thm:08.canon-mod-short}, and
\[
\abs{\Err_{\mathrm{norm}}(h)} \;\le\; \mathcal{E}_\Gamma[h;\chi] + \mathcal{E}'_\Gamma[h;\chi'].
\]
\end{theorem}

\begin{proof}
Decompose the difference into: (i) cutoff transport (Lemma~\ref{lem:08.cutoff-change} and
Theorem~\ref{thm:08.canon-mod-short}), (ii) quantization transport (Assumption~\ref{ass:08.quant-stability}),
(iii) refinement transport (Assumption~\ref{ass:08.ref-stability}), and absorb the remaining auxiliary changes into
the tail/continuous-spectrum ledgers already available from the main trace construction.
\end{proof}

\section{Drop-in corollaries for later chapters}
\label{sec:08.dropin}

\begin{corollary}[Drop-in rule]
\label{cor:08.dropin}
Any statement in Chapters~\ref{chap:09-local-weyl}--\ref{chap:10-applications} that depends on the trace formula may be
formulated solely in terms of $\mathcal{T}_\Gamma[h;\chi]$ and $\mathcal{E}_\Gamma[h;\chi]$, with auxiliary choices
appearing only through the explicit correction $\Delta_{\mathrm{st}}$ and the ledger bounds.
\end{corollary}

\begin{corollary}[Rescaled tests]
\label{cor:08.rescale}
Let $h_T(t)=h(t/T)$ with $T\ge 1$, and suppose $(h_T,\chi)$ is admissible in the regime of interest. Then
\[
T \longmapsto \mathcal{T}_\Gamma[h_T;\chi]
\]
is transport-stable under admissible auxiliary changes, with stability controlled by the ledger
$\mathcal{E}_\Gamma[h_T;\chi]$ (plus the explicit short-time correction).
\end{corollary}

\section{Bibliographic notes}
