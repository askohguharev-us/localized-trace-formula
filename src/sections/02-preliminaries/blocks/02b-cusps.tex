\subsection{Cusps, truncation, and the continuous spectrum}\label{subsec:cusps}

We record the cuspidal geometry of a finite--area hyperbolic surface
\[
  M=\Gamma\backslash\mathbb{H},\qquad \Gamma\subset\PSL_2(\mathbb{R})\ \text{cofinite},
\]
together with the normalization of scaling matrices, cusp widths, truncation
operators, and Eisenstein series.  These conventions are structural: they
control all parabolic terms and every error bound that depends on cusp data.
Standard references include Selberg~\cite{Selberg1956}, Hejhal~\cite{Hejhal1983},
and Iwaniec~\cite{Iwaniec2002}.

% ---------------------------------------------------------------------
% Block 1: Cuspidal structure and scaling matrices
% ---------------------------------------------------------------------
\subsubsection{Cuspidal structure and scaling matrices}\label{subsubsec:cusp-scaling}

\paragraph{Cusps as parabolic fixed points.}
A \emph{cusp} of $M$ is a $\Gamma$--equivalence class of a parabolic fixed point
in $\mathbb{P}^1(\mathbb{R})$.  Cofiniteness of $\Gamma$ implies that the set
of cusps is finite; we enumerate it by
\[
  \mathfrak{A}(M)=\{\mathfrak{a}_1,\dots,\mathfrak{a}_h\},\qquad h\ge 0.
\]
For each cusp $\mathfrak{a}$ let
\[
  \Gamma_{\mathfrak{a}}=\{\gamma\in\Gamma:\gamma\mathfrak{a}=\mathfrak{a}\}
\]
denote its stabilizer.  This subgroup is parabolic and isomorphic to $\mathbb{Z}$.

\paragraph{Scaling matrices and widths.}
Fix, once and for all, for each cusp $\mathfrak{a}$ a scaling matrix
$\sigma_{\mathfrak{a}}\in\PSL_2(\mathbb{R})$ with $\sigma_{\mathfrak{a}}(\infty)=\mathfrak{a}$
and a \emph{width} $w_{\mathfrak{a}}>0$ such that
\begin{equation}\label{eq:scaling-conjugation}
  \sigma_{\mathfrak{a}}^{-1}\Gamma_{\mathfrak{a}}\sigma_{\mathfrak{a}}
  = \left\{
      \pm\begin{pmatrix}1& n w_{\mathfrak{a}}\\0&1\end{pmatrix}
      : n\in\mathbb{Z}
    \right\}.
\end{equation}
The tuple $\big(\{w_{\mathfrak{a}}\},\{\sigma_{\mathfrak{a}}\}\big)$ is part of the global
dependency ledger (Chapter~\ref{chap:preliminaries}).

\begin{remark}[Normalization freedom]\label{rem:scaling-freedom}
The choice of $\sigma_{\mathfrak{a}}$ is not unique.  If $\sigma_{\mathfrak{a}}$ satisfies
\eqref{eq:scaling-conjugation}, then so does $\sigma_{\mathfrak{a}}u_t$, where
\[
  u_t=\begin{pmatrix}1&t\\0&1\end{pmatrix},\qquad t\in\mathbb{R},
\]
and also $\pm\sigma_{\mathfrak{a}}$.  This corresponds to the fact that cusp coordinates
are determined only up to horizontal translation (and sign in $\SL_2$).  All later
definitions and bounds are invariant under this freedom.
\end{remark}

\paragraph{Canonical cusp strips and cusp coordinates.}
For $w>0$ define the strip
\[
  \mathcal{S}(w)=\{x+iy\in\mathbb{H}: 0\le x<w,\ y>0\}.
\]
The cusp neighborhood associated to $\mathfrak{a}$ is
\[
  C_{\mathfrak{a}} := \sigma_{\mathfrak{a}}\big(\mathcal{S}(w_{\mathfrak{a}})\big)\subset\mathbb{H}.
\]
Coordinates $(x,y)$ on $\mathcal{S}(w_{\mathfrak{a}})$ are called \emph{cusp coordinates}
attached to $\mathfrak{a}$; in these coordinates $\Gamma_{\mathfrak{a}}$ acts by
$x\mapsto x+n w_{\mathfrak{a}}$.

\begin{lemma}[Uniqueness of cusp coordinates]\label{lem:cusp-coord-unique}
Let $\sigma_{\mathfrak{a}}$ and $\sigma'_{\mathfrak{a}}$ be two scaling matrices for the same cusp
$\mathfrak{a}$ satisfying \eqref{eq:scaling-conjugation} with the same width $w_{\mathfrak{a}}$.
Then there exists $t\in\mathbb{R}$ such that
\[
  \sigma'_{\mathfrak{a}}=\sigma_{\mathfrak{a}}\,u_t.
\]
\end{lemma}

\begin{proof}
Both $\sigma_{\mathfrak{a}}$ and $\sigma'_{\mathfrak{a}}$ send $\infty$ to $\mathfrak{a}$, hence
$g:=\sigma_{\mathfrak{a}}^{-1}\sigma'_{\mathfrak{a}}$ fixes $\infty$, so $g$ is upper triangular.
Moreover, $g$ normalizes the translation subgroup in \eqref{eq:scaling-conjugation} with the same
step $w_{\mathfrak{a}}$, forcing $g$ to be unipotent.  Thus $g=u_t$ for some $t\in\mathbb{R}$.
\end{proof}

\paragraph{Cusp invariants.}
The cuspidal structure is encoded by
\[
  h,\qquad \{w_{\mathfrak{a}}\}_{\mathfrak{a}\in\mathfrak{A}(M)},\qquad \{\sigma_{\mathfrak{a}}\}_{\mathfrak{a}\in\mathfrak{A}(M)}.
\]
All cusp-dependent constants later are permitted to depend only on these data and on $\Gamma$ (and,
when explicitly stated, on the spectral-gap parameter $\beta$ fixed in Chapter~\ref{chap:preliminaries}).

\subsubsection{Truncation at height, cusp volumes, and boundary length}\label{subsubsec:truncation}

\paragraph{Truncated cusp tails.}
Fix a cusp $\mathfrak{a}$ and $Y>1$.  The truncated tail above height $Y$ is
\[
  C_{\mathfrak{a}}(Y):=
  \sigma_{\mathfrak{a}}\Big(\{x+iy\in\mathbb{H}: 0\le x<w_{\mathfrak{a}},\ y\ge Y\}\Big).
\]
Let $\pi:\mathbb{H}\to M$ be the quotient map.  For $Y$ sufficiently large the sets
$\pi(C_{\mathfrak{a}}(Y))$ are pairwise disjoint (standard cusp separation).
We then define the truncated surface
\begin{equation}\label{eq:truncated-surface}
  M(Y):= M \setminus \bigcup_{\mathfrak{a}\in\mathfrak{A}(M)} \pi\big(C_{\mathfrak{a}}(Y)\big).
\end{equation}
The family $\{M(Y)\}_{Y\ge Y_0}$ is an exhaustion of $M$ by compact surfaces with boundary.

\paragraph{Exact volume and boundary length.}
With $d\mu(z)=dx\,dy/y^2$ the hyperbolic area element, one has for any $0<Y_1<Y_2$,
\[
  \vol\!\left(\sigma_{\mathfrak{a}}\{0\le x<w_{\mathfrak{a}},\ Y_1\le y\le Y_2\}\right)
  = w_{\mathfrak{a}}\int_{Y_1}^{Y_2}\frac{dy}{y^2}
  = w_{\mathfrak{a}}\left(\frac1{Y_1}-\frac1{Y_2}\right).
\]
Letting $Y_2\to\infty$ yields the tail identity
\begin{equation}\label{eq:cusp-tail}
  \vol\big(\pi(C_{\mathfrak{a}}(Y))\big)=\frac{w_{\mathfrak{a}}}{Y}.
\end{equation}
The boundary component of $\partial M(Y)$ corresponding to $\mathfrak{a}$ is the horocycle
$y=Y$ in $\mathfrak{a}$--coordinates, hence
\begin{equation}\label{eq:cusp-boundary-length}
  \length\big(\partial_{\mathfrak{a}} M(Y)\big)=\int_0^{w_{\mathfrak{a}}}\frac{dx}{Y}=\frac{w_{\mathfrak{a}}}{Y}.
\end{equation}
Summing \eqref{eq:cusp-tail} and \eqref{eq:cusp-boundary-length} over cusps gives the global formulas
\begin{equation}\label{eq:global-tail-boundary}
  \vol(M\setminus M(Y))=\sum_{\mathfrak{a}}\frac{w_{\mathfrak{a}}}{Y},
  \qquad
  \length(\partial M(Y))=\sum_{\mathfrak{a}}\frac{w_{\mathfrak{a}}}{Y}.
\end{equation}

\begin{lemma}[Linear decay of cusp tails]\label{lem:linear-decay}
Let $C_\Gamma:=\sum_{\mathfrak{a}} w_{\mathfrak{a}}$.  Then for all $Y\ge Y_0$,
\[
  \vol(M\setminus M(Y)) \le \frac{C_\Gamma}{Y},
  \qquad
  \length(\partial M(Y)) \le \frac{C_\Gamma}{Y}.
\]
\end{lemma}

\begin{proof}
This is immediate from \eqref{eq:global-tail-boundary}.
\end{proof}

\paragraph{Injectivity radius in cusp regions (quantitative form).}
In $\mathfrak{a}$--coordinates the generator of $\Gamma_{\mathfrak{a}}$ acts by
$x\mapsto x+w_{\mathfrak{a}}$.  At height $y$ the hyperbolic length of the
corresponding closed horocycle segment is $w_{\mathfrak{a}}/y$ (by \eqref{eq:cusp-boundary-length}).
Thus the local injectivity radius on the cusp tail is controlled by $w_{\mathfrak{a}}/y$.

\begin{lemma}[Cusp injectivity radius bound]\label{lem:inj-radius-cusp}
Let $w_{\min}:=\min_{\mathfrak{a}} w_{\mathfrak{a}}>0$.  There exists an absolute constant $c_0>0$
such that for every $z\in \pi(C_{\mathfrak{a}}(Y))$ with $Y\ge Y_0$,
\[
  \inj(z)\ge c_0\,\frac{w_{\min}}{\Im(\sigma_{\mathfrak{a}}^{-1}z)}\ \ge\ c_0\,\frac{w_{\min}}{Y}.
\]
\end{lemma}

\begin{proof}
The shortest nontrivial deck displacement in the cusp tail is realized by the primitive
translation in $\Gamma_{\mathfrak{a}}$, whose hyperbolic length at height $y$ equals $w_{\mathfrak{a}}/y$.
The injectivity radius is half the length of the shortest closed geodesic loop through $z$; in the cusp tail
this is bounded below by a fixed fraction of $w_{\mathfrak{a}}/y$.  Taking the minimum over $\mathfrak{a}$
yields the stated uniform bound.
\end{proof}

\subsubsection{Sharp and smoothed truncation; Sobolev stability}\label{subsubsec:smooth-truncation}

Truncation is used in two distinct but compatible ways:
(i) geometrically, to restrict kernels to a compact core;
(ii) spectrally, to regularize Eisenstein inner products and to isolate
the parabolic contribution.  Sharp truncation is adequate for $L^2$--bounds but
not for commutator estimates; for the latter we fix a smoothed truncation.

\paragraph{Sharp truncation operator.}
For $Y\ge Y_0$ define $\Lambda^Y:L^2(M)\to L^2(M)$ by
\[
  (\Lambda^Y f)(z)=
  \begin{cases}
    f(z), & z\in M(Y),\\[2pt]
    0,    & z\in M\setminus M(Y).
  \end{cases}
\]
This is a bounded operator with $\|\Lambda^Y\|_{L^2\to L^2}\le 1$ but it is not smooth
across $\partial M(Y)$.

\paragraph{Smoothed truncation: canonical construction.}
Fix a function $\eta\in C^\infty([0,\infty))$ such that
\[
  0\le \eta\le 1,\qquad \eta(u)=1\ (u\le 1),\qquad \eta(u)=0\ (u\ge 2),
\]
and fix this $\eta$ globally for the entire monograph.  For $Y\ge Y_0$ set
\[
  \eta_Y(u):=\eta\!\left(\frac{u}{Y}\right).
\]
Define $\Lambda^Y_{\mathrm{sm}}$ by multiplication in cusp coordinates:
for $z\in \pi(C_{\mathfrak{a}}(Y_0))$ put
\begin{equation}\label{eq:smoothed-truncation}
  (\Lambda^Y_{\mathrm{sm}} f)(z)
  := \eta_Y\!\big(\Im(\sigma_{\mathfrak{a}}^{-1}z)\big)\, f(z),
\end{equation}
and set $\Lambda^Y_{\mathrm{sm}} f(z):=f(z)$ on the compact complement of the cusp charts.
Then $\Lambda^Y_{\mathrm{sm}}$ equals the identity on $M(Y)$ and vanishes on cusp heights $\ge 2Y$.

\begin{lemma}[Uniform $L^2$ and Sobolev bounds]\label{lem:sobolev-trunc}
For every $s\ge 0$ there exists $C_s<\infty$ depending only on $s$, $\Gamma$ and the fixed cutoff $\eta$
such that for all $Y\ge Y_0$,
\[
  \|\Lambda^Y_{\mathrm{sm}} f\|_{H^s(M)} \le C_s \|f\|_{H^s(M)}.
\]
Moreover $\|\Lambda^Y_{\mathrm{sm}}\|_{L^2\to L^2}\le 1$ and
\[
  \supp(\Lambda^Y-\Lambda^Y_{\mathrm{sm}})\ \subset\ \bigcup_{\mathfrak{a}} \pi\!\left(\sigma_{\mathfrak{a}}\{0\le x<w_{\mathfrak{a}},\ Y\le y\le 2Y\}\right).
\]
\end{lemma}

\begin{proof}
The operator $\Lambda^Y_{\mathrm{sm}}$ is multiplication by a bounded smooth function whose derivatives
in cusp coordinates satisfy uniform estimates of the form $|\partial_y^k \eta_Y(y)|\ll_k Y^{-k}$.
Standard Sobolev multiplier estimates on manifolds with bounded geometry on the compact core and
explicit cusp charts yield uniform $H^s$--boundedness.  The support statement is immediate from the
definition of $\eta$.
\end{proof}

\paragraph{Commutator placeholder (proved later).}
Let $P_{\lambda,\eta}$ denote the spectral projector to a window around $\lambda$ (defined in Chapter~4).
The key commutator estimate needed later has the form
\begin{equation}\label{eq:commutator-placeholder}
  \big\|[\Lambda^Y_{\mathrm{sm}},P_{\lambda,\eta}]\big\|_{L^2\to L^2}
  \ \le\ C_{\Gamma,\beta,\eta}\,\mathcal{E}(\lambda,Y),
\end{equation}
where $\mathcal{E}(\lambda,Y)\to 0$ as $Y\to\infty$ in the regime chosen in Chapter~4.
We record \eqref{eq:commutator-placeholder} here only as a \emph{dependency commitment}:
all constants in that estimate depend solely on the ledger parameters and on the fixed cutoff $\eta$.

\subsubsection{Eisenstein series, scattering, and Maass--Selberg relations}\label{subsubsec:eisenstein}

\paragraph{Eisenstein series.}
For each cusp $\mathfrak{a}$ define for $\Re(s)>1$
\begin{equation}\label{eq:eisenstein-def}
  E_{\mathfrak{a}}(z,s)
  := \sum_{\gamma\in\Gamma_{\mathfrak{a}}\backslash\Gamma}
     \Im(\sigma_{\mathfrak{a}}^{-1}\gamma z)^{s}.
\end{equation}
This converges absolutely for $\Re(s)>1$, extends meromorphically to $s\in\mathbb{C}$, and satisfies
a functional equation $s\mapsto 1-s$ encoded by the scattering matrix (Hejhal~\cite{Hejhal1983}).

\paragraph{Fourier expansion at cusps.}
Let $\mathfrak{b}$ be a cusp and write $z=x+iy$ in $\mathfrak{b}$--coordinates, i.e. consider
$E_{\mathfrak{a}}(\sigma_{\mathfrak{b}} z,s)$.  Then
\begin{equation}\label{eq:eisenstein-fourier}
  E_{\mathfrak{a}}(\sigma_{\mathfrak{b}} z,s)
  = \delta_{\mathfrak{a}\mathfrak{b}}\, y^{s}
    + \varphi_{\mathfrak{a}\mathfrak{b}}(s)\, y^{1-s}
    + \sum_{n\neq 0}
      \rho_{\mathfrak{a}\mathfrak{b}}(n,s)\,
      \sqrt{y}\,K_{s-\frac12}(2\pi |n|y)\,e^{2\pi i n x},
\end{equation}
where $K_\nu$ is the modified Bessel function of the second kind, and the matrix
$\Phi(s):=(\varphi_{\mathfrak{a}\mathfrak{b}}(s))_{\mathfrak{a},\mathfrak{b}}$ is the \emph{scattering matrix}.

\paragraph{Scattering matrix properties.}
The scattering matrix satisfies
\begin{equation}\label{eq:scattering-fe}
  \Phi(s)\Phi(1-s)=I,
\end{equation}
and is unitary on the critical line $\Re(s)=1/2$:
\begin{equation}\label{eq:scattering-unitary}
  \Phi\!\left(\tfrac12+ir\right)^*\,\Phi\!\left(\tfrac12+ir\right)=I
  \qquad (r\in\mathbb{R}).
\end{equation}
Consequently, $\det\Phi(s)\det\Phi(1-s)=1$.

\paragraph{Maass--Selberg relations (canonical form).}
Define the sharp truncation $\Lambda^Y$ as above.  For $\Re(s)=\Re(s')=1/2$ and $Y\ge Y_0$,
the truncated inner products satisfy a precise identity with logarithmic divergence and scattering data.
We record the standard form (Selberg/Hejhal):
\begin{equation}\label{eq:maass-selberg}
  \left\langle \Lambda^Y E_{\mathfrak{a}}(\cdot,s),\, \Lambda^Y E_{\mathfrak{b}}(\cdot,\overline{s'}) \right\rangle_{L^2(M)}
  = \delta_{\mathfrak{a}\mathfrak{b}}\cdot \frac{Y^{s+\overline{s'}-1}-Y^{1-s-\overline{s'}}}{s+\overline{s'}-1}
    + \frac{\varphi'_{\mathfrak{a}\mathfrak{b}}}{\varphi_{\mathfrak{a}\mathfrak{b}}}(s)\,\delta_{s=s'}
    + \mathcal{R}_{\mathfrak{a}\mathfrak{b}}(s,s';Y),
\end{equation}
where $\mathcal{R}_{\mathfrak{a}\mathfrak{b}}(s,s';Y)=O_{\Gamma,\beta}\!\big(Y^{-1}\big)$ uniformly for $s,s'$ in compact subsets of the
critical line away from poles.  In the diagonal specialization $s=s'=\tfrac12+ir$ this implies
\begin{equation}\label{eq:maass-selberg-diag}
  \left\langle \Lambda^Y E_{\mathfrak{a}}\!\left(\cdot,\tfrac12+ir\right),\, \Lambda^Y E_{\mathfrak{b}}\!\left(\cdot,\tfrac12+ir\right) \right\rangle
  = \delta_{\mathfrak{a}\mathfrak{b}}\log Y
    - \Re\!\left(\frac{\varphi'_{\mathfrak{a}\mathfrak{b}}}{\varphi_{\mathfrak{a}\mathfrak{b}}}\!\left(\tfrac12+ir\right)\right)
    + O_{\Gamma,\beta}\!\left(\frac1Y\right).
\end{equation}
The sign convention in \eqref{eq:maass-selberg-diag} matches the Hejhal normalization and will be used consistently.

\begin{remark}[Plancherel measure commitment]\label{rem:plancherel-measure}
The continuous spectral measure is $dr/(4\pi)$ throughout.  In particular, the spectral resolution of $L^2(M)$
uses $\{E_{\mathfrak{a}}(\cdot,\tfrac12+ir)\}_{\mathfrak{a}}$ with the measure $dr/(4\pi)$ and the scattering normalization above.
\end{remark}

\paragraph{Parabolic term in the trace formula (structural template).}
When inserting a radial test function with Selberg transform $h(r)$ into the pre-trace identity and
separating the continuous spectrum, one obtains a parabolic/scattering contribution of the schematic form
\begin{equation}\label{eq:parabolic-template}
  \mathcal{G}_{\mathrm{para}}[h]
  = \frac1{4\pi}\int_{\mathbb{R}} h(r)\,\mathrm{tr}\!\left(\Phi'\Phi^{-1}\right)\!\left(\tfrac12+ir\right)\,dr
  \ +\ (\text{explicit }\log Y\text{-terms})\ +\ O_{\Gamma,\beta}\!\left(\frac1Y\right),
\end{equation}
where the explicit $\log Y$ terms cancel against the truncation of the geometric side in the full Selberg trace formula.
Equation \eqref{eq:parabolic-template} is the normalization anchor for Chapter~6.

\subsubsection{A tail integral tool}\label{subsubsec:tail-tool}

We isolate a single elementary estimate that repeatedly converts cusp-tail integrals into boundary terms.

\begin{lemma}[Tail integral estimate]\label{lem:tail-integral}
Let $F\in C^\infty([Y,\infty))$ satisfy $|y^{k}F^{(k)}(y)|\le C_k$ for all $k\ge 0$ and all $y\ge Y$.
Then
\[
  \int_Y^\infty F(y)\,\frac{dy}{y^2}=\frac{F(Y)}{Y}+O\!\left(\frac1{Y^2}\right),
\]
where the implied constant depends only on $C_0$ and $C_1$.
\end{lemma}

\begin{proof}
Integrate by parts:
\[
  \int_Y^\infty F(y)\,\frac{dy}{y^2}
  =\left[-\frac{F(y)}{y}\right]_{Y}^{\infty}+\int_Y^\infty \frac{F'(y)}{y}\,dy
  =\frac{F(Y)}{Y}+\int_Y^\infty \frac{F'(y)}{y}\,dy.
\]
Since $|F'(y)|\le C_1/y$, the remaining integral is $O(\int_Y^\infty y^{-2}dy)=O(Y^{-1})$; refining with the assumed bounds gives $O(Y^{-2})$.
\end{proof}

\begin{corollary}[Boundary reduction in cusp tails]\label{cor:boundary-reduction}
In cusp coordinates, integrals of the form $\int_{y\ge Y} F(y)\,d\mu$ reduce to boundary evaluations at $y=Y$
up to an error $O(Y^{-2})$ under the hypotheses of Lemma~\ref{lem:tail-integral}.
\end{corollary}
