\subsection{Cusps, truncation, and Eisenstein series}\label{subsec:cusps}

Let $\Gamma\subset\PSL_2(\R)$ be cofinite and $M=\Gamma\backslash\H$.
We use $ds^2=(dx^2+dy^2)/y^2$ and $d\mu(z)=dx\,dy/y^2$.
All implied constants in this subsection depend only on $\Gamma$ and the fixed cusp data
$\{\sigma_{\mathfrak a},w_{\mathfrak a}\}$, unless explicitly stated otherwise.

% ---------------------------------------------------------------------
% Cusp data and scaling
% ---------------------------------------------------------------------
Let $\mathfrak a_1,\dots,\mathfrak a_h$ be the cusps of $M$.
For each cusp $\mathfrak a$ let $\Gamma_{\mathfrak a}$ be its stabilizer in $\Gamma$.
Fix $\sigma_{\mathfrak a}\in\PSL_2(\R)$ with $\sigma_{\mathfrak a}(\infty)=\mathfrak a$
and a width $w_{\mathfrak a}>0$ such that
\begin{equation}\label{eq:scaling-conj}
\sigma_{\mathfrak a}^{-1}\Gamma_{\mathfrak a}\sigma_{\mathfrak a}
=
\left\{
\pm\begin{pmatrix}1& n w_{\mathfrak a}\\ 0&1\end{pmatrix} : n\in\Z
\right\}.
\end{equation}
Such $\sigma_{\mathfrak a}$ is unique up to right-multiplication by
$\bigl(\begin{smallmatrix}1&t\\0&1\end{smallmatrix}\bigr)$, $t\in\R$.

Set $w_{\min}:=\min_{\mathfrak a} w_{\mathfrak a}$ and $W_{\Gamma}:=\sum_{\mathfrak a} w_{\mathfrak a}$.

% ---------------------------------------------------------------------
% Height and truncation
% ---------------------------------------------------------------------
For $z\in M$ lying in the cusp chart of $\mathfrak a$, define the height
\begin{equation}\label{eq:height-function}
y_{\mathfrak a}(z):=\Im(\sigma_{\mathfrak a}^{-1}\tilde z),
\end{equation}
where $\tilde z\in\H$ is any lift of $z$.

For $Y>1$ define the truncated cusp tail and the truncated surface
\begin{equation}\label{eq:trunc-def}
C_{\mathfrak a}(Y):=\sigma_{\mathfrak a}\{x+iy:0\le x<w_{\mathfrak a},\ y\ge Y\},
\qquad
M(Y):=M\setminus\bigcup_{\mathfrak a}\pi\bigl(C_{\mathfrak a}(Y)\bigr).
\end{equation}
There exists $Y_0=Y_0(\Gamma)$ such that for all $Y\ge Y_0$ the sets $\pi(C_{\mathfrak a}(Y))$
are pairwise disjoint and $M(Y)$ is compact with horocyclic boundary.

% ---------------------------------------------------------------------
% Exact tail volume and boundary length
% ---------------------------------------------------------------------
\begin{lemma}[Cusp tail identities]\label{lem:cusp-tail-identities}
For each cusp $\mathfrak a$ and $Y\ge Y_0$,
\begin{equation}\label{eq:cusp-vol-length}
\vol\Bigl(\pi(C_{\mathfrak a})\setminus \pi(C_{\mathfrak a}(Y))\Bigr)=\frac{w_{\mathfrak a}}{Y},
\qquad
\length\bigl(\partial \pi(C_{\mathfrak a}(Y))\bigr)=\frac{w_{\mathfrak a}}{Y}.
\end{equation}
In particular,
\begin{equation}\label{eq:global-vol-length}
\vol(M\setminus M(Y))=\frac{W_{\Gamma}}{Y},
\qquad
\length(\partial M(Y))=\frac{W_{\Gamma}}{Y}.
\end{equation}
\end{lemma}

% ---------------------------------------------------------------------
% Truncation operators
% ---------------------------------------------------------------------
Define the sharp truncation $\Lambda^{Y}:L^{2}(M)\to L^{2}(M)$ by
\begin{equation}\label{eq:sharp-trunc}
(\Lambda^{Y}f)(z)=\mathbf 1_{M(Y)}(z)\,f(z).
\end{equation}

Fix $\eta\in C^{\infty}([0,\infty))$ with $0\le\eta\le 1$, $\eta(u)=1$ for $u\le 1$,
$\eta(u)=0$ for $u\ge 2$, and set $\eta_Y(u)=\eta(u/Y)$.
Define the smoothed truncation $\Lambda^{Y}_{\mathrm{sm}}$ by
\begin{equation}\label{eq:smooth-trunc}
(\Lambda^{Y}_{\mathrm{sm}}f)(z)=
\begin{cases}
\eta_Y\!\bigl(y_{\mathfrak a}(z)\bigr)\,f(z), & z\in \pi(C_{\mathfrak a}),\\
f(z), & z\in M\setminus\bigcup_{\mathfrak a}\pi(C_{\mathfrak a}).
\end{cases}
\end{equation}

\begin{lemma}[Sobolev stability]\label{lem:smooth-trunc-sobolev}
For each $s\ge 0$,
\[
\|\Lambda^{Y}_{\mathrm{sm}}\|_{H^{s}(M)\to H^{s}(M)}\le C_{s,\Gamma,\eta}
\quad\text{uniformly for }Y\ge 1,
\qquad
\|\Lambda^{Y}_{\mathrm{sm}}\|_{L^{2}\to L^{2}}\le 1.
\]
\end{lemma}

\begin{remark}[Tail extraction principle]\label{rem:tail-extraction}
If $F\in C^{2}([Y,\infty))$ satisfies
$\sup_{y\ge Y}\bigl(|F(y)|+y|F'(y)|+y^{2}|F''(y)|\bigr)\le A$,
then
\[
\int_{Y}^{\infty} F(y)\,\frac{dy}{y^{2}}
=
\frac{F(Y)}{Y}
+
O\!\left(\frac{A}{Y^{2}}\right),
\qquad (Y\ge 1).
\]
\end{remark}

% ---------------------------------------------------------------------
% Eisenstein series and scattering matrix
% ---------------------------------------------------------------------
For $\Re(s)>1$ define the Eisenstein series
\begin{equation}\label{eq:eisenstein-def}
E_{\mathfrak a}(z,s):=
\sum_{\gamma\in \Gamma_{\mathfrak a}\backslash \Gamma}
\Im\!\bigl(\sigma_{\mathfrak a}^{-1}\gamma z\bigr)^{s}.
\end{equation}
It continues meromorphically to $s\in\C$ and satisfies the functional equation $s\mapsto 1-s$.

At a cusp $\mathfrak b$ one has the Fourier expansion
\begin{equation}\label{eq:eis-fourier}
E_{\mathfrak a}(\sigma_{\mathfrak b}(x+iy),s)
=
\delta_{\mathfrak a\mathfrak b}\,y^{s}
+
\varphi_{\mathfrak a\mathfrak b}(s)\,y^{1-s}
+
\sum_{n\neq 0}\rho_{\mathfrak a\mathfrak b}(n,s)\,
\sqrt{y}\,K_{s-\frac12}(2\pi|n|y)\,e^{2\pi i n x}.
\end{equation}
Let $\Phi(s)=(\varphi_{\mathfrak a\mathfrak b}(s))_{\mathfrak a,\mathfrak b}$ be the scattering matrix.
Then $\Phi(s)\Phi(1-s)=I$ and $\Phi(\tfrac12+ir)$ is unitary for $r\in\R$.
The continuous Plancherel measure is fixed as $dr/(4\pi)$.

% ---------------------------------------------------------------------
% Maass--Selberg relations
% ---------------------------------------------------------------------
\begin{theorem}[Maass--Selberg relations]\label{thm:maass-selberg}
Let $s=\tfrac12+ir$ with $r\in\R$ and $Y\ge Y_0(\Gamma)$.
Then for any cusps $\mathfrak a,\mathfrak b$,
\begin{equation}\label{eq:maass-selberg}
\Big\langle \Lambda^{Y}E_{\mathfrak a}(\cdot,s),\,\Lambda^{Y}E_{\mathfrak b}(\cdot,\overline{s})\Big\rangle
=
\delta_{\mathfrak a\mathfrak b}\,\log Y
+
\Re\!\left(\frac{\varphi'_{\mathfrak a\mathfrak b}}{\varphi_{\mathfrak a\mathfrak b}}(s)\right)
+
O_{\Gamma}\!\left(\frac{1}{Y}\right).
\end{equation}
The same holds with $\Lambda^{Y}$ replaced by $\Lambda^{Y}_{\mathrm{sm}}$, with error $O_{\Gamma,\eta}(Y^{-1})$.
\end{theorem}

% ---------------------------------------------------------------------
% Parabolic spectral term
% ---------------------------------------------------------------------
For any admissible even test function $h(r)$,
\begin{equation}\label{eq:para-term}
G_{\mathrm{para}}[h]
=
\sum_{\mathfrak a,\mathfrak b}
\int_{-\infty}^{\infty}
h(r)\,
\Re\!\left(
\frac{\varphi'_{\mathfrak a\mathfrak b}}{\varphi_{\mathfrak a\mathfrak b}}
\!\left(\tfrac12+ir\right)
\right)\,
\frac{dr}{4\pi}.
\end{equation}

% ---------------------------------------------------------------------
% Block audit (reference certificate)
% ---------------------------------------------------------------------
\begin{remark}[Audit]\label{rem:cusps-audit}
Equations \eqref{eq:scaling-conj}--\eqref{eq:para-term} fix the cusp coordinate system,
truncation operators, Eisenstein normalization, and the Maass--Selberg renormalization.
All later parabolic and continuous-spectrum contributions are expressed in these conventions.
\end{remark}
