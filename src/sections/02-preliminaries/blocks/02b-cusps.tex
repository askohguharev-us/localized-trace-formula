\subsection{Cuspidal geometry, truncation, and the continuous spectrum}
\label{subsec:cusps-truncation-eisenstein}

\subsubsection*{Standing object and cusp ledger}
Let
\[
  M=\Gamma\backslash\mathbb H,\qquad \Gamma\subset \mathrm{PSL}_2(\mathbb R)
\]
be a cofinite Fuchsian group, and write $d\mu(z)=\frac{dx\,dy}{y^2}$ for the hyperbolic area measure on $\mathbb H$.
Cusps of $M$ are $\Gamma$--equivalence classes of parabolic fixed points in $\mathbb P^1(\mathbb R)$; there are finitely many,
\[
  \{\mathfrak a_1,\dots,\mathfrak a_h\}.
\]
For each cusp $\mathfrak a$ let
\[
  \Gamma_{\mathfrak a}=\{\gamma\in\Gamma:\gamma\mathfrak a=\mathfrak a\}
\]
denote its stabilizer, an infinite cyclic subgroup generated by a primitive parabolic element.

\medskip
\noindent
\textbf{Cusp data (persistent invariants).}
For each cusp $\mathfrak a$ we fix:
\begin{enumerate}[label=(C\arabic*),leftmargin=2.2em]
\item a \emph{scaling matrix} $\sigma_{\mathfrak a}\in \mathrm{PSL}_2(\mathbb R)$ with $\sigma_{\mathfrak a}(\infty)=\mathfrak a$;
\item a \emph{width} $w_{\mathfrak a}>0$ such that
\[
  \sigma_{\mathfrak a}^{-1}\Gamma_{\mathfrak a}\sigma_{\mathfrak a}
  =
  \Big\{
    \pm
    \begin{pmatrix}1&n\,w_{\mathfrak a}\\0&1\end{pmatrix}
    : n\in\mathbb Z
  \Big\};
\]
\item the associated \emph{cusp height function}
\[
  y_{\mathfrak a}(z):=\Im(\sigma_{\mathfrak a}^{-1}z),\qquad z\in \mathbb H.
\]
\end{enumerate}
All cusp-dependent constants in this monograph are allowed to depend \emph{only} on the finite dataset
\[
  \mathcal C(\Gamma):=\big(h;\ \{w_{\mathfrak a}\}_{\mathfrak a};\ \{\sigma_{\mathfrak a}\}_{\mathfrak a}\big),
\]
and on the fixed spectral-gap parameter $\beta$ introduced in \S\ref{sec:prelim-ledger}.

\begin{remark}[Uniqueness up to translation]\label{rem:scaling-ambiguity}
If $\sigma_{\mathfrak a}$ and $\sigma'_{\mathfrak a}$ are scaling matrices for the same cusp $\mathfrak a$ with the \emph{same} width $w_{\mathfrak a}$,
then there exists $t\in\mathbb R$ such that
\[
  \sigma'_{\mathfrak a}=\sigma_{\mathfrak a}\begin{pmatrix}1&t\\0&1\end{pmatrix}.
\]
In particular, the coordinate $x$ in $\sigma_{\mathfrak a}^{-1}z=x+iy$ is defined only up to horizontal translation; all constructions below are invariant under this freedom.
\end{remark}

\subsubsection*{Canonical cusp neighborhoods and disjointness at height}
Define the standard strip of width $w>0$,
\[
  \mathcal S(w)=\{x+iy\in\mathbb H:\ 0\le x<w,\ y>0\}.
\]
The set $\sigma_{\mathfrak a}(\mathcal S(w_{\mathfrak a}))$ projects to a neighborhood of the cusp $\mathfrak a$ in $M$.

\begin{lemma}[Disjoint cusp tails]\label{lem:disjoint-tails}
There exists $Y_0=Y_0(\Gamma)\ge 1$ such that for all $Y\ge Y_0$ the cusp-tail sets
\[
  \mathcal U_{\mathfrak a}(Y):=\pi\big(\{z\in\mathbb H:\ y_{\mathfrak a}(z)\ge Y\}\big)\subset M
\]
are pairwise disjoint as $\mathfrak a$ varies over cusps. Consequently the \emph{truncated surface}
\[
  M(Y):=M\setminus \bigcup_{\mathfrak a}\mathcal U_{\mathfrak a}(Y)
\]
is a compact surface with smooth (horocyclic) boundary for all $Y\ge Y_0$.
\end{lemma}

\begin{proof}[Proof sketch]
This is standard for cofinite Fuchsian groups: cusp neighborhoods form a finite family of embedded ends and become disjoint after raising the height parameter.
A proof can be extracted from the standard construction of a fundamental domain with cusp neighborhoods (e.g. \cite{Hejhal1983,Buser1992,Iwaniec2002}).
\end{proof}

% ---------------------------------------------------------------------
\subsubsection*{Exact volume and boundary length at height}
For $Y\ge Y_0$, each boundary component of $M(Y)$ is a horocycle at height $Y$ in the corresponding cusp coordinate.

\begin{lemma}[Cusp tail volume and boundary length]\label{lem:cusp-volume-length}
For every cusp $\mathfrak a$ and every $Y\ge Y_0$ one has
\[
  \mathrm{vol}\big(\mathcal U_{\mathfrak a}(Y)\big)=\frac{w_{\mathfrak a}}{Y},
  \qquad
  \mathrm{length}\big(\partial \mathcal U_{\mathfrak a}(Y)\big)=\frac{w_{\mathfrak a}}{Y}.
\]
Consequently,
\[
  \mathrm{vol}\big(M\setminus M(Y)\big)=\sum_{\mathfrak a}\frac{w_{\mathfrak a}}{Y},
  \qquad
  \mathrm{length}\big(\partial M(Y)\big)=\sum_{\mathfrak a}\frac{w_{\mathfrak a}}{Y}.
\]
\end{lemma}

\begin{proof}
In cusp coordinates $z=x+iy$ with $0\le x<w_{\mathfrak a}$ and $y\ge Y$,
\[
  d\mu=\frac{dx\,dy}{y^2},
\]
so
\[
  \mathrm{vol}\big(\mathcal U_{\mathfrak a}(Y)\big)=\int_Y^\infty\int_0^{w_{\mathfrak a}}\frac{dx\,dy}{y^2}
  =w_{\mathfrak a}\int_Y^\infty \frac{dy}{y^2}
  =\frac{w_{\mathfrak a}}{Y}.
\]
The boundary horocycle at height $Y$ has induced metric $ds=\frac{dx}{Y}$, hence length $\int_0^{w_{\mathfrak a}}\frac{dx}{Y}=\frac{w_{\mathfrak a}}{Y}$.
Summing over disjoint cusps gives the global identities.
\end{proof}

\begin{corollary}[Uniform decay]\label{cor:linear-decay}
Let $W_\Gamma:=\sum_{\mathfrak a} w_{\mathfrak a}$. Then for all $Y\ge Y_0$,
\[
  \mathrm{vol}(M\setminus M(Y))\le \frac{W_\Gamma}{Y},
  \qquad
  \mathrm{length}(\partial M(Y))\le \frac{W_\Gamma}{Y}.
\]
\end{corollary}

\subsubsection*{Exact displacement and injectivity radius in cusps}
The next lemma records the exact hyperbolic displacement under one primitive translation at height $y$.

\begin{lemma}[Exact horizontal displacement]\label{lem:exact-horiz-distance}
For $z=x+iy\in\mathbb H$ and $w>0$ one has
\[
  d_{\mathbb H}(z,z+w)=2\,\operatorname{arsinh}\!\Big(\frac{w}{2y}\Big).
\]
In particular,
\[
  d_{\mathbb H}(z,z+w)\asymp \min\Big\{1,\frac{w}{y}\Big\},
\]
with absolute implied constants.
\end{lemma}

\begin{proof}
This is a standard identity for the hyperbolic metric on $\mathbb H$; it follows from the formula
\[
  \cosh d_{\mathbb H}(z,z')=1+\frac{|z-z'|^2}{2\,\Im z\,\Im z'}
\]
applied to $z'=z+w$.
\end{proof}

\begin{lemma}[Injectivity radius in cusp tails]\label{lem:inj-cusp}
Let $Y\ge Y_0$ and $p\in \mathcal U_{\mathfrak a}(Y)$.
Then
\[
  \mathrm{inj}(p)\asymp \min\Big\{1,\frac{w_{\mathfrak a}}{y_{\mathfrak a}(p)}\Big\}
  \le c\,\frac{w_{\mathfrak a}}{Y}
\]
for an absolute constant $c>0$. In particular, if $w_{\min}:=\min_{\mathfrak a}w_{\mathfrak a}$ then
\[
  \mathrm{inj}(p)\ge c'\,\frac{w_{\min}}{y_{\mathfrak a}(p)}
  \ge c'\,\frac{w_{\min}}{Y}
\]
for some absolute $c'>0$.
\end{lemma}

\begin{proof}
In the cusp $\mathfrak a$, the shortest nontrivial loop is induced by the primitive translation $x\mapsto x+w_{\mathfrak a}$.
Applying Lemma~\ref{lem:exact-horiz-distance} and using the definition of injectivity radius yields the claim.
\end{proof}

\subsubsection*{Sharp and smoothed truncation operators}
Sharp truncation is convenient geometrically but not stable in Sobolev spaces; we therefore fix a canonical smoothed truncation.

\medskip
\noindent
\textbf{Sharp truncation.}
For $Y\ge Y_0$ define $\Lambda^Y:L^2(M)\to L^2(M)$ by
\[
  (\Lambda^Y f)(p)=
  \begin{cases}
    f(p), & p\in M(Y),\\
    0, & p\in M\setminus M(Y).
  \end{cases}
\]
Then $\|\Lambda^Y\|_{L^2\to L^2}\le 1$.

\medskip
\noindent
\textbf{Smoothed truncation.}
Fix once and for all a function $\eta\in C^\infty(\mathbb R_{\ge 0})$ such that
\[
  \eta(u)=1\ (u\le 1),\qquad \eta(u)=0\ (u\ge 2),\qquad 0\le \eta\le 1.
\]
For $Y\ge Y_0$ set $\eta_Y(u):=\eta(u/Y)$.
Define $\Lambda^Y_{\mathrm{sm}}:C^\infty(M)\to C^\infty(M)$ by
\begin{equation}\label{eq:smoothed-trunc}
  (\Lambda^Y_{\mathrm{sm}} f)(p)
  :=
  \begin{cases}
    \eta_Y\!\big(y_{\mathfrak a}(p)\big)\,f(p), & p\in \mathcal U_{\mathfrak a}(Y_0)\ \text{for some cusp }\mathfrak a,\\
    f(p), & p\in M(Y_0),
  \end{cases}
\end{equation}
which is well-defined because the $\mathcal U_{\mathfrak a}(Y_0)$ are disjoint.
Thus $\Lambda^Y_{\mathrm{sm}}$ equals $1$ on $M(Y)$, equals $0$ on each tail $\{y_{\mathfrak a}\ge 2Y\}$, and transitions smoothly in $Y\le y_{\mathfrak a}\le 2Y$.

\begin{lemma}[Basic operator bounds]\label{lem:smoothed-bounds}
For all $Y\ge Y_0$,
\[
  \|\Lambda^Y_{\mathrm{sm}}\|_{L^2\to L^2}\le 1.
\]
Moreover, for each integer $s\ge 0$ there exists $C_s=C_s(\Gamma,\eta)$ such that
\[
  \|\Lambda^Y_{\mathrm{sm}} f\|_{H^s(M)}\le C_s \|f\|_{H^s(M)}
  \qquad (f\in H^s(M)),
\]
uniformly for all $Y\ge Y_0$.
\end{lemma}

\begin{proof}[Proof sketch]
$\Lambda^Y_{\mathrm{sm}}$ is multiplication by a bounded smooth function; on $H^s$ the bound follows from Leibniz' rule and uniform control of derivatives of $\eta_Y(y_{\mathfrak a}(p))$ in the cusp charts. The constants depend on finitely many derivatives of $\eta$ and on the fixed cusp atlas determined by $\mathcal C(\Gamma)$.
\end{proof}

\begin{remark}[Support of the difference]
The operator $\Lambda^Y-\Lambda^Y_{\mathrm{sm}}$ is supported in the transition region
\[
  \bigcup_{\mathfrak a}\{p\in M:\ Y\le y_{\mathfrak a}(p)\le 2Y\},
\]
whose volume is $\asymp W_\Gamma/Y$ by Lemma~\ref{lem:cusp-volume-length}.
\end{remark}

\medskip
\noindent
\textbf{Recorded commutator bound (proved later).}
Let $P_{\lambda,\eta}$ denote the spectral projector onto the Laplace spectrum in $[\lambda-\eta,\lambda+\eta]$ (see Chapter~4).
We record the canonical estimate (proved in Chapter~4 by semiclassical propagation/Egorov):
there exists $C=C(\Gamma,\beta,\eta)$ such that for all $\lambda\ge 1$,
\begin{equation}\label{eq:commutator-record}
  \big\|[\Lambda^Y_{\mathrm{sm}},P_{\lambda,\eta}]\big\|_{L^2\to L^2}
  \le C\,\frac{\lambda^{1/2}}{Y},
  \qquad Y\ge Y_0.
\end{equation}
This is the quantitative bridge between truncation height and spectral localization.

\subsubsection*{A cusp tail calculus: boundary extraction by integration by parts}
We will repeatedly reduce cusp-tail integrals to boundary evaluations at $y=Y$.

\begin{lemma}[Tail extraction]\label{lem:tail-extraction}
Let $F\in C^\infty([Y,\infty))$ satisfy
\[
  \sup_{y\ge Y}\ |y^{k}F^{(k)}(y)|\le A_k \qquad (k=0,1,2),
\]
for some constants $A_k$. Then
\[
  \int_Y^\infty F(y)\,\frac{dy}{y^2}
  =\frac{F(Y)}{Y}+\int_Y^\infty \frac{F'(y)}{y}\,dy,
\]
and in particular
\[
  \int_Y^\infty F(y)\,\frac{dy}{y^2}
  =\frac{F(Y)}{Y}+O\!\Big(\frac{A_1}{Y}\Big)
  =\frac{F(Y)}{Y}+O\!\Big(\frac{A_2}{Y^2}\Big),
\]
where the implied constants are absolute.
\end{lemma}

\begin{proof}
Integrate by parts:
\[
  \int_Y^\infty F(y)\frac{dy}{y^2}
  =\Big[-\frac{F(y)}{y}\Big]_Y^\infty +\int_Y^\infty \frac{F'(y)}{y}\,dy
  =\frac{F(Y)}{Y}+\int_Y^\infty \frac{F'(y)}{y}\,dy.
\]
The stated bounds follow from the hypotheses.
\end{proof}

\subsubsection*{Eisenstein series and the scattering matrix}
For each cusp $\mathfrak a$ define the Eisenstein series (for $\Re(s)>1$) by
\begin{equation}\label{eq:eisenstein-def}
  E_{\mathfrak a}(z,s)
  :=
  \sum_{\gamma\in \Gamma_{\mathfrak a}\backslash \Gamma}
  \Im\big(\sigma_{\mathfrak a}^{-1}\gamma z\big)^s,
  \qquad \Re(s)>1.
\end{equation}
It converges absolutely for $\Re(s)>1$ and admits meromorphic continuation to $s\in\mathbb C$ with functional equation $s\mapsto 1-s$
(see \cite{Selberg1956,Hejhal1983,Iwaniec2002}).

\medskip
\noindent
\textbf{Fourier expansion and scattering coefficients.}
For cusps $\mathfrak a,\mathfrak b$ one has the Fourier expansion at $\mathfrak b$:
\begin{equation}\label{eq:eisenstein-fourier}
  E_{\mathfrak a}(\sigma_{\mathfrak b} z,s)
  =
  \delta_{\mathfrak a\mathfrak b}\,y^s
  +\varphi_{\mathfrak a\mathfrak b}(s)\,y^{1-s}
  +\sum_{n\ne 0}\rho_{\mathfrak a\mathfrak b}(n,s)\,
  \sqrt{y}\,K_{s-\frac12}(2\pi|n|y)\,e^{2\pi i n x}.
\end{equation}
The matrix
\[
  \Phi(s):=\big(\varphi_{\mathfrak a\mathfrak b}(s)\big)_{\mathfrak a,\mathfrak b}
\]
is the \emph{scattering matrix}. It satisfies
\begin{equation}\label{eq:scattering-fe}
  \Phi(s)\,\Phi(1-s)=I,
\end{equation}
and is unitary on the critical line:
\begin{equation}\label{eq:scattering-unitary}
  \Phi\!\Big(\frac12+ir\Big)^*=\Phi\!\Big(\frac12+ir\Big)^{-1}.
\end{equation}

\subsubsection*{Maass--Selberg relations (contract form)}
The Maass--Selberg relations quantify the truncation divergence of Eisenstein inner products and identify the finite part with the logarithmic derivative of scattering.

\begin{theorem}[Maass--Selberg, matrix form]\label{thm:maass-selberg}
Fix $Y\ge Y_0$ and $r\in\mathbb R$.
Let $E(\cdot,\tfrac12+ir)$ denote the $h$-vector of Eisenstein series $\big(E_{\mathfrak a}(\cdot,\tfrac12+ir)\big)_{\mathfrak a}$.
Then
\begin{equation}\label{eq:maass-selberg-matrix}
  \Big\langle \Lambda^Y E(\cdot,\tfrac12+ir),\ \Lambda^Y E(\cdot,\tfrac12+ir)\Big\rangle
  =
  (\log Y)\,I
  + \Re\!\Big(\Phi'(\tfrac12+ir)\,\Phi(\tfrac12+ir)^{-1}\Big)
  + O_{\Gamma}\!\Big(\frac{1}{Y}\Big),
\end{equation}
where $\langle\cdot,\cdot\rangle$ is the $L^2(M)$ inner product applied entrywise, and the $O_\Gamma(1/Y)$ error is uniform for $r$ in compact sets.
Equivalently, for each pair of cusps $\mathfrak a,\mathfrak b$,
\begin{equation}\label{eq:maass-selberg-entry}
  \big\langle \Lambda^Y E_{\mathfrak a}(\cdot,\tfrac12+ir),\ \Lambda^Y E_{\mathfrak b}(\cdot,\tfrac12-ir)\big\rangle
  =
  \delta_{\mathfrak a\mathfrak b}\,\log Y
  + \Re\!\Big(\big(\Phi' \Phi^{-1}\big)_{\mathfrak a\mathfrak b}\!\big(\tfrac12+ir\big)\Big)
  + O_{\Gamma}\!\Big(\frac{1}{Y}\Big).
\end{equation}
\end{theorem}

\begin{proof}[Proof sketch]
Insert the Fourier expansions \eqref{eq:eisenstein-fourier} into the truncated inner product and integrate termwise over cusp charts.
The divergence arises only from the constant terms $\delta_{\mathfrak a\mathfrak b}y^{1/2\pm ir}$ and $\varphi_{\mathfrak a\mathfrak b}(\tfrac12+ir)y^{1/2\mp ir}$, producing $\log Y$ and the finite term involving $\Phi' \Phi^{-1}$.
All nonconstant Fourier modes decay exponentially in $y$ via the $K$-Bessel factor, and the remaining tail contributions are $O_\Gamma(1/Y)$ by Lemma~\ref{lem:tail-extraction}.
A complete proof is standard; see \cite{Selberg1956,Hejhal1983,Iwaniec2002}.
\end{proof}

\begin{remark}[Smoothed truncation version]
Replacing $\Lambda^Y$ with $\Lambda^Y_{\mathrm{sm}}$ in \eqref{eq:maass-selberg-matrix} changes the left-hand side by $O_\Gamma(1/Y)$,
since $\Lambda^Y-\Lambda^Y_{\mathrm{sm}}$ is supported in the transition strip of volume $\asymp W_\Gamma/Y$ and Eisenstein series have controlled growth there.
Thus \eqref{eq:maass-selberg-matrix} and \eqref{eq:maass-selberg-entry} remain valid with $\Lambda^Y_{\mathrm{sm}}$ (possibly with a different implied constant).
\end{remark}

\subsubsection*{Parabolic (continuous-spectrum) contribution: recorded trace-formula shape}
Let $h(r)$ be an even test function of Paley--Wiener type (normalized as in the Selberg/Harish--Chandra transform block) associated to a radial kernel.
In the spectral side of the trace formula, the continuous spectrum contributes a parabolic/scattering term of the schematic form
\begin{equation}\label{eq:parabolic-term}
  G_{\mathrm{para}}(h)
  :=
  \frac{1}{4\pi}\int_{-\infty}^{\infty}
  h(r)\,
  \Re\!\Big(\mathrm{tr}\big(\Phi'(\tfrac12+ir)\,\Phi(\tfrac12+ir)^{-1}\big)\Big)\,dr,
\end{equation}
together with the explicit truncation divergence $(\log Y)\cdot \frac{h(0)}{4\pi}\cdot h$-fold multiplicity (made precise in Chapter~6 after fixing the pre-trace normalization).
The identity \eqref{eq:parabolic-term} is the analytic bridge between cusp geometry and the continuous spectrum; it is the term that is ultimately controlled via scattering theory and the spectral gap.
