\subsection{Hyperbolic geometry: normalizations, models, and global constants}
\label{subsec:geometry-diamond}

\subsubsection*{G0. Object and invariant ledger}
Throughout, the basic geometric object is the finite-area quotient
\[
  M=\Gamma\backslash \mathbb H,\qquad \Gamma\subset \mathrm{PSL}_2(\mathbb R)\ \text{discrete, cofinite}.
\]
All geometric constants in later chapters are allowed to depend \emph{only} on:
\[
  \mathcal G(\Gamma):=\big(\vol(M);\ \text{torsion orders } \{e_j\};\ \text{cusp data }\mathcal C(\Gamma);\ \text{a fixed compact core }M(Y_0)\big),
\]
where $\mathcal C(\Gamma)=(h;\{w_{\mathfrak a}\};\{\sigma_{\mathfrak a}\})$ is the cusp ledger fixed in \S\ref{subsec:cusps-truncation-eisenstein},
and $Y_0=Y_0(\Gamma)$ is a disjointness height (Lemma~\ref{lem:disjoint-tails}).
No constant in this block depends on later spectral-localization parameters $(\lambda,\eta)$.

% ---------------------------------------------------------------------
\subsubsection*{G1. The hyperbolic plane: metric, measure, distance}
Let
\[
  \mathbb H=\{z=x+iy\in\mathbb C:\ y>0\},
\qquad
  ds^2=\frac{dx^2+dy^2}{y^2},
\qquad
  d\mu(z)=\frac{dx\,dy}{y^2}.
\]
The distance $d_{\mathbb H}(z,w)$ is determined by the identity
\begin{equation}\label{eq:cosh-distance}
  \cosh d_{\mathbb H}(z,w)=1+\frac{|z-w|^2}{2\,\Im z\,\Im w}.
\end{equation}
In particular, for a horizontal shift $w>0$,
\begin{equation}\label{eq:horizontal-distance}
  d_{\mathbb H}(x+iy,\ (x+w)+iy)=2\,\operatorname{arsinh}\!\Big(\frac{w}{2y}\Big)
  \asymp \min\Big\{1,\frac{w}{y}\Big\}.
\end{equation}

\medskip
\noindent\textbf{Geodesics.}
Geodesics in $\mathbb H$ are vertical lines and semicircles orthogonal to $\mathbb R$.
For each pair of distinct boundary points $\alpha,\beta\in \mathbb R\cup\{\infty\}$ there is a unique geodesic with endpoints $\{\alpha,\beta\}$.

% ---------------------------------------------------------------------
\subsubsection*{G2. Isometries and classification; translation length}
$\mathrm{PSL}_2(\mathbb R)$ acts on $\mathbb H$ by Möbius maps
\[
  \gamma z=\frac{az+b}{cz+d},\qquad
  \gamma=\begin{pmatrix}a&b\\c&d\end{pmatrix}\in \mathrm{PSL}_2(\mathbb R),
\]
preserving $ds^2$ and $d\mu$.
Elements are classified by the absolute trace:
elliptic ($|\tr\gamma|<2$), parabolic ($|\tr\gamma|=2$), hyperbolic ($|\tr\gamma|>2$).

\medskip
\noindent\textbf{Hyperbolic translation length.}
If $\gamma$ is hyperbolic, it stabilizes a unique geodesic axis $\mathcal A_\gamma$ and acts on it as a translation by $\ell(\gamma)>0$.
One has the exact identity
\begin{equation}\label{eq:length-trace}
  |\tr(\gamma)|=2\cosh\!\Big(\frac{\ell(\gamma)}{2}\Big).
\end{equation}
If $\gamma=\gamma_0^k$ with $\gamma_0$ primitive hyperbolic, then $\ell(\gamma)=k\,\ell(\gamma_0)$.

% ---------------------------------------------------------------------
\subsubsection*{G3. Balls and exponential area growth}
Let $B_{\mathbb H}(z,R)$ be the hyperbolic ball of radius $R$.
Its area is independent of $z$ and equals
\begin{equation}\label{eq:ball-area}
  \vol\big(B_{\mathbb H}(R)\big)=2\pi(\cosh R-1)
  =\pi e^{R}\bigl(1+O(e^{-2R})\bigr)
  \qquad (R\to\infty).
\end{equation}
This exponential growth is the geometric origin of the ``logarithmic propagation scale'' $T\asymp \log\lambda$ in microlocal arguments.

% ---------------------------------------------------------------------
\subsubsection*{G4. Fundamental and Dirichlet domains; coarse counting law}
Let $\Gamma\subset \mathrm{PSL}_2(\mathbb R)$ be discrete and cofinite.
A (measurable) fundamental domain $\mathcal F$ satisfies $\mathbb H=\bigsqcup_{\gamma\in\Gamma}\gamma\mathcal F$ up to measure zero and
\[
  \vol(M)=\int_{\mathcal F} d\mu<\infty.
\]
Given $z_0\in\mathbb H$, the Dirichlet domain is
\[
  \mathcal D(z_0)=\{z\in\mathbb H:\ d_{\mathbb H}(z,z_0)\le d_{\mathbb H}(z,\gamma z_0)\ \forall \gamma\in\Gamma\}.
\]
It is convex and locally finite; $\vol(M)=\int_{\mathcal D(z_0)}d\mu$.

\medskip
\noindent\textbf{Coarse lattice-point growth (robust form).}
Define the orbit counting function
\[
  N_\Gamma(R;z_0):=\#\{\gamma\in\Gamma:\ d_{\mathbb H}(z_0,\gamma z_0)\le R\}.
\]
Then as $R\to\infty$,
\begin{equation}\label{eq:orbit-count-coarse}
  N_\Gamma(R;z_0)\asymp_\Gamma e^{R}.
\end{equation}
More precise asymptotics and remainder terms depend on spectral input; when needed, they will be invoked with explicit hypotheses and citations in the analytic chapters (cf. \cite{Hejhal1983,Iwaniec2002}).

\medskip
\noindent\textbf{Quantitative separation (local finiteness).}
There exists $\epsilon_\Gamma>0$ such that for any fixed compact set $K\subset\mathbb H$ one has
\begin{equation}\label{eq:separation-compact}
  \gamma K\cap \gamma'K=\varnothing\quad\text{whenever}\quad \gamma\neq\gamma'\ \text{and}\ d_{\mathbb H}(\gamma K,\gamma'K)<\epsilon_\Gamma.
\end{equation}
In particular, for kernels with compact support in the distance variable, orbital/Poincaré sums are locally finite and hence absolutely convergent term-by-term.

% ---------------------------------------------------------------------
\subsubsection*{G5. Closed geodesics and hyperbolic conjugacy}
Primitive closed geodesics on $M$ correspond to primitive hyperbolic conjugacy classes $[\gamma_0]$ in $\Gamma$; their lengths satisfy
\[
  \ell(\mathcal C_{\gamma_0})=\ell(\gamma_0).
\]
The prime geodesic theorem asserts
\begin{equation}\label{eq:PGT}
  \pi_M(R):=\#\{\text{primitive closed geodesics on }M \text{ with }\ell\le R\}
  \sim \frac{e^{R}}{R}\qquad (R\to\infty),
\end{equation}
with effective remainders depending on spectral input (see \cite{Hejhal1983,Iwaniec2002}).
We will use \eqref{eq:PGT} only as a \emph{black-box} bound for hyperbolic orbital sums, with hypotheses stated at the point of use.

% ---------------------------------------------------------------------
\subsubsection*{G6. Cusps, truncation, thick--thin, and collars}
Cusps, widths, scaling matrices, truncation $M(Y)$, and exact tail volume/length formulas are fixed in \S\ref{subsec:cusps-truncation-eisenstein}.
We recall only the structural consequences needed for analysis.

\medskip
\noindent\textbf{Injectivity radius in cusp tails.}
For $p\in \mathcal U_{\mathfrak a}(Y)$, Lemma~\ref{lem:inj-cusp} and \eqref{eq:horizontal-distance} imply
\[
  \mathrm{inj}(p)\asymp \min\Big\{1,\frac{w_{\mathfrak a}}{y_{\mathfrak a}(p)}\Big\}
  \le c\,\frac{w_{\mathfrak a}}{Y}.
\]
This is the precise cusp-scale that enters Sobolev constants and commutator bounds.

\medskip
\noindent\textbf{Thick--thin decomposition (Margulis).}
There exists a universal $\epsilon_*>0$ such that for any $\epsilon\in(0,\epsilon_*)$ the thin part
\[
  M_{<\epsilon}:=\{p\in M:\ \mathrm{inj}(p)<\epsilon\}
\]
is a disjoint union of: (i) cusp neighborhoods (ends), and (ii) embedded collars around short closed geodesics.
The thick part $M_{\ge\epsilon}$ is compact.

\medskip
\noindent\textbf{Collar lemma (embedded tubular neighborhoods).}
Let $\gamma$ be a simple closed geodesic on $M$ of length $\ell=\ell(\gamma)$.
Then there is an embedded collar of half-width
\begin{equation}\label{eq:collar-width}
  w(\ell)=\operatorname{arsinh}\!\Big(\frac{1}{\sinh(\ell/2)}\Big),
\end{equation}
and collars of distinct simple closed geodesics are disjoint.
As $\ell\to 0$ one has $w(\ell)=\log(2/\ell)+O(\ell^2)$.
These collars provide the geometric mechanism behind uniform local charts on the thick part and explicit degeneration control on the thin part.

\medskip
\noindent\textbf{Compact core.}
Fix $Y\ge Y_0(\Gamma)$ so cusp tails are disjoint. Then
\[
  M_{\mathrm{core}}:=M(Y)
\]
is a compact surface with (horocyclic) boundary, serving as the base region for elliptic regularity and Sobolev estimates; cusp contributions are then handled by the explicit model charts.

% ---------------------------------------------------------------------
\subsubsection*{G7. Laplacian normalization and spectral parameter}
On smooth functions on $\mathbb H$ we take the nonnegative Laplace--Beltrami operator
\begin{equation}\label{eq:laplacian}
  \Delta:=-y^2\Big(\partial_x^2+\partial_y^2\Big).
\end{equation}
The spectral parameterization is
\[
  \lambda=\frac14+t^2,\qquad t\in\mathbb R_{\ge 0},
\]
matching the spherical/Plancherel conventions used in the Selberg/Harish--Chandra transform block.

% ---------------------------------------------------------------------
\subsubsection*{G8. Spherical functions and Plancherel measure on \texorpdfstring{$\mathbb H$}{H}}
Let $\varphi_t(r)$ be the $K$--biinvariant spherical function on $\mathbb H$ with $\varphi_t(0)=1$.
It may be written as
\[
  \varphi_t(r)=P_{-1/2+it}(\cosh r),
\]
and admits the standard large-$r$ expansion
\begin{equation}\label{eq:spherical-asympt}
  \varphi_t(r)=c(t)e^{(it-\frac12)r}+c(-t)e^{(-it-\frac12)r}+O(e^{-3r/2}),
  \qquad
  c(t)=\frac{\Gamma(it)}{\Gamma(\frac12+it)}.
\end{equation}
The Plancherel density for radial analysis on $\mathbb H$ is
\begin{equation}\label{eq:plancherel-density}
  d\mu_{\mathrm{Pl}}(t)=\frac{1}{2\pi}\,t\,\tanh(\pi t)\,dt,
\end{equation}
consistent with the inversion formula for the Selberg/Harish--Chandra transform (next block).

% ---------------------------------------------------------------------
\subsubsection*{G9. Model kernels: heat, wave, resolvent; quotient kernels by summation}
\noindent\textbf{Heat kernel on $\mathbb H$.}
The heat kernel $K_\tau(z,w)$ solves $\partial_\tau K_\tau=-\Delta_z K_\tau$ and has spectral representation
\begin{equation}\label{eq:heat-kernel}
  K_\tau(z,w)=\int_0^\infty e^{-(\frac14+t^2)\tau}\,\varphi_t\!\big(d_{\mathbb H}(z,w)\big)\,d\mu_{\mathrm{Pl}}(t).
\end{equation}
As $\tau\to 0^+$ one has the standard Gaussian short-time asymptotic
\[
  K_\tau(z,w)\sim (4\pi\tau)^{-1}e^{-d_{\mathbb H}(z,w)^2/(4\tau)}\quad (\text{with curvature corrections}).
\]

\medskip
\noindent\textbf{Wave kernel on $\mathbb H$.}
The even wave group $\cos\big(\tau\sqrt{\Delta-\frac14}\big)$ has kernel
\begin{equation}\label{eq:wave-kernel}
  W_\tau(z,w)=\int_0^\infty \cos(\tau t)\,\varphi_t\!\big(d_{\mathbb H}(z,w)\big)\,d\mu_{\mathrm{Pl}}(t).
\end{equation}
Precise dispersive estimates are classical; we will use only that smooth compactly supported spectral cutoffs yield kernels with rapid off-diagonal decay and controlled propagation on times $\tau\ll \log\lambda$ in semiclassical regimes.

\medskip
\noindent\textbf{Resolvent on $\mathbb H$.}
For $s\in\mathbb C$ with $\Re(s)>\frac12$ let $R(s)=(\Delta-s(1-s))^{-1}$.
Its kernel admits the spectral integral representation
\begin{equation}\label{eq:resolvent-kernel}
  R(s;z,w)=\int_0^\infty \frac{\varphi_t\!\big(d_{\mathbb H}(z,w)\big)}{t^2+(s-\frac12)^2}\,d\mu_{\mathrm{Pl}}(t),
\end{equation}
and extends meromorphically in $s$.

\medskip
\noindent\textbf{Quotient kernels by Poincaré summation.}
For a $\Gamma$--invariant kernel built from a model kernel $K(z,w)$ on $\mathbb H$, define
\begin{equation}\label{eq:poincare-kernel}
  K^\Gamma(z,w):=\sum_{\gamma\in\Gamma} K(z,\gamma w),
\end{equation}
whenever the sum is absolutely convergent (or in the sense of analytic continuation).
Local finiteness \eqref{eq:separation-compact}, together with decay of $K$ in $d_{\mathbb H}(z,w)$ (e.g.\ for heat kernels $\tau>0$ and for compactly supported wave packets),
ensures that \eqref{eq:poincare-kernel} defines smooth $\Gamma$--invariant kernels on $\mathbb H$ and hence kernels on $M$.

\subsubsection*{G10. Sobolev/elliptic constants: dependence statement}
Fix $\epsilon\in(0,\epsilon_*)$.
On the thick part $M_{\ge\epsilon}$, standard elliptic theory yields Sobolev embeddings with constants depending only on $(\epsilon,\vol(M))$ and finitely many atlas charts.
On $M$ as a whole, Sobolev constants may additionally depend on cusp widths through the cusp model charts; injectivity control is furnished by Lemma~\ref{lem:inj-cusp}.
We record the dependency rule:
\begin{equation}\label{eq:sobolev-dependence}
  \|f\|_{L^\infty(M(Y))}\ \ll_{\Gamma,Y}\ \|f\|_{H^2(M(Y))},
  \qquad
  \|f\|_{L^\infty(M)}\ \ll_{\mathcal G(\Gamma)}\ \|f\|_{H^2(M)},
\end{equation}
with the understanding that \eqref{eq:sobolev-dependence} is invoked later only after fixing a truncation height $Y\ge Y_0$ and specifying how $Y$ scales with spectral parameters.

\subsubsection*{G11. Gauss--Bonnet and torsion bookkeeping}
If $\Gamma$ is torsion-free, $M$ is a surface of genus $g$ with $n$ cusps and
\[
  \vol(M)=2\pi(2g-2+n).
\]
If $\Gamma$ has elliptic elements, $M$ is a hyperbolic orbifold with cone points of orders $e_j\ge 2$, and Gauss--Bonnet reads
\begin{equation}\label{eq:gauss-bonnet-orb}
  \vol(M)=2\pi\Big(2g-2+n+\sum_j (1-1/e_j)\Big).
\end{equation}
Elliptic conjugacy classes yield explicit additional terms in geometric trace identities; when torsion is present, these will be included systematically in Chapter~6.
