\subsection{Hyperbolic geometry: normalizations, models, and global constants}
\label{subsec:geometry-diamond}

\subsubsection*{G0. Object lock and invariant ledger}
Throughout, the geometric object is the finite-area quotient
\[
  M=\Gamma\backslash \mathbb H,\qquad \Gamma\subset \mathrm{PSL}_2(\mathbb R)\ \text{discrete, cofinite}.
\]
All geometric constants used later are allowed to depend \emph{only} on the ledger
\[
  \mathcal G(\Gamma):=
  \Big(
     \vol(M);\ \text{torsion orders } \{e_j\};\ \text{cusp ledger } \mathcal C(\Gamma);\ \text{a fixed core height } Y_0
  \Big),
\]
where the cusp ledger is fixed once and for all in Section~\ref{subsec:cusps-truncation-eisenstein},
\[
  \mathcal C(\Gamma)=\big(\#\{\text{cusps}\};\{w_{\mathfrak a}\};\{\sigma_{\mathfrak a}\}\big),
\]
and $Y_0=Y_0(\Gamma)$ is any height for which the cusp tails are pairwise disjoint
(cf.\ Lemma~\ref{lem:disjoint-tails} in the cusp block).
\emph{No constant in this block depends on later localization parameters $(\lambda,\eta)$}.

% ---------------------------------------------------------------------
\subsubsection*{G1. The hyperbolic plane: metric, measure, distance}
Let
\[
  \mathbb H=\{z=x+iy\in\mathbb C:\ y>0\},
\qquad
  ds^2=\frac{dx^2+dy^2}{y^2},
\qquad
  d\mu(z)=\frac{dx\,dy}{y^2}.
\]
The hyperbolic distance $d_{\mathbb H}(z,w)$ is characterized by
\begin{equation}\label{eq:cosh-distance}
  \cosh d_{\mathbb H}(z,w)=1+\frac{|z-w|^2}{2\,\Im z\,\Im w}.
\end{equation}
In particular, for a horizontal shift $w>0$,
\begin{equation}\label{eq:horizontal-distance}
  d_{\mathbb H}(x+iy,\ (x+w)+iy)
  =2\,\operatorname{arsinh}\!\Big(\frac{w}{2y}\Big)
  \asymp \min\Big\{1,\frac{w}{y}\Big\},
\end{equation}
with implied constants absolute.

\medskip
\noindent\textbf{Geodesics.}
Geodesics in $\mathbb H$ are vertical lines and semicircles orthogonal to $\mathbb R$.
Equivalently, for any two distinct boundary points $\alpha\neq\beta\in\mathbb R\cup\{\infty\}$,
there is a unique geodesic with endpoints $\{\alpha,\beta\}$.

% ---------------------------------------------------------------------
\subsubsection*{G2. Isometries and classification; translation length}
$\mathrm{PSL}_2(\mathbb R)$ acts on $\mathbb H$ by fractional linear maps
\[
  \gamma z=\frac{az+b}{cz+d},\qquad
  \gamma=\begin{pmatrix}a&b\\c&d\end{pmatrix}\in \mathrm{PSL}_2(\mathbb R),
\]
preserving $ds^2$ and $d\mu$.
Elements are classified by the absolute trace:
elliptic ($|\tr\gamma|<2$), parabolic ($|\tr\gamma|=2$), hyperbolic ($|\tr\gamma|>2$).

\medskip
\noindent\textbf{Hyperbolic translation length.}
If $\gamma$ is hyperbolic, it stabilizes a unique geodesic axis $\mathcal A_\gamma$ and acts on it as
a translation by $\ell(\gamma)>0$.
One has the exact identity
\begin{equation}\label{eq:length-trace}
  |\tr(\gamma)|=2\cosh\!\Big(\frac{\ell(\gamma)}{2}\Big).
\end{equation}
If $\gamma=\gamma_0^k$ with $\gamma_0$ primitive hyperbolic and $k\in\mathbb N$, then
\[
  \ell(\gamma)=k\,\ell(\gamma_0).
\]

% ---------------------------------------------------------------------
\subsubsection*{G3. Balls and exponential area growth}
Let $B_{\mathbb H}(z,R)$ be the hyperbolic ball of radius $R$.
Its area is independent of $z$ and equals
\begin{equation}\label{eq:ball-area}
  \vol\big(B_{\mathbb H}(R)\big)=2\pi(\cosh R-1)
  =\pi e^{R}\bigl(1+O(e^{-2R})\bigr)\qquad (R\to\infty).
\end{equation}
This exponential growth is the geometric source of the logarithmic propagation scale
$T\asymp \log\lambda$ in semiclassical/microlocal regimes.

% ---------------------------------------------------------------------
\subsubsection*{G4. Fundamental/Dirichlet domains; local finiteness; coarse orbit growth}
Let $\Gamma\subset \mathrm{PSL}_2(\mathbb R)$ be discrete and cofinite.
A measurable fundamental domain $\mathcal F$ satisfies $\mathbb H=\bigsqcup_{\gamma\in\Gamma}\gamma\mathcal F$
up to measure zero and
\[
  \vol(M)=\int_{\mathcal F} d\mu<\infty.
\]
Given $z_0\in\mathbb H$, the Dirichlet domain is
\[
  \mathcal D(z_0)=\{z\in\mathbb H:\ d_{\mathbb H}(z,z_0)\le d_{\mathbb H}(z,\gamma z_0)\ \forall \gamma\in\Gamma\}.
\]
It is convex and locally finite, and $\vol(M)=\int_{\mathcal D(z_0)}d\mu$.

\medskip
\begin{lemma}[Proper discontinuity / local finiteness]\label{lem:local-finiteness}
For any compact set $K\subset\mathbb H$ and any $R>0$, the set
\[
  \{\gamma\in\Gamma:\ d_{\mathbb H}(K,\gamma K)\le R\}
\]
is finite. In particular, for any fixed $z,w\in\mathbb H$ and any $R>0$,
\[
  \#\{\gamma\in\Gamma:\ d_{\mathbb H}(z,\gamma w)\le R\}<\infty.
\]
Consequently, if a kernel $K(z,w)=k(d_{\mathbb H}(z,w))$ has compact support in the distance variable,
then the Poincar\'e sum $\sum_{\gamma\in\Gamma}K(z,\gamma w)$ is \emph{termwise finite} for each $(z,w)$.
\end{lemma}

\begin{proof}
Proper discontinuity of the $\Gamma$-action implies $\{\gamma:\ \gamma K\cap K\neq\varnothing\}$ is finite for compact $K$.
Apply this to the compact thickening $K_R:=\{u\in\mathbb H:\ d_{\mathbb H}(u,K)\le R\}$:
if $d_{\mathbb H}(K,\gamma K)\le R$, then $\gamma K\cap K_R\neq\varnothing$, hence only finitely many $\gamma$ occur.
\end{proof}

\medskip
\noindent\textbf{Coarse orbit growth (robust, no remainder claimed).}
Define the orbit counting function
\[
  N_\Gamma(R;z_0):=\#\{\gamma\in\Gamma:\ d_{\mathbb H}(z_0,\gamma z_0)\le R\}.
\]
For cofinite $\Gamma$ one has the robust exponential growth law
\begin{equation}\label{eq:orbit-count-coarse}
  N_\Gamma(R;z_0)\asymp_\Gamma e^{R}\qquad (R\to\infty),
\end{equation}
with implied constants depending only on $\Gamma$ (and on $z_0$ through a bounded perturbation inside a fixed compact core).
Sharper asymptotics and explicit error terms require spectral input (e.g.\ gaps / mixing rates) and will be invoked
only at the point of use, with hypotheses and citations (see \cite{Hejhal1983,Iwaniec2002}).

% ---------------------------------------------------------------------
\subsubsection*{G5. Closed geodesics and hyperbolic conjugacy}
Primitive closed geodesics on $M$ correspond to primitive hyperbolic conjugacy classes $[\gamma_0]$ in $\Gamma$; their lengths satisfy
\[
  \ell(\mathcal C_{\gamma_0})=\ell(\gamma_0).
\]
The prime geodesic theorem asserts
\begin{equation}\label{eq:PGT}
  \pi_M(R):=\#\{\text{primitive closed geodesics on }M \text{ with }\ell\le R\}
  \sim \frac{e^{R}}{R}\qquad (R\to\infty),
\end{equation}
with effective remainders depending on spectral data (see \cite{Hejhal1983,Iwaniec2002}).
In this monograph \eqref{eq:PGT} is used only as a \emph{black-box} upper bound for hyperbolic orbital sums,
with any additional hypotheses stated explicitly where needed.

\subsubsection*{G6. Cusps, truncation, injectivity radius, thick--thin, collars}
Cusps, widths, scaling matrices, truncation $M(Y)$, and exact tail volume/length formulas are fixed in
Section~\ref{subsec:cusps-truncation-eisenstein}. We record only the structural consequences needed later.

\medskip
\noindent\textbf{Injectivity radius: definition.}
For $p\in M$ define the injectivity radius by
\begin{equation}\label{eq:inj-def}
  \inj(p):=\frac12\inf_{\gamma\in\Gamma\setminus\{1\}} d_{\mathbb H}(\tilde p,\gamma \tilde p),
\end{equation}
where $\tilde p\in\mathbb H$ is any lift of $p$ (the value is independent of the lift).

\medskip
\noindent\textbf{Cusp-scale in tails.}
In cusp coordinates $p=\sigma_{\mathfrak a}(x+iy)$, the shortest nontrivial displacement in $\Gamma_{\mathfrak a}$
is the horizontal shift by $w_{\mathfrak a}$, hence by \eqref{eq:horizontal-distance} and \eqref{eq:inj-def},
\begin{equation}\label{eq:inj-cusp-asymp}
  \inj(p)\asymp \min\Big\{1,\frac{w_{\mathfrak a}}{y}\Big\},
  \qquad (y\ \text{large}),
\end{equation}
with implied constants depending only on the cusp ledger $\{w_{\mathfrak a}\}$.
In particular, on the truncated tail $y\ge Y$ one has $\inj(p)\ll w_{\mathfrak a}/Y$.

\medskip
\noindent\textbf{Thick--thin decomposition (Margulis).}
There exists a universal $\epsilon_*>0$ such that for any $\epsilon\in(0,\epsilon_*)$ the thin part
\[
  M_{<\epsilon}:=\{p\in M:\ \inj(p)<\epsilon\}
\]
is a disjoint union of: (i) cusp neighborhoods, and (ii) embedded collars around short closed geodesics.
The thick part $M_{\ge\epsilon}:=M\setminus M_{<\epsilon}$ is compact.

\medskip
\noindent\textbf{Collar lemma (embedded tubular neighborhoods).}
Let $\gamma$ be a simple closed geodesic on $M$ of length $\ell=\ell(\gamma)$.
Then there is an embedded collar of half-width
\begin{equation}\label{eq:collar-width}
  w(\ell)=\operatorname{arsinh}\!\Big(\frac{1}{\sinh(\ell/2)}\Big),
\end{equation}
and collars of distinct simple closed geodesics are disjoint.
As $\ell\to 0$ one has $w(\ell)=\log(2/\ell)+O(\ell^2)$.

\medskip
\noindent\textbf{Compact core.}
Fix $Y\ge Y_0(\Gamma)$ so cusp tails are disjoint. Then
\[
  M_{\mathrm{core}}:=M(Y)
\]
is a compact surface with horocyclic boundary; it is the base region for elliptic regularity and Sobolev estimates,
while the cusp ends are handled by explicit model charts.

\subsubsection*{G7. Laplacian normalization and spectral parameter}
On smooth functions on $\mathbb H$ we take the nonnegative Laplace--Beltrami operator
\begin{equation}\label{eq:laplacian}
  \Delta:=-y^2\Big(\partial_x^2+\partial_y^2\Big).
\end{equation}
The spherical spectral parameterization is
\[
  \lambda=\frac14+t^2,\qquad t\in\mathbb R_{\ge 0},
\]
aligned with the Selberg/Harish--Chandra transform conventions in Section~\ref{sec:selberg-transform}.

\subsubsection*{G8. Spherical functions and Plancherel measure on $\mathbb H$}
Let $\varphi_t(r)$ be the $K$--biinvariant spherical function on $\mathbb H$ with $\varphi_t(0)=1$.
It may be written as
\[
  \varphi_t(r)=P_{-1/2+it}(\cosh r),
\]
and admits the standard large-$r$ expansion
\begin{equation}\label{eq:spherical-asympt}
  \varphi_t(r)=c(t)e^{(it-\frac12)r}+c(-t)e^{(-it-\frac12)r}+O(e^{-3r/2}),
  \qquad
  c(t)=\frac{\Gamma(it)}{\Gamma(\frac12+it)}.
\end{equation}
The Plancherel density for radial analysis on $\mathbb H$ is
\begin{equation}\label{eq:plancherel-density}
  d\mu_{\mathrm{Pl}}(t)=\frac{1}{2\pi}\,t\,\tanh(\pi t)\,dt,
\end{equation}
so that the usual even integral on $\mathbb R$ with factor $1/(4\pi)$ matches \eqref{eq:plancherel-density} after restriction to $t\ge 0$.

\subsubsection*{G9. Model kernels: heat, wave, resolvent; quotient kernels by Poincar\'e summation}
\noindent\textbf{Heat kernel on $\mathbb H$.}
The heat kernel $K_\tau(z,w)$ solves $\partial_\tau K_\tau=-\Delta_z K_\tau$ and has the spectral representation
\begin{equation}\label{eq:heat-kernel}
  K_\tau(z,w)=\int_0^\infty e^{-(\frac14+t^2)\tau}\,\varphi_t\!\big(d_{\mathbb H}(z,w)\big)\,d\mu_{\mathrm{Pl}}(t),
  \qquad \tau>0.
\end{equation}
As $\tau\to 0^+$ one has the standard Gaussian short-time behavior
\[
  K_\tau(z,w)\sim (4\pi\tau)^{-1}e^{-d_{\mathbb H}(z,w)^2/(4\tau)},
\]
with the usual curvature corrections in the full asymptotic expansion.

\medskip
\noindent\textbf{Wave kernel on $\mathbb H$.}
The even wave group $\cos\!\big(\tau\sqrt{\Delta-\frac14}\big)$ has kernel
\begin{equation}\label{eq:wave-kernel}
  W_\tau(z,w)=\int_0^\infty \cos(\tau t)\,\varphi_t\!\big(d_{\mathbb H}(z,w)\big)\,d\mu_{\mathrm{Pl}}(t).
\end{equation}
We will use only the \emph{stable} consequence relevant for localized analysis:
smooth compactly supported spectral cutoffs (in $t$) yield kernels which are smooth and rapidly decaying off the diagonal,
and whose effective propagation is controlled on times $\tau\ll \log\lambda$ when combined with semiclassical localization
(precise dispersive bounds are classical and may be cited when needed).

\medskip
\noindent\textbf{Resolvent on $\mathbb H$.}
For $s\in\mathbb C$ with $\Re(s)>\frac12$ let $R(s)=(\Delta-s(1-s))^{-1}$.
Its kernel admits the spectral integral representation
\begin{equation}\label{eq:resolvent-kernel}
  R(s;z,w)=\int_0^\infty \frac{\varphi_t\!\big(d_{\mathbb H}(z,w)\big)}{t^2+(s-\frac12)^2}\,d\mu_{\mathrm{Pl}}(t),
\end{equation}
and extends meromorphically in $s$.

\medskip
\noindent\textbf{Quotient kernels by Poincar\'e summation.}
Given a model kernel $K(z,w)$ on $\mathbb H$, define the $\Gamma$--invariant kernel
\begin{equation}\label{eq:poincare-kernel}
  K^\Gamma(z,w):=\sum_{\gamma\in\Gamma} K(z,\gamma w),
\end{equation}
whenever the sum converges absolutely, or else by analytic continuation/regularization in the standard cases
(e.g.\ resolvents and Eisenstein/Green kernels).
By Lemma~\ref{lem:local-finiteness}, if $K$ is compactly supported in the distance variable then the sum is locally finite.
If $K$ has sufficient decay in $d_{\mathbb H}(z,w)$ (e.g.\ heat kernels for $\tau>0$), then \eqref{eq:poincare-kernel}
converges absolutely and defines a smooth kernel on $M$.

\subsubsection*{G10. Sobolev/elliptic constants: dependency rule}
Fix $\epsilon\in(0,\epsilon_*)$.
On the thick part $M_{\ge\epsilon}$, standard elliptic theory yields Sobolev embeddings with constants depending only on
$(\epsilon,\vol(M))$ and finitely many atlas charts.
On $M$ as a whole, Sobolev constants may additionally depend on cusp widths through cusp model charts;
injectivity control is furnished by \eqref{eq:inj-cusp-asymp}.
We record the dependency rule in the form used later:
\begin{equation}\label{eq:sobolev-dependence}
  \|f\|_{L^\infty(M(Y))}\ \ll_{\Gamma,Y}\ \|f\|_{H^2(M(Y))},
  \qquad
  \|f\|_{L^\infty(M)}\ \ll_{\mathcal G(\Gamma)}\ \|f\|_{H^2(M)}.
\end{equation}
Whenever \eqref{eq:sobolev-dependence} is invoked, the truncation height $Y\ge Y_0(\Gamma)$ will be fixed
and its scaling (if any) with spectral parameters will be stated explicitly.

\subsubsection*{G11. Gauss--Bonnet and torsion bookkeeping}
If $\Gamma$ is torsion-free, $M$ is a surface of genus $g$ with $n$ cusps and
\[
  \vol(M)=2\pi(2g-2+n).
\]
If $\Gamma$ has elliptic elements, $M$ is a hyperbolic orbifold with cone points of orders $e_j\ge 2$, and Gauss--Bonnet reads
\begin{equation}\label{eq:gauss-bonnet-orb}
  \vol(M)=2\pi\Big(2g-2+n+\sum_j (1-1/e_j)\Big).
\end{equation}
Elliptic conjugacy classes yield explicit additional terms in geometric trace identities; when torsion is present,
they will be included systematically in Chapter~6 (geometric expansion).
