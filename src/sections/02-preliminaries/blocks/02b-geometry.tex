\subsection{Hyperbolic geometry and spherical analysis}\label{subsec:geometry}

Throughout this monograph the underlying geometric object is the finite--area quotient
\[
M=\Gamma\backslash\Hh,\qquad \Gamma\subset\PSL_2(\R)\ \text{discrete and cofinite}.
\]
All geometric constants are allowed to depend only on $\Gamma$ and the fixed cusp data
introduced in Section~\ref{subsec:cusps}. In particular, no dependence on later
localization parameters occurs in this block.

% ---------------------------------------------------------------------
\subsubsection{The hyperbolic plane: metric, measure, and distance}

We work on
\[
\H=\{z=x+iy\in\C:\ y>0\},
\qquad
ds^2=\frac{dx^2+dy^2}{y^2},
\qquad
d\mu(z)=\frac{dx\,dy}{y^2}.
\]
The hyperbolic distance $d_{\H}(z,w)$ is characterized by the identity
\begin{equation}\label{eq:cosh-distance}
\cosh d_{\H}(z,w)=1+\frac{|z-w|^2}{2\,\Im z\,\Im w}.
\end{equation}
In particular, for a horizontal shift $u>0$ one has
\begin{equation}\label{eq:horizontal-distance}
d_{\H}(x+iy,(x+u)+iy)
=
2\,\arsinh\!\Big(\frac{u}{2y}\Big)
\asymp
\min\Big\{1,\frac{u}{y}\Big\},
\end{equation}
with absolute implied constants.

Geodesics in $\H$ are vertical lines and semicircles orthogonal to $\R$.

% ---------------------------------------------------------------------
\subsubsection{Isometries and translation length}

The group $\PSL_2(\R)$ acts by fractional linear transformations
\[
\gamma z=\frac{az+b}{cz+d},\qquad
\gamma=\begin{pmatrix}a&b\\c&d\end{pmatrix}\in\PSL_2(\R),
\]
preserving $ds^2$ and $d\mu$. Elements are classified by their trace:
elliptic ($|\tr\gamma|<2$), parabolic ($|\tr\gamma|=2$), hyperbolic ($|\tr\gamma|>2$).

If $\gamma$ is hyperbolic, it stabilizes a unique geodesic axis and acts as a translation by
a length $\ell(\gamma)>0$. One has the exact identity
\begin{equation}\label{eq:length-trace}
|\tr(\gamma)|=2\cosh\!\Big(\frac{\ell(\gamma)}{2}\Big).
\end{equation}
If $\gamma=\gamma_0^k$ with $\gamma_0$ primitive hyperbolic and $k\in\N$, then
$\ell(\gamma)=k\,\ell(\gamma_0)$.

Primitive closed geodesics on $M$ are in bijection with primitive hyperbolic conjugacy
classes in $\Gamma$, with matching length $\ell(\mathcal C_{\gamma_0})=\ell(\gamma_0)$.

\subsubsection{Ball volume growth}

Let $B_{\H}(z,R)$ denote the hyperbolic ball of radius $R$. Its area is independent of $z$ and equals
\begin{equation}\label{eq:ball-area}
\vol\big(B_{\H}(R)\big)=2\pi(\cosh R-1)
=\pi e^{R}\bigl(1+O(e^{-2R})\bigr),
\qquad (R\to\infty).
\end{equation}

% ---------------------------------------------------------------------
\subsubsection{Fundamental domains and local finiteness}

A measurable fundamental domain $\mathcal F$ for $\Gamma$ satisfies
\[
\H=\bigsqcup_{\gamma\in\Gamma}\gamma\mathcal F
\quad\text{(disjoint up to measure zero)},
\qquad
\vol(M)=\int_{\mathcal F}d\mu<\infty.
\]
For $z_0\in\H$, the Dirichlet domain
\[
\mathcal D(z_0):=\{z\in\H:\ d_{\H}(z,z_0)\le d_{\H}(z,\gamma z_0)\ \forall \gamma\in\Gamma\}
\]
is convex and locally finite.

\begin{lemma}[Proper discontinuity]\label{lem:local-finiteness}
For any compact set $K\subset\H$ and any $R>0$, the set
\[
\{\gamma\in\Gamma:\ d_{\H}(K,\gamma K)\le R\}
\]
is finite. In particular, for fixed $z,w\in\H$ and $R>0$,
\[
\#\{\gamma\in\Gamma:\ d_{\H}(z,\gamma w)\le R\}<\infty.
\]
Consequently, if a kernel $k(d_{\H}(z,w))$ is compactly supported in the distance variable,
then the Poincar\'e sum $\sum_{\gamma\in\Gamma}k(d_{\H}(z,\gamma w))$ is termwise finite for each $(z,w)$.
\end{lemma}

\subsubsection{Injectivity radius and thick--thin decomposition}

For $p\in M$ define the injectivity radius by
\begin{equation}\label{eq:inj-def}
\inj(p):=\frac12\inf_{\gamma\in\Gamma\setminus\{1\}} d_{\H}(\tilde p,\gamma\tilde p),
\end{equation}
where $\tilde p\in\H$ is any lift.

There exists a universal $\epsilon_*>0$ (Margulis constant) such that for each $\epsilon\in(0,\epsilon_*)$
the thin part
\[
M_{<\epsilon}:=\{p\in M:\ \inj(p)<\epsilon\}
\]
is a disjoint union of cusp neighborhoods and collars about short closed geodesics.
The thick part $M_{\ge\epsilon}:=M\setminus M_{<\epsilon}$ is compact.

If $\gamma$ is a simple closed geodesic of length $\ell(\gamma)$, then it admits an embedded collar
of half-width
\begin{equation}\label{eq:collar-width}
w(\ell)=\arsinh\!\Big(\frac{1}{\sinh(\ell/2)}\Big),
\end{equation}
and collars of distinct simple closed geodesics are disjoint.

% ---------------------------------------------------------------------
\subsubsection{Laplacian convention and spectral parameter}

On $\H$ we take the nonnegative Laplace--Beltrami operator
\begin{equation}\label{eq:laplacian}
\Delta=-y^{2}\bigl(\partial_x^{2}+\partial_y^{2}\bigr).
\end{equation}
The standard spectral parameterization is
\[
\lambda=\frac14+t^2,\qquad t\in\R_{\ge 0}.
\]

% ---------------------------------------------------------------------
\subsubsection{Spherical functions and Plancherel measure on $\H$}

Let $\varphi_t(r)$ be the $K$--biinvariant spherical function on $\H$ normalized by $\varphi_t(0)=1$.
It may be expressed as
\[
\varphi_t(r)=P_{-1/2+it}(\cosh r).
\]
The Plancherel density for radial analysis on $\H$ is
\begin{equation}\label{eq:plancherel-density}
d\mu_{\mathrm{Pl}}(t)=\frac{1}{2\pi}\,t\,\tanh(\pi t)\,dt.
\end{equation}

% ---------------------------------------------------------------------
\subsubsection{Model kernels on $\H$}

\paragraph{Heat kernel.}
The heat kernel $K_\tau(z,w)$ on $\H$ admits the spectral representation
\begin{equation}\label{eq:heat-kernel}
K_\tau(z,w)=\int_0^\infty e^{-(\frac14+t^2)\tau}\,
\varphi_t\!\bigl(d_{\H}(z,w)\bigr)\,d\mu_{\mathrm{Pl}}(t),
\qquad \tau>0.
\end{equation}

\paragraph{Wave kernel.}
The even wave kernel is given by
\begin{equation}\label{eq:wave-kernel}
W_\tau(z,w)=\int_0^\infty \cos(\tau t)\,
\varphi_t\!\bigl(d_{\H}(z,w)\bigr)\,d\mu_{\mathrm{Pl}}(t).
\end{equation}

\paragraph{Resolvent kernel.}
For $\Re(s)>\frac12$ the resolvent $R(s)=(\Delta-s(1-s))^{-1}$ has kernel
\begin{equation}\label{eq:resolvent-kernel}
R(s;z,w)=\int_0^\infty
\frac{\varphi_t\!\bigl(d_{\H}(z,w)\bigr)}{t^2+(s-\frac12)^2}\,
d\mu_{\mathrm{Pl}}(t),
\end{equation}
which extends meromorphically in $s$.

% ---------------------------------------------------------------------
\subsubsection{Poincar\'e summation and quotient kernels}

Given a kernel $K(z,w)$ on $\H$, define its $\Gamma$--periodization
\begin{equation}\label{eq:poincare-kernel}
K^\Gamma(z,w):=\sum_{\gamma\in\Gamma} K(z,\gamma w),
\end{equation}
whenever the series converges.
By Lemma~\ref{lem:local-finiteness}, if $K(z,w)$ is compactly supported in the distance variable,
then \eqref{eq:poincare-kernel} is locally finite and defines a smooth kernel on $M$.
For kernels with sufficient decay in $d_{\H}(z,w)$ (e.g.\ heat kernels for $\tau>0$),
the sum converges absolutely.

% ---------------------------------------------------------------------
\subsubsection{Gauss--Bonnet and torsion bookkeeping}

If $\Gamma$ is torsion-free, $M$ has genus $g$ and $n$ cusps, and
\[
\vol(M)=2\pi(2g-2+n).
\]
If $\Gamma$ has elliptic elements with cone point orders $e_j\ge 2$, then
\begin{equation}\label{eq:gauss-bonnet-orb}
\vol(M)=2\pi\Big(2g-2+n+\sum_j (1-1/e_j)\Big).
\end{equation}

% ---------------------------------------------------------------------
\subsubsection*{Audit}\label{rem:geometry-audit}
Equations \eqref{eq:cosh-distance}--\eqref{eq:gauss-bonnet-orb} fix the hyperbolic
normalizations, Laplacian sign convention, spherical Plancherel measure, and the
canonical model kernels used later. The geometric inputs are restricted to robust
structural statements; any refined counting asymptotics requiring spectral
information will be invoked only with explicit hypotheses at the point of use.
