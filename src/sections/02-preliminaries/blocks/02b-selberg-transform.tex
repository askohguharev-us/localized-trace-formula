\subsection{The Selberg/Harish--Chandra transform}\label{subsec:selberg-transform}

This block fixes the transform conventions relating radial kernels on $\Hh$
to spectral multipliers of $\Delta$. All constants depend only on the
ambient hyperbolic normalization; no statement below involves later
localization parameters.

% ---------------------------------------------------------------------
\subsubsection{Spherical functions and Plancherel density}\label{subsec:spherical-functions}

We realize the hyperbolic plane as
\[
\Hh=\{z=x+iy\in\C:\ y>0\},
\qquad
ds^{2}=y^{-2}(dx^{2}+dy^{2}),
\qquad
d\mu(z)=\frac{dx\,dy}{y^{2}}.
\]
We use the nonnegative Laplace--Beltrami operator
\begin{equation}\label{eq:laplace-Hh}
\Delta=-y^{2}\bigl(\partial_{x}^{2}+\partial_{y}^{2}\bigr).
\end{equation}
The spectral parameter is written as
\[
\lambda=\frac14+r^{2},\qquad r\in\C,
\]
so that the continuous spectrum on $\Hh$ corresponds to $r\in\R$.

For $r\in\C$ let $\varphi_{r}$ be the $K$--biinvariant spherical function normalized by
\begin{equation}\label{eq:spherical-eig}
\Delta\,\varphi_{r}=\Bigl(\frac14+r^{2}\Bigr)\varphi_{r},
\qquad
\varphi_{r}(i)=1.
\end{equation}
Writing $u=d_{\H}(z,i)$, one has $\varphi_{r}(z)=\varphi_{r}(u)$ and
\begin{equation}\label{eq:spherical-legendre}
\varphi_{r}(u)=P_{-\frac12+ir}(\cosh u).
\end{equation}
The Harish--Chandra $c$--function is
\begin{equation}\label{eq:c-function}
c(r)=\frac{\Gamma(ir)}{\Gamma(\frac12+ir)}.
\end{equation}
For $r$ in compact subsets of $\R$, the standard large--$u$ expansion reads
\begin{equation}\label{eq:spherical-asymp}
\varphi_{r}(u)
=
c(r)e^{(ir-\frac12)u}
+
c(-r)e^{(-ir-\frac12)u}
+
O\!\bigl(e^{-3u/2}\bigr),
\qquad u\to\infty.
\end{equation}

The Plancherel measure for radial analysis on $\Hh$ is
\begin{equation}\label{eq:plancherel-density}
d\mu_{\mathrm{Pl}}(r)=\frac{1}{4\pi}\,r\,\tanh(\pi r)\,dr,
\qquad r\in\R .
\end{equation}

% ---------------------------------------------------------------------
\subsubsection{Radial kernels and the spherical transform}\label{subsec:radial-kernels}

A \emph{radial kernel} on $\Hh$ is a function of the distance:
\[
K(z,w)=k\!\bigl(d_{\H}(z,w)\bigr),
\qquad
k:[0,\infty)\to\C.
\]
It defines the convolution operator
\begin{equation}\label{eq:radial-operator}
(T_{k}f)(z)=\int_{\Hh}k\!\bigl(d_{\H}(z,w)\bigr)\,f(w)\,d\mu(w),
\end{equation}
whenever the integral converges.

\begin{definition}[Spherical transform]\label{def:spherical-transform}
Assume that $k$ is measurable and that $u\mapsto k(u)\sinh u$ is integrable on $(0,\infty)$.
The spherical (Selberg/Harish--Chandra) transform of $k$ is the even function $h$ on $\R$ defined by
\begin{equation}\label{eq:selberg-transform}
h(r)
=
\int_{\Hh}k\!\bigl(d_{\H}(z,i)\bigr)\,\varphi_{r}(z)\,d\mu(z)
=
2\pi\int_{0}^{\infty}k(u)\,\varphi_{r}(u)\,\sinh u\,du,
\qquad r\in\R.
\end{equation}
If $k\in C_c^\infty([0,\infty))$, then $h$ extends to an even entire function.
\end{definition}

\begin{proposition}[Multiplier property]\label{prop:multiplier-property}
Let $\psi$ be a smooth eigenfunction of $\Delta$ on $\Hh$ with
$\Delta\psi=(\frac14+r^{2})\psi$ for some $r\in\R$.
If $k\in C_c^\infty([0,\infty))$ and $h$ is given by \eqref{eq:selberg-transform}, then
\begin{equation}\label{eq:multiplier}
T_k\psi=h(r)\,\psi.
\end{equation}
\end{proposition}

\subsubsection{Inversion, Plancherel, and convolution}\label{subsec:inversion-plancherel}

For admissible even $h$, the inverse transform is
\begin{equation}\label{eq:spherical-inversion}
k(u)
=
\int_{-\infty}^{\infty} h(r)\,\varphi_{r}(u)\,d\mu_{\mathrm{Pl}}(r)
=
\frac{1}{4\pi}\int_{-\infty}^{\infty} h(r)\,\varphi_{r}(u)\,r\tanh(\pi r)\,dr,
\qquad u\ge 0.
\end{equation}

\begin{lemma}[Plancherel identity]\label{lem:selberg-plancherel}
If $k_1,k_2\in C_c^\infty([0,\infty))$ have transforms $h_1,h_2$, then
\begin{equation}\label{eq:plancherel}
2\pi\int_{0}^{\infty} k_1(u)\,\overline{k_2(u)}\,\sinh u\,du
=
\int_{-\infty}^{\infty} h_1(r)\,\overline{h_2(r)}\,d\mu_{\mathrm{Pl}}(r).
\end{equation}
\end{lemma}

\begin{lemma}[Convolution]\label{lem:selberg-convolution}
Let $k_1,k_2\in C_c^\infty([0,\infty))$ have transforms $h_1,h_2$.
Then the hyperbolic radial convolution satisfies
\begin{equation}\label{eq:conv-mult}
h_{1*2}(r)=h_1(r)\,h_2(r),
\qquad r\in\R,
\end{equation}
equivalently $T_{k_1}\circ T_{k_2}=T_{k_1*k_2}$.
\end{lemma}

\subsubsection{Admissible spectral class and Paley--Wiener}\label{subsec:admissible-paley-wiener}

\begin{definition}[Admissible spectral class]\label{def:admissible-Hh}
An even function $h:\C\to\C$ is \emph{admissible} if there exists $\varepsilon>0$ such that:
\begin{itemize}
\item[(H1)] $h$ is holomorphic in the strip $|\Im r|<\frac12+\varepsilon$;
\item[(H2)] there exists $\delta>0$ with
\[
h(r)\ll (1+|r|)^{-2-\delta}
\quad\text{uniformly for }|\Im r|<\frac12+\varepsilon;
\]
\item[(H3)] the same bound holds uniformly on the lines $\Im r=\pm(\frac12+\varepsilon')$
for each $\varepsilon'<\varepsilon$.
\end{itemize}
\end{definition}

If $h$ is admissible, then \eqref{eq:spherical-inversion} defines a continuous radial profile $k$
with exponential decay
\[
k(u)\ll_{\varepsilon'} e^{-(\frac12+\varepsilon')u}
\qquad\text{for each }\varepsilon'<\varepsilon.
\]

\begin{theorem}[Paley--Wiener]\label{thm:paley-wiener}
Let $R>0$.
\begin{itemize}
\item[(PW$\Rightarrow$)] If $k\in C_c^\infty([0,\infty))$ is supported in $[0,R]$, then $h$ extends to an even entire function and,
for each $N>0$,
\[
|h(r)|\ll_{N}(1+|r|)^{-N}e^{R|\Im r|}
\qquad (r\in\C).
\]
\item[(PW$\Leftarrow$)] Conversely, if $h$ is even entire of exponential type $R$ and rapidly decaying on $\R$, then the inverse transform $k$
is supported in $[0,R]$.
\end{itemize}
\end{theorem}

\subsubsection{Examples}\label{subsec:selberg-examples}

\paragraph{Heat kernel multiplier.}
For $t>0$, the heat semigroup $e^{-t\Delta}$ corresponds to
\[
h_t(r)=e^{-(\frac14+r^2)t}.
\]

\paragraph{Even wave multiplier.}
For $\tau\in\R$, the even wave group $\cos(\tau\sqrt{\Delta-\frac14})$ corresponds to
\[
h_\tau(r)=\cos(\tau r).
\]

\paragraph{Windowed multipliers.}
Let $\chi\in C_c^\infty(\R)$ be even. For $R>0$ and $\eta>0$ define
\begin{equation}\label{eq:windowed-h}
h_{R,\eta}(r)
=
\chi\Bigl(\frac{r-R}{\eta}\Bigr)
+
\chi\Bigl(\frac{r+R}{\eta}\Bigr),
\end{equation}
an even entire function; the associated profile $k_{R,\eta}$ is defined by \eqref{eq:spherical-inversion}.

\subsubsection{Interface with $\Gamma\backslash\Hh$}\label{subsec:pretrace-interface}

Let $M=\Gamma\backslash\Hh$ with $\Gamma$ cofinite.
Given a radial profile $k$ on $\Hh$, define the automorphic kernel
\begin{equation}\label{eq:automorphic-kernel}
K^\Gamma(z,w):=\sum_{\gamma\in\Gamma} k\!\bigl(d_{\H}(z,\gamma w)\bigr),
\end{equation}
whenever the sum converges absolutely. (In the presence of cusps, a renormalized
interpretation is used together with truncation; this is fixed in the cusp block.)

If $\phi$ is an $L^2$--eigenfunction on $M$ with $\Delta\phi=(\frac14+r^2)\phi$, then under admissibility and truncation
\begin{equation}\label{eq:multiplier-quotient}
T_k^\Gamma\phi=h(r)\,\phi,
\end{equation}
where $T_k^\Gamma$ is the integral operator with kernel $K^\Gamma$ and $h$ is the transform of $k$.

% ---------------------------------------------------------------------
\subsubsection*{Audit}\label{rem:selberg-audit}
Equations \eqref{eq:selberg-transform}--\eqref{eq:windowed-h} fix the Selberg/Harish--Chandra normalization,
the Plancherel density \eqref{eq:plancherel-density}, and inversion \eqref{eq:spherical-inversion}.
Definition~\ref{def:admissible-Hh} records the admissible spectral strip used throughout.
The quotient interface \eqref{eq:automorphic-kernel}--\eqref{eq:multiplier-quotient} fixes the operator normalization.
