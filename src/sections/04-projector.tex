\chapter{The Spectral Projector}
\label{ch:spectral-projector}

\section{Setting and spectral resolution}
\label{sec:sp-setting}

Let $M=\Gamma\backslash\mathbb{H}$ be a finite-area hyperbolic surface, where
$\Gamma\subset \mathrm{PSL}_2(\mathbb{R})$ is discrete and cofinite.
Let $\Delta$ be the nonnegative Laplace--Beltrami operator on $M$.
It is essentially self-adjoint on $C_c^\infty(M)$ and admits a unique self-adjoint extension
(still denoted $\Delta$) on $L^2(M)$; see, e.g., \cite[Ch.~3]{Iwaniec2002} and \cite[Ch.~5]{Borthwick2016}.

\subsection{Spectral parameterization and Plancherel measure}

We use the standard parameterization
\[
  \lambda_j=\tfrac14+r_j^2,\qquad r_j\in\mathbb{R}_{\ge 0},
\]
for the discrete spectrum (counted with multiplicity), and write the continuous spectrum
in terms of Eisenstein series $E_{\mathfrak a}(z,\tfrac12+ir)$ attached to cusps $\mathfrak a$.
Throughout, Eisenstein series are normalized so that the continuous part of the spectral decomposition
uses the Plancherel measure $dr/(4\pi)$; see \cite[Vol.~I, Ch.~6]{Hejhal1983} and \cite[Ch.~7]{Iwaniec2002}.

For $f\in L^2(M)$ one has the decomposition
\begin{equation}\label{eq:sp-plancherel}
  f
  = \sum_{j} \langle f,\phi_j\rangle \phi_j
  + \frac{1}{4\pi}\sum_{\mathfrak a}\int_{\mathbb R}
    \langle f,E_{\mathfrak a}(\cdot,\tfrac12+ir)\rangle\,E_{\mathfrak a}(\cdot,\tfrac12+ir)\,dr,
\end{equation}
with convergence in $L^2(M)$, where $\{\phi_j\}$ is an orthonormal basis of Maass cusp forms.

\subsection{Functional calculus for $\sqrt{\Delta-\tfrac14}$}

Let
\[
  A:=\sqrt{\Delta-\tfrac14}
\]
be defined by the spectral theorem (on the orthogonal complement of the residual spectrum if present;
for simplicity we keep the notation $A$ and refer to standard treatments).
For any bounded Borel function $h:\mathbb{R}\to\mathbb{C}$ that is even, the operator $h(A)$
is well-defined and bounded on $L^2(M)$, self-adjoint whenever $h$ is real-valued.
Moreover
\begin{equation}\label{eq:sp-multiplier}
  h(A)f
  = \sum_j h(r_j)\langle f,\phi_j\rangle\phi_j
  + \frac{1}{4\pi}\sum_{\mathfrak a}\int_{\mathbb R} h(r)
    \langle f,E_{\mathfrak a}(\cdot,\tfrac12+ir)\rangle\,E_{\mathfrak a}(\cdot,\tfrac12+ir)\,dr,
\end{equation}
for all $f\in L^2(M)$.

\section{The localized spectral window and the projector}
\label{sec:sp-window}

\subsection{A compactly supported time cutoff}

Fix an even cutoff $\widehat\chi\in C_c^\infty(\mathbb{R})$, $\widehat\chi(-t)=\widehat\chi(t)$, such that
\begin{equation}\label{eq:sp-chi-hat}
  \mathrm{supp}\,\widehat\chi\subset (-1,1),\qquad \widehat\chi(0)>0.
\end{equation}
Let $\chi$ be its inverse Fourier transform
\begin{equation}\label{eq:sp-chi}
  \chi(u):=\frac{1}{2\pi}\int_{\mathbb{R}}\widehat\chi(t)e^{itu}\,dt .
\end{equation}
Then $\chi$ extends to an entire function and is Schwartz on $\mathbb{R}$
(Paley--Wiener).

\subsection{Spectral localization around frequency $\lambda$}

Let $\lambda\ge 1$ and $\eta\in(0,1]$.
Define the even spectral multiplier
\begin{equation}\label{eq:sp-h}
  h_{\lambda,\eta}(r)
  := \chi\!\left(\frac{r-\lambda}{\eta}\right)+\chi\!\left(\frac{r+\lambda}{\eta}\right),
  \qquad r\in\mathbb{R}.
\end{equation}
The parameter $\lambda$ is the central frequency and $\eta$ is the window width.
Note that $h_{\lambda,\eta}$ is bounded and rapidly decaying away from $r=\pm\lambda$.

\begin{definition}[Spectral projector]\label{def:spectral-projector}
The (smooth) spectral projector at $(\lambda,\eta)$ is the bounded operator
\begin{equation}\label{eq:sp-projector}
  P_{\lambda,\eta}:=h_{\lambda,\eta}(A)=h_{\lambda,\eta}\!\left(\sqrt{\Delta-\tfrac14}\right)
  \qquad\text{on }L^2(M).
\end{equation}
\end{definition}

By \eqref{eq:sp-multiplier}, $P_{\lambda,\eta}$ acts as the multiplier $h_{\lambda,\eta}$ on both
the discrete and continuous spectral parameters.

\subsection{Basic operator properties}

\begin{lemma}[Self-adjointness and positivity]\label{lem:sp-sa-pos}
If $h_{\lambda,\eta}$ is real-valued and even, then $P_{\lambda,\eta}$ is self-adjoint on $L^2(M)$.
If, moreover, $h_{\lambda,\eta}(r)\ge 0$ for all $r$, then $P_{\lambda,\eta}$ is positive semidefinite.
\end{lemma}

\begin{proof}
This follows immediately from the spectral theorem and \eqref{eq:sp-multiplier}.
\end{proof}

\begin{lemma}[$L^2$-operator norm]\label{lem:sp-L2norm}
One has
\[
  \|P_{\lambda,\eta}\|_{L^2\to L^2}\le \|h_{\lambda,\eta}\|_{L^\infty(\mathbb R)}.
\]
In particular, if $\|h_{\lambda,\eta}\|_\infty\le 1$ then $\|P_{\lambda,\eta}\|_{2\to2}\le 1$.
\end{lemma}

\begin{proof}
By \eqref{eq:sp-multiplier}, $P_{\lambda,\eta}$ is a spectral multiplier with multiplier $h_{\lambda,\eta}$.
The operator norm equals the essential supremum of the multiplier.
\end{proof}

\section{Wave representation and finite propagation}
\label{sec:sp-wave}

\subsection{A compactly supported wave packet}

Define the length-side kernel
\begin{equation}\label{eq:sp-g-def}
  g_{\lambda,\eta}(t):=\frac{1}{2\pi}\int_{\mathbb{R}} h_{\lambda,\eta}(r)\,e^{irt}\,dr .
\end{equation}
A direct change of variables using \eqref{eq:sp-chi} gives the explicit formula
\begin{equation}\label{eq:sp-g-explicit}
  g_{\lambda,\eta}(t)=2\eta\,\widehat\chi(\eta t)\cos(\lambda t).
\end{equation}
In particular,
\begin{equation}\label{eq:sp-g-support}
  \mathrm{supp}\,g_{\lambda,\eta}\subset [-\eta^{-1},\eta^{-1}].
\end{equation}

Let $U(t):=\cos(tA)$ denote the even wave group associated with $A=\sqrt{\Delta-\tfrac14}$.
By Fourier inversion and the evenness of $h_{\lambda,\eta}$ one obtains the representation
\begin{equation}\label{eq:sp-wave-rep}
  P_{\lambda,\eta}
  = \int_{\mathbb R} g_{\lambda,\eta}(t)\,U(t)\,dt ,
\end{equation}
where the integral converges in the strong operator topology on $L^2(M)$.

\subsection{Kernel on the universal cover and propagation speed}

Let $K_{\lambda,\eta}(z,w)$ be the Schwartz kernel of $P_{\lambda,\eta}$ on $M$.
Let $\widetilde K_{\lambda,\eta}(\tilde z,\tilde w)$ denote the corresponding $\Gamma$-invariant kernel on $\mathbb H$,
so that
\begin{equation}\label{eq:sp-periodization}
  K_{\lambda,\eta}(z,w)=\sum_{\gamma\in\Gamma}\widetilde K_{\lambda,\eta}(\tilde z,\gamma\tilde w),
\end{equation}
for any lifts $\tilde z,\tilde w\in\mathbb H$ of $z,w\in M$ (absolute convergence holds away from the diagonal; see \cite{Hejhal1983,Borthwick2016}).

Finite propagation for the wave equation implies that the kernel of $U(t)$ on $\mathbb H$ vanishes when
$d(\tilde z,\tilde w)>|t|$. Combining this with \eqref{eq:sp-wave-rep} and \eqref{eq:sp-g-support} yields:

\begin{lemma}[Geometric support]\label{lem:sp-support}
On the universal cover $\mathbb H$ one has
\[
  \widetilde K_{\lambda,\eta}(\tilde z,\tilde w)=0
  \qquad\text{whenever}\qquad d(\tilde z,\tilde w)>\eta^{-1}.
\]
Consequently, for fixed $z,w\in M$, the sum in \eqref{eq:sp-periodization} is finite:
only those $\gamma\in\Gamma$ with $d(\tilde z,\gamma\tilde w)\le \eta^{-1}$ contribute.
\end{lemma}

\begin{proof}
The support property of the wave kernel is standard (finite propagation speed).
Since $g_{\lambda,\eta}$ is supported in $|t|\le \eta^{-1}$, the integral \eqref{eq:sp-wave-rep}
only averages wave propagators $U(t)$ with $|t|\le \eta^{-1}$, hence the kernel vanishes when
$d(\tilde z,\tilde w)>\eta^{-1}$.
\end{proof}

\section{Sobolev mapping and parameter regime}
\label{sec:sp-sobolev}

\subsection{Sobolev mapping bounds from spectral calculus}

Let $H^s(M)$ be the standard Sobolev spaces defined by $(I+\Delta)^{s/2}$.
Since $P_{\lambda,\eta}$ is a spectral multiplier, mapping properties follow from elementary spectral estimates.

\begin{lemma}[Sobolev mapping]\label{lem:sp-sobolev}
For each $s\ge 0$,
\[
  \|P_{\lambda,\eta}\|_{H^s\to H^s}\ \le\
  \sup_{r\in\mathbb R} |h_{\lambda,\eta}(r)|,
\]
and, more generally,
\[
  \|P_{\lambda,\eta}\|_{H^s\to H^{s+m}}
  \ \le\ \sup_{r\in\mathbb R}(1+r^2)^{m/2}|h_{\lambda,\eta}(r)|,
  \qquad m\ge 0.
\]
\end{lemma}

\begin{proof}
Insert \eqref{eq:sp-multiplier} into the definition of $\|\,\cdot\,\|_{H^s}$ and use that
$(I+\Delta)^{m/2}$ acts as multiplication by $(1+\tfrac14+r^2)^{m/2}$ in the spectral parameter.
\end{proof}

\subsection{Admissible regimes of $(\lambda,\eta)$}

In the sequel we work with asymptotic frequency $\lambda\to\infty$ and window width $\eta=\eta(\lambda)\in(0,1]$.
Two regimes are relevant:

\begin{itemize}
  \item \emph{Polynomial windows:} $\eta\ge \lambda^{-\theta}$ for some $0<\theta<\theta_0$.
  This ensures that derivatives of $h_{\lambda,\eta}$ scale at most polynomially in $\lambda$.
  \item \emph{Ehrenfest windows:} $\eta\ge (c_E\log\lambda)^{-1}$ for some constant $c_E>0$.
  This is the regime naturally compatible with semiclassical propagation up to Ehrenfest time
  for hyperbolic dynamics; see \cite{Anantharaman2008,NonnenmacherZworski2009,Zworski2012}.
\end{itemize}

The constant $c_E$ and the threshold $\theta_0$ will be fixed once and for all (depending only on $M$ and on the chosen cutoff $\widehat\chi$)
when we enter the semiclassical analysis in Chapter~5.

\section{Microlocal formulation (Ehrenfest scale)}
\label{sec:sp-microlocal}

This section records the microlocal framework in which $P_{\lambda,\eta}$ will be analyzed.
The detailed parametrix and error bounds are proved in Chapter~5.

\subsection{Semiclassical parameterization}

We set the semiclassical parameter
\[
  h:=\lambda^{-1}.
\]
Let $\Op_h(a)$ denote a standard semiclassical quantization of a symbol $a$ supported in a fixed compact set of $T^*M$
in local coordinates; see \cite{Zworski2012} for conventions.
The geodesic flow on $T^*M$ is denoted $g^t$.

\subsection{Egorov theorem up to Ehrenfest time}

Let $U(t)=e^{itA}$ denote the unitary wave group.
A semiclassical Egorov theorem on manifolds with Anosov geodesic flow gives, for times $|t|\le c\log(1/h)$,
\begin{equation}\label{eq:sp-egorov}
  U(-t)\,\Op_h(a)\,U(t)=\Op_h(a\circ g^t)+\mathcal{R}_h(t),
\end{equation}
where $\|\mathcal{R}_h(t)\|_{L^2\to L^2}\le C\,h^{1-\varepsilon}$ for any fixed $\varepsilon>0$,
with $C$ depending on finitely many seminorms of $a$ and on the geometry of $M$;
see \cite{Anantharaman2008,NonnenmacherZworski2009,Zworski2012}.

We will combine \eqref{eq:sp-egorov} with the compact time support \eqref{eq:sp-g-support} of $g_{\lambda,\eta}$.
In particular, when $\eta\ge (c_E\log\lambda)^{-1}$ one has $|t|\le \eta^{-1}\ll \log(1/h)$,
and therefore Egorov-type propagation applies throughout the full time range present in the wave packet
representation \eqref{eq:sp-wave-rep}. This is the mechanism behind the ``Ehrenfest barrier''
appearing later in the remainder analysis.

\section{Chapter summary and forward links}
\label{sec:sp-summary}

We have fixed a canonical family of smooth spectral projectors $P_{\lambda,\eta}=h_{\lambda,\eta}(\sqrt{\Delta-\tfrac14})$,
built from a compactly supported time cutoff $\widehat\chi$.
The key structural features are:

\begin{itemize}
  \item \emph{Spectral localization:} $P_{\lambda,\eta}$ acts as the multiplier $h_{\lambda,\eta}(r)$ on both discrete and continuous spectrum.
  \item \emph{Wave packet representation:} $P_{\lambda,\eta}$ is an average of $\cos(t\sqrt{\Delta-\tfrac14})$ with weight $g_{\lambda,\eta}(t)$.
  \item \emph{Finite propagation:} the kernel on $\mathbb H$ is supported in $d(\tilde z,\tilde w)\le \eta^{-1}$.
  \item \emph{Regimes:} polynomial windows $\eta\ge \lambda^{-\theta}$ and Ehrenfest windows $\eta\ge (c_E\log\lambda)^{-1}$.
\end{itemize}

\noindent\textbf{Forward links.}
Chapter~5 constructs the microlocal parametrix for the wave kernel and quantifies the remainder terms at Ehrenfest scale.
Chapter~6 applies the same cutoff to the Selberg trace formula and expands the geometric side.
Chapter~7 combines the spectral and geometric expansions to obtain the localized trace formula with a final remainder bound.
