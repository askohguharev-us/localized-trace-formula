\chapter{The spectral projector}\label{ch:spectral-projector}

\section{Object lock, spectral resolution, and functional calculus}\label{sec:sp-setting}

\subsection{Geometric object and operator conventions}\label{subsec:sp-object}

Let $M=\Gamma\backslash\H$ be a finite-area hyperbolic surface (possibly with cusps and torsion), with
hyperbolic measure $d\mu$ and nonnegative Laplace--Beltrami operator $\Delta$ normalized as in
\eqref{eq:laplacian} (Chapter~2B). The closure of $\Delta$ on $C_c^\infty(M)$ is essentially self-adjoint and
defines a nonnegative self-adjoint operator on $L^2(M)$; see, e.g., \cite[Ch.~3]{Iwaniec2002} and \cite[Ch.~5]{Borthwick2016}.
All spectral expansions below are understood on $L^2(M)$ with the standard cuspidal/continuous decomposition,
and with residual terms inserted only if present (cf.\ \cite[Vol.~I, Ch.~6]{Hejhal1983}).

\subsection{Spectral parameterization and Plancherel normalization}\label{subsec:sp-plancherel}

We parameterize eigenvalues by
\[
\lambda_j=\frac14+r_j^2,\qquad r_j\in\R_{\ge 0},
\]
for the discrete spectrum (counted with multiplicity). The continuous spectrum is expressed via Eisenstein series
$E_{\mathfrak a}(z,\tfrac12+ir)$ attached to cusps $\mathfrak a$, normalized so that the continuous part of the
Plancherel decomposition carries the measure $dr/(4\pi)$; see \cite[Vol.~I, Ch.~6]{Hejhal1983} and \cite[Ch.~7]{Iwaniec2002}.
Thus for $f\in L^2(M)$,
\begin{equation}\label{eq:sp-plancherel}
f
=\sum_j \langle f,\phi_j\rangle\,\phi_j
+\frac{1}{4\pi}\sum_{\mathfrak a}\int_{\R}
\langle f,E_{\mathfrak a}(\cdot,\tfrac12+ir)\rangle\,E_{\mathfrak a}(\cdot,\tfrac12+ir)\,dr,
\end{equation}
with convergence in $L^2(M)$, where $\{\phi_j\}$ is an orthonormal basis of Maass cusp forms.

\subsection{Functional calculus for $A=\sqrt{\Delta-\tfrac14}$}\label{subsec:sp-fc}

Define
\[
A:=\sqrt{\Delta-\tfrac14}
\]
by the spectral theorem (on the orthogonal complement of any residual spectrum, if present; the formulae below
extend by inserting residual summands when required). For any bounded Borel function $h:\R\to\C$ that is even,
the operator $h(A)$ is bounded on $L^2(M)$ and self-adjoint when $h$ is real-valued. Moreover,
\begin{equation}\label{eq:sp-multiplier}
h(A)f
=\sum_j h(r_j)\langle f,\phi_j\rangle\,\phi_j
+\frac{1}{4\pi}\sum_{\mathfrak a}\int_{\R} h(r)\,
\langle f,E_{\mathfrak a}(\cdot,\tfrac12+ir)\rangle\,E_{\mathfrak a}(\cdot,\tfrac12+ir)\,dr.
\end{equation}

\section{The windowed multiplier and the projector}\label{sec:sp-window}

\subsection{A fixed compactly supported time cutoff}\label{subsec:sp-cutoff}

Fix once and for all an even cutoff $\widehat\chi\in C_c^\infty(\R)$ such that
\begin{equation}\label{eq:sp-chi-hat}
\supp(\widehat\chi)\subset (-1,1),\qquad \widehat\chi(0)>0,\qquad \widehat\chi(-t)=\widehat\chi(t).
\end{equation}
Let $\chi$ be its inverse Fourier transform
\begin{equation}\label{eq:sp-chi}
\chi(u):=\frac{1}{2\pi}\int_{\R}\widehat\chi(t)\,e^{itu}\,dt.
\end{equation}
By Paley--Wiener, $\chi$ extends to an entire function and is Schwartz on $\R$.

\subsection{Spectral localization at frequency $\lambda$ with width $\eta$}\label{subsec:sp-window-def}

Let $\lambda\ge 1$ and $\eta\in(0,1]$. Define the even spectral multiplier
\begin{equation}\label{eq:sp-h}
h_{\lambda,\eta}(r)
:=\chi\!\left(\frac{r-\lambda}{\eta}\right)+\chi\!\left(\frac{r+\lambda}{\eta}\right),
\qquad r\in\R.
\end{equation}
The parameter $\lambda$ is the central frequency and $\eta$ is the window width. The choice \eqref{eq:sp-h}
is canonical given \eqref{eq:sp-chi-hat}; in particular, all constants below may depend on $\widehat\chi$ but not on $(\lambda,\eta)$.

\begin{definition}[Spectral projector]\label{def:spectral-projector}
The (smooth) spectral projector at $(\lambda,\eta)$ is
\begin{equation}\label{eq:sp-projector}
P_{\lambda,\eta}:=h_{\lambda,\eta}(A)=h_{\lambda,\eta}\!\left(\sqrt{\Delta-\tfrac14}\right)
\qquad\text{as a bounded operator on }L^2(M).
\end{equation}
\end{definition}

By \eqref{eq:sp-multiplier}, $P_{\lambda,\eta}$ acts as the multiplier $h_{\lambda,\eta}$ on the discrete and
continuous spectral parameters.

\subsection{Immediate operator properties}\label{subsec:sp-basic}

\begin{lemma}[Self-adjointness and positivity]\label{lem:sp-sa-pos}
If $h_{\lambda,\eta}$ is real-valued and even, then $P_{\lambda,\eta}$ is self-adjoint on $L^2(M)$.
If in addition $h_{\lambda,\eta}(r)\ge 0$ for all $r\in\R$, then $P_{\lambda,\eta}$ is positive semidefinite.
\end{lemma}

\begin{lemma}[$L^2$ operator norm]\label{lem:sp-L2norm}
One has
\begin{equation}\label{eq:sp-L2norm}
\|P_{\lambda,\eta}\|_{L^2\to L^2}\le \|h_{\lambda,\eta}\|_{L^\infty(\R)}.
\end{equation}
\end{lemma}

\section{Wave representation and geometric support}\label{sec:sp-wave}

\subsection{Wave packet associated to the window}\label{subsec:sp-wavepacket}

Define the (even) time-side profile
\begin{equation}\label{eq:sp-g-def}
g_{\lambda,\eta}(t):=\frac{1}{2\pi}\int_{\R} h_{\lambda,\eta}(r)\,e^{irt}\,dr.
\end{equation}
A change of variables using \eqref{eq:sp-chi} yields the explicit identity
\begin{equation}\label{eq:sp-g-explicit}
g_{\lambda,\eta}(t)=2\eta\,\widehat\chi(\eta t)\cos(\lambda t).
\end{equation}
In particular,
\begin{equation}\label{eq:sp-g-support}
\supp(g_{\lambda,\eta})\subset[-\eta^{-1},\eta^{-1}].
\end{equation}

Let $U(t):=\cos(tA)$ denote the even wave group associated with $A=\sqrt{\Delta-\tfrac14}$.
By Fourier inversion and the evenness of $h_{\lambda,\eta}$,
\begin{equation}\label{eq:sp-wave-rep}
P_{\lambda,\eta}=\int_{\R} g_{\lambda,\eta}(t)\,U(t)\,dt,
\end{equation}
where the integral converges in the strong operator topology on $L^2(M)$.

\subsection{Kernel on $\H$ and finite propagation speed (FACT)}\label{subsec:sp-propagation}

Let $K_{\lambda,\eta}(z,w)$ be the Schwartz kernel of $P_{\lambda,\eta}$ on $M$ and let
$\widetilde K_{\lambda,\eta}(\tilde z,\tilde w)$ denote the corresponding $\Gamma$--invariant kernel on $\H$.
Then
\begin{equation}\label{eq:sp-periodization}
K_{\lambda,\eta}(z,w)=\sum_{\gamma\in\Gamma}\widetilde K_{\lambda,\eta}(\tilde z,\gamma\tilde w),
\end{equation}
for any lifts $\tilde z,\tilde w\in\H$.

Finite propagation for the wave equation on $\H$ implies that the kernel of $U(t)$ on $\H$ vanishes when
$d_\H(\tilde z,\tilde w)>|t|$. Combined with \eqref{eq:sp-wave-rep}--\eqref{eq:sp-g-support} this gives:

\begin{lemma}[Geometric support of $\widetilde K_{\lambda,\eta}$]\label{lem:sp-support}
On $\H$ one has
\[
\widetilde K_{\lambda,\eta}(\tilde z,\tilde w)=0
\qquad\text{whenever}\qquad d_\H(\tilde z,\tilde w)>\eta^{-1}.
\]
Consequently, for fixed $z,w\in M$, the sum \eqref{eq:sp-periodization} is pointwise finite: only
$\gamma\in\Gamma$ with $d_\H(\tilde z,\gamma\tilde w)\le \eta^{-1}$ contribute.
\end{lemma}

\section{Sobolev mapping and window regimes}\label{sec:sp-sobolev}

\subsection{Sobolev mapping from spectral calculus}\label{subsec:sp-sobolev-map}

Let $H^s(M)$ be defined by $(I+\Delta)^{s/2}$. Since $P_{\lambda,\eta}$ is a spectral multiplier,
one obtains the following uniform mapping bounds.

\begin{lemma}[Sobolev bounds]\label{lem:sp-sobolev}
For $s\ge 0$ and $m\ge 0$,
\begin{equation}\label{eq:sp-sobolev-0}
\|P_{\lambda,\eta}\|_{H^s\to H^s}\le \sup_{r\in\R}|h_{\lambda,\eta}(r)|,
\end{equation}
and
\begin{equation}\label{eq:sp-sobolev-m}
\|P_{\lambda,\eta}\|_{H^s\to H^{s+m}}
\le
\sup_{r\in\R}\bigl(1+\tfrac14+r^2\bigr)^{m/2}\,|h_{\lambda,\eta}(r)|.
\end{equation}
\end{lemma}

\subsection{Admissible regimes for $(\lambda,\eta)$}\label{subsec:sp-regimes}

We work with $\lambda\to\infty$ and $\eta=\eta(\lambda)\in(0,1]$. Two regimes are recorded for later use.

\begin{itemize}
\item \textbf{Polynomial windows.} $\eta\ge \lambda^{-\theta}$ for some fixed $0<\theta<\theta_0$.
This ensures that derivatives of $h_{\lambda,\eta}$ (and of the induced kernels) scale at most polynomially in $\lambda$.
\item \textbf{Ehrenfest windows.} $\eta\ge (c_E\log\lambda)^{-1}$ for some $c_E>0$.
This is the window size naturally compatible with semiclassical propagation up to Ehrenfest times in hyperbolic dynamics.
\end{itemize}
The constants $\theta_0$ and $c_E$ are fixed once and for all (depending only on $M$ and on the choice of $\widehat\chi$)
when semiclassical analysis is activated in Chapter~5.

\section{Microlocal formulation at Ehrenfest scale (interface only)}\label{sec:sp-microlocal}

This section fixes notation for the semiclassical analysis performed in Chapter~5 and states only the interface
result needed to connect \eqref{eq:sp-wave-rep} with Egorov-type propagation. Detailed parametrices and error bounds
are proved later and will not be duplicated here.

\subsection{Semiclassical parameter and quantization}\label{subsec:sp-semiclassical}

Set
\[
h:=\lambda^{-1}.
\]
Let $\Op_h(a)$ denote a standard semiclassical quantization of a compactly supported symbol $a$ on $T^*M$ in local charts.
Let $g^t$ denote the geodesic flow on $T^*M$.

\subsection{Egorov up to logarithmic time (Hypothesis + citation)}\label{subsec:sp-egorov}

Let $V(t):=e^{itA}$ be the unitary wave group. For Anosov geodesic flow, a semiclassical Egorov theorem yields,
for $|t|\le c\log(1/h)$,
\begin{equation}\label{eq:sp-egorov}
V(-t)\,\Op_h(a)\,V(t)=\Op_h(a\circ g^t)+\mathcal R_h(t),
\end{equation}
with $\|\mathcal R_h(t)\|_{L^2\to L^2}\le C\,h^{1-\varepsilon}$ for any fixed $\varepsilon>0$,
where $C$ depends on finitely many seminorms of $a$ and on geometric data of $M$;
see \cite{Anantharaman2008,NonnenmacherZworski2009,Zworski2012}.
We combine \eqref{eq:sp-egorov} with the compact time support \eqref{eq:sp-g-support}. In particular, if
$\eta\ge (c_E\log\lambda)^{-1}$ and $c_E$ is chosen so that $\eta^{-1}\le c\log(1/h)$, then the entire time range
present in \eqref{eq:sp-wave-rep} falls within the Egorov validity interval.

\section{Summary and forward links}\label{sec:sp-summary}

We have fixed a canonical family of smooth spectral projectors
\[
P_{\lambda,\eta}=h_{\lambda,\eta}\!\left(\sqrt{\Delta-\tfrac14}\right),
\]
built from a single compactly supported time cutoff $\widehat\chi$.

\begin{itemize}
\item \textbf{Spectral localization.} $P_{\lambda,\eta}$ acts as the multiplier $h_{\lambda,\eta}$ on discrete and continuous spectrum
by \eqref{eq:sp-multiplier}.
\item \textbf{Wave representation.} $P_{\lambda,\eta}$ admits the exact wave packet formula \eqref{eq:sp-wave-rep} with explicit
$g_{\lambda,\eta}$ given by \eqref{eq:sp-g-explicit}.
\item \textbf{Finite propagation (FACT).} The kernel on $\H$ is supported in $d_\H(\tilde z,\tilde w)\le \eta^{-1}$
(Lemma~\ref{lem:sp-support}), hence the periodization sum \eqref{eq:sp-periodization} is pointwise finite.
\item \textbf{Regimes.} Polynomial windows and Ehrenfest windows are recorded in Section~\ref{subsec:sp-regimes} for later activation.
\end{itemize}

\noindent\textbf{Forward links.}
Chapter~5 constructs the microlocal parametrix and quantifies the remainder at Ehrenfest scale.
Chapter~6 applies the same cutoff within the Selberg trace identity to expand the geometric side.
Chapter~7 combines spectral and geometric expansions into the localized trace formula.
