\section{Truncated Selberg kernels}\label{sec:truncated-kernels}

\subsection{Setup and admissible dependencies}\label{subsec:ch3-setup}

Let $M=\Gamma\backslash\H$ be a cofinite quotient, with metric and measure fixed in
Section~\ref{subsec:geometry-diamond}. Cusps, scaling matrices, height functions, and the
smoothed truncation $\Lambda_{\sm}^Y$ are fixed in
Section~\ref{subsec:cusps-diamond}. We write
\[
\mathscr C_\Gamma=\bigl(h;\{w_{\mathfrak a}\};\{\sigma_{\mathfrak a}\}\bigr),\qquad
Y\ge Y_0(\mathscr C_\Gamma),
\]
where $Y_0$ is any height for which the truncated cusp tails are pairwise disjoint.

\medskip
\noindent\textbf{Dependency rule.}
Unless explicitly stated otherwise, all implied constants in Chapter~\ref{sec:truncated-kernels} may depend
only on $\mathscr C_\Gamma$, on $Y_0(\mathscr C_\Gamma)$, on the fixed cutoff profile $\eta$ used to define
$\Lambda_{\sm}^Y$, and on the support radius $R$ of the radial profile $q$ below.
In particular, \emph{no constant depends on later localization parameters $(\lambda,\eta)$} (where $\eta$
will later denote a spectral window; here $\eta$ denotes the fixed smooth cutoff profile from Chapter~2).

% ---------------------------------------------------------------------
\subsection{Definition of the truncated kernel}\label{subsec:block3.1}

Fix $R>0$ and a radial profile $q\in C_c^\infty([0,\infty))$ supported in $[0,R]$.
Define the $\PSL_2(\R)$--invariant kernel on $\H$ by
\[
k_q(z,w):=q\bigl(d_\H(z,w)\bigr).
\]
For $z,w\in\H$ set
\begin{equation}\label{eq:K-def}
K(z,w):=\sum_{\gamma\in\Gamma} k_q(z,\gamma w)
       =\sum_{\gamma\in\Gamma} q\!\bigl(d_\H(z,\gamma w)\bigr).
\end{equation}
The sum is \emph{pointwise finite}: since $\supp(q)\subset[0,R]$ and the $\Gamma$--action is properly discontinuous,
for each fixed $(z,w)$ there are only finitely many $\gamma$ with $d_\H(z,\gamma w)\le R$.
Thus $K$ descends to a smooth kernel on $M\times M$.

Let $\Lambda^Y_{\sm}$ be the smoothed truncation multiplier fixed in \eqref{eq:smooth-trunc}.
Define the truncated kernel on $M\times M$ by
\begin{equation}\label{eq:KY-def}
K_Y(z,w):=\Lambda^Y_{\sm}(z)\,\Lambda^Y_{\sm}(w)\,K(z,w),
\qquad z,w\in M.
\end{equation}

\begin{proposition}[Operator identity and quadratic-form compatibility]\label{prop:KY-factor}
Let $T$ (resp.\ $T_Y$) be the integral operator on $L^2(M)$ with kernel $K$ (resp.\ $K_Y$).
Then
\begin{equation}\label{eq:KY-factor}
T_Y=\Lambda^Y_{\sm}\,T\,\Lambda^Y_{\sm}
\end{equation}
as operators on $L^2(M)$, and in particular for all $f,g\in L^2(M)$,
\begin{equation}\label{eq:KY-quadratic}
\langle T_Y f,g\rangle_{L^2(M)}
=
\langle T(\Lambda^Y_{\sm}f),\,\Lambda^Y_{\sm}g\rangle_{L^2(M)}.
\end{equation}
\end{proposition}

\noindent
\emph{Convention.} We never use (and explicitly forbid) any eigenfunction identity of the form
$T_Y\phi_j=h(r_j)\phi_j$; the only spectral compatibility we use is the quadratic-form statement
\eqref{eq:KY-quadratic}.

\subsection{Geometric representation and orbit counting}\label{subsec:block3.2}

Define for $z,w\in\H$ and $R>0$
\[
\mathcal N(z,w;R):=\{\gamma\in\Gamma:\ d_\H(z,\gamma w)\le R\}.
\]
Then \eqref{eq:K-def} and \eqref{eq:KY-def} may be written as a finite sum
\begin{equation}\label{eq:KY-gammasum}
K_Y(z,w)=\Lambda^Y_{\sm}(z)\Lambda^Y_{\sm}(w)\sum_{\gamma\in\mathcal N(z,w;R)} q\!\bigl(d_\H(z,\gamma w)\bigr).
\end{equation}

\begin{lemma}[Uniform counting bound on compacta]\label{lem:latticepoint-upper}
Fix $Y\ge Y_0(\mathscr C_\Gamma)$. There exists $C=C(\Gamma,Y)$ such that for all $z,w\in M(Y)$,
\begin{equation}\label{eq:latticepoint-upper}
\#\mathcal N(\tilde z,\tilde w;R)\le C\,e^{R},
\end{equation}
where $\tilde z,\tilde w$ are any lifts of $z,w$ to $\H$.
\end{lemma}

\noindent
\emph{Remark.} One may choose $C$ depending on a lower bound for $\inj$ on $M(Y)$ (packing by disjoint balls),
but we will only use the existence of a bound of the form \eqref{eq:latticepoint-upper}.

\begin{corollary}[Pointwise bound]\label{cor:KY-pointwise}
For $Y\ge Y_0(\mathscr C_\Gamma)$ and $z,w\in M(Y)$,
\begin{equation}\label{eq:KY-pointwise}
|K_Y(z,w)|\le C(\Gamma,Y)\,e^{R}\,\|q\|_{L^\infty}\,.
\end{equation}
\end{corollary}

\begin{proposition}[Quadratic-form spectral decomposition]\label{prop:KY-spectral-qf}
Let $T=h(\Delta)$ be the untruncated Selberg operator associated to $q$ via the transform conventions of
Section~\ref{sec:selberg-transform}. Then for $f,g\in L^2(M)$,
\begin{equation}\label{eq:KY-qf-spectral}
\langle T_Y f,g\rangle
=\sum_{j} h(r_j)\,\langle \Lambda^Y_{\sm}f,\phi_j\rangle\,\overline{\langle \Lambda^Y_{\sm}g,\phi_j\rangle}
+\frac{1}{4\pi}\sum_{\mathfrak a}\int_{-\infty}^{\infty}
h(r)\,
\langle \Lambda^Y_{\sm}f,E_{\mathfrak a}(\cdot,\tfrac12+ir)\rangle\,
\overline{\langle \Lambda^Y_{\sm}g,E_{\mathfrak a}(\cdot,\tfrac12+ir)\rangle}\,dr,
\end{equation}
with residual terms inserted only if present in the spectral decomposition of $L^2(M)$.
\end{proposition}

\subsection{Support control and geometric localization}\label{subsec:block3.3}

\begin{lemma}[Support criterion]\label{lem:support-criterion}
Assume $\supp(q)\subset[0,R]$. If $K_Y(z,w)\neq 0$, then:
\[
z,w\in \supp(\Lambda^Y_{\sm})\subset M\setminus \bigcup_{\mathfrak a}\pi(C_{\mathfrak a}(2Y))
\quad\text{and}\quad
\exists\gamma\in\Gamma:\ d_\H(\tilde z,\gamma\tilde w)\le R,
\]
for some (equivalently, any) lifts $\tilde z,\tilde w$ of $z,w$.
\end{lemma}

\begin{corollary}[Core stability]\label{cor:core-stability}
If $z,w\in M(Y)$, then $\Lambda^Y_{\sm}(z)=\Lambda^Y_{\sm}(w)=1$ and hence
\[
K_Y(z,w)=K(z,w).
\]
\end{corollary}

\noindent
\textbf{Cusp-height control under distance restriction (safe form).}
If $z,w$ lie in the same cusp chart and $d_\H(\tilde z,\tilde w)\le R$, then their heights satisfy
$y(w)\ll_{\mathscr C_\Gamma} e^{R}y(z)$ (a direct consequence of the hyperbolic metric in cusp coordinates).
We use this only to preclude uncontrolled excursions in cusp height when distance is bounded.

\subsection{A priori operator bounds}\label{subsec:block3.4}

Let $T$ (resp.\ $T_Y$) be the operators with kernels $K$ (resp.\ $K_Y$).

\begin{lemma}[$L^2$ operator norm]\label{lem:L2-opnorm}
One has
\begin{equation}\label{eq:L2-opnorm}
\|T_Y\|_{L^2\to L^2}\le \|T\|_{L^2\to L^2}\le \|h\|_{L^\infty(\R)}.
\end{equation}
\end{lemma}

\begin{lemma}[Hilbert--Schmidt bound]\label{lem:HS-bound}
For each fixed $Y\ge Y_0(\mathscr C_\Gamma)$, the operator $T_Y$ is Hilbert--Schmidt, and
\begin{equation}\label{eq:HS-bound}
\|T_Y\|_{HS}^2
=\iint_{M\times M}|K_Y(z,w)|^2\,d\mu(z)\,d\mu(w)
\le \vol\!\big(M\setminus{\textstyle\bigcup_{\mathfrak a}}\pi(C_{\mathfrak a}(2Y))\big)^2
\cdot \sup_{z,w}|K_Y(z,w)|^2,
\end{equation}
hence in particular $\|T_Y\|_{HS}<\infty$.
\end{lemma}

\begin{proposition}[Sobolev mapping via functional calculus]\label{prop:sobolev-KY}
Fix $s\ge 0$. Assume $h$ satisfies $\sup_{r\in\R}(1+|r|)^s|h(r)|<\infty$.
Then
\begin{equation}\label{eq:sobolev-KY}
\|T_Y f\|_{H^s(M)}
\le C_{s,\mathscr C_\Gamma,\eta}\,
\sup_{r\in\R}(1+|r|)^s|h(r)|\ \|f\|_{H^s(M)}.
\end{equation}
\end{proposition}

\begin{proposition}[Truncation tail in Hilbert--Schmidt norm]\label{prop:tail-HS}
Let $S_Y:=\supp(1-\Lambda^Y_{\sm})\subset M\setminus M(Y)$.
Then $\vol(S_Y)\ll_{\mathscr C_\Gamma,\eta} Y^{-1}$, and
\begin{equation}\label{eq:tail-HS}
\|T-T_Y\|_{HS}\ \ll_{\Gamma,\mathscr C_\Gamma,\eta}\ e^{R}\|q\|_{L^\infty}\,Y^{-1/2}.
\end{equation}
\end{proposition}

\noindent
\textbf{Dichotomy enforced.}
Exponential decay in $Y$ is available for individual cusp forms in cusp coordinates, but no such uniform
decay holds for Eisenstein series; continuous-spectrum tail control is handled via Maass--Selberg and is
polynomial in $Y$. We enforce this dichotomy throughout.

\subsection*{Chapter audit (Kernel construction)}\label{audit:ch3-annals}

\begin{itemize}
\item[(K1)] The kernel $K$ is defined by the pointwise finite sum \eqref{eq:K-def} (compact distance support + proper discontinuity).
\item[(K2)] The truncated kernel $K_Y$ is defined by \eqref{eq:KY-def} using the fixed cutoff $\Lambda^Y_{\sm}$.
\item[(K3)] The only spectral compatibility statement used is the operator/quadratic-form identity
      \eqref{eq:KY-factor}--\eqref{eq:KY-quadratic}; no eigenfunction multiplier identity is asserted for $T_Y$.
\item[(K4)] Support control is isolated by Lemma~\ref{lem:support-criterion} and core stability by Corollary~\ref{cor:core-stability}.
\item[(K5)] Robust bounds: $L^2$ norm \eqref{eq:L2-opnorm}; Hilbert--Schmidt finiteness \eqref{eq:HS-bound};
      Sobolev mapping \eqref{eq:sobolev-KY}; truncation tail \eqref{eq:tail-HS} with correct $Y$--scale.
\end{itemize}
