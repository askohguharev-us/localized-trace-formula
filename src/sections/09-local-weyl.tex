\chapter{A Local Weyl Law with Quantified Remainders}
\label{chap:09-local-weyl}

\epigraph{A Weyl law is not an asymptotic; it is a stability statement about what survives every admissible normalization.}{}

\section{Purpose and statement of the problem}
\label{sec:09.purpose}

This chapter establishes a local Weyl law for the spectral data associated with the geometric setting
fixed in Chapters~\ref{chap:01-introduction}--\ref{chap:08-normalization}. The central goal is to extract,
from the normalized trace functional of Chapter~\ref{chap:08-normalization}, a canonical leading term
together with a remainder bound that is stable under all admissible auxiliary choices.

Throughout, the underlying operator and the corresponding spectral resolution are those fixed earlier
(e.g.\ the Laplace--Beltrami operator on the relevant quotient, or the automorphic Laplacian in the presence
of a character). We write the spectrum in the standard spectral parameterization
\[
\lambda_j=\frac14+r_j^2,
\qquad r_j\ge 0,
\]
and denote by $\{\varphi_j\}$ an orthonormal basis of eigenfunctions.

The fundamental object of this chapter is the (possibly weighted) spectral measure
\[
\mu_{\Gamma,\chi} := \sum_{j} \delta_{r_j},
\qquad
\mu_{\Gamma,\chi}^A := \sum_{j} \ip{A\varphi_j}{\varphi_j}\,\delta_{r_j},
\]
where $A$ is a fixed zeroth-order pseudodifferential operator (or a microlocal projector) as introduced
in Chapter~\ref{chap:05-microlocal}. The notation indicates that the construction of the localized trace
depends on admissible auxiliary data, but the theorems below show that the final asymptotic is canonical
modulo explicit corrections and a quantified error ledger.

\section{Test functions and smoothing conventions}
\label{sec:09.tests}

\subsection{Admissible smoothing kernels}
\label{subsec:09.admissible-smoothing}

Let $h:\R\to\C$ be an admissible test function in the sense of Chapter~\ref{chap:08-normalization}
(typically Schwartz, or Paley--Wiener of prescribed exponential type). We shall systematically
consider rescaled families
\[
h_T(t):=h\!\left(\frac{t}{T}\right),
\qquad T\ge 1.
\]
We write $\widehat h$ for the Fourier transform with the convention
\[
\widehat h(\xi)=\int_{\R} h(t)e^{-it\xi}\,dt.
\]

The analysis of this chapter is based on the normalized trace functional
\[
\mathcal{T}_\Gamma[h;\chi]=\mathcal{L}_\Gamma[h-h(0)\chi]
\]
constructed in Chapter~\ref{chap:08-normalization}. In particular, the dependence on $\chi$ is controlled
explicitly by Theorem~\ref{thm:08.normalization-stability}.

\subsection{Even--odd decomposition and canonical parity reduction}
\label{subsec:09.parity}

For any admissible $h$ we define
\[
h^{\mathrm{ev}}(t)=\frac12\bigl(h(t)+h(-t)\bigr),
\qquad
h^{\mathrm{odd}}(t)=\frac12\bigl(h(t)-h(-t)\bigr),
\]
so that $h=h^{\mathrm{ev}}+h^{\mathrm{odd}}$.

The identity contribution in trace formulas is parity-even. Consequently, the odd component provides a
canonical defect channel which isolates contributions that cannot appear in the principal Weyl term.

\begin{lemma}[Parity reduction of the principal term]
\label{lem:09.parity-reduction}
Let $\mathcal{T}_\Gamma[h;\chi]$ be the canonical normalized trace functional of
Chapter~\ref{chap:08-normalization}. Then the principal Weyl term depends only on $h^{\mathrm{ev}}$.
More precisely, for any admissible $(h,\chi)$ one has
\[
\mathcal{T}_\Gamma[h;\chi]
=
\mathcal{T}_\Gamma[h^{\mathrm{ev}};\chi]
+
\mathcal{T}_\Gamma[h^{\mathrm{odd}};\chi],
\]
and the contribution of $\mathcal{T}_\Gamma[h^{\mathrm{odd}};\chi]$ is purely non-identity and hence belongs
to the remainder ledger.
\end{lemma}

\begin{proof}
The decomposition is tautological by linearity. The parity claim follows from the fact that the identity
term in the localized trace formula is an even functional of the test variable, as in the standard
parametrix computation for the wave kernel and the Selberg trace formula; the odd part therefore has
no identity contribution and must be accounted for entirely by non-identity and remainder terms.
\end{proof}

\section{Spectral counting functions and smoothed Weyl laws}
\label{sec:09.smoothed-weyl}

\subsection{Smoothed counting function}
\label{subsec:09.smoothed-count}

Let $w\in \mathcal{S}(\R)$ be a fixed nonnegative Schwartz function with $\int_{\R} w(t)\,dt=1$.
For $\delta>0$ we set
\[
w_\delta(t):=\delta^{-1}w(t/\delta).
\]
We define the smoothed counting function
\[
N_\Gamma(r;\delta)
:=
(\mu_{\Gamma,\chi}*w_\delta)(r)
=
\sum_{j} w_\delta(r-r_j),
\]
and, more generally, for a bounded operator $A$ of order $0$,
\[
N_{\Gamma,A}(r;\delta)
:=
(\mu_{\Gamma,\chi}^A*w_\delta)(r)
=
\sum_{j}\ip{A\varphi_j}{\varphi_j}\,w_\delta(r-r_j).
\]

The principal asymptotic of $N_\Gamma(r;\delta)$ as $r\to\infty$ is independent of $\delta$ as long as
$\delta$ remains in an admissible regime (e.g.\ $\delta\ge r^{-\theta}$ for some fixed $\theta>0$),
and the dependence on $\delta$ is fully absorbed into the Tauberian remainder.

\subsection{The local Weyl law: statement}
\label{subsec:09.statement}

We denote by $\vol(X)$ the Riemannian volume of the underlying manifold/orbifold $X$ and by $n=\dim X$.
In the classical compact surface case one has $n=2$ and the principal term is linear in $r$.
To keep the statement uniform across admissible geometries, we write the principal term in the standard
phase-volume form.

\begin{theorem}[Smoothed local Weyl law]
\label{thm:09.smoothed-weyl}
Assume the standing hypotheses of Chapters~\ref{chap:01-introduction}--\ref{chap:08-normalization}.
Let $w$ be as above and let $\delta\in(0,1]$ be fixed. Then there exists an explicit constant $C_X>0$,
depending only on the geometric data of $X$, such that
\[
N_\Gamma(r;\delta)
=
C_X\,r^{n-1}
+
\mathcal{R}_\Gamma(r;\delta),
\qquad r\to+\infty,
\]
where the remainder satisfies the quantified bound
\[
\abs{\mathcal{R}_\Gamma(r;\delta)}
\le
\mathcal{E}_{\Gamma}^{\mathrm{TF}}(r;\delta)
+
\mathcal{E}_{\Gamma}^{\mathrm{Tab}}(r;\delta).
\]
Here:
\begin{enumerate}
  \item $\mathcal{E}_{\Gamma}^{\mathrm{TF}}(r;\delta)$ is the trace-formula remainder (geometric and spectral),
  controlled by the error ledger of Chapter~\ref{chap:08-normalization};
  \item $\mathcal{E}_{\Gamma}^{\mathrm{Tab}}(r;\delta)$ is the Tauberian smoothing remainder, depending only on
  $\delta$ and on the decay properties of $w$.
\end{enumerate}
In particular, the principal term is canonical and depends only on the phase-volume invariant of $X$.
\end{theorem}

\begin{remark}
\label{rem:09.principal-term}
In dimension $n=2$ (hyperbolic surfaces), the principal coefficient is explicitly
\[
C_X=\frac{\vol(X)}{2\pi}.
\]
More generally, $C_X$ is the universal phase-volume constant
\[
C_X=\frac{1}{(2\pi)^n}\vol\bigl(S^*X\bigr),
\]
as in the standard local Weyl law.
\end{remark}

\section{Derivation from the normalized trace formula}
\label{sec:09.derivation}

\subsection{Spectral side and normalization}
\label{subsec:09.spectral-side}

Let $h$ be an admissible test function. The raw localized trace functional has the spectral expansion
\[
\mathcal{L}_\Gamma[h]
=
\sum_{j} h(r_j)\;+\;\mathcal{C}_\Gamma[h],
\]
where $\mathcal{C}_\Gamma[h]$ denotes the continuous-spectrum contribution when present (and is identically
zero in the compact case). The normalized trace functional of Chapter~\ref{chap:08-normalization} is
\[
\mathcal{T}_\Gamma[h;\chi]
=
\mathcal{L}_\Gamma[h-h(0)\chi].
\]
The subtraction removes the spurious zero-mode dependence and isolates the part of the trace that is stable
under admissible cutoffs.

\subsection{Identity term and phase volume}
\label{subsec:09.identity-term}

The identity term of the trace formula has the universal form
\[
\mathcal{I}_\Gamma[h]
=
\int_{\R} h(r)\,\rho_X(r)\,dr,
\]
where $\rho_X(r)$ is the Plancherel density determined by the geometry (for hyperbolic surfaces,
$\rho_X(r)=\frac{\vol(X)}{2\pi}r\tanh(\pi r)$ up to conventional normalization).
In particular, the leading Weyl term arises from the asymptotic expansion of $\rho_X(r)$ as $r\to\infty$.

\begin{lemma}[Asymptotic expansion of the identity contribution]
\label{lem:09.identity-asymptotic}
Let $h_T(t)=h(t/T)$ be an admissible rescaled family. Then the identity contribution satisfies
\[
\mathcal{I}_\Gamma[h_T]
=
C_X\,T^{n-1}\int_{\R} h(u)\,du
+
O\!\left(T^{n-2}\right),
\qquad T\to+\infty,
\]
where $C_X$ is the phase-volume constant of Remark~\ref{rem:09.principal-term}.
\end{lemma}

\begin{proof}
This is the standard stationary phase expansion for the local spectral density, equivalently the leading
term in the Fourier inversion of the wave kernel parametrix. The constant $C_X$ is determined by the
principal symbol of the Laplacian and is therefore canonical. The lower-order term is controlled by the
standard remainder estimates in the parametrix construction.
\end{proof}

\subsection{Non-identity contributions and the trace-formula remainder}
\label{subsec:09.nonidentity}

The remaining geometric contributions (hyperbolic, elliptic, parabolic/cusp terms, and the scattering
determinant where applicable) define the trace-formula remainder. The normalization theorem of
Chapter~\ref{chap:08-normalization} implies that these contributions are stable under admissible changes
of cutoff profiles, quantization conventions, and microlocal partitions, up to the explicit identity-term
correction and a quantified ledger.

\begin{proposition}[Trace-formula remainder bound]
\label{prop:09.trace-formula-remainder}
Let $h$ be admissible. Then
\[
\mathcal{T}_\Gamma[h;\chi]
=
\mathcal{I}_\Gamma[h]
+
\mathcal{G}_\Gamma[h]
+
\mathcal{R}_\Gamma^{\mathrm{TF}}[h;\chi],
\]
where $\mathcal{G}_\Gamma[h]$ denotes the non-identity geometric contributions, and the remainder satisfies
\[
\abs{\mathcal{R}_\Gamma^{\mathrm{TF}}[h;\chi]}
\le
\mathcal{E}_\Gamma[h;\chi],
\]
with $\mathcal{E}_\Gamma[h;\chi]$ the consolidated normalization ledger of
Definition~\ref{def:08.total-ledger}.
\end{proposition}

\begin{proof}
This is a direct restatement of the normalization stability theorem
(Theorem~\ref{thm:08.normalization-stability}), together with the decomposition of the raw trace formula
established in Chapter~\ref{chap:07-main-results}.
\end{proof}

\section{A two-component remainder decomposition}
\label{sec:09.two-dust}

\subsection{Trace remainder versus Tauberian remainder}
\label{subsec:09.dust-split}

The passage from smoothed traces to smoothed counting functions introduces an additional analytic error
which is conceptually distinct from the geometric remainder. It is convenient to separate these two
sources of error.

\begin{definition}[Two-component remainder decomposition]
\label{def:09.two-component}
For $r>0$ and $\delta\in(0,1]$ define the decomposition
\[
\mathcal{R}_\Gamma(r;\delta)
=
\mathcal{R}_\Gamma^{\mathrm{TF}}(r;\delta)
+
\mathcal{R}_\Gamma^{\mathrm{Tab}}(r;\delta),
\]
where:
\begin{enumerate}
  \item $\mathcal{R}_\Gamma^{\mathrm{TF}}(r;\delta)$ denotes the trace-formula remainder arising from the
  non-identity contributions and from normalization errors;
  \item $\mathcal{R}_\Gamma^{\mathrm{Tab}}(r;\delta)$ denotes the Tauberian remainder arising from smoothing,
  i.e.\ from the replacement of a sharp cutoff by convolution with $w_\delta$.
\end{enumerate}
\end{definition}

This separation is canonical: $\mathcal{R}_\Gamma^{\mathrm{TF}}$ depends on the geometry and on the trace formula,
whereas $\mathcal{R}_\Gamma^{\mathrm{Tab}}$ depends only on the smoothing kernel $w$ and on the scale $\delta$.

\subsection{Quantified Tauberian control}
\label{subsec:09.tauberian}

We recall the elementary smoothing inequality: for any nondecreasing function $F$ and any mollifier $w_\delta$,
one has
\[
F(r-\delta)
\;\le\;
(F*w_\delta)(r)
\;\le\;
F(r+\delta),
\]
whence the deviation between $F$ and its smoothing is controlled by the increment of $F$ on a boundary layer
of size $\delta$.

In our setting $F(r)$ is the sharp counting function
\[
N_\Gamma(r):=\#\{j:\ r_j\le r\}.
\]
Thus the Tauberian error is bounded by the number of spectral points in a window of size $\delta$ around $r$.

\begin{lemma}[Tauberian remainder bound]
\label{lem:09.tauberian-bound}
Let $w$ be a nonnegative mollifier with $\int w=1$. Then for all $r>0$ and $\delta\in(0,1]$,
\[
\abs{N_\Gamma(r;\delta)-N_\Gamma(r)}
\le
N_\Gamma(r+\delta)-N_\Gamma(r-\delta).
\]
In particular,
\[
\abs{\mathcal{R}_\Gamma^{\mathrm{Tab}}(r;\delta)}
\le
N_\Gamma(r+\delta)-N_\Gamma(r-\delta).
\]
\end{lemma}

\begin{proof}
This is the standard monotonicity estimate for convolution with a nonnegative kernel of mass one.
\end{proof}

\section{Weighted local Weyl laws}
\label{sec:09.weighted}

\subsection{Microlocal weights and pseudodifferential observables}
\label{subsec:09.observable}

Let $A\in\Psi^0(X)$ be a zeroth-order pseudodifferential operator with principal symbol $\sigma_0(A)$.
The weighted counting function is
\[
N_{\Gamma,A}(r;\delta)=\sum_j \ip{A\varphi_j}{\varphi_j}\,w_\delta(r-r_j).
\]

\begin{theorem}[Weighted smoothed local Weyl law]
\label{thm:09.weighted-weyl}
Let $A\in\Psi^0(X)$ be fixed. Then there exists an explicit constant $C_{X,A}$ such that
\[
N_{\Gamma,A}(r;\delta)
=
C_{X,A}\,r^{n-1}
+
\mathcal{R}_{\Gamma,A}(r;\delta),
\qquad r\to+\infty,
\]
where
\[
C_{X,A}
=
\frac{1}{(2\pi)^n}\int_{S^*X} \sigma_0(A)\,d\omega,
\]
and the remainder satisfies
\[
\abs{\mathcal{R}_{\Gamma,A}(r;\delta)}
\le
\mathcal{E}_{\Gamma,A}^{\mathrm{TF}}(r;\delta)
+
\mathcal{E}_{\Gamma,A}^{\mathrm{Tab}}(r;\delta),
\]
with $\mathcal{E}_{\Gamma,A}^{\mathrm{TF}}$ controlled by the normalization ledger of Chapter~\ref{chap:08-normalization}.
\end{theorem}

\begin{proof}
This is the standard local Weyl law for pseudodifferential observables, derived by inserting $A$ into the
trace formula and applying the symbolic calculus of Chapter~\ref{chap:05-microlocal}. The principal term
is the phase-space average of the principal symbol, and the remainder follows from the same error ledger
as in the unweighted case, together with the Tauberian bound of Lemma~\ref{lem:09.tauberian-bound}.
\end{proof}

\begin{remark}
\label{rem:09.quantization-independence}
The coefficient $C_{X,A}$ depends only on the principal symbol of $A$ and is therefore invariant under
admissible changes of quantization conventions. All quantization-dependent contributions are of lower
order and are absorbed into the trace-formula remainder ledger.
\end{remark}

\section{Odd-channel defect estimate}
\label{sec:09.odd-defect}

The even--odd splitting of Section~\ref{subsec:09.parity} provides an intrinsic diagnostic for the principal term.

\begin{proposition}[Odd-channel bound]
\label{prop:09.odd-channel}
Let $h$ be admissible. Then
\[
\mathcal{T}_\Gamma[h^{\mathrm{odd}};\chi]
=
\mathcal{G}_\Gamma[h^{\mathrm{odd}}]
+
\mathcal{R}_\Gamma^{\mathrm{TF}}[h^{\mathrm{odd}};\chi],
\]
where $\mathcal{G}_\Gamma$ denotes the non-identity geometric contributions, and
\[
\abs{\mathcal{R}_\Gamma^{\mathrm{TF}}[h^{\mathrm{odd}};\chi]}
\le
\mathcal{E}_\Gamma[h^{\mathrm{odd}};\chi].
\]
In particular, no identity-type Weyl main term occurs in the odd channel.
\end{proposition}

\begin{proof}
The identity contribution is even in the test variable, hence vanishes on $h^{\mathrm{odd}}$.
The remainder estimate is the normalization stability theorem of Chapter~\ref{chap:08-normalization}.
\end{proof}

\section{From smoothed laws to sharp counting}
\label{sec:09.sharp}

\subsection{A Tauberian upgrade}
\label{subsec:09.tauberian-upgrade}

The sharp Weyl law follows by choosing an admissible family of mollifiers $w_\delta$ with $\delta=\delta(r)$
tending to zero slowly enough that the Tauberian boundary layer remains negligible compared to the principal term.

\begin{theorem}[Sharp Weyl law from smoothed estimates]
\label{thm:09.sharp-weyl}
Assume the smoothed Weyl law of Theorem~\ref{thm:09.smoothed-weyl} holds uniformly for
$\delta\in[r^{-\theta},1]$ for some fixed $\theta\in(0,1)$. Then the sharp counting function satisfies
\[
N_\Gamma(r)
=
C_X\,r^{n-1}
+
o(r^{n-1}),
\qquad r\to+\infty.
\]
More precisely, one has the bound
\[
N_\Gamma(r)
=
C_X\,r^{n-1}
+
O\!\bigl(\mathcal{E}_\Gamma^{\mathrm{TF}}(r;\delta(r))\bigr)
+
O\!\bigl(N_\Gamma(r+\delta(r))-N_\Gamma(r-\delta(r))\bigr),
\]
where $\delta(r)\ge r^{-\theta}$ is any admissible smoothing scale.
\end{theorem}

\begin{proof}
Combine Theorem~\ref{thm:09.smoothed-weyl} with Lemma~\ref{lem:09.tauberian-bound} and choose $\delta(r)$
so that the Tauberian increment term is negligible compared to $r^{n-1}$. The trace-formula remainder is
controlled uniformly by the normalization ledger, hence yields the stated bound.
\end{proof}

\section{Uniformity in auxiliary data}
\label{sec:09.uniformity}

A central feature of the Weyl law established above is that it is stable under admissible changes of
localization profiles, quantization conventions, and microlocal refinements. This is an immediate
consequence of the normalization theory of Chapter~\ref{chap:08-normalization}.

\begin{corollary}[Stability under admissible auxiliary changes]
\label{cor:09.stability}
Let $\mathcal{T}_\Gamma[h;\chi]$ and $\mathcal{T}'_\Gamma[h;\chi']$ be normalized trace functionals constructed
from two admissible sets of auxiliary choices. Then the Weyl coefficients $C_X$ and $C_{X,A}$ in
Theorems~\ref{thm:09.smoothed-weyl} and \ref{thm:09.weighted-weyl} coincide, and the difference between the
corresponding remainders is bounded by the sum of the associated normalization ledgers.
\end{corollary}

\begin{proof}
The principal term arises from the identity contribution and is therefore canonical.
The remainder stability is Theorem~\ref{thm:08.normalization-stability}.
\end{proof}

\section{Bibliographic notes}
\label{sec:09.bib-notes}

The local Weyl law in its classical and microlocal forms is standard; see \cite{HormanderI,Sogge2014,Zworski2012}.
In the automorphic and hyperbolic settings, the principal term and the spectral parameterization
are consistent with the Selberg trace formula tradition; see \cite{Hejhal1983,Buser1992,Iwaniec2002,Borthwick2016}.
Weighted versions for pseudodifferential observables follow from the symbolic calculus and the local trace expansion,
cf.\ \cite{DuistermaatGuillemin1975,Ivrii1998,DyatlovZworski2019}.
