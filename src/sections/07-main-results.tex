\chapter{Spectral Expansion, Normalization, and Comparison for the Localized Trace Formula}

\noindent\textbf{Scope and purpose.}
Let \(M=\Gamma\backslash\mathbb{H}\) be a finite-area hyperbolic surface, \(\Delta\) the (positive) Laplace--Beltrami operator,
and let
\[
  P_{\lambda,\eta}=h_{\lambda,\eta}\!\left(\sqrt{\Delta-\tfrac14}\right)
\]
be the localized spectral projector introduced in Chapters~5--6.
This chapter constructs the \emph{spectral side} of the localized Selberg trace formula, fixes the \emph{normalization locks}
(Fourier conventions, Selberg transforms, cusp truncation), and performs the \emph{comparison-locked} identity with the geometric side.
In particular, we:
\begin{enumerate}
\item[(S7.1)] write \(\Tr^{\mathrm{reg}}(P_{\lambda,\eta})\) as a discrete \(+\) continuous \(+\) (possible) residual contribution;
\item[(S7.2)] isolate all normalization-dependent linear functionals needed for cofinite \(\Gamma\) (cusps);
\item[(S7.3)] extract the localized Weyl main term and record a remainder hierarchy compatible with \(\eta\ge \lambda^{-\theta}\);
\item[(S7.4)] identify the analytic portals where \(\varphi'/\varphi\) (scattering) and \(Z_\Gamma'/Z_\Gamma\) (Selberg zeta) enter.
\end{enumerate}
Standard references are Selberg~\cite{Selberg1956}, Hejhal~\cite{Hejhal1983}, and Iwaniec~\cite{Iwaniec2002}.

% ==========================================================
\section{OBJ-LOCK and normalization locks (no substitution)}
% ==========================================================

\noindent\textbf{OBJ-LOCK (single object).}
The single object compared throughout Chapters~6--7 is the same regularized trace
\[
  \Tr^{\mathrm{reg}}(P_{\lambda,\eta}),
\]
and, where microlocal cutoffs are inserted, its conjugated version \(\Tr^{\mathrm{reg}}(\Psi P_{\lambda,\eta}\Psi)\).
No substitution is permitted between:
\begin{itemize}
\item the Selberg trace formula with a test function \((h,g)\),
\item the wave trace distribution \(\Tr(\cos(t\sqrt{\Delta-\tfrac14}))\),
\item and a truncated/regularized trace,
\end{itemize}
unless the modification is stated explicitly and all normalization constants are recorded.

\medskip
\noindent\textbf{Fourier conventions (LOCK-F).}
Fix an even cutoff \(\widehat{\chi}\in C_c^\infty(\mathbb{R})\) with \(\supp\widehat{\chi}\subset(-1,1)\) and \(\widehat{\chi}(0)>0\), and define
\begin{equation}\label{eq:7.chi.def}
  \chi(u)=\frac{1}{2\pi}\int_{\mathbb{R}}\widehat{\chi}(t)\,e^{itu}\,dt.
\end{equation}
Let \(\lambda\to\infty\) and \(\eta=\eta(\lambda)\in(0,1)\), and define the even spectral window
\begin{equation}\label{eq:7.h.def}
  h_{\lambda,\eta}(r)=\chi\!\left(\frac{r-\lambda}{\eta}\right)+\chi\!\left(\frac{r+\lambda}{\eta}\right).
\end{equation}

\medskip
\noindent\textbf{Selberg transform convention (LOCK-\(g\)).}
We adopt the Selberg transform convention consistent with Chapter~6:
\begin{equation}\label{eq:7.g.def}
  g_{\lambda,\eta}(u)=\frac{1}{2\pi}\int_{\mathbb{R}} h_{\lambda,\eta}(r)\,e^{iru}\,dr.
\end{equation}
A change of variables using \eqref{eq:7.chi.def} yields the explicit formula
\begin{equation}\label{eq:7.g.explicit}
  g_{\lambda,\eta}(u)=2\eta\,\widehat{\chi}(\eta u)\,\cos(\lambda u).
\end{equation}
In particular,
\begin{equation}\label{eq:7.g.support}
  \supp g_{\lambda,\eta}\subset[-\eta^{-1},\eta^{-1}].
\end{equation}

\medskip
\noindent\textbf{Localization regime (LOCK-\(\eta\)).}
We impose
\begin{equation}\label{eq:7.eta.regime}
  \eta\ge \lambda^{-\theta},\qquad 0<\theta<\theta_0,
\end{equation}
where \(\theta_0\) will be fixed later so that all remainders (hyperbolic and cusp terms) are power-saving in \(\lambda\).

\medskip
\noindent\textbf{Functional calculus representation (LOCK-P).}
Define the localized projector by
\begin{equation}\label{eq:7.P.fc}
  P_{\lambda,\eta}=h_{\lambda,\eta}\!\left(\sqrt{\Delta-\tfrac14}\right).
\end{equation}
Equivalently (wave representation from Chapter~5),
\begin{equation}\label{eq:7.P.wave}
  P_{\lambda,\eta}
  =\frac{1}{2\pi}\int_{\mathbb{R}} e^{-it\lambda}\,\widehat{\chi}(\eta t)\,
  \cos\!\bigl(t\sqrt{\Delta-\tfrac14}\bigr)\,dt.
\end{equation}

\medskip
\noindent\textbf{Normalization lock \(\widehat{\chi}(0)\) (LOCK-\(\chi_0\)).}
By \eqref{eq:7.chi.def}, \(\int_{\mathbb{R}}\chi(u)\,du=\widehat{\chi}(0)\).
This constant is the unique normalization bridge between:
(i) the microlocal Weyl main term (Chapter~5),
(ii) the identity orbital integral main term (Chapter~6),
and (iii) the localized spectral main term (this chapter).
Any drift in this constant indicates a mismatch in Fourier conventions or trace normalization.

\section{Spectral decomposition of \(L^2(M)\) and scattering data}
% ==========================================================

\noindent\textbf{Discrete spectrum.}
Let \(\{u_j\}\) be an orthonormal basis of \(L^2\)-eigenfunctions of \(\Delta\) with eigenvalues
\begin{equation}\label{eq:7.eigs}
  \Delta u_j=\left(\tfrac14+r_j^2\right)u_j,\qquad r_j\in \mathbb{R}\cup i(0,\tfrac12),
\end{equation}
listed with multiplicity.

\medskip
\noindent\textbf{Continuous spectrum (Eisenstein series).}
Let \(\kappa\) denote the number of cusps. For each cusp \(\mathfrak a\), let \(E_{\mathfrak a}(z,s)\) be the Eisenstein series,
normalized so that the vector \(E(z,s)=(E_{\mathfrak a}(z,s))_{\mathfrak a}\) satisfies the functional equation
\begin{equation}\label{eq:7.eis.func.eq}
  E(z,s)=\Phi(s)\,E(z,1-s),
\end{equation}
where \(\Phi(s)\) is the \(\kappa\times\kappa\) scattering matrix.
Define the scattering determinant
\begin{equation}\label{eq:7.phi.det}
  \varphi(s):=\det \Phi(s).
\end{equation}
We record the standard identities \(\Phi(\overline{s})^\ast=\Phi(s)\) and \(\Phi(s)\Phi(1-s)=I\), hence
\begin{equation}\label{eq:7.varphi.sym}
  \varphi(s)\varphi(1-s)=1,\qquad \overline{\varphi(\tfrac12+ir)}=\varphi(\tfrac12-ir)^{-1}.
\end{equation}

\medskip
\noindent\textbf{Spectral resolution (Plancherel).}
The cofinite spectral theorem yields, for \(f\in L^2(M)\),
\begin{equation}\label{eq:7.plancherel}
  f=\sum_j \langle f,u_j\rangle u_j
  +\frac{1}{4\pi}\sum_{\mathfrak a=1}^\kappa \int_{\mathbb{R}}
    \langle f,E_{\mathfrak a}(\cdot,\tfrac12+ir)\rangle E_{\mathfrak a}(\cdot,\tfrac12+ir)\,dr,
\end{equation}
with convergence in \(L^2\); see Hejhal~\cite{Hejhal1983} or Iwaniec~\cite{Iwaniec2002}.

\section{Action of \(P_{\lambda,\eta}\) on the spectrum}
% ==========================================================

\begin{lemma}[Diagonal action]\label{lem:7.diagonal.action}
The operator \(P_{\lambda,\eta}\) acts by scalar multiplication on:
\begin{itemize}
\item the discrete eigenspaces: \(P_{\lambda,\eta}u_j=h_{\lambda,\eta}(r_j)u_j\);
\item the Eisenstein continuum: \(P_{\lambda,\eta}E_{\mathfrak a}(\cdot,\tfrac12+ir)=h_{\lambda,\eta}(r)\,E_{\mathfrak a}(\cdot,\tfrac12+ir)\).
\end{itemize}
\end{lemma}

\begin{proof}
Immediate from functional calculus \eqref{eq:7.P.fc} and the spectral parametrization \eqref{eq:7.eigs}.
\end{proof}

\begin{proposition}[Discrete spectral contribution]\label{prop:7.discrete}
The discrete contribution to the trace equals
\begin{equation}\label{eq:7.S.disc}
  \mathcal{S}_{\lambda,\eta}^{\mathrm{disc}}=\sum_{j} h_{\lambda,\eta}(r_j).
\end{equation}
\end{proposition}

\begin{proof}
By Lemma~\ref{lem:7.diagonal.action}, the trace over the discrete spectral subspace is the sum of the eigenvalues \(h_{\lambda,\eta}(r_j)\).
\end{proof}

\section{Regularized trace and the continuous spectral contribution}
% ==========================================================

\noindent
Because \(M\) is non-compact, the trace of integral kernels requires truncation/regularization.
We adopt a standard cusp truncation \(M(Y)\) at height \(Y\) and the associated regularized trace \(\Tr^{\mathrm{reg}}\).
Under this convention the continuous spectrum contributes through the Maass--Selberg relations and the scattering determinant \(\varphi\).

\medskip
\noindent\textbf{Maass--Selberg input (structural form).}
For each cusp \(\mathfrak a\) and \(r\in\mathbb{R}\), truncation yields an identity of the form
\begin{equation}\label{eq:7.maass.selberg.struct}
  \int_{M(Y)} \big|E_{\mathfrak a}(z,\tfrac12+ir)\big|^2\,d\mu(z)
  = 2\log Y \;+\; \frac{\varphi'(\tfrac12+ir)}{\varphi(\tfrac12+ir)}
  \;+\; \mathcal{U}_\Gamma(r)\;+\;O(Y^{-1}),
\end{equation}
where \(\mathcal{U}_\Gamma(r)\) is an explicit universal term determined by the chosen Eisenstein normalization and scaling matrices at cusps.
Precise closed forms are given in Hejhal~\cite[Vol.~I, Ch.~2]{Hejhal1983}.

\medskip
\begin{definition}[Continuous spectral contribution]\label{def:7.S.cont}
Define the continuous contribution to the regularized trace by
\begin{equation}\label{eq:7.S.cont}
  \mathcal{S}^{\mathrm{cont}}_{\lambda,\eta}
  := \frac{1}{4\pi}\int_{\mathbb{R}} h_{\lambda,\eta}(r)\,
      \Bigl(-\frac{\varphi'(\tfrac12+ir)}{\varphi(\tfrac12+ir)}\Bigr)\,dr
      \;+\; \mathcal{N}_\Gamma[h_{\lambda,\eta}],
\end{equation}
where \(\mathcal{N}_\Gamma[h]\) is the explicit normalization functional induced by the truncation convention (it removes the \(2\log Y\) divergence
and incorporates \(\mathcal{U}_\Gamma(r)\) and cusp constants).
\end{definition}

\begin{remark}[Normalization functional \(\mathcal{N}_\Gamma[h]\)]\label{rem:7.NGamma}
In standard conventions (e.g. Hejhal), \(\mathcal{N}_\Gamma[h]\) is an explicitly computable linear functional of \(h\) (equivalently of \(g\)),
built from:
\begin{itemize}
\item the value \(g(0)=\frac{1}{2\pi}\int_{\mathbb{R}} h(r)\,dr\),
\item integrals of \(h(r)\) against Gamma/digamma-factor terms,
\item cusp-dependent constants determined by the chosen scaling matrices.
\end{itemize}
We isolate \(\mathcal{N}_\Gamma\) so that the compared object remains fixed while all convention-dependence is audited explicitly.
\end{remark}

\medskip
\begin{definition}[Residual contribution]\label{def:7.S.res}
Let \(\mathcal{S}^{\mathrm{res}}_{\lambda,\eta}\) denote the residual part coming from poles of Eisenstein series (equivalently poles of \(\Phi(s)\))
in \(s\in(1/2,1]\), when present. In many standard cases \(\mathcal{S}^{\mathrm{res}}_{\lambda,\eta}\) is absent; otherwise it is a finite, explicit sum.
\end{definition}

\medskip
\begin{definition}[Full spectral side]\label{def:7.S.full}
Define the full spectral side by
\begin{equation}\label{eq:7.S.full}
  \mathcal{S}_{\lambda,\eta}
  := \mathcal{S}^{\mathrm{disc}}_{\lambda,\eta}
   + \mathcal{S}^{\mathrm{cont}}_{\lambda,\eta}
   + \mathcal{S}^{\mathrm{res}}_{\lambda,\eta}.
\end{equation}
\end{definition}

% ==========================================================
\section{Exact comparison identity with the geometric side}
% ==========================================================

\noindent
Let \(\mathcal{G}_{\lambda,\eta}\) be the geometric side defined in Chapter~6:
\begin{equation}\label{eq:7.G.recall}
  \mathcal{G}_{\lambda,\eta}
  = I_{\lambda,\eta}+H_{\lambda,\eta}+P^{\mathrm{para}}_{\lambda,\eta}+E^{\mathrm{ell}}_{\lambda,\eta},
\end{equation}
with \(I_{\lambda,\eta}\) the identity term, \(H_{\lambda,\eta}\) the hyperbolic term, \(P^{\mathrm{para}}_{\lambda,\eta}\) the parabolic (cusp) term,
and \(E^{\mathrm{ell}}_{\lambda,\eta}\) the elliptic term (if \(\Gamma\) has torsion).

\begin{theorem}[Localized Selberg trace formula: exact identity]\label{thm:7.selberg.exact}
For every cofinite \(\Gamma\) and every \((\lambda,\eta)\), one has the exact identity
\begin{equation}\label{eq:7.selberg.exact}
  \mathcal{S}_{\lambda,\eta}=\mathcal{G}_{\lambda,\eta},
\end{equation}
where \(\mathcal{S}_{\lambda,\eta}\) is defined by \eqref{eq:7.S.full} and \(\mathcal{G}_{\lambda,\eta}\) by \eqref{eq:7.G.recall},
with both sides computed under the same cusp normalization convention.
\end{theorem}

\begin{proof}
This is the Selberg trace formula applied to the test function \(h_{\lambda,\eta}\), together with the standard cofinite regularization.
See Selberg~\cite{Selberg1956}, Hejhal~\cite{Hejhal1983}, Iwaniec~\cite{Iwaniec2002}.
\end{proof}

\begin{remark}[Normalization lock]\label{rem:7.lock.no.mismatch}
The identity \eqref{eq:7.selberg.exact} holds only if the cusp truncation/normalization conventions match on both sides.
This is why cusp-dependent linear functionals are isolated explicitly (Chapter~6 for \(\mathcal{C}_\Gamma\), here for \(\mathcal{N}_\Gamma\)):
they must be checked term-by-term under the chosen convention.
\end{remark}

\section{Localized Weyl main term and remainder hierarchy}
% ==========================================================

\noindent
We extract the main term using the geometric-side Weyl extraction from Chapter~6 and the exact identity
\(\mathcal{S}_{\lambda,\eta}=\mathcal{G}_{\lambda,\eta}\).

\begin{theorem}[Localized Weyl main term (locked)]\label{thm:7.weyl.lock}
Assume \eqref{eq:7.eta.regime}. Then, as \(\lambda\to\infty\),
\begin{equation}\label{eq:7.weyl.main}
  \mathcal{S}_{\lambda,\eta}
  = \frac{\vol(M)}{2\pi}\,\widehat{\chi}(0)\,\lambda\eta
    \;+\; H_{\lambda,\eta}
    \;+\; \mathcal{R}^{\mathrm{cusp}}_{\lambda,\eta}
    \;+\; \mathcal{R}^{\mathrm{ell}}_{\lambda,\eta}
    \;+\; O_{\widehat{\chi}}(\eta^2)
    \;+\; O\!\big(e^{-2\pi\lambda}\lambda\eta\big),
\end{equation}
where \(\mathcal{R}^{\mathrm{cusp}}_{\lambda,\eta}\) collects cusp-dependent lower-order terms (parabolic and normalization),
and \(\mathcal{R}^{\mathrm{ell}}_{\lambda,\eta}\) collects elliptic terms when present.
\end{theorem}

\begin{proof}
By Theorem~\ref{thm:7.selberg.exact}, \(\mathcal{S}_{\lambda,\eta}=\mathcal{G}_{\lambda,\eta}\).
Insert the geometric-side extraction from Chapter~6 (Lemma~6.1.2 and Proposition~6.6.1), and absorb the remaining cusp/elliptic terms
into \(\mathcal{R}^{\mathrm{cusp}}_{\lambda,\eta}\) and \(\mathcal{R}^{\mathrm{ell}}_{\lambda,\eta}\).
\end{proof}

\begin{remark}[Windowed counting]\label{rem:7.window.count}
The discrete sum \(\sum_j h_{\lambda,\eta}(r_j)\) is a smoothed local counting function at resolution \(\eta\).
The identity \eqref{eq:7.weyl.main} shows that the principal mass at scale \(\eta\) is Weyl, while oscillations are governed by
closed geodesics (hyperbolic term) and cusp scattering (continuous spectrum).
\end{remark}

\section{Growth control for \(\varphi'/\varphi\) and admissible \(\theta_0\)}
% ==========================================================

\begin{lemma}[Polynomial growth of \(\varphi'/\varphi\)]\label{lem:7.phi.growth}
There exists \(C_\Gamma>0\) such that
\[
  \frac{\varphi'(\tfrac12+ir)}{\varphi(\tfrac12+ir)} \ll_\Gamma (1+|r|)^{C_\Gamma},
  \qquad r\in\mathbb{R}.
\]
\end{lemma}

\begin{proof}
Standard bounds on scattering determinants and Eisenstein series for cofinite groups; see Hejhal~\cite{Hejhal1983} and Iwaniec~\cite{Iwaniec2002}.
\end{proof}

\begin{proposition}[Cusp contribution under localization]\label{prop:7.cusp.size}
Uniformly for \(\eta\) satisfying \eqref{eq:7.eta.regime},
\[
  \mathcal{S}^{\mathrm{cont}}_{\lambda,\eta}
  \ll_{\Gamma,\widehat{\chi}}
  \eta\,(1+\lambda)^{C_\Gamma} + \big|\mathcal{N}_\Gamma[h_{\lambda,\eta}]\big|.
\]
In particular, if \(\theta<1/C_\Gamma\), then \(\mathcal{S}^{\mathrm{cont}}_{\lambda,\eta}=o(\lambda\eta)\) as \(\lambda\to\infty\).
\end{proposition}

\begin{proof}
The window \(h_{\lambda,\eta}\) is supported where \(r=\pm\lambda+O(\eta)\) and \(\int_{\mathbb{R}}|h_{\lambda,\eta}(r)|\,dr\asymp\eta\).
Combine with Lemma~\ref{lem:7.phi.growth}. The functional \(\mathcal{N}_\Gamma[h]\) is explicit and of comparable size under the same localization bounds.
\end{proof}

\section{Selberg zeta as the intrinsic geodesic--spectrum portal}
% ==========================================================

\noindent
The intrinsic hyperbolic object tying together the length spectrum and the Laplace spectrum is the Selberg zeta function \(Z_\Gamma(s)\).
We record the classical log-derivative relation.

\begin{proposition}[Log-derivative of Selberg zeta]\label{prop:7.selberg.zeta}
Let \(Z_\Gamma(s)\) be the Selberg zeta function associated to \(\Gamma\). For \(\Re s>1\),
\[
  \frac{Z_\Gamma'(s)}{Z_\Gamma(s)}
  = \sum_{[\gamma_0]\in\mathcal{P}}\sum_{k\ge1}
    \frac{L(\gamma_0)\,e^{-ksL(\gamma_0)}}{2\sinh\!\big(\tfrac{kL(\gamma_0)}{2}\big)}.
\]
Moreover \(Z_\Gamma(s)\) admits meromorphic continuation, and its zeros encode the Laplace spectrum and resonances.
\end{proposition}

\begin{proof}
Expand \(\log Z_\Gamma(s)\) as an Euler product over primitive geodesics and differentiate termwise; see Hejhal~\cite{Hejhal1983}.
\end{proof}

\begin{remark}[Meaning for the localized trace formula]\label{rem:7.zeta.selberg.meaning}
The hyperbolic term \(H_{\lambda,\eta}\) from Chapter~6 is a windowed prime-geodesic sum, with window \(g_{\lambda,\eta}(kL)\)
enforcing the length cutoff \(kL\le \eta^{-1}\). This is the hyperbolic analogue of the explicit formula linking primes to zeros in the Riemann setting.
\end{remark}

\section{Where the Riemann zeta function enters (arithmetic \(\Gamma\))}
% ==========================================================

\noindent
For a general cofinite \(\Gamma\), the continuous spectrum is governed by \(\varphi(s)\).
In arithmetic cases, \(\varphi(s)\) factors into completed \(L\)-functions, and \(\zeta\) appears explicitly.

\begin{example}[Modular surface]\label{ex:7.modular}
For \(\Gamma=\mathrm{PSL}_2(\mathbb{Z})\), one has
\[
  \varphi(s)=\frac{\xi(2s-1)}{\xi(2s)},\qquad
  \xi(s)=\pi^{-s/2}\Gamma\!\Big(\frac{s}{2}\Big)\zeta(s).
\]
Hence
\[
  -\frac{\varphi'(s)}{\varphi(s)}
  = -2\frac{\xi'(2s-1)}{\xi(2s-1)}+2\frac{\xi'(2s)}{\xi(2s)}.
\]
Substituting \(s=\tfrac12+ir\) into \eqref{eq:7.S.cont} shows that the continuous spectral term contains logarithmic derivatives of \(\zeta\),
together with explicit Gamma-factor derivatives.
\end{example}

\begin{remark}[Interpretation]\label{rem:7.zeta.forced}
In the non-compact case, the trace splits into discrete and continuous spectrum.
The continuous spectrum is governed by scattering, and scattering is governed by \(\varphi'/\varphi\).
In arithmetic cases, \(\varphi(s)\) is an explicit ratio of completed \(L\)-functions; thus \(\zeta\) enters as a structural component of the trace identity.
\end{remark}

\section*{Chapter 7 Audit (Diamond / Control)}

\noindent\textbf{Locks (must hold exactly).}
\begin{itemize}
\item[(L7.1)] Fourier lock: \(\chi(u)=\frac{1}{2\pi}\int\widehat{\chi}(t)e^{itu}dt\).
\item[(L7.2)] Window lock: \(h_{\lambda,\eta}(r)=\chi(\frac{r-\lambda}{\eta})+\chi(\frac{r+\lambda}{\eta})\).
\item[(L7.3)] Transform lock: \(g_{\lambda,\eta}(u)=\frac{1}{2\pi}\int h_{\lambda,\eta}(r)e^{iru}dr=2\eta\widehat{\chi}(\eta u)\cos(\lambda u)\).
\item[(L7.4)] Main-term lock: coefficient \(\frac{\vol(M)}{2\pi}\widehat{\chi}(0)\lambda\eta\) matches Chapter~6.
\item[(L7.5)] Cusp lock: the same truncation convention must be used on both sides; \(\mathcal{N}_\Gamma\) and cusp constants must match.
\item[(L7.6)] Object lock: Selberg trace (test function) is not the wave trace distribution.
\end{itemize}

\medskip
\noindent\textbf{Failure modes.}
\begin{itemize}
\item[(F7.1)] \(g\)-normalization mismatch \(\Rightarrow\) wrong length cutoff and wrong hyperbolic weights.
\item[(F7.2)] Missing \(\mathcal{N}_\Gamma[h]\) \(\Rightarrow\) cusp divergence not cancelled (regularization drift).
\item[(F7.3)] Mixed cusp conventions \(\Rightarrow\) constant drift in the continuous term.
\item[(F7.4)] Treating wave-trace singularities as Selberg orbital weights \(\Rightarrow\) object substitution.
\end{itemize}

\chapter{Local Weyl Law from the Localized Trace Formula}

\section{Smoothed local counting and the localized projector}

Let \(P_{\lambda,\eta}=h_{\lambda,\eta}(\sqrt{\Delta-\tfrac14})\) be the localized projector \eqref{eq:7.P.fc}.
Define the smoothed local spectral counting functional
\begin{equation}\label{eq:8.Nsmooth.def}
\mathcal N(\lambda,\eta)
:= \sum_{j} h_{\lambda,\eta}(r_j)
\;+\; \frac{1}{4\pi}\int_{\mathbb{R}} h_{\lambda,\eta}(r)\,
\Bigl(-\frac{\varphi'(\tfrac12+ir)}{\varphi(\tfrac12+ir)}\Bigr)\,dr
\;+\; \mathcal{N}_\Gamma[h_{\lambda,\eta}]
\;+\; \mathcal{S}^{\mathrm{res}}_{\lambda,\eta},
\end{equation}
so that \(\mathcal N(\lambda,\eta)=\mathcal{S}_{\lambda,\eta}\) by Definition~\ref{def:7.S.full}.

\begin{lemma}[Trace equals smoothed counting]\label{lem:8.trace_equals_counting}
With the fixed normalization convention,
\[
\mathcal N(\lambda,\eta)=\Tr^{\mathrm{reg}}(P_{\lambda,\eta})=\mathcal{S}_{\lambda,\eta}=\mathcal{G}_{\lambda,\eta}.
\]
\end{lemma}

\begin{proof}
This is Theorem~\ref{thm:7.selberg.exact}, together with the definition of the regularized trace for cofinite \(\Gamma\).
\end{proof}

\section{Main term extraction}

\begin{proposition}[Weyl main term for the smoothed count]\label{prop:8.main_term}
Assume \eqref{eq:7.eta.regime}. Then, as \(\lambda\to\infty\),
\[
\mathcal N(\lambda,\eta)
= \frac{\vol(M)}{2\pi}\widehat{\chi}(0)\lambda\eta
\;+\;\Err(\lambda,\eta),
\]
where \(\Err(\lambda,\eta)\) collects \(H_{\lambda,\eta}\), cusp/elliptic lower-order terms, residual contributions, and the explicit
\(O_{\widehat{\chi}}(\eta^2)\) and \(O(e^{-2\pi\lambda}\lambda\eta)\) errors in \eqref{eq:7.weyl.main}.
\end{proposition}

\begin{proof}
Immediate from Theorem~\ref{thm:7.weyl.lock}, since \(\mathcal N(\lambda,\eta)=\mathcal{S}_{\lambda,\eta}\).
\end{proof}

\section{From smoothed to sharp local counting: a Tauberian transfer}

Define the sharp local counting function in the \(r\)-parameter by
\begin{equation}\label{eq:8.Nsharp.def}
N(\lambda,\eta):=\#\{j:\ |r_j-\lambda|\le \eta\}.
\end{equation}
Under mild positivity assumptions on the fixed bump \(\chi\) (which can be arranged by choosing \(\widehat{\chi}\ge0\)),
the smoothed sum \(\sum_j h_{\lambda,\eta}(r_j)\) controls \(N(\lambda,\eta)\) up to absolute constants depending only on the fixed window profile.

\begin{lemma}[Local Tauberian comparison]\label{lem:8.tauberian}
Assume \(\widehat{\chi}\ge0\), \(\widehat{\chi}\not\equiv 0\), and \(\supp\widehat{\chi}\subset(-1,1)\).
Then there exist constants \(c_1,c_2,C>0\) depending only on \(\widehat{\chi}\) such that for all \(\lambda\) and \(\eta\in(0,1)\),
\[
c_1\,N(\lambda,\eta)\ \le\ \sum_j h_{\lambda,\eta}(r_j)\ \le\ c_2\,N(\lambda,C\eta).
\]
\end{lemma}

\begin{proof}
The function \(h_{\lambda,\eta}\) is a fixed-profile bump of width \(\asymp\eta\) centered at \(\pm\lambda\).
Under \(\widehat{\chi}\ge0\), \(\chi\) is positive-definite and one may choose the profile so that \(h_{\lambda,\eta}\) dominates
a characteristic function on \([\lambda-\eta,\lambda+\eta]\) up to constants and is supported in \([\lambda-C\eta,\lambda+C\eta]\).
This yields the stated two-sided comparison.
\end{proof}

\section{A quantitative local Weyl statement (conditional remainder input)}

The strength of a \emph{quantitative} local Weyl law depends on the strongest remainder control available for \(\Err(\lambda,\eta)\),
which in our framework comes from combining:
(i) the microlocal parametrix error from Chapter~5,
(ii) hyperbolic cutoff and amplitude control from Chapter~6,
(iii) cusp normalization control (explicit \(\mathcal{N}_\Gamma\)).

\begin{theorem}[Quantitative local Weyl law: transfer form]\label{thm:8.local_weyl.transfer}
Assume that for some \(\delta>0\) one has the uniform estimate
\[
\mathcal N(\lambda,\eta)=\frac{\vol(M)}{2\pi}\widehat{\chi}(0)\lambda\eta + O(\lambda^{1-\delta})
\]
uniformly for \(\eta\in[\lambda^{-\theta},1]\). Then
\[
N(\lambda,\eta)=\frac{\vol(M)}{2\pi}\lambda\eta + O(\lambda^{1-\delta})
\]
uniformly for the same \(\eta\)-range, with implied constants depending only on \(M\) and the fixed window profile.
\end{theorem}

\begin{proof}
