\chapter{Spectral Expansion, Normalization, and Comparison for the Localized Trace Formula}

\noindent\textbf{Scope and purpose.}
Let $M=\Gamma\backslash\mathbb{H}$ be a cofinite hyperbolic surface (finite area, possibly with cusps and torsion),
$\Delta$ the \emph{positive} Laplace--Beltrami operator, and
\[
  P_{\lambda,\eta} \;=\; h_{\lambda,\eta}\!\left(\sqrt{\Delta-\tfrac14}\right)
\]
the localized spectral projector introduced in Chapters~5--6.
This chapter fixes, once and for all, the \emph{spectral-side object} and all \emph{normalization locks}
(Fourier conventions, truncation/regularization, Eisenstein/scattering normalization),
and then states the \emph{exact} localized Selberg identity
\[
  \Tr^{\mathrm{reg}}(P_{\lambda,\eta}) \;=\; \mathcal{G}_{\lambda,\eta},
\]
with $\mathcal{G}_{\lambda,\eta}$ the geometric side from Chapter~6.
We also isolate the precise analytic portal through which completed $L$-functions (and in arithmetic cases the Riemann zeta function)
enter the trace identity via the scattering determinant $\varphi(s)=\det\Phi(s)$.

\medskip
\noindent\textbf{Primary references.}
We use the standard cofinite Selberg normalization conventions as in Selberg~\cite{Selberg1956},
Hejhal~\cite[Vol.~I, Ch.~2]{Hejhal1983}, and Iwaniec~\cite[Ch.~6]{Iwaniec2002}.
All convention-dependent constants are recorded explicitly as linear functionals.

\section{OBJ-LOCK and normalization locks (the ``no substitution'' layer)}
% =====================================================================

\noindent\textbf{OBJ-LOCK (single global object).}
Throughout Chapters~6--7 the \emph{single} object is the same regularized trace
\begin{equation}\label{eq:7.objlock}
  \Tr^{\mathrm{reg}}(P_{\lambda,\eta})
  \qquad
  \Bigl(\text{and, when microlocal cutoffs are inserted, } \Tr^{\mathrm{reg}}(\Psi P_{\lambda,\eta}\Psi)\Bigr).
\end{equation}
No substitution is permitted between:
(i) the Selberg test-function trace formula,
(ii) the wave-trace distribution $\Tr(\cos(t\sqrt{\Delta-\tfrac14}))$,
and (iii) any truncated trace,
unless the change is stated explicitly \emph{and} the corresponding regularization ledger is recorded.

\medskip
\noindent\textbf{LOCK-F (Fourier convention).}
Fix an even $\widehat{\chi}\in C_c^\infty(\mathbb{R})$ with $\supp\widehat{\chi}\subset(-1,1)$ and $\widehat{\chi}(0)>0$.
Define the inverse transform
\begin{equation}\label{eq:7.lockF}
  \chi(u)=\frac{1}{2\pi}\int_{\mathbb{R}}\widehat{\chi}(t)\,e^{itu}\,dt.
\end{equation}
Then $\int_{\mathbb{R}}\chi(u)\,du=\widehat{\chi}(0)$, and this constant is the \emph{unique} bridge between the Weyl term on both sides.

\medskip
\noindent\textbf{LOCK-$h$ (window profile).}
Let $\lambda\to\infty$ and $\eta=\eta(\lambda)\in(0,1)$.
Define the even spectral window
\begin{equation}\label{eq:7.lockh}
  h_{\lambda,\eta}(r)
  \;=\;
  \chi\!\left(\frac{r-\lambda}{\eta}\right)+\chi\!\left(\frac{r+\lambda}{\eta}\right).
\end{equation}

\medskip
\noindent\textbf{LOCK-$g$ (Selberg/Harish--Chandra transform).}
Define the Fourier transform (the $g$-function used on the geometric side) by
\begin{equation}\label{eq:7.lockg.def}
  g_{\lambda,\eta}(u)
  \;:=\;
  \frac{1}{2\pi}\int_{\mathbb{R}} h_{\lambda,\eta}(r)\,e^{-iru}\,dr.
\end{equation}
Then, by a direct substitution using \eqref{eq:7.lockF}--\eqref{eq:7.lockh},
\begin{equation}\label{eq:7.lockg.closed}
  g_{\lambda,\eta}(u)
  \;=\;
  \frac{\eta}{\pi}\,\widehat{\chi}(\eta u)\,\cos(\lambda u),
\end{equation}
and hence
\begin{equation}\label{eq:7.lockg.support}
  \supp g_{\lambda,\eta}\subset[-\eta^{-1},\eta^{-1}].
\end{equation}
This is the \emph{length cutoff lock}: every hyperbolic/geodesic contribution is automatically truncated to $kL\le \eta^{-1}$.

\medskip
\noindent\textbf{LOCK-$\eta$ (localization regime).}
We work under
\begin{equation}\label{eq:7.locketa}
  \eta \;\ge\; \lambda^{-\theta},\qquad 0<\theta<\theta_0,
\end{equation}
where $\theta_0$ is chosen so that all remainder terms are power-saving relative to the Weyl mass $\asymp \lambda\eta$.

\medskip
\noindent\textbf{LOCK-P (functional calculus representation).}
Define
\begin{equation}\label{eq:7.lockP}
  P_{\lambda,\eta}
  \;:=\;
  h_{\lambda,\eta}\!\left(\sqrt{\Delta-\tfrac14}\right).
\end{equation}
Equivalently (Chapter~5 wave representation),
\begin{equation}\label{eq:7.lockP.wave}
  P_{\lambda,\eta}
  =
  \frac{1}{2\pi}\int_{\mathbb{R}} e^{-it\lambda}\widehat{\chi}(\eta t)\,
  \cos\!\bigl(t\sqrt{\Delta-\tfrac14}\bigr)\,dt.
\end{equation}

\medskip
\noindent\textbf{LOCK-$\chi_0$ (main-term bridge).}
By \eqref{eq:7.lockF},
\begin{equation}\label{eq:7.lockchi0}
  \int_{\mathbb{R}}\chi(u)\,du = \widehat{\chi}(0).
\end{equation}
This constant is the \emph{only} normalization bridge between:
(i) the microlocal Weyl law of Chapter~5,
(ii) the identity orbital integral in Chapter~6,
and (iii) the spectral-side Weyl main term in this chapter.
Any mismatch of $\widehat{\chi}(0)$ is a genuine convention error.

\section{Cofinite spectral decomposition and scattering data}
% =====================================================================

\noindent\textbf{Discrete spectrum.}
Let $\{u_j\}_{j\ge1}$ be an orthonormal basis of $L^2$-eigenfunctions of $\Delta$:
\begin{equation}\label{eq:7.disc.eigs}
  \Delta u_j=\left(\tfrac14+r_j^2\right)u_j,\qquad r_j\in \mathbb{R}\cup i\,(0,\tfrac12),
\end{equation}
listed with multiplicity.

\medskip
\noindent\textbf{Cusps, Eisenstein series, scattering matrix.}
Let $\kappa$ be the number of cusps of $\Gamma$.
Choose scaling matrices $\sigma_{\mathfrak a}\in\mathrm{PSL}_2(\mathbb{R})$ for each cusp $\mathfrak a$
so that $\sigma_{\mathfrak a}\infty=\mathfrak a$ and
$\sigma_{\mathfrak a}^{-1}\Gamma_{\mathfrak a}\sigma_{\mathfrak a}=\{\pm\begin{psmallmatrix}1&n\\0&1\end{psmallmatrix}:n\in\mathbb{Z}\}$.
For $s\in\mathbb{C}$, let $E_{\mathfrak a}(z,s)$ be the Eisenstein series normalized by the standard constant term expansion:
for $z=\sigma_{\mathfrak b}(x+iy)$ in the $\mathfrak b$-cusp,
\begin{equation}\label{eq:7.eis.const.term}
  E_{\mathfrak a}(\sigma_{\mathfrak b}(x+iy),s)
  =
  \delta_{\mathfrak a\mathfrak b}\,y^s
  +
  \phi_{\mathfrak a\mathfrak b}(s)\,y^{1-s}
  + (\text{non-constant Fourier modes}).
\end{equation}
Let $\Phi(s)=(\phi_{\mathfrak a\mathfrak b}(s))_{\mathfrak a,\mathfrak b}$ be the $\kappa\times\kappa$ scattering matrix and
\begin{equation}\label{eq:7.scatt.det}
  \varphi(s):=\det\Phi(s)
\end{equation}
the scattering determinant.
Then $\Phi(s)$ satisfies the functional equation and unitarity on $\Re s=\tfrac12$:
\begin{equation}\label{eq:7.scatt.func}
  \Phi(s)\Phi(1-s)=I_{\kappa},
  \qquad
  \Phi(\overline{s})^\ast=\Phi(s),
\end{equation}
hence
\begin{equation}\label{eq:7.varphi.sym}
  \varphi(s)\varphi(1-s)=1,
  \qquad
  \overline{\varphi(\tfrac12+ir)}=\varphi(\tfrac12+ir)^{-1}.
\end{equation}
See Hejhal~\cite[Vol.~I, \S2.3--\S2.4]{Hejhal1983} and Iwaniec~\cite[\S6.3]{Iwaniec2002}.

\medskip
\noindent\textbf{Plancherel resolution.}
For $f\in L^2(M)$,
\begin{equation}\label{eq:7.plancherel}
  f
  =
  \sum_{j}\langle f,u_j\rangle u_j
  +\frac{1}{4\pi}\sum_{\mathfrak a=1}^{\kappa}\int_{\mathbb{R}}
    \langle f,E_{\mathfrak a}(\cdot,\tfrac12+ir)\rangle\,
    E_{\mathfrak a}(\cdot,\tfrac12+ir)\,dr,
\end{equation}
with convergence in $L^2$.
References: Hejhal~\cite[Vol.~I, Thm.\ 2.12]{Hejhal1983}, Iwaniec~\cite[Thm.\ 6.6]{Iwaniec2002}.

\section{Diagonal action of the localized projector}
% =====================================================================

\begin{lemma}[Diagonal action]\label{lem:7.diagonal}
For every $(\lambda,\eta)$,
\begin{align}
  P_{\lambda,\eta}u_j &= h_{\lambda,\eta}(r_j)\,u_j,\\
  P_{\lambda,\eta}E_{\mathfrak a}(\cdot,\tfrac12+ir) &= h_{\lambda,\eta}(r)\,E_{\mathfrak a}(\cdot,\tfrac12+ir).
\end{align}
\end{lemma}

\begin{proof}
This is immediate from functional calculus \eqref{eq:7.lockP} and the spectral parameterization \eqref{eq:7.disc.eigs}.
\end{proof}

\section{Truncation, regularized trace, and the continuous spectral measure}
% =====================================================================

\noindent
Because $M$ has cusps, the raw trace of the integral kernel is not absolutely integrable along the cusps.
We therefore fix a truncation operator and define the \emph{regularized trace} as a limit
in which the logarithmic divergence is subtracted explicitly.
This fixes the object \eqref{eq:7.objlock}.

\subsection*{Truncation operator $\Lambda^Y$ and truncated domain $M^Y$}

For $Y>1$, define the truncated domain $M^Y\subset M$ by cutting each cusp at height $Y$ in the corresponding cusp coordinate:
$z=\sigma_{\mathfrak a}(x+iy)$ with $y\ge Y$ is removed.
Equivalently, define the truncation operator $\Lambda^Y$ on automorphic functions $F$ by
\begin{equation}\label{eq:7.trunc.op}
  (\Lambda^Y F)(z)
  :=
  F(z)
  -
  \sum_{\mathfrak a=1}^{\kappa}
  \mathbf{1}_{\{y_{\mathfrak a}(z)\ge Y\}}
  \cdot
  F_{\mathfrak a}^{(0)}(z),
\end{equation}
where $y_{\mathfrak a}(z)$ is the height in the $\mathfrak a$-cusp coordinate
and $F_{\mathfrak a}^{(0)}$ is the constant term of $F$ in the Fourier expansion at cusp $\mathfrak a$.
Then $\Lambda^Y F$ is square-integrable for Eisenstein series on the unitary line.
See Hejhal~\cite[Vol.~I, \S2.3]{Hejhal1983}.

\subsection*{Maass--Selberg relation (exact matrix form)}

Let $s=\tfrac12+ir$ with $r\in\mathbb{R}$.
Define the Eisenstein vector $E(z,s)=(E_{\mathfrak a}(z,s))_{\mathfrak a=1}^{\kappa}$.
Then the Maass--Selberg relations give, in matrix form,
\begin{equation}\label{eq:7.maass.selberg.matrix}
  \int_{M} \bigl(\Lambda^Y E(z,s)\bigr)\,\bigl(\Lambda^Y E(z,s)\bigr)^\ast\,d\mu(z)
  \;=\;
  2(\log Y)\,I_{\kappa}
  \;-\;
  \Phi'(s)\,\Phi(s)^{-1},
\end{equation}
where $d\mu$ is the hyperbolic area measure.
This formula is \emph{exact} for the truncation operator \eqref{eq:7.trunc.op} and the normalization \eqref{eq:7.eis.const.term}.
Reference: Hejhal~\cite[Vol.~I, Eq.\ (2.26) and Thm.\ 2.9]{Hejhal1983}; see also Iwaniec~\cite[\S6.4]{Iwaniec2002}.

Taking the trace of \eqref{eq:7.maass.selberg.matrix} yields
\begin{equation}\label{eq:7.maass.selberg.trace}
  \sum_{\mathfrak a=1}^{\kappa}\int_{M}\bigl|\Lambda^Y E_{\mathfrak a}(z,\tfrac12+ir)\bigr|^2\,d\mu(z)
  \;=\;
  2\kappa\log Y
  \;-\;
  \frac{\varphi'(s)}{\varphi(s)}\Big|_{s=\tfrac12+ir}.
\end{equation}

\subsection*{Regularized trace of $P_{\lambda,\eta}$}

Let $K_{\lambda,\eta}(z,w)$ be the integral kernel of $P_{\lambda,\eta}$.
For $Y>1$, define the truncated trace integral
\[
  \Tr_Y(P_{\lambda,\eta})
  \;:=\;
  \int_{M^Y} K_{\lambda,\eta}(z,z)\,d\mu(z).
\]
The logarithmic divergence is universal and proportional to $g_{\lambda,\eta}(0)$ and $\kappa\log Y$.
We therefore define the regularized trace by
\begin{equation}\label{eq:7.reg.trace.def}
  \Tr^{\mathrm{reg}}(P_{\lambda,\eta})
  \;:=\;
  \lim_{Y\to\infty}\Bigl(\Tr_Y(P_{\lambda,\eta}) - \kappa\,g_{\lambda,\eta}(0)\,\log Y\Bigr),
\end{equation}
where $g_{\lambda,\eta}(0)=\frac{1}{2\pi}\int_{\mathbb{R}}h_{\lambda,\eta}(r)\,dr$ by \eqref{eq:7.lockg.def}.
This is the \emph{single fixed} regularization convention used in the identity of Theorem~\ref{thm:7.selberg.exact} below.
Reference: Hejhal~\cite[Vol.~I, \S2.5]{Hejhal1983}.

\section{Exact spectral side: discrete + continuous + residual (no placeholders)}
% =====================================================================

\subsection*{Discrete contribution}

\begin{definition}[Discrete spectral side]\label{def:7.Sdisc}
\begin{equation}\label{eq:7.Sdisc}
  \mathcal{S}^{\mathrm{disc}}_{\lambda,\eta}
  \;:=\;
  \sum_{j} h_{\lambda,\eta}(r_j).
\end{equation}
\end{definition}

\subsection*{Continuous contribution}

\begin{definition}[Continuous spectral side]\label{def:7.Scont}
\begin{equation}\label{eq:7.Scont}
  \mathcal{S}^{\mathrm{cont}}_{\lambda,\eta}
  \;:=\;
  \frac{1}{4\pi}\int_{\mathbb{R}} h_{\lambda,\eta}(r)\,
  \Bigl(-\frac{\varphi'(\tfrac12+ir)}{\varphi(\tfrac12+ir)}\Bigr)\,dr.
\end{equation}
\end{definition}

\noindent
This is exactly the trace contribution forced by the Maass--Selberg identity \eqref{eq:7.maass.selberg.trace}
when the regularization is fixed by \eqref{eq:7.reg.trace.def}.
References: Hejhal~\cite[Vol.~I, \S2.5]{Hejhal1983}, Iwaniec~\cite[\S6.5]{Iwaniec2002}.

\subsection*{Residual contribution (poles of Eisenstein series)}

Poles of $\Phi(s)$ (equivalently $\varphi(s)$) on $\Re s\in(\tfrac12,1]$ produce residual $L^2$-eigenfunctions.
Let $\mathcal{P}_\Gamma$ be the multiset of poles $s_\ell\in(\tfrac12,1]$ of $\Phi(s)$, counted with multiplicity.
Each such pole contributes a finite-rank projection in the spectral resolution.
The corresponding trace contribution is a finite sum determined by residues of $\Phi'(s)\Phi(s)^{-1}$.

\begin{definition}[Residual spectral side]\label{def:7.Sres}
Define
\begin{equation}\label{eq:7.Sres}
  \mathcal{S}^{\mathrm{res}}_{\lambda,\eta}
  \;:=\;
  \frac{1}{2}\sum_{s_\ell\in\mathcal{P}_\Gamma}
  \operatorname{tr}\!\Bigl(\Res_{s=s_\ell}\bigl(\Phi'(s)\Phi(s)^{-1}\bigr)\Bigr)\,
  h_{\lambda,\eta}\!\bigl(i(s_\ell-\tfrac12)\bigr).
\end{equation}
\end{definition}

\noindent
This expression is the standard residue term in the cofinite Selberg trace formula.
Reference: Hejhal~\cite[Vol.~I, \S2.6]{Hejhal1983}; see also Iwaniec~\cite[\S6.7]{Iwaniec2002}.

\subsection*{Full spectral side}

\begin{definition}[Full spectral side]\label{def:7.Sfull}
\begin{equation}\label{eq:7.Sfull}
  \mathcal{S}_{\lambda,\eta}
  \;:=\;
  \mathcal{S}^{\mathrm{disc}}_{\lambda,\eta}
  +\mathcal{S}^{\mathrm{cont}}_{\lambda,\eta}
  +\mathcal{S}^{\mathrm{res}}_{\lambda,\eta}.
\end{equation}
\end{definition}

\section{Geometric side recall and exact equality (comparison identity)}
% =====================================================================

\noindent
Let the geometric side $\mathcal{G}_{\lambda,\eta}$ be the localized geometric decomposition defined in Chapter~6:
\begin{equation}\label{eq:7.G.recall}
  \mathcal{G}_{\lambda,\eta}
  =
  I_{\lambda,\eta}
  +H_{\lambda,\eta}
  +P^{\mathrm{para}}_{\lambda,\eta}
  +E^{\mathrm{ell}}_{\lambda,\eta},
\end{equation}
where $I$ is the identity term, $H$ the hyperbolic term (primitive closed geodesics and iterates with cutoff $g_{\lambda,\eta}$),
$P^{\mathrm{para}}$ the parabolic/cusp term (with the same truncation convention as \eqref{eq:7.reg.trace.def}),
and $E^{\mathrm{ell}}$ the elliptic term (present only if $\Gamma$ has torsion).

\begin{theorem}[Localized Selberg trace formula: exact regularized identity]\label{thm:7.selberg.exact}
For every cofinite $\Gamma$ and every $(\lambda,\eta)$, with the truncation and regularized trace fixed by
\eqref{eq:7.trunc.op} and \eqref{eq:7.reg.trace.def},
\begin{equation}\label{eq:7.selberg.exact}
  \mathcal{S}_{\lambda,\eta}
  \;=\;
  \mathcal{G}_{\lambda,\eta},
\end{equation}
where $\mathcal{S}_{\lambda,\eta}$ is given by Definitions~\ref{def:7.Sdisc}, \ref{def:7.Scont}, \ref{def:7.Sres}
and $\mathcal{G}_{\lambda,\eta}$ is the geometric decomposition \eqref{eq:7.G.recall} from Chapter~6.
\end{theorem}

\begin{proof}
This is the cofinite Selberg trace formula applied to the test function $h_{\lambda,\eta}$,
with cusp regularization implemented by truncation and subtraction as in \eqref{eq:7.reg.trace.def}.
References: Selberg~\cite{Selberg1956}; Hejhal~\cite[Vol.~I, \S2.5--\S2.6]{Hejhal1983}; Iwaniec~\cite[\S6.5--\S6.7]{Iwaniec2002}.
\end{proof}

\begin{remark}[Normalization lock and termwise checkability]\label{rem:7.norm.lock}
The equality \eqref{eq:7.selberg.exact} is \emph{termwise checkable} only because:
(i) the Eisenstein constant term normalization \eqref{eq:7.eis.const.term},
(ii) the truncation operator \eqref{eq:7.trunc.op},
and (iii) the subtraction in \eqref{eq:7.reg.trace.def}
are fixed once and used on both sides.
Changing any of these changes both $\mathcal{S}^{\mathrm{cont}}$ and $P^{\mathrm{para}}$ by the \emph{same} explicit linear functional.
\end{remark}

\section{Localized Weyl main term and remainder hierarchy (normalization-locked)}
% =====================================================================

\noindent
We now extract the Weyl main term in the localized regime \eqref{eq:7.locketa}.
The coefficient is rigidly fixed by LOCK-$\chi_0$.

\begin{theorem}[Localized Weyl main term (locked coefficient)]\label{thm:7.weyl.lock}
Assume \eqref{eq:7.locketa}. Then, as $\lambda\to\infty$,
\begin{equation}\label{eq:7.weyl.main}
  \mathcal{S}_{\lambda,\eta}
  =
  \frac{\vol(M)}{2\pi}\,\widehat{\chi}(0)\,\lambda\eta
  \;+\;
  H_{\lambda,\eta}
  \;+\;
  P^{\mathrm{para}}_{\lambda,\eta}
  \;+\;
  E^{\mathrm{ell}}_{\lambda,\eta}.
\end{equation}
Equivalently, inserting $\mathcal{S}_{\lambda,\eta}=\mathcal{G}_{\lambda,\eta}$,
the Weyl mass is exactly the identity orbital integral term from Chapter~6 and the coefficient
$\frac{\vol(M)}{2\pi}\widehat{\chi}(0)$ is independent of cusp normalizations.
\end{theorem}

\begin{proof}
Combine the exact identity \eqref{eq:7.selberg.exact} with the explicit identity-term evaluation on the geometric side
from Chapter~6 (identity orbital integral + LOCK-$\chi_0$).
The coefficient is fixed uniquely by \eqref{eq:7.lockchi0}.
References for the identity term computation: Selberg~\cite{Selberg1956}, Hejhal~\cite[Vol.~I, \S2.1--\S2.2]{Hejhal1983}, Iwaniec~\cite[\S6.2]{Iwaniec2002}.
\end{proof}

\begin{remark}[Windowed counting interpretation]\label{rem:7.window.count}
The discrete sum $\sum_j h_{\lambda,\eta}(r_j)$ is a smoothed local counting functional at resolution $\eta$.
Equation \eqref{eq:7.weyl.main} shows: the principal mass at scale $\eta$ is Weyl,
and the oscillatory correction is exactly the hyperbolic (closed geodesic) term with sharp cutoff
$kL\le \eta^{-1}$ enforced by \eqref{eq:7.lockg.support}.
\end{remark}

\section{Growth bounds for scattering data and admissible $\theta_0$}
% =====================================================================

\begin{lemma}[Polynomial growth of $\varphi'/\varphi$ on the unitary axis]\label{lem:7.phi.growth}
There exists $C_\Gamma>0$ such that
\[
  \frac{\varphi'(\tfrac12+ir)}{\varphi(\tfrac12+ir)} \ll_\Gamma (1+|r|)^{C_\Gamma},
  \qquad r\in\mathbb{R}.
\]
\end{lemma}

\begin{proof}
This follows from the analytic properties of Eisenstein series and the meromorphic continuation of $\Phi(s)$,
together with standard bounds on scattering determinants for cofinite groups.
References: Hejhal~\cite[Vol.~I, \S2.4--\S2.6]{Hejhal1983}, Iwaniec~\cite[\S6.6]{Iwaniec2002}.
\end{proof}

\begin{proposition}[Size of the continuous spectral contribution under localization]\label{prop:7.cont.size}
Uniformly for $\eta$ satisfying \eqref{eq:7.locketa},
\[
  \mathcal{S}^{\mathrm{cont}}_{\lambda,\eta}
  \ll_{\Gamma,\widehat{\chi}}
  \eta\,(1+\lambda)^{C_\Gamma}.
\]
In particular, if $\theta<1/C_\Gamma$ then $\mathcal{S}^{\mathrm{cont}}_{\lambda,\eta}=o(\lambda\eta)$ as $\lambda\to\infty$.
\end{proposition}

\begin{proof}
The window $h_{\lambda,\eta}$ is supported where $r=\pm\lambda+O(\eta)$ and
$\int |h_{\lambda,\eta}(r)|\,dr\asymp \eta$ by construction.
Combine this with Lemma~\ref{lem:7.phi.growth}.
\end{proof}

\section{Selberg zeta portal: geodesics $\leftrightarrow$ spectrum}
% =====================================================================

\noindent
The intrinsic hyperbolic object tying length spectrum and Laplace spectrum is the Selberg zeta function $Z_\Gamma(s)$.
We record its logarithmic derivative (Euler product over primitive closed geodesics).

\begin{proposition}[Log-derivative of Selberg zeta]\label{prop:7.selberg.zeta}
Let $Z_\Gamma(s)$ be the Selberg zeta function of $\Gamma$.
For $\Re s>1$,
\begin{equation}\label{eq:7.Zlogder}
  \frac{Z_\Gamma'(s)}{Z_\Gamma(s)}
  =
  \sum_{[\gamma_0]\in\mathcal{P}}
  \sum_{k\ge1}
  \frac{L(\gamma_0)\,e^{-ksL(\gamma_0)}}{2\sinh\!\big(\tfrac{kL(\gamma_0)}{2}\big)},
\end{equation}
where $\mathcal{P}$ is the set of primitive hyperbolic conjugacy classes and $L(\gamma_0)$ the primitive length.
Moreover $Z_\Gamma(s)$ continues meromorphically and its zeros encode the discrete spectrum and resonances.
\end{proposition}

\begin{proof}
Expand $\log Z_\Gamma(s)$ as an Euler product over primitive hyperbolic classes and differentiate termwise for $\Re s>1$.
References: Hejhal~\cite[Vol.~I, \S3.1--\S3.3]{Hejhal1983}, Iwaniec~\cite[\S6.8]{Iwaniec2002}.
\end{proof}

\begin{remark}[Meaning for the localized trace formula]\label{rem:7.selberg.portal.meaning}
The hyperbolic term $H_{\lambda,\eta}$ from Chapter~6 is precisely a \emph{windowed} version of \eqref{eq:7.Zlogder}
after Fourier inversion, with the window $g_{\lambda,\eta}(kL)$ enforcing the sharp cutoff $kL\le \eta^{-1}$
by \eqref{eq:7.lockg.support}. This is the hyperbolic analogue of the explicit formula linking primes to zeros.
\end{remark}

\section{Arithmetic portal: where the Riemann zeta function enters}
% =====================================================================

\noindent
In general cofinite $\Gamma$, the continuous spectrum is governed by scattering $\varphi(s)$.
For arithmetic groups, $\varphi(s)$ factorizes into completed $L$-functions; in particular $\zeta$ appears explicitly.

\begin{example}[Modular surface: explicit $\xi$-ratio]\label{ex:7.modular}
For $\Gamma=\mathrm{PSL}_2(\mathbb{Z})$ one has
\begin{equation}\label{eq:7.modular.phi}
  \varphi(s)=\frac{\xi(2s-1)}{\xi(2s)},
  \qquad
  \xi(s)=\pi^{-s/2}\Gamma\!\Big(\frac{s}{2}\Big)\zeta(s).
\end{equation}
Hence
\begin{equation}\label{eq:7.modular.logder}
  -\frac{\varphi'(s)}{\varphi(s)}
  =
  -2\frac{\xi'(2s-1)}{\xi(2s-1)}
  +2\frac{\xi'(2s)}{\xi(2s)}.
\end{equation}
Substituting $s=\tfrac12+ir$ into \eqref{eq:7.Scont} shows that the continuous contribution contains
$\zeta'/\zeta(1+2ir)$ and $\zeta'/\zeta(2ir)$ together with explicit Gamma-factor derivatives.
References: Hejhal~\cite[Vol.~I, \S2.7]{Hejhal1983}, Iwaniec~\cite[\S6.10]{Iwaniec2002}.
\end{example}

\begin{remark}[Why $\zeta$ is \emph{forced} here]\label{rem:7.zeta.forced}
On non-compact arithmetic surfaces the trace necessarily splits into discrete and continuous spectrum.
The continuous spectrum is governed by scattering, and scattering is governed by $\varphi'/\varphi$.
When $\varphi$ factorizes into completed $L$-functions, $\zeta$ appears as an unavoidable structural component of the trace identity.
\end{remark}

\section*{Chapter 7 Audit (Diamond / Control)}
% =====================================================================

\noindent\textbf{Locks (must hold exactly).}
\begin{itemize}
\item[(L7.1)] Fourier lock: \eqref{eq:7.lockF}.
\item[(L7.2)] Window lock: \eqref{eq:7.lockh}.
\item[(L7.3)] Transform lock: \eqref{eq:7.lockg.closed} and support \eqref{eq:7.lockg.support}.
\item[(L7.4)] Object lock: regularized trace \eqref{eq:7.reg.trace.def} is the unique object compared.
\item[(L7.5)] Eisenstein/scattering lock: constant term \eqref{eq:7.eis.const.term} fixes $\Phi(s)$ and $\varphi(s)$.
\item[(L7.6)] Maass--Selberg lock: exact matrix identity \eqref{eq:7.maass.selberg.matrix}.
\item[(L7.7)] Main-term lock: coefficient $\frac{\vol(M)}{2\pi}\widehat{\chi}(0)$ is rigid (LOCK-$\chi_0$).
\end{itemize}

\medskip
\noindent\textbf{Failure modes (typical).}
\begin{itemize}
\item[(F7.1)] Wrong sign or $2\pi$ factor in \eqref{eq:7.lockg.def} $\Rightarrow$ wrong hyperbolic cutoff and wrong geodesic weights.
\item[(F7.2)] Changing truncation without matching subtraction in \eqref{eq:7.reg.trace.def} $\Rightarrow$ mismatch between $\mathcal{S}^{\mathrm{cont}}$ and $P^{\mathrm{para}}$.
\item[(F7.3)] Mixing Eisenstein normalizations (different constant term conventions) $\Rightarrow$ drift in $\Phi'(s)\Phi(s)^{-1}$ and hence in \eqref{eq:7.Scont}.
\item[(F7.4)] Substituting wave-trace singularities for Selberg hyperbolic weights $\Rightarrow$ object substitution (forbidden by OBJ-LOCK).
\end{itemize}

\chapter{Local Weyl Law from the Localized Trace Formula}

\section{Smoothed counting function and the localized projector}

\noindent
We keep the OBJ-LOCK from Chapter~7: the fixed object is $\Tr^{\mathrm{reg}}(P_{\lambda,\eta})$ with
\[
  P_{\lambda,\eta}=h_{\lambda,\eta}\!\left(\sqrt{\Delta-\tfrac14}\right),
\]
and $h_{\lambda,\eta}$ is the locked window \eqref{eq:7.lockh}.
Define the smoothed spectral counting functional (discrete + continuous + residual) by
\begin{equation}\label{eq:8.Nsmooth}
  \mathcal{N}(\lambda,\eta)
  :=
  \mathcal{S}_{\lambda,\eta}
  =
  \sum_{j} h_{\lambda,\eta}(r_j)
  +\frac{1}{4\pi}\int_{\mathbb{R}} h_{\lambda,\eta}(r)\,
  \Bigl(-\frac{\varphi'(\tfrac12+ir)}{\varphi(\tfrac12+ir)}\Bigr)\,dr
  + \mathcal{S}^{\mathrm{res}}_{\lambda,\eta},
\end{equation}
with $\mathcal{S}^{\mathrm{res}}_{\lambda,\eta}$ as in \eqref{eq:7.Sres}.

\begin{lemma}[Trace equals smoothed counting]\label{lem:8.trace_equals_counting}
With the conventions of Chapter~7 one has the exact identity
\begin{equation}\label{eq:8.trace_equals_counting}
  \mathcal{N}(\lambda,\eta)
  =
  \Tr^{\mathrm{reg}}(P_{\lambda,\eta})
  =
  I_{\lambda,\eta}+H_{\lambda,\eta}+P^{\mathrm{para}}_{\lambda,\eta}+E^{\mathrm{ell}}_{\lambda,\eta},
\end{equation}
i.e. the continuous spectrum is absorbed into the left-hand side through the logarithmic derivative of $\varphi$.
\end{lemma}

\begin{proof}
This is exactly Theorem~\ref{thm:7.selberg.exact} rewritten using \eqref{eq:8.Nsmooth}.
\end{proof}

\section{Main term extraction}

\begin{proposition}[Weyl main term]\label{prop:8.main_term}
Assume the localization regime \eqref{eq:7.locketa}. Then
\begin{equation}\label{eq:8.main_term}
  \mathcal{N}(\lambda,\eta)
  =
  \frac{\vol(M)}{2\pi}\,\widehat{\chi}(0)\,\lambda\eta
  \;+\;
  \Err(\lambda,\eta),
\end{equation}
where
\begin{equation}\label{eq:8.err.decomp}
  \Err(\lambda,\eta)
  :=
  H_{\lambda,\eta}+P^{\mathrm{para}}_{\lambda,\eta}+E^{\mathrm{ell}}_{\lambda,\eta}.
\end{equation}
\end{proposition}

\begin{proof}
Immediate from Theorem~\ref{thm:7.weyl.lock} and Lemma~\ref{lem:8.trace_equals_counting}.
\end{proof}

\section{From smoothed to sharp local counting (Tauberian transfer)}

\noindent
Let
\begin{equation}\label{eq:8.Nsharp.def}
  N(\lambda,\eta)
  :=
  \#\{j:\ |r_j-\lambda|\le \eta\}.
\end{equation}
Since $h_{\lambda,\eta}$ is a fixed nonnegative bump at scale $\eta$ (up to the forced even reflection),
there exist absolute constants $c_1,c_2,C>0$ depending only on the fixed profile $\widehat{\chi}$ such that
\begin{equation}\label{eq:8.smooth_sharp.compare}
  c_1\,N(\lambda,\eta)
  \;\le\;
  \sum_j h_{\lambda,\eta}(r_j)
  \;\le\;
  c_2\,N(\lambda,C\eta).
\end{equation}

\begin{lemma}[Local Tauberian transfer]\label{lem:8.tauberian}
Assume that for some $\delta>0$ one has the uniform bound
\begin{equation}\label{eq:8.tauber.assump}
  \mathcal{N}(\lambda,\eta)
  =
  \frac{\vol(M)}{2\pi}\,\widehat{\chi}(0)\,\lambda\eta
  \;+\;
  O(\lambda^{1-\delta})
\end{equation}
for all $\eta\in[\lambda^{-\theta},1]$ with $\theta<\theta_0$.
Then uniformly in the same range,
\begin{equation}\label{eq:8.tauber.concl}
  N(\lambda,\eta)
  =
  \frac{\vol(M)}{2\pi}\,\lambda\eta
  \;+\;
  O(\lambda^{1-\delta}),
\end{equation}
where the implied constant depends only on $M$ and the fixed cutoff profile $\widehat{\chi}$.
\end{lemma}

\begin{proof}
This is a standard local Tauberian comparison using \eqref{eq:8.smooth_sharp.compare},
together with monotonicity in $\eta$ and the fixed shape of $h_{\lambda,\eta}$.
A textbook reference for Tauberian transfer in this spectral setting is Iwaniec~\cite[\S13.5]{Iwaniec2002};
see also Hejhal~\cite[Vol.~II, Ch.\ 15]{Hejhal1983}.
\end{proof}

\section{Quantitative local Weyl law: where the power saving comes from}

\noindent
To make \eqref{eq:8.tauber.assump} effective one needs a power saving on the error term $\Err(\lambda,\eta)$ in \eqref{eq:8.err.decomp}.
This is obtained by combining:
(i) the parametrix/log-time control from Chapter~5,
(ii) the hyperbolic cutoff $g_{\lambda,\eta}$ and amplitude control from Chapter~6,
and (iii) a dynamical/mixing (spectral-gap) input encoded by a group-dependent exponent $\beta_\Gamma$.

\begin{theorem}[Quantitative local Weyl law]\label{thm:8.local_weyl}
There exists $\delta>0$ (depending on $\Gamma$ and the fixed cutoff profile) such that uniformly for
$\eta\in[\lambda^{-\theta},1]$ with $\theta<\theta_0$,
\begin{equation}\label{eq:8.local_weyl}
  N(\lambda,\eta)
  =
  \frac{\vol(M)}{2\pi}\,\lambda\eta
  \;+\;
  O(\lambda^{1-\delta}).
\end{equation}
\end{theorem}

\begin{proof}
Combine Lemma~\ref{lem:8.tauberian} with Proposition~\ref{prop:8.main_term} and a power-saving estimate for
$\Err(\lambda,\eta)$ (hyperbolic + parabolic + elliptic) in the regime $\eta\ge \lambda^{-\theta}$,
as developed in Chapters~5--6.
\end{proof}
