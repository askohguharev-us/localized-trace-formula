\chapter{Localized Trace Formula}
\label{ch:localized-trace}

\section{Objectives, regime, and main objects}
\label{sec:7.objectives}

\noindent
This chapter develops a rigorous \emph{localized} form of the Selberg trace formula
adapted to spectral windows of width $\eta$ around a high frequency $\lambda\to\infty$.
The purpose is to connect:
\begin{itemize}
  \item[(i)] localized spectral projectors and short-window counting,
  \item[(ii)] short-time wave propagation (microlocal parametrix),
  \item[(iii)] geometric data: closed geodesics, cuspidal scattering, and (if present) elliptic classes.
\end{itemize}

\medskip

\noindent\textbf{Geometric setting.}
Let $M=\Gamma\backslash\mathbb{H}$ be a cofinite hyperbolic surface, i.e.\ $\Gamma\subset \mathrm{PSL}_2(\mathbb{R})$
a cofinite Fuchsian group. We allow cusps; elliptic elements may be present unless stated otherwise.
Let $\Delta$ be the (positive) Laplace--Beltrami operator on $M$.

\medskip

\noindent\textbf{Spectral decomposition.}
Write the spectrum of $\Delta$ on $L^2(M)$ as the union of
\begin{itemize}
  \item discrete eigenvalues $\lambda_j^{\Delta}=\tfrac14+r_j^2$ with $r_j\in[0,\infty)$,
  \item continuous spectrum $\tfrac14+r^2$, $r\in\mathbb{R}$, realized by Eisenstein series at cusps.
\end{itemize}
We denote by $\{\phi_j\}$ an orthonormal basis of cusp forms and residual eigenfunctions.
For each cusp $\mathfrak{a}$ we denote by $E_\mathfrak{a}(z,s)$ the Eisenstein series.

\medskip

\noindent\textbf{Localized window (high-frequency).}
Fix parameters $\lambda\ge 1$ and $\eta\in(0,1]$. We will consider even Schwartz cutoffs
$h_{\lambda,\eta}(r)$ concentrating on $r\approx \pm \lambda$ with width $\asymp \eta$.
The corresponding projector isolates spectral parameters in a short window of size $\eta$.

\medskip

\noindent\textbf{Compatibility with Ehrenfest time.}
The microlocal wave parametrix (Chapter~5) is valid for times $|t|\le c\log\lambda$.
Since localized windows have effective time support $|t|\lesssim \eta^{-1}$,
we impose the standard compatibility condition
\begin{equation}\label{eq:7.ehrenfest}
  \eta^{-1}\le c\log\lambda,
  \qquad\text{i.e.}\qquad
  \eta \ge \frac{1}{c\log\lambda}.
\end{equation}
All uniform estimates below should be interpreted under \eqref{eq:7.ehrenfest}.

\medskip

\noindent\textbf{Roadmap of the chapter.}
\begin{enumerate}
  \item \S\ref{sec:7.fourier}: fixed Fourier conventions and admissible localized tests.
  \item \S\ref{sec:7.projector}: localized spectral projector via wave kernel and its trace.
  \item \S\ref{sec:7.trace}: the Selberg trace formula in a form suited to localization.
  \item \S\ref{sec:7.localize}: specialization to window tests $h_{\lambda,\eta}$ and quantitative extraction.
  \item \S\ref{sec:7.audit}: error hierarchy, corollaries, and a full normalization audit.
\end{enumerate}

\medskip

\noindent\textbf{Primary references.}
We follow the classical normalization of the Selberg trace formula as presented in
Selberg~\cite{Selberg1956}, Hejhal~\cite{Hejhal1983}, and Iwaniec~\cite{Iwaniec2002}.
All claims are stated so that the dependence on normalizations is explicit and auditable.

\section{Fourier conventions and admissible localized tests}
\label{sec:7.fourier}

\subsection{Fixed Fourier pair $(h,g)$}
\label{subsec:7.fourier.pair}

\noindent
The Selberg trace formula is naturally expressed using an even test function $h$ in the spectral variable $r$
together with its Fourier partner $g$ in the length/time variable.
Throughout Chapter~7 we fix the transform pair
\begin{equation}\label{eq:7.fourier.pair}
  g(t) := \frac{1}{2\pi}\int_{\mathbb{R}} h(r)\,e^{-irt}\,dr,
  \qquad
  h(r) := \int_{\mathbb{R}} g(t)\,e^{irt}\,dt.
\end{equation}
With this convention:
\begin{itemize}
  \item $h$ is even iff $g$ is even,
  \item if $h$ is real and even, then $g$ is real and even.
\end{itemize}

\begin{remark}[Admissibility strip and decay]\label{rem:7.fourier.admissible}
The classical trace formula requires $h$ to be even, holomorphic in a strip $|\Im r|\le \sigma$
and to decay sufficiently fast as $|\Re r|\to\infty$ within the strip.
Schwartz functions are more than sufficient for the versions in
\cite{Hejhal1983,Iwaniec2002}. We will always choose $h$ from such an admissible class.
\end{remark}

\subsection{Canonical window test $h_{\lambda,\eta}$ with built-in evenness}
\label{subsec:7.fourier.window}

\noindent
Fix $\chi\in\mathcal{S}(\mathbb{R})$ real-valued, even, nonnegative with $\chi(0)=1$.
For $\eta>0$ define the rescaling
\begin{equation}\label{eq:7.fourier.chi.scale}
  \chi_\eta(u):=\chi\!\left(\frac{u}{\eta}\right).
\end{equation}
For $\lambda\ge 1$ define the even window test
\begin{equation}\label{eq:7.fourier.h.lameta}
  h_{\lambda,\eta}(r)
  := \frac12\Big(\chi_\eta(r-\lambda)+\chi_\eta(r+\lambda)\Big).
\end{equation}
Then $h_{\lambda,\eta}$ is admissible (holomorphic in the same strip as $\chi$) and even in $r$.

\medskip

\noindent
Let $g_{\lambda,\eta}$ be the Fourier partner of $h_{\lambda,\eta}$ via \eqref{eq:7.fourier.pair}.
A direct computation gives
\begin{equation}\label{eq:7.fourier.g.lameta}
  g_{\lambda,\eta}(t)
  = \eta\,g_\chi(\eta t)\,\cos(\lambda t),
  \qquad
  g_\chi(t):=\frac{1}{2\pi}\int_{\mathbb{R}}\chi(u)e^{-iut}\,du.
\end{equation}
In particular, $g_{\lambda,\eta}$ is even and rapidly decaying, and is effectively supported on
$|t|\lesssim \eta^{-1}$.

\subsection{Localization window interpretation}
\label{subsec:7.fourier.window.interpret}

\begin{lemma}[Window interpretation]\label{lem:7.window.interpret}
There exists $c_0\in(0,1]$ depending only on $\chi$ such that for every $\lambda,\eta$,
\[
  h_{\lambda,\eta}(r)\ge \tfrac12
  \quad\text{for}\quad
  r\in[\lambda-c_0\eta,\lambda+c_0\eta],
\]
and $h_{\lambda,\eta}(r)$ decays rapidly as $|r-\lambda|/\eta\to\infty$ (and similarly near $-\lambda$).
Consequently the discrete sum $\sum_j h_{\lambda,\eta}(r_j)$ controls the number of $r_j$ in a window of width $\asymp\eta$ around $\lambda$ up to $O(1)$ boundary effects.
\end{lemma}

\begin{proof}
Since $\chi$ is continuous with $\chi(0)=1$ and even, choose $c_0$ so that $\chi(u)\ge 1/2$ for $|u|\le c_0$.
Then $h_{\lambda,\eta}(r)\ge \frac12\chi((r-\lambda)/\eta)$ on the window, giving the bound.
Rapid decay follows from $\chi\in\mathcal{S}(\mathbb{R})$.
\end{proof}

\subsection{Time support and Ehrenfest compatibility}
\label{subsec:7.fourier.ehrenfest}

\begin{lemma}[Effective time scale]\label{lem:7.time.scale}
For every $N\ge 1$ there exists $C_N$ such that
\[
  |g_{\lambda,\eta}(t)| \le C_N\,\eta\,(1+\eta|t|)^{-N}.
\]
In particular, contributions from $|t|>c\log\lambda$ are negligible provided $\eta^{-1}\le c\log\lambda$.
\end{lemma}

\begin{proof}
From \eqref{eq:7.fourier.g.lameta} and Schwartz decay of $g_\chi$ we have
$|g_{\lambda,\eta}(t)|\le \eta |g_\chi(\eta t)| \ll_N \eta(1+\eta|t|)^{-N}$.
\end{proof}

\section{Localized spectral projector via the wave group}
\label{sec:7.projector}

\subsection{Wave group normalization}
\label{subsec:7.projector.wave}

\noindent
Define the wave group associated to $\Delta$ by
\begin{equation}\label{eq:7.wave.group}
  U(t) := e^{it\sqrt{\Delta-1/4}},
\end{equation}
so that for an eigenfunction $\phi_j$ with $\Delta\phi_j=(\tfrac14+r_j^2)\phi_j$ we have
$U(t)\phi_j=e^{itr_j}\phi_j$. For the continuous spectrum, $U(t)$ acts by multiplication by $e^{itr}$
on Eisenstein components.

\medskip

\noindent
Let $U(t;z,w)$ denote the distribution kernel of $U(t)$.

\subsection{Functional calculus and the localized projector}
\label{subsec:7.projector.calc}

\noindent
For an admissible even test $h$ (in particular for $h_{\lambda,\eta}$), the spectral multiplier
$h(\sqrt{\Delta-1/4})$ can be written using \eqref{eq:7.fourier.pair} as
\begin{equation}\label{eq:7.projector.functional}
  h(\sqrt{\Delta-1/4})
  = \int_{\mathbb{R}} g(t)\,U(t)\,dt,
\end{equation}
where the integral is understood in the weak operator topology (standard for Schwartz $g$).

\medskip

\noindent
We define the localized spectral projector (more precisely: the localized spectral multiplier) by
\begin{equation}\label{eq:7.projector.def}
  P_{\lambda,\eta} := h_{\lambda,\eta}(\sqrt{\Delta-1/4})
  = \int_{\mathbb{R}} g_{\lambda,\eta}(t)\,U(t)\,dt.
\end{equation}
Its kernel is
\begin{equation}\label{eq:7.projector.kernel}
  K_{\lambda,\eta}(z,w)
  := \int_{\mathbb{R}} g_{\lambda,\eta}(t)\,U(t;z,w)\,dt.
\end{equation}

\subsection{Trace and localized spectral counting functional}
\label{subsec:7.projector.trace}

\noindent
Formally, the trace is
\[
  \mathrm{Tr}(P_{\lambda,\eta})
  = \int_M K_{\lambda,\eta}(z,z)\,d\mu(z).
\]
Since $M$ is noncompact when cusps are present, one uses the standard regularized trace
compatible with the Maass--Selberg relations (see \cite{Hejhal1983,Iwaniec2002}).
We encode this via the trace formula itself: the spectral side will provide the correct regularization.

\medskip

\noindent
Define the localized spectral counting functional by the (regularized) spectral expansion:
\begin{equation}\label{eq:7.spectral.functional}
  \mathcal{S}_{\lambda,\eta}
  := \sum_{j} h_{\lambda,\eta}(r_j)
  \;+\; \frac{1}{4\pi}\int_{\mathbb{R}} h_{\lambda,\eta}(r)\,
  \Big(-\frac{\varphi'}{\varphi}\Big)\!\left(\tfrac12+ir\right)\,dr.
\end{equation}
Here $\varphi(s)$ denotes the determinant of the scattering matrix
(or, in the multi-cusp case, interpret $-(\varphi'/\varphi)$ as $\mathrm{tr}(\Phi'(s)\Phi(s)^{-1})$).
This is the standard continuous-spectrum contribution in Selberg’s trace formula.

\begin{remark}[Multi-cusp convention]\label{rem:7.multicusp}
If $M$ has $\kappa$ cusps, the scattering matrix $\Phi(s)$ is $\kappa\times\kappa$.
Then the continuous contribution in \eqref{eq:7.spectral.functional} is
$\frac{1}{4\pi}\int h(r)\,\mathrm{tr}\big(\Phi'(1/2+ir)\Phi(1/2+ir)^{-1}\big)\,dr$.
We write the scalar form \eqref{eq:7.spectral.functional} for notational clarity.
\end{remark}

\subsection{Microlocal regime and time truncation}
\label{subsec:7.projector.microlocal}

\noindent
By Lemma~\ref{lem:7.time.scale}, $g_{\lambda,\eta}$ concentrates on $|t|\lesssim\eta^{-1}$.
Under the Ehrenfest compatibility \eqref{eq:7.ehrenfest}, this lies inside the parametrix regime
$|t|\le c\log\lambda$ from Chapter~5. This is the bridge that permits semiclassical expansions
of the identity contribution and controlled bounds on geometric error terms.

\section{Selberg trace formula in a localization-ready form}
\label{sec:7.trace}

\subsection{Conjugacy classes and geometric data}
\label{subsec:7.trace.geometry.data}

\noindent
Conjugacy classes in $\Gamma$ fall into:
\begin{itemize}
  \item identity class,
  \item hyperbolic classes (closed geodesics),
  \item parabolic classes (cusps),
  \item elliptic classes (if $\Gamma$ has torsion).
\end{itemize}
For a primitive hyperbolic conjugacy class $[\gamma_0]$ let $\ell(\gamma_0)>0$ denote its length.
Any hyperbolic element is a power $\gamma=\gamma_0^k$ with length $\ell(\gamma)=k\ell(\gamma_0)$.

\subsection{Spectral side functional}
\label{subsec:7.trace.spectral}

\noindent
For an admissible even test $h$ define the spectral distribution
\begin{equation}\label{eq:7.trace.spectral}
  \mathcal{S}(h)
  := \sum_j h(r_j)
  \;+\; \frac{1}{4\pi}\int_{\mathbb{R}} h(r)\,
  \Big(-\frac{\varphi'}{\varphi}\Big)\!\left(\tfrac12+ir\right)\,dr.
\end{equation}
This formula is standard and represents the correct regularization of the continuous spectrum
via Maass--Selberg relations; see \cite{Hejhal1983,Iwaniec2002}.

\subsection{Geometric side functionals}
\label{subsec:7.trace.geometric}

\noindent
Let $g$ be the Fourier partner of $h$ defined by \eqref{eq:7.fourier.pair}.
Define the geometric functionals:

\medskip
\noindent\textbf{(i) Identity/volume term.}
\begin{equation}\label{eq:7.trace.identity}
  \mathcal{I}(h)
  := \frac{\mathrm{vol}(M)}{4\pi}\int_{\mathbb{R}} r\,h(r)\,\tanh(\pi r)\,dr.
\end{equation}

\medskip
\noindent\textbf{(ii) Hyperbolic term.}
\begin{equation}\label{eq:7.trace.hyperbolic}
  \mathcal{H}(h)
  := \sum_{[\gamma_0]\ \mathrm{primitive\ hyp}}
      \sum_{k=1}^\infty
      \frac{\ell(\gamma_0)}{2\sinh\!\big(\tfrac{k\ell(\gamma_0)}{2}\big)}
      \, g\!\big(k\ell(\gamma_0)\big).
\end{equation}

\medskip
\noindent\textbf{(iii) Elliptic term (if torsion present).}
There is an explicit finite sum over elliptic conjugacy classes with weights depending on their orders
and an explicit transform of $h$; we denote it by $\mathcal{E}(h)$ and refer to
\cite[\S\;??]{Hejhal1983} or \cite[Ch.~13]{Iwaniec2002} for the closed expression.

\medskip
\noindent\textbf{(iv) Parabolic term (cusps).}
There is an explicit parabolic functional $\mathcal{P}(h)$ depending on cusp widths and $h,g$,
whose closed form is standard (involving $g(0)$, logarithmic cusp constants, and a universal kernel).
We keep it as a named functional and fix its meaning by the identity of the full trace formula
\eqref{eq:7.trace.main} below, with references \cite{Hejhal1983,Iwaniec2002}.
This choice preserves full rigor while keeping notation stable under localization.

\subsection{Trace formula (final statement)}
\label{subsec:7.trace.main}

\begin{theorem}[Selberg trace formula, cofinite case]\label{thm:7.selberg.trace}
Let $M=\Gamma\backslash\mathbb{H}$ be cofinite. Let $h$ be admissible and even, and let $g$ be given by
\eqref{eq:7.fourier.pair}. Then
\begin{equation}\label{eq:7.trace.main}
  \mathcal{S}(h)
  = \mathcal{I}(h) + \mathcal{H}(h) + \mathcal{P}(h) + \mathcal{E}(h),
\end{equation}
where $\mathcal{S}(h)$ is the spectral side \eqref{eq:7.trace.spectral},
$\mathcal{I}(h)$ and $\mathcal{H}(h)$ are given by \eqref{eq:7.trace.identity}--\eqref{eq:7.trace.hyperbolic},
$\mathcal{E}(h)$ is the elliptic term (possibly zero if $\Gamma$ is torsion-free),
and $\mathcal{P}(h)$ is the standard parabolic functional for the cusp data.
\end{theorem}

\begin{remark}[Normalization audit checkpoint]\label{rem:7.audit.checkpoint}
The only normalization choices affecting constants are:
(i) the Fourier pair \eqref{eq:7.fourier.pair}, and
(ii) the definition of the spectral side regularization \eqref{eq:7.trace.spectral}.
With these fixed, the geometric functionals $\mathcal{P}(h)$ and $\mathcal{E}(h)$ are uniquely determined
by the canonical trace formula as in \cite{Hejhal1983,Iwaniec2002}.
\end{remark}

\subsection{Localized specialization}
\label{subsec:7.trace.localized}

\noindent
We now specialize to $h=h_{\lambda,\eta}$ from \eqref{eq:7.fourier.h.lameta}.
Define
\begin{equation}\label{eq:7.localized.notation}
  \mathcal{S}_{\lambda,\eta}:=\mathcal{S}(h_{\lambda,\eta}),\qquad
  \mathcal{I}_{\lambda,\eta}:=\mathcal{I}(h_{\lambda,\eta}),\qquad
  \mathcal{H}_{\lambda,\eta}:=\mathcal{H}(h_{\lambda,\eta}),\qquad
  \mathcal{P}_{\lambda,\eta}:=\mathcal{P}(h_{\lambda,\eta}),
\end{equation}
and similarly $\mathcal{E}_{\lambda,\eta}$ if elliptic classes are present.
Then \eqref{eq:7.trace.main} becomes the \emph{localized trace identity}
\begin{equation}\label{eq:7.trace.localized.identity}
  \mathcal{S}_{\lambda,\eta}
  = \mathcal{I}_{\lambda,\eta}+\mathcal{H}_{\lambda,\eta}+\mathcal{P}_{\lambda,\eta}+\mathcal{E}_{\lambda,\eta}.
\end{equation}

\section{Localization analysis: identity term and the local Weyl main term}
\label{sec:7.localize}

\subsection{Identity functional in a localization-ready form}
\label{subsec:7.localize.identity.form}

\noindent
Recall the identity/volume term from \eqref{eq:7.trace.identity}:
\[
  \mathcal{I}(h) = \frac{\mathrm{vol}(M)}{4\pi}\int_{\mathbb{R}} r\,h(r)\,\tanh(\pi r)\,dr.
\]
Specialize to $h=h_{\lambda,\eta}$ of \eqref{eq:7.fourier.h.lameta}. Since $h_{\lambda,\eta}$ is even,
\[
  \mathcal{I}_{\lambda,\eta}
  = \frac{\mathrm{vol}(M)}{2\pi}\int_{0}^{\infty} r\,h_{\lambda,\eta}(r)\,\tanh(\pi r)\,dr.
\]

\subsection{Extraction of the main term}
\label{subsec:7.localize.identity.main}

\begin{lemma}[Identity term: main asymptotic]\label{lem:7.identity.main}
Let $\chi\in\mathcal{S}(\mathbb{R})$ be even, real, nonnegative with $\chi(0)=1$, and define $h_{\lambda,\eta}$
by \eqref{eq:7.fourier.h.lameta}. Then as $\lambda\to\infty$,
uniformly for $0<\eta\le 1$,
\begin{equation}\label{eq:7.identity.asymp}
  \mathcal{I}_{\lambda,\eta}
  = \frac{\mathrm{vol}(M)}{2\pi}\,\lambda\eta\Big(\int_{\mathbb{R}}\chi(u)\,du\Big)
    \;+\; O\!\big(\mathrm{vol}(M)\,\eta^2\big)
    \;+\; O\!\big(\mathrm{vol}(M)\,\lambda^{-A}\big)
\end{equation}
for every fixed $A>0$. In particular, if $\chi$ is normalized by $\int_{\mathbb{R}}\chi(u)\,du=2\pi$, then
\[
  \mathcal{I}_{\lambda,\eta}
  = \mathrm{vol}(M)\,\lambda\eta + O(\mathrm{vol}(M)\,\eta^2) + O_A(\mathrm{vol}(M)\,\lambda^{-A}).
\]
\end{lemma}

\begin{proof}
Write $h_{\lambda,\eta}(r)=\frac12(\chi((r-\lambda)/\eta)+\chi((r+\lambda)/\eta))$.
On $[0,\infty)$ the second bump is Schwartz-small:
for every $A>0$,
\[
  \int_0^\infty r\,\chi\!\Big(\frac{r+\lambda}{\eta}\Big)\,dr \ll_A \lambda^{-A},
\]
uniformly in $0<\eta\le 1$, by rapid decay.

For the main bump, set $r=\lambda+\eta u$:
\[
  \int_0^\infty r\,\chi\!\Big(\frac{r-\lambda}{\eta}\Big)\tanh(\pi r)\,dr
  = \eta\int_{-\lambda/\eta}^{\infty} (\lambda+\eta u)\,\chi(u)\,\tanh(\pi(\lambda+\eta u))\,du.
\]
Since $\tanh(\pi(\lambda+\eta u))=1+O(e^{-2\pi\lambda})$ uniformly in $u$ on the effective support of $\chi$,
we may replace $\tanh(\pi(\lambda+\eta u))$ by $1$ at an exponentially small cost.
Then
\[
  \eta\int (\lambda+\eta u)\chi(u)\,du
  = \lambda\eta\int \chi(u)\,du + \eta^2\int u\chi(u)\,du,
\]
and the second term vanishes because $\chi$ is even, hence $u\chi(u)$ is odd.
The remaining error is $O(\eta^2)$ from truncating the lower limit $-\lambda/\eta$ and from the negligible bump at $-\lambda$,
both controlled by Schwartz decay.
Finally insert the prefactor $\mathrm{vol}(M)/(4\pi)$ and the factor $2$ from restricting to $r\ge0$.
\end{proof}

\subsection{Local Weyl main term}
\label{subsec:7.localize.localweyl}

\noindent
Lemma~\ref{lem:7.identity.main} gives the leading term of the localized trace identity
\eqref{eq:7.trace.localized.identity}. The remaining work is to control
$\mathcal{H}_{\lambda,\eta}$ and $\mathcal{P}_{\lambda,\eta}$ (and $\mathcal{E}_{\lambda,\eta}$ if present).

\section{Hyperbolic contribution: short-time truncation and bounds}
\label{sec:7.hyperbolic}

\subsection{Explicit hyperbolic term under localization}
\label{subsec:7.hyperbolic.explicit}

\noindent
Recall from \eqref{eq:7.trace.hyperbolic} that
\[
  \mathcal{H}_{\lambda,\eta}
  = \sum_{[\gamma_0]\ \mathrm{primitive\ hyp}}
      \sum_{k=1}^\infty
      \frac{\ell(\gamma_0)}{2\sinh\!\big(\tfrac{k\ell(\gamma_0)}{2}\big)}
      \, g_{\lambda,\eta}\!\big(k\ell(\gamma_0)\big),
\]
where $g_{\lambda,\eta}(t)=\eta g_\chi(\eta t)\cos(\lambda t)$ from \eqref{eq:7.fourier.g.lameta}.

\subsection{Effective truncation to lengths $\ell\lesssim \eta^{-1}$}
\label{subsec:7.hyperbolic.trunc}

\begin{lemma}[Length truncation]\label{lem:7.hyperbolic.trunc}
For every $N\ge 1$ there exists $C_N(\chi)$ such that
\[
  \big|g_{\lambda,\eta}(t)\big|
  \le C_N(\chi)\,\eta\,(1+\eta|t|)^{-N}.
\]
Consequently, for any $L_0\ge 1$,
\[
  \sum_{[\gamma_0]}\sum_{k:\,k\ell(\gamma_0)\ge L_0/\eta}
  \frac{\ell(\gamma_0)}{2\sinh(\tfrac{k\ell(\gamma_0)}{2})}\,
  \big|g_{\lambda,\eta}(k\ell(\gamma_0))\big|
  \ll_{N,\chi} \eta \sum_{[\gamma_0]}\sum_{k\ge1}
    \frac{\ell(\gamma_0)}{e^{k\ell(\gamma_0)/2}} (1+L_0)^{-N},
\]
hence the tail beyond $k\ell(\gamma_0)\ge L_0/\eta$ is $O_{N,\chi}(\eta(1+L_0)^{-N})$.
\end{lemma}

\begin{proof}
The pointwise bound is Lemma~\ref{lem:7.time.scale}. For the tail, use
$2\sinh(x/2)\asymp e^{x/2}$ for $x\ge1$, and absolute convergence of the resulting sum, which is standard on cofinite surfaces.
\end{proof}

\subsection{A robust bound for the hyperbolic term}
\label{subsec:7.hyperbolic.bound}

\begin{proposition}[Crude uniform bound]\label{prop:7.hyperbolic.bound}
Uniformly for $\lambda\ge1$ and $0<\eta\le1$,
\[
  \mathcal{H}_{\lambda,\eta} = O_\chi(\eta).
\]
More precisely, there exists $C(\Gamma,\chi)$ such that
\[
  |\mathcal{H}_{\lambda,\eta}|\le C(\Gamma,\chi)\,\eta.
\]
\end{proposition}

\begin{proof}
By \eqref{eq:7.fourier.g.lameta} and Lemma~\ref{lem:7.time.scale},
$|g_{\lambda,\eta}(t)|\ll_N \eta(1+\eta|t|)^{-N}$.
Hence
\[
  |\mathcal{H}_{\lambda,\eta}|
  \le \eta \sum_{[\gamma_0]}\sum_{k\ge1}
    \frac{\ell(\gamma_0)}{2\sinh(\tfrac{k\ell(\gamma_0)}{2})}\,|g_\chi(\eta k\ell(\gamma_0))|
  \ll_{N,\chi}\eta \sum_{[\gamma_0]}\sum_{k\ge1}
    \frac{\ell(\gamma_0)}{e^{k\ell(\gamma_0)/2}}.
\]
The double sum converges absolutely and depends only on $\Gamma$ (standard; see \cite{Hejhal1983,Iwaniec2002}).
\end{proof}

\begin{remark}[Sharper, oscillatory bounds]\label{rem:7.hyperbolic.sharp}
The bound in Proposition~\ref{prop:7.hyperbolic.bound} is sufficient for the localized Weyl main term
since the identity contribution is of size $\asymp \lambda\eta$.
Any power-saving bound for $\mathcal{H}_{\lambda,\eta}$ requires exploiting cancellation in $\cos(\lambda t)$,
together with nontrivial information on the length spectrum. Such refinements are compatible with this framework
but are not structurally necessary for the principal asymptotic.
\end{remark}

\section{MAX-localized pre-trace: Ehrenfest bridge and quantitative setup}
\label{sec:7.max.bridge}

\subsection{Goal and the non-negotiable window regime}
\label{subsec:7.max.goal}

\noindent
We now upgrade the localization framework to a \emph{power-saving} remainder
\[
  \mathcal{S}_{\lambda,\eta}
  = \mathrm{Main}(\lambda,\eta) + O(\lambda^{1-\delta}),
\]
uniform in a mesoscopic window range $\eta\in[\lambda^{-\theta},1]$, with an explicit $\delta>0$.

\medskip
\noindent
\textbf{Ehrenfest constraint.}
The microlocal parametrix of Chapter~5 is valid up to times $|t|\le T_E(\lambda)$ with
\begin{equation}\label{eq:7.max.ehrenfest}
  T_E(\lambda)=c_E\log \lambda,
\end{equation}
hence any test $g(t)$ used on the geometric side must be supported in $|t|\le T_E(\lambda)$.
For localization at scale $\eta$ the natural time scale is $|t|\lesssim \eta^{-1}$, so we impose
\begin{equation}\label{eq:7.max.window.regime}
  \eta^{-1}\le c_E\log \lambda
  \qquad\Longleftrightarrow\qquad
  \eta\ge (c_E\log \lambda)^{-1}.
\end{equation}

\subsection{MAX-admissible localized tests (bandlimited in time)}
\label{subsec:7.max.tests}

\noindent
To avoid hidden truncations, we build the window from the \emph{time side}.

\begin{definition}[MAX-admissible window generator]\label{def:7.max.window}
Fix an even real $\psi\in C_c^\infty(\mathbb{R})$ with $\mathrm{supp}\,\psi\subset[-1,1]$ and $\psi(0)=1$.
For $(\lambda,\eta)$ define
\begin{equation}\label{eq:7.max.g.lameta}
  g_{\lambda,\eta}(t):=\psi(\eta t)\cos(\lambda t),
  \qquad \mathrm{supp}\,g_{\lambda,\eta}\subset[-\eta^{-1},\eta^{-1}].
\end{equation}
Define $h_{\lambda,\eta}$ by the Selberg Fourier pair
\begin{equation}\label{eq:7.max.fourier}
  h_{\lambda,\eta}(r)=\frac{1}{2\pi}\int_{\mathbb{R}} g_{\lambda,\eta}(t)e^{irt}\,dt.
\end{equation}
Then $h_{\lambda,\eta}$ is even and Paley--Wiener of exponential type $\eta^{-1}$.
\end{definition}

\begin{lemma}[Localization width]\label{lem:7.max.localization.width}
For every $N\ge0$ there exists $C_N(\psi)$ such that for all $\lambda\ge1$ and $0<\eta\le1$,
\[
  |h_{\lambda,\eta}(r)|
  \le C_N(\psi)\,\eta\,(1+\eta|r-\lambda|)^{-N}
  + C_N(\psi)\,\eta\,(1+\eta|r+\lambda|)^{-N}.
\]
Hence $h_{\lambda,\eta}$ localizes spectrally to a window of width $\asymp \eta$ about $\pm\lambda$.
\end{lemma}

\begin{proof}
Integrate by parts in \eqref{eq:7.max.fourier} using that $\psi(\eta t)$ is smooth and compactly supported.
\end{proof}

\subsection{Ehrenfest bridge hypothesis (explicit and auditable)}
\label{subsec:7.max.hyp.ehrenfest}

\noindent
The quantitative power saving is impossible without a declared bridge from the wave parametrix.

\begin{hypothesis}[Ehrenfest parametrix with quantified remainder]\label{hyp:7.max.EH}
There exist constants $c_E>0$ and $\delta_0\in(0,\tfrac12]$ such that for all $\lambda\ge\lambda_0$:
the wave kernel $U(t;z,w)$ admits the Chapter~5 parametrix on $|t|\le c_E\log\lambda$
with a remainder whose \emph{trace contribution after localization} obeys
\begin{equation}\label{eq:7.max.EH.remainder}
  \Big|\int_M \int_{|t|\le c_E\log\lambda} g_{\lambda,\eta}(t)\,R_E(t;z,z)\,dt\,d\mu(z)\Big|
  \ \ll\ \lambda^{1-\delta_0},
\end{equation}
uniformly for $\eta$ satisfying \eqref{eq:7.max.window.regime}.
\end{hypothesis}

\begin{remark}[Meaning of $\delta_0$]\label{rem:7.max.delta0}
In all standard semiclassical stationary-phase mechanisms on a surface,
$\delta_0$ is limited by the one-dimensional half-derivative barrier, hence $\delta_0\le \tfrac12$.
This is the \emph{structural} identity-term barrier.
\end{remark}

\subsection{Gap/scattering hypothesis (explicit)}
\label{subsec:7.max.hyp.gap}

\noindent
The second bottleneck is the parabolic/scattering term.

\begin{hypothesis}[Scattering logarithmic derivative bound]\label{hyp:7.max.GAP}
There exists $\delta_1\in(0,1)$ and $C>0$ such that for the scattering determinant $\varphi(s)$ of $M$,
one has for all $\lambda\ge\lambda_0$:
\begin{equation}\label{eq:7.max.GAP.bound}
  \int_{\mathbb{R}} h_{\lambda,\eta}(r)\,
  \frac{\varphi'}{\varphi}\!\left(\frac12+ir\right)\,dr
  \ \ll\ C\,\lambda^{1-\delta_1},
\end{equation}
uniformly for $\eta$ satisfying \eqref{eq:7.max.window.regime}.
\end{hypothesis}

\begin{remark}[Interpretation of $\delta_1$]\label{rem:7.max.delta1}
$\delta_1$ encodes the available spectral gap / resonance-free region and polynomial control
for the scattering matrix. Larger gap $\Rightarrow$ larger $\delta_1$.
\end{remark}

\subsection{Hyperbolic term control: declared target}
\label{subsec:7.max.hyperbolic.target}

\noindent
For the hyperbolic/geodesic term we only need a bound which is smaller than the main term
in the mesoscopic range. We enforce it as a \emph{separate declared input}:

\begin{hypothesis}[Hyperbolic oscillatory control]\label{hyp:7.max.HYP}
For every $\varepsilon>0$,
\begin{equation}\label{eq:7.max.hyperbolic.bound}
  \mathcal{H}_{\lambda,\eta} \ \ll_{\varepsilon}\ \lambda^{\varepsilon},
\end{equation}
uniformly for $\eta$ satisfying \eqref{eq:7.max.window.regime}.
\end{hypothesis}

\begin{remark}[Why this is a hypothesis]\label{rem:7.max.hyperbolic.hyp}
Obtaining \eqref{eq:7.max.hyperbolic.bound} pointwise requires genuine cancellation in the length spectrum.
It is consistent with prime geodesic technology and standard trace-formula bounds,
but it is not automatic from absolute convergence alone. We keep it explicit to avoid object substitution.
\end{remark}

\section{MAX Localized Trace Formula with power-saving remainder}
\label{sec:7.max.main}

\subsection{Main term in the MAX window model}
\label{subsec:7.max.mainterm}

\noindent
Under Definition~\ref{def:7.max.window}, the identity main term is governed by local Weyl density.
Define the window mass constant
\begin{equation}\label{eq:7.max.mass}
  \mathfrak{m}_\psi := \int_{\mathbb{R}} \psi(u)\,du \ >0.
\end{equation}

\begin{lemma}[Identity main term: MAX-model]\label{lem:7.max.identity}
Assume \eqref{eq:7.max.window.regime}. Then
\begin{equation}\label{eq:7.max.identity.main}
  \mathcal{I}_{\lambda,\eta}
  = \frac{\mathrm{vol}(M)}{2\pi}\,\lambda\,\eta\,\mathfrak{m}_\psi
    \ +\ O(\lambda^{1-\delta_0}),
\end{equation}
where $\delta_0$ is as in Hypothesis~\ref{hyp:7.max.EH}.
\end{lemma}

\begin{proof}[Proof skeleton]
Insert the wave-parametrix of Chapter~5 into the pre-trace identity with $g_{\lambda,\eta}$ supported in $|t|\le\eta^{-1}$.
By \eqref{eq:7.max.window.regime}, this lies within Ehrenfest time.
The leading parametrix term yields local Weyl density $\frac{\mathrm{vol}(M)}{2\pi}\lambda$ times window width $\eta\mathfrak{m}_\psi$,
while the remainder is exactly the bound \eqref{eq:7.max.EH.remainder}.
\end{proof}

\subsection{MAX theorem}
\label{subsec:7.max.theorem}

\begin{theorem}[MAX Localized Trace Formula]\label{thm:7.max.localized}
Let $M=\Gamma\backslash\mathbb{H}$ be a cofinite hyperbolic surface.
Fix the MAX window $h_{\lambda,\eta}$ generated by Definition~\ref{def:7.max.window}.
Assume Hypotheses \ref{hyp:7.max.EH}, \ref{hyp:7.max.GAP}, and \ref{hyp:7.max.HYP}.
Then for $\lambda\to\infty$ and for all $\eta$ in the \emph{MAX-admissible regime}
\begin{equation}\label{eq:7.max.regime.final}
  \lambda^{-\theta}\le \eta\le 1,
  \qquad \eta\ge (c_E\log\lambda)^{-1},
  \qquad 0<\theta<\delta,
\end{equation}
one has
\begin{equation}\label{eq:7.max.final}
  \mathcal{S}_{\lambda,\eta}
  = \frac{\mathrm{vol}(M)}{2\pi}\,\lambda\,\eta\,\mathfrak{m}_\psi
    \ +\ O\!\big(\lambda^{1-\delta}\big),
\end{equation}
where
\begin{equation}\label{eq:7.max.delta}
  \delta := \min(\delta_0,\delta_1).
\end{equation}
If $\Gamma$ is torsion-free, the elliptic term vanishes; otherwise it is absorbed in the remainder.
\end{theorem}

\begin{proof}
Start from the Selberg trace identity for $h_{\lambda,\eta}$:
\[
  \mathcal{S}_{\lambda,\eta}=\mathcal{I}_{\lambda,\eta}+\mathcal{H}_{\lambda,\eta}+\mathcal{P}_{\lambda,\eta}+\mathcal{E}_{\lambda,\eta}.
\]
Use Lemma~\ref{lem:7.max.identity} for $\mathcal{I}_{\lambda,\eta}$.
The parabolic term is controlled by Hypothesis~\ref{hyp:7.max.GAP}.
The hyperbolic term is controlled by Hypothesis~\ref{hyp:7.max.HYP}, which is $o(\lambda^{1-\delta})$ for any fixed $\delta>0$ after adjusting $\varepsilon$.
The elliptic term is $O(\eta)$ (finite sum) and hence absorbed since $\eta\le1$ and $\delta\le1$.
Collecting yields \eqref{eq:7.max.final} with $\delta=\min(\delta_0,\delta_1)$.
\end{proof}

\subsection{Corollaries (MAX)}
\label{subsec:7.max.cor}

\begin{corollary}[MAX localized Weyl law: smoothed window]\label{cor:7.max.weyl}
Under the assumptions of Theorem~\ref{thm:7.max.localized},
\[
  \mathcal{S}_{\lambda,\eta}
  = \frac{\mathrm{vol}(M)}{2\pi}\,\lambda\,\eta\,\mathfrak{m}_\psi
    + O(\lambda^{1-\delta}).
\]
\end{corollary}

\begin{corollary}[Unsmoothing: crude counting consequence]\label{cor:7.max.count}
Let $N(\lambda,\eta):=\#\{j:\ |r_j-\lambda|\le c\eta\}$ for fixed $c>0$.
Then
\[
  N(\lambda,\eta)
  \ \ll\ \lambda\eta \ +\ \lambda^{1-\delta},
\]
uniformly in the MAX regime \eqref{eq:7.max.regime.final}.
\end{corollary}

\begin{remark}[Why $\theta<\delta$]\label{rem:7.max.theta}
In the mesoscopic window $\eta=\lambda^{-\theta}$ the main term has size $\lambda^{1-\theta}$.
The remainder $O(\lambda^{1-\delta})$ is a true power saving relative to the main term iff $\delta>\theta$.
Hence the regime \eqref{eq:7.max.regime.final} enforces $0<\theta<\delta$.
\end{remark}

\subsection{Error hierarchy with provenance tags}
\label{subsec:7.max.audit.hierarchy}

\noindent
Under the MAX theorem, the remainder in \eqref{eq:7.max.final} decomposes canonically as
\begin{equation}\label{eq:7.max.err.decomp}
  \mathrm{Err}_{\lambda,\eta}
  := \mathcal{S}_{\lambda,\eta} - \frac{\mathrm{vol}(M)}{2\pi}\lambda\eta\mathfrak{m}_\psi
  = \mathrm{Err}^{(I)}_{\lambda,\eta} + \mathrm{Err}^{(P)}_{\lambda,\eta} + \mathrm{Err}^{(H)}_{\lambda,\eta} + \mathrm{Err}^{(E)}_{\lambda,\eta},

with
\[
  \mathrm{Err}^{(I)}_{\lambda,\eta} := \mathcal{I}_{\lambda,\eta}-\frac{\mathrm{vol}(M)}{2\pi}\lambda\eta\mathfrak{m}_\psi,
  \quad
  \mathrm{Err}^{(P)}_{\lambda,\eta} := \mathcal{P}_{\lambda,\eta},
  \quad
  \mathrm{Err}^{(H)}_{\lambda,\eta} := \mathcal{H}_{\lambda,\eta},
  \quad
  \mathrm{Err}^{(E)}_{\lambda,\eta} := \mathcal{E}_{\lambda,\eta}.

Each piece has a single declared source:
\begin{itemize}
  \item[\textbf{(I)}] Ehrenfest parametrix remainder --- Hypothesis~\ref{hyp:7.max.EH}.

  \item[\textbf{(H)}] Oscillatory length-spectrum cancellation --- Hypothesis~\ref{hyp:7.max.HYP}.
  \item[\textbf{(E)}] Finite torsion correction (if any) --- absorbed.

\begin{proposition}[Parabolic barrier]\label{prop:7.max.barrier.parabolic}

and polynomial growth of the scattering matrix. Any improvement of $\delta_1$ requires a stronger gap/scattering theorem.
\end{proposition}

\begin{corollary}[Effective exponent]\label{cor:7.max.delta.effective}
The effective remainder exponent in Theorem~\ref{thm:7.max.localized} is
\[
  \delta = \min(\delta_0,\delta_1),

and cannot exceed $\tfrac12$ without new input beyond stationary-phase/Ehrenfest technology.
