\chapter{Spectral expansion, normalization ledger, and comparison for the localized trace formula}
\label{chap:ch7-spectral-comparison}

\section*{Orientation and main deliverables}\label{sec:ch7-orientation}

\noindent\textbf{Setting.}
Let $M=\Gamma\backslash\H$ be a finite-area hyperbolic surface, $\Delta\ge 0$ the nonnegative Laplace--Beltrami operator,
and
\[
A:=\sqrt{\Delta-\tfrac14}.
\]
Let $P_{\lambda,\eta}$ be the localized spectral projector introduced in Chapter~5 and used on the geometric side in
Chapter~6.

\medskip
\noindent\textbf{Purpose.}
This chapter fixes the \emph{spectral side} of the localized trace formula in a convention-locked form and compares it to
the geometric side of Chapter~6. Concretely:
\begin{itemize}
\item[(S7.1)] a spectral decomposition of $\Tr^{\mathrm{reg}}(P_{\lambda,\eta})$ into discrete, continuous, and residual parts,
with an explicit \emph{normalization ledger} for cusp regularization;
\item[(S7.2)] an exact identity $\mathcal S_{\lambda,\eta}=\mathcal G_{\lambda,\eta}$ with Chapter~6 under the same ledger,
with no object substitution;
\item[(S7.3)] extraction of the localized Weyl main term with the unique bridge constant $\widehat\chi(0)$;
\item[(S7.4)] the intrinsic ``zeta portals'': Selberg zeta $Z_\Gamma$ in general, and $\zeta$ for arithmetic $\Gamma$ via scattering;
\item[(S7.5)] \textbf{CONTROL module:} a microlocal \emph{anti-trace} (time-reversal odd) which kills the Weyl term and reads
\emph{oriented} orbit information.
\end{itemize}

\medskip
\noindent\textbf{References.}
For cofinite trace formula and scattering normalization see Selberg~\cite{Selberg1956},
Hejhal~\cite{Hejhal1983}, Iwaniec~\cite{Iwaniec2002}. For microlocal trace and wave/FIO methods see
Duistermaat--Guillemin~\cite{DuistermaatGuillemin1975}, H\"ormander~\cite{Hormander1983,Hormander1985},
Sogge~\cite{Sogge2014}, Dyatlov--Zworski~\cite{DyatlovZworski2019}.

\section{OBJ-LOCK and convention locks}\label{sec:ch7-objlock}
% =====================================================================

\subsection{The object (and what is \emph{not} allowed)}\label{subsec:ch7-object}

\noindent\textbf{OBJ-LOCK.}
Throughout Chapters~6--7 the invariant object is
\begin{equation}\label{eq:7.obj}
\Tr^{\mathrm{reg}}(P_{\lambda,\eta}),
\qquad
P_{\lambda,\eta}=h_{\lambda,\eta}(A),
\end{equation}
and, when microlocal cutoffs are inserted, the declared variant
\begin{equation}\label{eq:7.obj.microlocal}
\Tr^{\mathrm{reg}}(\Psi P_{\lambda,\eta}\Psi),
\end{equation}
with $\Psi$ stated explicitly.

\medskip
\noindent\textbf{No substitution.}
No identification is permitted between:
(i) Selberg trace formula for a test pair $(h,g)$,
(ii) wave-trace distributions $\Tr(\cos(tA))$,
(iii) truncated kernel traces $\Tr_{M(Y)}(\cdot)$,
unless the change is explicitly stated and the corresponding normalization ledger is recorded
(\S\ref{sec:ch7-regtrace}).

\subsection{Fourier and window locks}\label{subsec:ch7-locks}

\noindent\textbf{LOCK-F (Fourier).}
Fix an even cutoff $\widehat\chi\in C_c^\infty(\R)$ with
\begin{equation}\label{eq:7.chihat.lock}
\widehat\chi(-t)=\widehat\chi(t),\qquad
\supp(\widehat\chi)\subset(-1,1),\qquad
\widehat\chi(0)>0,
\end{equation}
and define
\begin{equation}\label{eq:7.chi.def}
\chi(u):=\frac{1}{2\pi}\int_{\R}\widehat\chi(t)\,e^{itu}\,dt.
\end{equation}

\medskip
\noindent\textbf{LOCK-$(\lambda,\eta)$ (window).}
Let $\lambda\to\infty$ and $\eta=\eta(\lambda)\in(0,1]$. Define the even spectral window
\begin{equation}\label{eq:7.h.window}
h_{\lambda,\eta}(r):=
\chi\!\left(\frac{r-\lambda}{\eta}\right)+
\chi\!\left(\frac{r+\lambda}{\eta}\right),
\qquad r\in\R.
\end{equation}

\medskip
\noindent\textbf{LOCK-$\eta$ (localization regime).}
We work under
\begin{equation}\label{eq:7.eta.regime}
\eta\ge \lambda^{-\theta},\qquad 0<\theta<\theta_0,
\end{equation}
where $\theta_0>0$ is chosen so that later remainder bounds admit a power saving in $\lambda$.

\subsection{Selberg transform lock}\label{subsec:ch7-glock}

\noindent\textbf{LOCK-$g$.}
Define the Selberg transform kernel by
\begin{equation}\label{eq:7.g.def}
g_{\lambda,\eta}(t):=\frac{1}{2\pi}\int_{\R} h_{\lambda,\eta}(r)\,e^{-irt}\,dr.
\end{equation}
Then, exactly as in Chapter~6,
\begin{equation}\label{eq:7.g.explicit}
g_{\lambda,\eta}(t)=\frac{\eta}{\pi}\,\widehat\chi(\eta t)\,\cos(\lambda t),
\end{equation}
and hence
\begin{equation}\label{eq:7.g.support}
\supp(g_{\lambda,\eta})\subset[-\eta^{-1},\,\eta^{-1}].
\end{equation}

\begin{remark}[Consistency note]\label{rem:7.g.consistency}
The sign in the exponential in \eqref{eq:7.g.def} is fixed so that \eqref{eq:7.g.explicit} coincides with
Lemma~\ref{lem:6.g.explicit} in Chapter~6. This is the first and most frequent source of $2\pi$-level errors.
\end{remark}

\subsection{Functional calculus lock and wave representation}\label{subsec:ch7-plock}

\noindent\textbf{LOCK-P.}
Define
\begin{equation}\label{eq:7.P.fc}
P_{\lambda,\eta}:=h_{\lambda,\eta}(A).
\end{equation}
Equivalently, with $W(t)=e^{itA}$,
\begin{equation}\label{eq:7.P.wave.W}
P_{\lambda,\eta}
=\frac{1}{2\pi}\int_{\R} e^{-it\lambda}\,\widehat\chi(\eta t)\,W(t)\,dt,
\end{equation}
and taking even part (since $\widehat\chi$ is even),
\begin{equation}\label{eq:7.P.wave.cos}
P_{\lambda,\eta}
=\frac{1}{2\pi}\int_{\R} e^{-it\lambda}\,\widehat\chi(\eta t)\,\cos(tA)\,dt.
\end{equation}

\subsection{The unique bridge constant}\label{subsec:ch7-chi0}

\noindent\textbf{LOCK-$\chi_0$.}
By \eqref{eq:7.chi.def},
\begin{equation}\label{eq:7.chi0}
\int_{\R}\chi(u)\,du=\widehat\chi(0).
\end{equation}
This constant is the unique normalization bridge between:
the microlocal Weyl term (Chapter~5), the identity term (Chapter~6), and the spectral main term (this chapter).
A mismatch in $\widehat\chi(0)$ is a hard convention error.

\section{Spectral resolution and scattering data}\label{sec:ch7-spectral}
% =====================================================================

\subsection{Discrete spectrum}\label{subsec:ch7-discrete}

Let $\{u_j\}$ be an orthonormal basis of $L^2$-eigenfunctions of $\Delta$ with
\begin{equation}\label{eq:7.eigs}
\Delta u_j=\Big(\tfrac14+r_j^2\Big)u_j,\qquad r_j\in\R\cup i(0,\tfrac12),
\end{equation}
listed with multiplicity.

\subsection{Continuous spectrum (Eisenstein series)}\label{subsec:ch7-continuous}

Let $\kappa$ be the number of cusps. For each cusp $\mathfrak a$, let $E_{\mathfrak a}(z,s)$ be the Eisenstein series,
normalized so that the vector $E(z,s)=(E_{\mathfrak a}(z,s))_{\mathfrak a}$ satisfies the functional equation
\begin{equation}\label{eq:7.eisenstein.functional}
E(z,s)=\Phi(s)\,E(z,1-s),
\end{equation}
where $\Phi(s)$ is the $\kappa\times\kappa$ scattering matrix.
Define the scattering determinant
\begin{equation}\label{eq:7.varphi.def}
\varphi(s):=\det\Phi(s).
\end{equation}
One has the standard relations
\begin{equation}\label{eq:7.varphi.sym}
\Phi(s)\Phi(1-s)=I,\qquad \Phi(\overline{s})^\ast=\Phi(s),
\end{equation}
hence
\begin{equation}\label{eq:7.varphi.sym2}
\varphi(s)\varphi(1-s)=1,
\qquad
\overline{\varphi(\tfrac12+ir)}=\varphi(\tfrac12-ir)^{-1}.
\end{equation}
See \cite{Hejhal1983,Iwaniec2002}.

\subsection{Plancherel resolution}\label{subsec:ch7-plancherel}

For $f\in L^2(M)$,
\begin{equation}\label{eq:7.plancherel}
f=\sum_j \langle f,u_j\rangle u_j
+\frac{1}{4\pi}\sum_{\mathfrak a=1}^\kappa \int_{\R}
\langle f,E_{\mathfrak a}(\cdot,\tfrac12+ir)\rangle\,E_{\mathfrak a}(\cdot,\tfrac12+ir)\,dr,
\end{equation}
with convergence in $L^2$. See \cite{Hejhal1983,Iwaniec2002}.

\section{Diagonal action of $P_{\lambda,\eta}$ and the discrete trace}\label{sec:ch7-diagonal}
% =====================================================================

\begin{lemma}[Diagonal action]\label{lem:7.diagonal.action}
The operator $P_{\lambda,\eta}=h_{\lambda,\eta}(A)$ acts by scalars:
\begin{itemize}
\item $P_{\lambda,\eta}u_j=h_{\lambda,\eta}(r_j)\,u_j$ on each $L^2$-eigenspace;
\item $P_{\lambda,\eta}E_{\mathfrak a}(\cdot,\tfrac12+ir)=h_{\lambda,\eta}(r)\,E_{\mathfrak a}(\cdot,\tfrac12+ir)$ on the continuous spectrum.
\end{itemize}
\end{lemma}

\begin{proof}
This is immediate from the functional calculus for the self-adjoint operator $A$. See \cite{Hormander1983,Sogge2014}.
\end{proof}

\begin{proposition}[Discrete spectral contribution]\label{prop:7.discrete.trace}
The discrete contribution to $\Tr^{\mathrm{reg}}(P_{\lambda,\eta})$ equals
\begin{equation}\label{eq:7.S.disc}
\mathcal S^{\mathrm{disc}}_{\lambda,\eta}=\sum_{j} h_{\lambda,\eta}(r_j).
\end{equation}
\end{proposition}

\begin{proof}
On the discrete spectral subspace, $P_{\lambda,\eta}$ has eigenvalues $h_{\lambda,\eta}(r_j)$, and the trace is their sum.
See \cite{Hejhal1983,Iwaniec2002}.
\end{proof}

\section{Regularized trace and the continuous contribution (normalization ledger)}
\label{sec:ch7-regtrace}
% =====================================================================

\noindent
Because $M$ is noncompact, kernel traces require truncation/regularization.
We fix once and for all a standard cusp truncation $M(Y)$ and the associated regularized trace $\Tr^{\mathrm{reg}}$
in the sense of Hejhal~\cite[Vol.~I, Ch.~2]{Hejhal1983} (equivalently Iwaniec~\cite{Iwaniec2002} with compatible cusp scalings).

\subsection{Maass--Selberg input}\label{subsec:ch7-maass-selberg}

For each cusp $\mathfrak a$ and $r\in\R$, the truncated $L^2$ mass satisfies a Maass--Selberg relation of the form
\begin{equation}\label{eq:7.maass.selberg}
\int_{M(Y)} \big|E_{\mathfrak a}(z,\tfrac12+ir)\big|^2\,d\mu(z)
=2\log Y
+\frac{\varphi'(\tfrac12+ir)}{\varphi(\tfrac12+ir)}
+\mathcal U_\Gamma(r)
+O(Y^{-1}),
\end{equation}
where $\mathcal U_\Gamma(r)$ is an explicit term determined by the chosen cusp normalization
(Gamma/digamma factors and cusp constants). See \cite{Hejhal1983,Iwaniec2002}.

\subsection{Ledger functional}\label{subsec:ch7-ledger}

\begin{definition}[Normalization ledger functional]\label{def:7.ledger.NGamma}
Let $\mathcal N_\Gamma[h]$ denote the explicit linear functional of $h$ (equivalently of $g$) which:
(i) removes the divergence $2\log Y$ arising from \eqref{eq:7.maass.selberg} under the fixed truncation convention,
and (ii) records the contribution of $\mathcal U_\Gamma(r)$ and cusp constants in the same convention.
\end{definition}

\begin{remark}[Ledger contents]\label{rem:7.ledger.contents}
In the standard Hejhal normalization, $\mathcal N_\Gamma[h]$ is a finite combination of:
\begin{itemize}
\item $g(0)=\frac{1}{2\pi}\int_\R h(r)\,dr$,
\item integrals $\int_\R h(r)\,\psi(\tfrac12+ir)\,dr$ against the digamma function $\psi=\Gamma'/\Gamma$,
\item cusp-dependent constants determined by the fixed scaling matrices at each cusp.
\end{itemize}
We keep $\mathcal N_\Gamma$ isolated to enforce OBJ-LOCK while making all convention-dependence explicit. See \cite{Hejhal1983}.
\end{remark}

\subsection{Continuous and residual contributions}\label{subsec:ch7-cont-res}

\begin{definition}[Continuous contribution]\label{def:7.S.cont}
Define
\begin{equation}\label{eq:7.S.cont}
\mathcal S^{\mathrm{cont}}_{\lambda,\eta}
:=\frac{1}{4\pi}\int_{\R} h_{\lambda,\eta}(r)\,
\frac{\varphi'(\tfrac12+ir)}{\varphi(\tfrac12+ir)}\,dr
+\mathcal N_\Gamma[h_{\lambda,\eta}].
\end{equation}
\end{definition}

\begin{definition}[Residual contribution]\label{def:7.S.res}
Let $\mathcal S^{\mathrm{res}}_{\lambda,\eta}$ denote the (possibly empty) residual term arising from poles of Eisenstein series
in $s\in(1/2,1]$ (depending on $\Gamma$ and normalization). When present, $\mathcal S^{\mathrm{res}}_{\lambda,\eta}$ is explicit and finite rank.
See \cite{Hejhal1983,Iwaniec2002}.
\end{definition}

\begin{definition}[Full spectral side]\label{def:7.S.full}
Define
\begin{equation}\label{eq:7.S.full}
\mathcal S_{\lambda,\eta}
:=\mathcal S^{\mathrm{disc}}_{\lambda,\eta}
+\mathcal S^{\mathrm{cont}}_{\lambda,\eta}
+\mathcal S^{\mathrm{res}}_{\lambda,\eta}.
\end{equation}
With the fixed truncation convention, $\mathcal S_{\lambda,\eta}=\Tr^{\mathrm{reg}}(P_{\lambda,\eta})$.
\end{definition}

\section{Exact comparison with the geometric side and the Weyl bridge}\label{sec:ch7-comparison}
% =====================================================================

Let $\mathcal G_{\lambda,\eta}$ be the geometric side from Chapter~6:
\begin{equation}\label{eq:7.G.recall}
\mathcal G_{\lambda,\eta}
=I_{\lambda,\eta}+H_{\lambda,\eta}+P_{\lambda,\eta}^{\mathrm{para}}+E_{\lambda,\eta}^{\mathrm{ell}},
\end{equation}
with the same cusp ledger convention bundled into $P_{\lambda,\eta}^{\mathrm{para}}$.

\begin{theorem}[Localized Selberg trace formula (exact identity; OBJ-LOCK)]\label{thm:7.selberg.exact}
For every cofinite $\Gamma$ and every $(\lambda,\eta)$,
\begin{equation}\label{eq:7.selberg.exact}
\mathcal S_{\lambda,\eta}=\mathcal G_{\lambda,\eta},
\end{equation}
where $\mathcal S_{\lambda,\eta}$ is given by Definitions~\ref{def:7.S.cont}--\ref{def:7.S.full} and
$\mathcal G_{\lambda,\eta}$ by \eqref{eq:7.G.recall}, with matching cusp ledgers under the fixed truncation convention.
\end{theorem}

\begin{proof}
This is the cofinite Selberg trace formula applied to $h_{\lambda,\eta}$ with standard regularization and identical cusp normalization
on both sides. See \cite{Selberg1956,Hejhal1983,Iwaniec2002}.
\end{proof}

\begin{remark}[Ledger is not optional]\label{rem:7.ledger.not.optional}
Any change in cusp scaling matrices or truncation convention changes $\mathcal N_\Gamma[h]$ and the corresponding geometric
parabolic ledger. Such changes must be recorded; otherwise \eqref{eq:7.selberg.exact} is not an identity of the same object.
\end{remark}

\begin{theorem}[Localized Weyl main term (bridge by $\widehat\chi(0)$)]\label{thm:7.weyl.main}
Assume \eqref{eq:7.eta.regime}. Then, as $\lambda\to\infty$,
\begin{equation}\label{eq:7.weyl.main}
\mathcal S_{\lambda,\eta}
=\frac{\vol(M)}{2\pi}\,\widehat\chi(0)\,\lambda\eta
+H_{\lambda,\eta}
+\mathcal R^{\mathrm{cusp}}_{\lambda,\eta}
+\mathcal R^{\mathrm{ell}}_{\lambda,\eta}
+O_{\widehat\chi}(\eta^2)
+O\!\big(e^{-2\pi\lambda}\lambda\eta\big),
\end{equation}
where $\mathcal R^{\mathrm{cusp}}_{\lambda,\eta}$ collects cusp-dependent lower-order terms (including the ledger $\mathcal N_\Gamma[h_{\lambda,\eta}]$
and any residual contribution), and $\mathcal R^{\mathrm{ell}}_{\lambda,\eta}$ collects elliptic terms when present.
\end{theorem}

\begin{proof}
By Theorem~\ref{thm:7.selberg.exact}, $\mathcal S_{\lambda,\eta}=\mathcal G_{\lambda,\eta}$.
Insert the Weyl extraction of Chapter~6 (Lemma~\ref{lem:6.identity.asymp} and Proposition~\ref{prop:6.main.term})
and use the shared bridge constant $\widehat\chi(0)$ from \eqref{eq:7.chi0}.
\end{proof}

\section{Growth control for $\varphi'/\varphi$ and admissible $\theta_0$}\label{sec:ch7-growth}
% =====================================================================

\begin{lemma}[Polynomial growth of $\varphi'/\varphi$]\label{lem:7.phi.growth}
There exists $C_\Gamma>0$ such that
\[
\frac{\varphi'(\tfrac12+ir)}{\varphi(\tfrac12+ir)}\ll_\Gamma (1+|r|)^{C_\Gamma},
\qquad r\in\R.
\]
\end{lemma}

\begin{proof}
Standard analytic bounds for cofinite scattering determinants. See \cite{Hejhal1983,Iwaniec2002}.
\end{proof}

\begin{proposition}[Size bound for the continuous side under localization]\label{prop:7.cont.size}
Uniformly for $\eta$ satisfying \eqref{eq:7.eta.regime},
\[
\mathcal S^{\mathrm{cont}}_{\lambda,\eta}
\ll_{\Gamma,\widehat\chi}
\eta(1+\lambda)^{C_\Gamma}+|\mathcal N_\Gamma[h_{\lambda,\eta}]|.
\]
In particular, if $\theta<1/C_\Gamma$, then $\mathcal S^{\mathrm{cont}}_{\lambda,\eta}=o(\lambda\eta)$ as $\lambda\to\infty$.
\end{proposition}

\begin{proof}
The window $h_{\lambda,\eta}$ is supported on $r=\pm\lambda+O(\eta)$ and satisfies
$\int_\R |h_{\lambda,\eta}(r)|\,dr\asymp_{\widehat\chi}\eta$.
Combine with Lemma~\ref{lem:7.phi.growth}. See \cite{Hejhal1983}.
\end{proof}

\section{Selberg zeta: the intrinsic ``geodesics $\leftrightarrow$ spectrum'' portal}
\label{sec:ch7-selbergzeta}
% =====================================================================

\begin{proposition}[Log-derivative of Selberg zeta]\label{prop:7.ZGamma.logder}
Let $Z_\Gamma(s)$ be the Selberg zeta function attached to $\Gamma$. For $\Re s>1$,
\[
\frac{Z_\Gamma'(s)}{Z_\Gamma(s)}
=\sum_{[\gamma_0]\in\mathcal P}\sum_{k\ge1}
\frac{L(\gamma_0)\,e^{-ksL(\gamma_0)}}{2\sinh(\tfrac{kL(\gamma_0)}{2})},
\]
and $Z_\Gamma(s)$ continues meromorphically to $\C$, with zeros encoding the discrete spectrum and resonances.
See \cite{Hejhal1983,Iwaniec2002}.
\end{proposition}

\begin{remark}[Meaning for the localized trace formula]\label{rem:7.ZGamma.meaning}
The hyperbolic term $H_{\lambda,\eta}$ from Chapter~6 is a windowed prime-geodesic expansion of $\frac{Z_\Gamma'}{Z_\Gamma}$,
with the cutoff $g_{\lambda,\eta}(kL)$ enforcing $kL\le \eta^{-1}$.
\end{remark}

\section{Where the Riemann zeta function enters (arithmetic groups)}\label{sec:ch7-zeta}
% =====================================================================

\begin{example}[Modular surface]\label{ex:7.modular.surface}
For $\Gamma=\PSL_2(\Z)$ one has
\[
\varphi(s)=\frac{\xi(2s-1)}{\xi(2s)},\qquad
\xi(s)=\pi^{-s/2}\Gamma\!\Big(\frac{s}{2}\Big)\zeta(s),
\]
hence
\[
\frac{\varphi'(s)}{\varphi(s)}
=2\frac{\xi'(2s-1)}{\xi(2s-1)}-2\frac{\xi'(2s)}{\xi(2s)}.
\]
Substituting $s=\tfrac12+ir$ into \eqref{eq:7.S.cont} shows that the continuous contribution contains
$\zeta'/\zeta(1+2ir)$ and $\zeta'/\zeta(2ir)$, together with explicit Gamma-factor derivatives.
See \cite{Hejhal1983,Iwaniec2002}.
\end{example}

\begin{remark}[Why $\zeta$ is forced]\label{rem:7.zeta.forced}
In the noncompact case, the continuous spectrum is governed by scattering; scattering is governed by $\varphi'/\varphi$;
and for arithmetic $\Gamma$ the scattering determinant is an explicit ratio of completed $L$-functions. Thus $\zeta$ enters
structurally in the trace identity.
\end{remark}

\section{CONTROL module: microlocal anti-trace and orbit reading}\label{sec:ch7-antitrace}
% =====================================================================

\subsection{Time reversal and odd microlocal weights}\label{subsec:ch7-time-reversal}

Let $S^\ast M$ be the cosphere bundle. Define the time-reversal involution
\[
\mathcal R:S^\ast M\to S^\ast M,\qquad \mathcal R(x,\xi)=(x,-\xi).
\]

\begin{definition}[Odd microlocal cutoff]\label{def:7.psi.odd}
Let $\Psi\in\Psi^0_{\mathrm{c}}(M)$ be properly supported of order $0$, chosen (if needed) to be supported away from cusp infinity
to ensure that $\Psi P_{\lambda,\eta}\Psi$ admits a regularized trace in the fixed convention.
Let $\psi\in C_c^\infty(S^\ast M)$ be its principal symbol and assume the \emph{oddness condition}
\begin{equation}\label{eq:7.psi.odd}
\psi\circ\mathcal R=-\psi.
\end{equation}
\end{definition}

\subsection{Anti-trace observable and Weyl cancellation}\label{subsec:ch7-antitrace-def}

\begin{definition}[Anti-trace observable]\label{def:7.antitrace}
Define the (regularized) anti-trace by
\begin{equation}\label{eq:7.antitrace.def}
\mathcal A_{\lambda,\eta}(\Psi):=\Tr^{\mathrm{reg}}(\Psi P_{\lambda,\eta}\Psi).
\end{equation}
\end{definition}

\begin{lemma}[Weyl cancellation for time-reversal odd weights]\label{lem:7.weyl.cancellation}
Assume \eqref{eq:7.psi.odd}. Then the leading Weyl term in $\mathcal A_{\lambda,\eta}(\Psi)$ vanishes.
More precisely, the semiclassical trace expansion of $\Tr^{\mathrm{reg}}(\Psi P_{\lambda,\eta}\Psi)$ has zero principal (phase-space)
term, so that $\mathcal A_{\lambda,\eta}(\Psi)$ is governed by oscillatory orbit contributions and lower-order microlocal remainders.
\end{lemma}

\begin{proof}[Proof sketch]
The leading trace term is obtained by integrating the principal symbol of the operator over $S^\ast M$ against Liouville measure.
Liouville measure is invariant under $\mathcal R$, while the principal contribution is linear in the (time-reversal odd) component induced
by \eqref{eq:7.psi.odd}; hence the principal term cancels. This is a standard symmetry consequence of microlocal trace expansions.
See \cite{DuistermaatGuillemin1975,Hormander1983,Sogge2014}.
\end{proof}

\begin{remark}[Control, not substitution]\label{rem:7.antitrace.objlock}
The anti-trace $\mathcal A_{\lambda,\eta}(\Psi)$ is a \emph{new declared observable} built from the same projector $P_{\lambda,\eta}$.
It does not replace $\Tr^{\mathrm{reg}}(P_{\lambda,\eta})$; it is admissible under OBJ-LOCK precisely because the insertion of $\Psi$
is explicitly recorded in \eqref{eq:7.antitrace.def}.
\end{remark}

\subsection{Orbit reading principle}\label{subsec:ch7-orbit-reading}

\begin{proposition}[Orbit reading principle]\label{prop:7.orbit.reading}
Assume $\Psi$ is as in Definition~\ref{def:7.psi.odd}. Then the hyperbolic contribution to $\mathcal A_{\lambda,\eta}(\Psi)$ is a weighted
sum over closed geodesics in which the weight is an orbit integral of the principal symbol transported by the geodesic flow.
In particular, $\mathcal A_{\lambda,\eta}(\Psi)$ distinguishes oriented orbit information that is invisible to the raw trace
$\Tr^{\mathrm{reg}}(P_{\lambda,\eta})$.
\end{proposition}

\begin{proof}[Proof sketch]
This is the standard microlocal refinement of trace formulas: insertion of $\Psi$ modifies each periodic-orbit contribution by the principal
symbol integrated along the orbit. The oddness \eqref{eq:7.psi.odd} kills the time-reversal even component, leaving an oriented/signed sensitivity.
See \cite{DuistermaatGuillemin1975,DyatlovZworski2019}.
\end{proof}

\medskip
\noindent\textbf{Typical failure modes.}
\begin{itemize}
\item[(F7.1)] Wrong Fourier sign or missing $2\pi$ in \eqref{eq:7.g.def} $\Rightarrow$ wrong length cutoff and hyperbolic weights.
\item[(F7.2)] Omitting $\mathcal N_\Gamma[h]$ $\Rightarrow$ divergence not cancelled; object not defined.
\item[(F7.3)] Mixing cusp scalings (Hejhal vs.\ Iwaniec without conversion) $\Rightarrow$ constant drift in continuous/parabolic terms.
\item[(F7.4)] Treating wave-trace singularities as Selberg hyperbolic weights $\Rightarrow$ object substitution.
\end{itemize}
