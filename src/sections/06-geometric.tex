\chapter{Geometric expansion of the localized Selberg trace formula}
\label{ch:selberg-geometric-localized}

\section*{Orientation, object lock, and conventions}\label{sec:ch6-orientation}

\noindent\textbf{Scope.}
Let $M=\Gamma\backslash\H$ be a finite-area hyperbolic surface with $\Gamma\subset\PSL_2(\R)$ discrete and cofinite,
and let $\Delta\ge 0$ be the nonnegative Laplace--Beltrami operator on $M$.
This chapter constructs the \emph{geometric side} of the Selberg trace formula for a \emph{localized} spectral test
function. The output is an explicit decomposition into identity, hyperbolic (closed geodesic), parabolic
(cusp/scattering), and elliptic (torsion) contributions, together with a clean extraction of the localized Weyl
main term and a strict bookkeeping of the constants to be matched with the spectral side in Chapter~7.

\medskip
\noindent\textbf{Reference model.}
We use the cofinite Selberg trace formula in the normalization of
Selberg~\cite{Selberg1956}, Hejhal~\cite[Vol.~I, Ch.~2]{Hejhal1983}, and Iwaniec~\cite{Iwaniec2002}.
All Fourier-transform conventions are fixed below and remain unchanged for the rest of the monograph.

% ---------------------------------------------------------------------
\section{Localized test functions and the Selberg transform}
\label{sec:ch6-test-selberg}

\subsection{Fourier normalization and Paley--Wiener cutoff}\label{subsec:ch6-fourier}

Fix an even cutoff $\widehat\chi\in C_c^\infty(\R)$ with
\begin{equation}\label{eq:6.chi.hat.lock}
\widehat\chi(-t)=\widehat\chi(t),\qquad
\supp(\widehat\chi)\subset(-1,1),\qquad
\widehat\chi(0)>0.
\end{equation}
Define the inverse Fourier transform by
\begin{equation}\label{eq:6.chi.def}
\chi(u):=\frac{1}{2\pi}\int_{\R}\widehat\chi(t)e^{itu}\,dt.
\end{equation}
Then $\chi$ is even, entire, and Schwartz on $\R$ (Paley--Wiener).

\subsection{Localized spectral window}\label{subsec:ch6-window}

Let $\lambda\to\infty$ and $\eta=\eta(\lambda)\in(0,1]$. Define the even test function
\begin{equation}\label{eq:6.h.window}
h_{\lambda,\eta}(r):=
\chi\!\left(\frac{r-\lambda}{\eta}\right)+
\chi\!\left(\frac{r+\lambda}{\eta}\right),
\qquad r\in\R.
\end{equation}
We impose the localization regime
\begin{equation}\label{eq:6.eta.regime}
\eta\ge \lambda^{-\theta},\qquad 0<\theta<\theta_0,
\end{equation}
where $\theta_0>0$ will be fixed once and for all at the point where a power-saving remainder is extracted.

\subsection{Selberg transform and explicit $g$--kernel}\label{subsec:ch6-selberg-transform}

For even $h$, the Selberg trace formula uses the associated kernel $g$ defined by the Fourier pair
\begin{equation}\label{eq:6.g.def}
g_{\lambda,\eta}(u):=\frac{1}{2\pi}\int_{\R} h_{\lambda,\eta}(r)\,e^{-iru}\,dr,
\qquad u\in\R.
\end{equation}

\begin{lemma}[Explicit localized $g$--kernel]\label{lem:6.g.explicit}
With $h_{\lambda,\eta}$ as in \eqref{eq:6.h.window} and $g_{\lambda,\eta}$ as in \eqref{eq:6.g.def}, one has
\begin{equation}\label{eq:6.g.explicit}
g_{\lambda,\eta}(u)=\frac{\eta}{\pi}\,\widehat\chi(\eta u)\,\cos(\lambda u).
\end{equation}
In particular,
\begin{equation}\label{eq:6.g.support}
\supp(g_{\lambda,\eta})\subset[-\eta^{-1},\,\eta^{-1}].
\end{equation}
\end{lemma}

\begin{proof}
Write $h_{\lambda,\eta}=h_++h_-$ with
$h_+(r)=\chi\!\big(\frac{r-\lambda}{\eta}\big)$ and $h_-(r)=\chi\!\big(\frac{r+\lambda}{\eta}\big)$.
For $h_+$ set $r=\lambda+\eta v$ and use \eqref{eq:6.chi.def}; the $h_-$ term produces the conjugate contribution.
Evenness of $\widehat\chi$ yields \eqref{eq:6.g.explicit}. The support statement follows from
$\supp(\widehat\chi)\subset(-1,1)$.
\end{proof}

\subsection{The associated functional-calculus operator}\label{subsec:ch6-projector}

Let $A=\sqrt{\Delta-\tfrac14}$ and define
\begin{equation}\label{eq:6.projector.fc}
P_{\lambda,\eta}:=h_{\lambda,\eta}(A)=h_{\lambda,\eta}\!\left(\sqrt{\Delta-\tfrac14}\right).
\end{equation}
Equivalently, by Fourier inversion and the even wave group $U(t)=\cos(tA)$,
\begin{equation}\label{eq:6.projector.wave}
P_{\lambda,\eta}
=\frac{1}{2\pi}\int_{\R} e^{-it\lambda}\,\widehat\chi(\eta t)\,
\cos\!\big(t\sqrt{\Delta-\tfrac14}\big)\,dt,
\end{equation}
which is the geometric counterpart of the microlocal wave representation used in Chapter~5.

\section{Identity contribution and Weyl extraction}
\label{sec:ch6-identity}

\begin{lemma}[Identity term (exact form)]\label{lem:6.identity.exact}
The identity contribution on the geometric side equals
\begin{equation}\label{eq:6.identity.exact}
I_{\lambda,\eta}
=\frac{\vol(M)}{4\pi}\int_{\R} r\,\tanh(\pi r)\,h_{\lambda,\eta}(r)\,dr.
\end{equation}
\end{lemma}

\begin{proof}
This is the standard identity orbital integral in the cofinite Selberg trace formula for the pair $(h_{\lambda,\eta},g_{\lambda,\eta})$;
see \cite{Selberg1956,Hejhal1983}.
\end{proof}

\begin{lemma}[Localized Weyl main term]\label{lem:6.identity.asymp}
As $\lambda\to\infty$, uniformly for $\eta$ satisfying \eqref{eq:6.eta.regime}, one has
\begin{equation}\label{eq:6.identity.asymp}
I_{\lambda,\eta}
=\frac{\vol(M)}{2\pi}\,\widehat\chi(0)\,\lambda\eta
\;+\;O_{\widehat\chi}(\eta^2)
\;+\;O\!\big(e^{-2\pi\lambda}\,\lambda\eta\big).
\end{equation}
\end{lemma}

\begin{proof}
Insert \eqref{eq:6.h.window} into \eqref{eq:6.identity.exact}.
On the support of $h_{\lambda,\eta}$ one has $\tanh(\pi r)=\sgn(r)+O(e^{-2\pi\lambda})$, hence
$r\tanh(\pi r)=|r|+O(e^{-2\pi\lambda}|r|)$ and
\[
I_{\lambda,\eta}
=\frac{\vol(M)}{4\pi}\int_\R |r|\,h_{\lambda,\eta}(r)\,dr
+O\!\big(e^{-2\pi\lambda}\,\lambda\eta\big).
\]
Split $h_{\lambda,\eta}=h_++h_-$ as in Lemma~\ref{lem:6.g.explicit}.
In the $h_+$ integral set $r=\lambda+\eta u$ and in the $h_-$ integral set $r=-\lambda+\eta u$.
Since $\chi$ is Schwartz, the region where $|\lambda+\eta u|$ changes sign contributes
$O_{\widehat\chi,N}(\eta^N\lambda^{1-N})$ for any $N$, hence negligible.
On the remaining region the linear $\eta u$ corrections cancel between the $h_+$ and $h_-$ contributions,
and the first surviving correction is $O_{\widehat\chi}(\eta^2)$.
Finally, $\int_\R \chi(u)\,du=\widehat\chi(0)$ by \eqref{eq:6.chi.def}, yielding \eqref{eq:6.identity.asymp}.
\end{proof}

\begin{remark}[Normalization lock]\label{rem:6.norm.lock}
The coefficient $\widehat\chi(0)$ is the unique normalization parameter that must match the leading term on the
spectral/microlocal side in Chapter~7. Any discrepancy indicates an inconsistency in Fourier conventions or in the
definition of the localized window.
\end{remark}

\section{Hyperbolic contribution (closed geodesics)}
\label{sec:ch6-hyperbolic}

Let $\mathcal P$ denote the set of primitive hyperbolic conjugacy classes in $\Gamma$.
For $[\gamma_0]\in\mathcal P$ let $L(\gamma_0)>0$ denote the primitive length, and write $\gamma=\gamma_0^k$, $k\ge 1$.

\begin{lemma}[Hyperbolic term (exact Selberg form)]\label{lem:6.hyperbolic.exact}
The hyperbolic contribution equals
\begin{equation}\label{eq:6.hyperbolic.exact}
H_{\lambda,\eta}
=\sum_{[\gamma_0]\in\mathcal P}\ \sum_{k=1}^{\infty}
\frac{L(\gamma_0)}{2\sinh\!\big(\tfrac{kL(\gamma_0)}{2}\big)}\;
g_{\lambda,\eta}\!\big(kL(\gamma_0)\big),
\end{equation}
where $g_{\lambda,\eta}$ is given by \eqref{eq:6.g.explicit}.
\end{lemma}

\begin{proof}
This is the standard evaluation of hyperbolic orbital integrals in the cofinite Selberg trace formula; see
\cite{Selberg1956,Hejhal1983}.
\end{proof}

\begin{corollary}[Exact length cutoff]\label{cor:6.length.cutoff}
By \eqref{eq:6.g.support}, one has $g_{\lambda,\eta}(u)=0$ for $|u|>\eta^{-1}$. Hence only those pairs
$([\gamma_0],k)$ with
\[
kL(\gamma_0)\le \eta^{-1}
\]
contribute to \eqref{eq:6.hyperbolic.exact}. In particular, for each fixed $(\lambda,\eta)$ the double sum is finite.
\end{corollary}

\begin{lemma}[Crude uniform bound]\label{lem:6.hyperbolic.bound}
For fixed $\widehat\chi$ one has
\begin{equation}\label{eq:6.hyperbolic.bound}
|H_{\lambda,\eta}|
\le C_{\widehat\chi}\,\eta
\sum_{[\gamma_0]\in\mathcal P}\ \sum_{\substack{k\ge1\\ kL(\gamma_0)\le \eta^{-1}}}
\frac{L(\gamma_0)}{2\sinh\!\big(\tfrac{kL(\gamma_0)}{2}\big)}.
\end{equation}
In particular, all dependence on $\lambda$ is oscillatory through the factor $\cos(\lambda\cdot)$ in \eqref{eq:6.g.explicit}.
\end{lemma}

\begin{proof}
Combine \eqref{eq:6.hyperbolic.exact} with \eqref{eq:6.g.explicit} and the bound
$|\widehat\chi(\eta u)|\le \|\widehat\chi\|_\infty$.
\end{proof}

\section{Parabolic contribution (cusps) and regularized functional}
\label{sec:ch6-parabolic}

Assume $M$ has $\kappa\ge 1$ cusps. Let $\Phi(s)$ denote the $\kappa\times\kappa$ scattering matrix in the fixed
Eisenstein normalization, and let $\varphi(s):=\det\Phi(s)$ be the scattering determinant.
The parabolic contribution is defined via truncation and the Maass--Selberg relations and can be written as an explicit
linear functional of $h$; see \cite{Hejhal1983,Iwaniec2002}.

\begin{definition}[Parabolic functional]\label{def:6.parabolic.functional}
Let $P_\Gamma[h]$ denote the (standard) parabolic contribution in the cofinite Selberg trace formula for $\Gamma$,
defined by truncation at height $Y$, subtraction of the logarithmic divergence, and passage to the limit $Y\to\infty$.
For the localized window we set
\begin{equation}\label{eq:6.parabolic.def}
P_{\lambda,\eta}^{\mathrm{para}}:=P_\Gamma[h_{\lambda,\eta}].
\end{equation}
\end{definition}

\begin{remark}[Structure and normalization dependence]\label{rem:6.parabolic.structure}
The functional $P_\Gamma[h]$ is explicit and consists of:
\begin{itemize}
\item[(i)] a term involving $\varphi'/\varphi$ on $\Re s=\tfrac12$,
\item[(ii)] a finite set of universal terms determined by the cusp normalization (Gamma/digamma factors and logarithmic constants),
\item[(iii)] a term proportional to $g(0)$ (equivalently, $\int h$) reflecting the subtraction convention.
\end{itemize}
Since the constants in (ii)--(iii) depend on the truncation convention, we keep them bundled into $P_\Gamma[h]$ and
fix the convention once and for all; Chapter~7 uses the identical convention on the spectral side.
\end{remark}

\begin{lemma}[Polynomial growth of the scattering logarithmic derivative]\label{lem:6.scattering.growth}
There exists $C_\Gamma>0$ such that
\[
\frac{\varphi'(\tfrac12+ir)}{\varphi(\tfrac12+ir)}\ll_\Gamma (1+|r|)^{C_\Gamma},
\qquad r\in\R.
\]
\end{lemma}

\begin{proof}
This is standard for cofinite scattering determinants; see \cite{Hejhal1983,Iwaniec2002}.
\end{proof}

\begin{proposition}[Size bound under localization]\label{prop:6.parabolic.bound}
Uniformly for $\eta$ satisfying \eqref{eq:6.eta.regime}, one has
\begin{equation}\label{eq:6.parabolic.bound}
P_{\lambda,\eta}^{\mathrm{para}}
\ll_{\Gamma,\widehat\chi}
\eta\,(1+\lambda)^{C_\Gamma}
\;+\;\big|P_\Gamma^{\mathrm{norm}}[h_{\lambda,\eta}]\big|,
\end{equation}
where $P_\Gamma^{\mathrm{norm}}[h]$ denotes the explicit cusp-normalization constant terms (fixed once and for all).
In particular, if $\theta<1/C_\Gamma$, then $P_{\lambda,\eta}^{\mathrm{para}}=o(\lambda\eta)$ as $\lambda\to\infty$.
\end{proposition}

\begin{proof}
The window $h_{\lambda,\eta}$ is supported on $r=\pm\lambda+O(\eta)$ and satisfies
$\int_\R |h_{\lambda,\eta}(r)|\,dr\asymp_{\widehat\chi}\eta$.
Combine this with Lemma~\ref{lem:6.scattering.growth} and the fact that the remaining normalization terms in
$P_\Gamma[h]$ are explicit linear functionals of $h$ whose coefficients are independent of $\lambda$ and at most
polynomial in $\lambda$.
\end{proof}

\section{Elliptic contribution (torsion)}
\label{sec:ch6-elliptic}

If $\Gamma$ is torsion-free, there are no elliptic conjugacy classes and the elliptic contribution vanishes.
Otherwise, let $\{R\}$ denote elliptic conjugacy classes and let $m_R\ge 2$ denote their orders.

\begin{lemma}[Elliptic term (standard Selberg form)]\label{lem:6.elliptic.form}
The elliptic contribution equals a finite explicit linear functional of $h_{\lambda,\eta}$:
\begin{equation}\label{eq:6.elliptic.form}
E_{\lambda,\eta}^{\mathrm{ell}}
=\sum_{\{R\}} \mathcal E_R[h_{\lambda,\eta}],
\end{equation}
where each $\mathcal E_R$ depends only on $m_R$ and the fixed normalization of the trace formula.
\end{lemma}

\begin{proof}
Standard in the Selberg trace formula with torsion; see \cite[Vol.~I, Ch.~2]{Hejhal1983}.
\end{proof}

\begin{remark}[Lower-order nature]\label{rem:6.elliptic.lower.order}
Elliptic terms are compactly supported (no long-length growth) and do not contribute to the Weyl main term.
Once the convention for $\mathcal E_R$ is fixed, they may be absorbed into the global remainder bookkeeping.
\end{remark}

\section{Assembly of the geometric side and forward link}
\label{sec:ch6-assembly}

\begin{definition}[Full geometric side]\label{def:6.G.full}
Define
\begin{equation}\label{eq:6.G.full}
\mathcal G_{\lambda,\eta}
:=I_{\lambda,\eta}+H_{\lambda,\eta}+P_{\lambda,\eta}^{\mathrm{para}}+E_{\lambda,\eta}^{\mathrm{ell}}.
\end{equation}
\end{definition}

\begin{theorem}[Geometric side (Selberg decomposition)]\label{thm:6.selberg.geometric}
For any cofinite $\Gamma$ and any $(\lambda,\eta)$, the quantity $\mathcal G_{\lambda,\eta}$ equals the geometric side
of the Selberg trace formula associated with the test function $h_{\lambda,\eta}$, i.e.\ the sum of:
\begin{itemize}
\item the identity term \eqref{eq:6.identity.exact},
\item the hyperbolic term \eqref{eq:6.hyperbolic.exact} with $g_{\lambda,\eta}$ given by \eqref{eq:6.g.explicit},
\item the parabolic term $P_\Gamma[h_{\lambda,\eta}]$ as in Definition~\ref{def:6.parabolic.functional},
\item and, if present, the elliptic term \eqref{eq:6.elliptic.form}.
\end{itemize}
\end{theorem}

\begin{proof}
This is precisely the cofinite Selberg trace formula with the Fourier convention \eqref{eq:6.g.def}; see
\cite{Selberg1956,Hejhal1983,Iwaniec2002}.
\end{proof}

\begin{proposition}[Weyl main term on the geometric side]\label{prop:6.main.term}
Assume \eqref{eq:6.eta.regime}. Then, as $\lambda\to\infty$,
\begin{equation}\label{eq:6.main.term}
\mathcal G_{\lambda,\eta}
=\frac{\vol(M)}{2\pi}\,\widehat\chi(0)\,\lambda\eta
\;+\;H_{\lambda,\eta}
\;+\;P_{\lambda,\eta}^{\mathrm{para}}
\;+\;E_{\lambda,\eta}^{\mathrm{ell}}
\;+\;O_{\widehat\chi}(\eta^2)
\;+\;O\!\big(e^{-2\pi\lambda}\,\lambda\eta\big).
\end{equation}
\end{proposition}

\begin{proof}
Insert Lemma~\ref{lem:6.identity.asymp} into \eqref{eq:6.G.full} and collect the remaining contributions.
\end{proof}

\medskip
\noindent\textbf{Forward link to Chapter 7.}
Chapter~7 writes the spectral side of $\Tr(P_{\lambda,\eta})$ (discrete plus continuous spectrum) in the same Fourier
convention and matches its leading constant $\widehat\chi(0)$ with \eqref{eq:6.main.term}. The hyperbolic sum
\eqref{eq:6.hyperbolic.exact} becomes the oscillatory geometric component against which the localized spectral
remainder will be compared.

\subsection*{Audit (Geometric certificate)}\label{audit:ch6}
\begin{itemize}
\item[(G1)] Fourier lock: \eqref{eq:6.chi.def} and \eqref{eq:6.g.def} are the only transforms used.
\item[(G2)] Window lock: $h_{\lambda,\eta}$ fixed by \eqref{eq:6.h.window}, regime by \eqref{eq:6.eta.regime}.
\item[(G3)] Weyl constant lock: $\widehat\chi(0)$ extracted in Lemma~\ref{lem:6.identity.asymp} and declared decisive in
Remark~\ref{rem:6.norm.lock}.
\item[(G4)] Hyperbolic cutoff: finiteness is a \emph{support} consequence of \eqref{eq:6.g.support}.
\item[(G5)] Parabolic convention: bundled into $P_\Gamma[h]$ with fixed truncation normalization (Definition~\ref{def:6.parabolic.functional}).
\item[(G6)] Torsion safety: elliptic contribution is explicit finite functional and lower order.
\end{itemize}
