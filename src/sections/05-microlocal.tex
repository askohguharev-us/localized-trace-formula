\subsection{Standing conventions and the semiclassical wave group}\label{subsec:microlocal-conv}

\noindent\textbf{Geometry and Laplacian.}
Let $M=\Gamma\backslash\Hh$ be a finite-area hyperbolic surface (possibly with cusps),
with hyperbolic metric $ds^{2}=y^{-2}(dx^{2}+dy^{2})$ and Laplacian $\Delta\ge 0$ normalized as in Chapter~2.
Set the self-adjoint operator
\[
P\;:=\;\sqrt{\Delta-\tfrac14}\quad\text{on }L^{2}(M).
\]

\medskip
\noindent\textbf{Semiclassical parameter and the correct propagator.}
We work at frequency $\lambda\to+\infty$ with
\[
h=\lambda^{-1}\downarrow 0.
\]
The \emph{semiclassical wave group} is
\begin{equation}\label{eq:Uh-def}
U_h(t)\;:=\;\exp\!\Big(\frac{i t}{h}P\Big)\qquad(t\in\R).
\end{equation}
Equivalently, if $U(t)=e^{itP}$ denotes the physical-time group, then $U_h(t)=U(t/h)$.
This scaling is essential: it is exactly what makes the microlocal phases appear as $e^{i(\cdot)/h}$.

\medskip
\noindent\textbf{Cusp truncation as OBJ-LOCK.}
On noncompact $M$, we insert a fixed smooth cutoff $\Lambda_Y$ (Chapter~2) and work with
\[
U_{h,Y}(t):=\Lambda_Y\,U_h(t)\,\Lambda_Y,
\]
postponing $Y\to\infty$ to the global trace step (Chapter~7).
This prevents hidden renormalizations: all trace statements are made for $U_{h,Y}(t)$.

\subsection{Semiclassical parametrix up to Ehrenfest time}\label{subsec:wave-parametrix}

\noindent\textbf{Ehrenfest time.}
Let
\begin{equation}\label{eq:Ehrenfest-time}
T_E(h)\;=\;c_E\log(1/h),
\end{equation}
where $c_E>0$ is chosen \emph{strictly below the Egorov/derivative-loss threshold}
so that exponential derivative growth of the geodesic flow is compensated by powers of $h$
(see \Cref{thm:egorov-log}).

\medskip
\noindent\textbf{Orbit-aware log window (the key upgrade).}
A blind cutoff $|t|\le T_E(h)$ may \emph{slice periodic packets}.
We therefore introduce a smooth \emph{orbit-aware time window}:
there exists a smooth function
\[
T_{\mathrm{orb}}(\rho;h)\in(0,T_E(h)],\qquad \rho\in S^{*}M,
\]
such that:
\begin{itemize}
\item $T_{\mathrm{orb}}(\rho;h)=T_E(h)$ on the aperiodic region (away from short periodic tubes);
\item on a $\delta$-neighborhood of periodic geodesics of length $\le T_E(h)$,
$T_{\mathrm{orb}}(\rho;h)$ is reduced in a smooth way so that the microlocal packet supported near a periodic orbit
is either \emph{fully propagated} through one return (Poincaré) cycle or is \emph{excluded coherently} (no slicing).
\end{itemize}

Concretely, we fix a smooth partition of unity on $S^{*}M$:
\[
1=\chi_{\mathrm{per}}+\chi_{\mathrm{aper}},
\]
where $\chi_{\mathrm{per}}$ is supported in a finite union of small tubular neighborhoods of periodic orbits
up to length $T_E(h)$, and $\chi_{\mathrm{aper}}$ is supported on the complement.
All subsequent $t$-integrals are understood with the orbit-aware restriction:
\[
|t|\le T_{\mathrm{orb}}(\rho;h)\quad\text{on the microlocal support at }\rho.
\]
This is exactly the mechanism that prevents “log cutting orbits”.

\medskip
\noindent\textbf{Local model on the universal cover.}
On $\Hh$, $U_h(t)$ is a semiclassical Fourier integral operator associated with the geodesic flow.
For $|t|$ small and $r=d(z,w)$ one has a Hadamard/FIO parametrix
\begin{equation}\label{eq:hadamard-small-time-fixed}
U_{h,\Hh}(t;z,w)
=\frac{1}{2\pi h}\Big(e^{\frac{i}{h}(r-t)}\,b_{+}(z,w,t;h)\;+\;e^{\frac{i}{h}(-r-t)}\,b_{-}(z,w,t;h)\Big),
\end{equation}
with classical amplitudes $b_\pm\sim\sum_{j\ge0}h^j b_{\pm,j}$ determined by transport
and normalized by $b_{\pm,0}(z,z,0)=1$.

\medskip
\noindent\textbf{Parametrix on $M$ up to Ehrenfest time.}

\begin{theorem}[Semiclassical parametrix up to Ehrenfest time]\label{thm:parametrix-logtime}
There exist $c_E>0$ and classical amplitudes $a_{\pm}(z,w,t;h)\sim\sum_{j\ge0}h^j a_{\pm,j}$
such that for all $|t|\le T_E(h)=c_E\log(1/h)$ one has
\begin{equation}\label{eq:parametrix-log-fixed}
U_h(t;z,w)
=\frac{1}{2\pi h}\Big(e^{\frac{i}{h}(d(z,w)-t)}\,a_{+}(z,w,t;h)+
e^{\frac{i}{h}(-d(z,w)-t)}\,a_{-}(z,w,t;h)\Big)+R_h(t;z,w),
\end{equation}
where the remainder satisfies an operator bound of the form
\begin{equation}\label{eq:R-bound-exp}
\|R_h(t)\|_{L^{2}\to L^{2}}\;\le\;C_N\,h^{N}\,e^{\Gamma |t|}\qquad(\forall N\in\N),
\end{equation}
with geometric constants $C_N,\Gamma$ depending only on $M$.
Consequently, for $|t|\le T_E(h)$,
\[
\|R_h(t)\|_{2\to2}\;\le\;C'_N\,h^{N-\Gamma c_E}.
\]
In particular, choosing $c_E$ small enough yields $\|R_h(t)\|_{2\to2}\le C''_N h^{N_0}$ uniformly on $|t|\le T_E(h)$,
for any prescribed $N_0$.
All constants are uniform under cusp truncation in the sense that the same bounds hold for $U_{h,Y}(t)$, up to tails $O(Y^{-1})$.
\end{theorem}

\begin{proof}[Proof sketch]
Construct a global FIO parametrix by propagating the local phase/amplitude by the Hamilton flow of $p(x,\xi)=|\xi|_g$.
Negative curvature gives exponential bounds on derivatives of the flow, producing $e^{\Gamma|t|}$ in \eqref{eq:R-bound-exp}.
Restricting to $|t|\le c_E\log(1/h)$ turns $e^{\Gamma|t|}$ into a power of $h$.
\end{proof}

\medskip
\noindent\textbf{Periodization (local finiteness).}
On $M$, one may write the kernel as a periodization over $\Gamma$:
\[
U_h(t;z,w)=\sum_{\gamma\in\Gamma}U_{h,\Hh}(t;z,\gamma w),
\]
and for $|t|\le T_E(h)$ the sum is effectively finite because only $\gamma$ with $d(z,\gamma w)\approx |t|$ contribute,
while the number of such elements grows like $e^{|t|}$ and is balanced by the hyperbolic dispersive decay encoded in the amplitudes.

\medskip
\noindent\textbf{Canonical relation and propagation of singularities.}
Microlocally $U_h(t)$ is an $h$--FIO associated with the graph of $g^t$ on $T^*M$,
hence for distributions $f$,
\begin{equation}\label{eq:WF-propagation-fixed}
\WF\big(U_h(t)f\big)=g^t\big(\WF(f)\big)\qquad(|t|\le T_E(h)).
\end{equation}

\subsection{Egorov's theorem at logarithmic times}\label{subsec:egorov}

\noindent\textbf{Quantization.}
Let $a\in S^0(T^*M)$ and write $\Op_h(a)$ for a fixed semiclassical quantization (Chapter~2 conventions).
We will use the fact that derivatives of $a\circ g^t$ grow like $e^{\Gamma|t|}$.

\begin{theorem}[Egorov up to Ehrenfest time]\label{thm:egorov-log}
Let $A_h=\Op_h(a)$ with $a\in S^0(T^*M)$.
Then for all $|t|\le T_E(h)=c_E\log(1/h)$ one has
\begin{equation}\label{eq:egorov-fixed}
U_h(-t)\,A_h\,U_h(t)=\Op_h(a\circ g^t)+\mathcal{O}_{L^2\to L^2}\!\big(h\,e^{\Gamma|t|}\big).
\end{equation}
In particular, if $c_E<1/\Gamma$ then
\[
U_h(-t)\,A_h\,U_h(t)=\Op_h(a\circ g^t)+\mathcal{O}_{2\to2}\!\big(h^{1-\Gamma c_E}\big)
\quad\text{uniformly for }|t|\le T_E(h).
\]
\end{theorem}

\begin{proof}[Proof sketch]
Differentiate $U_h(-t)A_hU_h(t)$ in $t$ and use the commutator expansion
$\frac{d}{dt}(U_h(-t)A_hU_h(t))=\frac{i}{h}U_h(-t)[P,A_h]U_h(t)$.
Symbolic calculus gives $[P,A_h]=\frac{h}{i}\Op_h(\{p,a\})+\mathcal{O}(h^2)$,
and derivative growth along $g^t$ produces $e^{\Gamma|t|}$.
\end{proof}

\medskip
\noindent\textbf{Orbit-aware Egorov.}
If $A_h$ is microlocalized by $\chi_{\mathrm{per}}$ or $\chi_{\mathrm{aper}}$, then
\eqref{eq:egorov-fixed} holds with $|t|\le T_{\mathrm{orb}}(\rho;h)$ on the corresponding microlocal support.
This is the precise way the calculus avoids slicing periodic packets.

\subsection{Stationary phase templates}\label{subsec:stationary-phase}

\noindent\textbf{Template.}
For oscillatory integrals $\int e^{i\varphi/h}a$, we use the standard stationary phase expansion with explicit
uniformity requirements; we only record that every remainder inherits an $e^{\Gamma|t|}$-loss coming from flow derivatives,
hence must be restricted to Ehrenfest/orbit-aware times.

\medskip
\noindent\textbf{Practical rule.}
Whenever the phase depends on $t$ through the flow, all “uniform in $t$” statements are made only for
\[
|t|\le T_{\mathrm{orb}}(\rho;h)\le T_E(h).
\]

\subsection{Matching with the spectral projector}\label{subsec:projector-matching}

\noindent\textbf{Spectral window and Fourier representation.}
Let $\chi\in\mathcal{S}(\R)$ be even, with $\widehat{\chi}\in C_c^\infty(\R)$.
Define the \emph{windowed spectral projector} (center $\lambda$, width $\eta$):
\[
P_{\lambda,\eta}:=\chi\!\Big(\frac{P-\lambda}{\eta}\Big).
\]
Then (functional calculus) gives
\begin{equation}\label{eq:proj-fourier-fixed}
P_{\lambda,\eta}
=\frac{1}{2\pi}\int_{\R} e^{-it\lambda}\,\eta\,\widehat{\chi}(\eta t)\,e^{itP}\,dt
=\frac{1}{2\pi}\int_{\R} e^{-it}\,\widehat{\chi}(\eta t)\,U_h(t)\,dt,
\end{equation}
where in the last identity we used $e^{itP}=U_h(th)$ and $\lambda=1/h$.
The effective support is $|t|\lesssim \eta^{-1}$.

\medskip
\noindent\textbf{Ehrenfest compatibility.}
We impose
\begin{equation}\label{eq:eta-window-fixed}
\eta^{-1}\le T_E(h)=c_E\log(1/h),
\end{equation}
or, equivalently, $\eta\ge (\log(1/h))^{-1}$ (and in practice $\eta\ge h^\theta$ is more than enough).
In the orbit-aware setting, we integrate with the microlocal time restriction $|t|\le T_{\mathrm{orb}}(\rho;h)$.

\medskip
\noindent\textbf{Microlocal structure of $P_{\lambda,\eta}$.}
On the diagonal (local Weyl regime), one has the correct scaling
\[
P_{\lambda,\eta}(z,z)\;\asymp\; \lambda\,\eta
\qquad(\text{up to geometry-dependent constants}),
\]
so any bound of the form $h^{-1}\eta^{-1}$ is dimensionally incorrect and must be avoided.

\medskip
\noindent\textbf{Projector parametrix (schematic).}
Combining \Cref{thm:parametrix-logtime} with \eqref{eq:proj-fourier-fixed} yields that $P_{\lambda,\eta}$
is an $h$--FIO microlocally supported near the energy shell $p=1$ (in $h$-scaled variables),
with canonical relation induced by short geodesic arcs of length $|t|\lesssim \eta^{-1}$.
All remainders inherit the Ehrenfest/orbit-aware restriction.

\section{Ehrenfest remainder in Hilbert--Schmidt norm and localized trace}
\label{sec:5.ehrenfest.HS}

\subsection{OBJ-LOCK: cusp cutoff and truncated operator}
\label{subsec:5.ehrenfest.objlock}

Fix a smooth cutoff $\Lambda_Y\in C_c^\infty(M)$ with $\Lambda_Y\equiv 1$ on the $Y/2$-thick part and
$\supp\Lambda_Y\subset M(Y)$. Choose $Y=Y(\lambda)=\lambda^{A}$ for fixed $A>0$ and define
\[
U_{h,Y}(t):=\Lambda_Y U_h(t)\Lambda_Y.
\]

\subsection{Ehrenfest decomposition}
\label{subsec:5.ehrenfest.decomp}

Assume that for $|t|\le T_E(h)$ we have
\[
U_{h,Y}(t)=U_{h,Y,\mathrm{main}}(t)+R_{E,Y}(t),
\]
where $U_{h,Y,\mathrm{main}}(t)$ is the microlocal main parametrix contribution and $R_{E,Y}(t)$ the remainder.

\subsection{KEY estimate: HS control of the remainder}
\label{subsec:5.ehrenfest.HSbound}

\begin{proposition}[KEY: Hilbert--Schmidt remainder bound up to Ehrenfest time]
\label{prop:5.EH.HS}
Fix $\varepsilon>0$ and $A>0$. There exist constants $c_E>0$ and $\lambda_0(\varepsilon,A)$ such that for all
$\lambda\ge \lambda_0$ and all $|t|\le c_E\log\lambda$,
\[
\|R_{E,Y(\lambda)}(t)\|_{HS(M)} \ \ll_{\varepsilon,A}\ \lambda^{1/2+\varepsilon}.
\]
\end{proposition}

\begin{remark}[Structural ceiling]
The exponent $1/2$ is the stationary-phase ceiling for wave-trace type methods on surfaces.
Any systematic improvement beyond $\lambda^{1/2+\varepsilon}$ requires new input beyond the Chapter~5 mechanism.
\end{remark}

\subsection{Trace extraction under a localized time window}
\label{subsec:5.ehrenfest.trace}

Let $\psi\in C_c^\infty(\R)$ be even with $\supp\psi\subset[-1,1]$.
Define
\[
g_{\lambda,\eta}(t):=\psi(\eta t)\cos(\lambda t),\qquad \supp g_{\lambda,\eta}\subset[-\eta^{-1},\eta^{-1}],
\]
and assume the Ehrenfest compatibility $\eta^{-1}\le c_E\log\lambda$.

Set
\[
\mathcal R_{\lambda,\eta}:=\int_{\R} g_{\lambda,\eta}(t)\,R_{E,Y(\lambda)}(t)\,dt.
\]

\begin{lemma}[HS$\Rightarrow$trace bound under localization]
\label{lem:5.HS.to.trace}
Under Proposition~\ref{prop:5.EH.HS}, for every $\varepsilon>0$,
\[
\big|\Tr(\mathcal R_{\lambda,\eta})\big|
\ \ll_{\varepsilon}\ \lambda^{1/2+\varepsilon},
\]
uniformly for $\eta^{-1}\le c_E\log\lambda$.
\end{lemma}

\begin{proof}
Use $|\Tr(A)|\le \|A\|_{HS}$ for trace-class/Hilbert--Schmidt operators and Minkowski:
\[
\|\mathcal R_{\lambda,\eta}\|_{HS}\le \int |g_{\lambda,\eta}(t)|\,\|R_{E,Y}(t)\|_{HS}\,dt
\ll \eta^{-1}\sup_{|t|\le c_E\log\lambda}\|R_{E,Y}(t)\|_{HS}.
\]
Then apply Proposition~\ref{prop:5.EH.HS} and absorb $\log\lambda$ into $\lambda^\varepsilon$.
\end{proof}
