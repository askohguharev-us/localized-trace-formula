\chapter{Microlocal analysis of the wave kernel and a localized spectral projector}
\label{chap:microlocal-wave-projector}

\section*{Orientation and standing conventions}\label{sec:ch5-overview}

\noindent\textbf{Purpose.}
This chapter supplies the microlocal input used in the localized trace formula: a log-time semiclassical
parametrix for the shifted wave group generated by $\sqrt{\Delta-\tfrac14}$, an Egorov theorem up to
Ehrenfest times, and a microlocal pseudodifferential description of the windowed projector
$P_{T,\eta}$. The output is stated in a form compatible with the Selberg normalization fixed earlier.

\medskip
\noindent\textbf{Geometric object lock.}
Let $M=\Gamma\backslash\H$ be a finite-area hyperbolic surface (possibly with cusps), and let
$\Delta\ge 0$ be the nonnegative Laplace--Beltrami operator on $M$. We work with the shifted operator
\[
A:=\sqrt{\Delta-\tfrac14},
\]
defined by the spectral theorem on $L^2(M)$ (with residual spectrum inserted if present; we suppress it
in the notation). Spectral parameterization uses
\[
\lambda=\tfrac14+t^2,\qquad t\in\R,
\]
with Eisenstein measure $dt/(4\pi)$ as in Chapters~3--4.

\medskip
\noindent\textbf{Semiclassical parameter and time scale.}
Fix a large frequency $T\ge 1$ and set
\[
h:=T^{-1}.
\]
Let $G^t$ denote the geodesic flow on $T^*M$ (equivalently, the Hamiltonian flow of the principal symbol
$p(x,\xi)=|\xi|_g$ of $A$). On negative curvature, derivatives of $G^t$ grow exponentially in $|t|$; hence
microlocal bookkeeping is restricted to Ehrenfest times
\begin{equation}\label{eq:ehrenfest-time-A}
|t|\le T_E(h):=c_0\log(1/h),
\end{equation}
with $c_0>0$ chosen sufficiently small (see Remark~\ref{rem:ehrenfest-c0-A}).

\medskip
\noindent\textbf{Wave group.}
Define the unitary wave group and its even part by
\begin{equation}\label{eq:wave-group-A}
W(t):=e^{itA},\qquad
U(t):=\cos(tA)=\tfrac12\big(W(t)+W(-t)\big).
\end{equation}
All microlocal statements are proved for $W(t)$ and then inherited by $U(t)$.

\medskip
\noindent\textbf{Window profile and projector.}
Fix once and for all an even cutoff $\widehat\chi\in C_c^\infty(\R)$ with
\begin{equation}\label{eq:chi-hat-A}
\supp(\widehat\chi)\subset(-1,1),\qquad \widehat\chi(0)>0,\qquad \widehat\chi(-s)=\widehat\chi(s),
\end{equation}
and define its inverse Fourier transform
\begin{equation}\label{eq:chi-def-A}
\chi(u):=\frac{1}{2\pi}\int_{\R}\widehat\chi(s)e^{isu}\,ds.
\end{equation}
For $(T,\eta)$ with $T\ge 1$ and $\eta\in(0,1]$, set
\begin{equation}\label{eq:multiplier-A}
m_{T,\eta}(t):=\chi\!\left(\frac{t-T}{\eta}\right)+\chi\!\left(\frac{t+T}{\eta}\right),
\end{equation}
and define the localized spectral projector by functional calculus
\begin{equation}\label{eq:projector-def-A}
P_{T,\eta}:=m_{T,\eta}(A).
\end{equation}
Unless stated otherwise, we assume the \emph{polynomial window regime}
\begin{equation}\label{eq:eta-regime-A}
T^{-\theta}\le \eta\le 1,\qquad 0<\theta<\theta_0,
\end{equation}
for a fixed $\theta_0>0$ depending only on geometric constants and on the choice of $\widehat\chi$.

\medskip
\noindent\textbf{Wave representation of $P_{T,\eta}$.}
Fourier inversion yields the exact operator identity
\begin{equation}\label{eq:projector-wave-A}
P_{T,\eta}
=\frac{1}{2\pi}\int_{\R} e^{-isT}\,\widehat\chi(\eta s)\,W(s)\,ds,
\end{equation}
with strong convergence on $L^2(M)$. Since $\supp(\widehat\chi)\subset(-1,1)$, only times $|s|\le \eta^{-1}$
enter.

\medskip
\noindent\textbf{References.}
We rely on standard FIO/pseudodifferential calculus and log-time semiclassical dynamics; see
\cite{Hormander1983,Hormander1985,DuistermaatGuillemin1975,Sogge2014,Zworski2012,DyatlovZworski2019}.

\section{Log-time parametrix for the shifted wave kernel}\label{sec:ch5-parametrix}

\subsection{Local FIO structure and canonical relation}\label{subsec:ch5-fio}

Let $p(x,\xi)=|\xi|_g$ denote the principal symbol of $A$ in local coordinates. Its Hamiltonian flow equals the
geodesic flow on $T^*M\setminus 0$; we denote it by $G^t$. For each fixed $t$, the wave group $W(t)$ is a Fourier
integral operator associated with the canonical graph
\[
\mathcal C_t=\{(x,\xi;y,\eta)\in T^*M\times T^*M:\ (x,\xi)=G^t(y,\eta)\}.
\]
Microlocally in a coordinate patch and away from conjugate points, the Schwartz kernel admits an oscillatory
representation of the standard form
\begin{equation}\label{eq:local-fio-A}
W(t;z,w)
=(2\pi h)^{-1}\int_{\R} e^{\frac{i}{h}\Phi(t,z,w,\vartheta)}\,a(t,z,w,\vartheta;h)\,d\vartheta
\;+\;R(t;z,w;h),
\end{equation}
where $\Phi$ parametrizes $\mathcal C_t$ and $a$ is a classical symbol solving transport equations along the flow
\cite[Vol.~IV]{Hormander1985}.

\subsection{Ehrenfest restriction and symbol growth}\label{subsec:ch5-ehrenfest}

On negative curvature, derivatives of $G^t$ satisfy bounds of the form
$\|\partial^\alpha G^t\|\le C_\alpha e^{C_\alpha|t|}$, implying exponential-in-$|t|$ growth of propagated symbol
seminorms. Restricting to \eqref{eq:ehrenfest-time-A} converts these factors into at most polynomial losses in $h^{-1}$.

\begin{theorem}[Log-time semiclassical parametrix]\label{thm:logtime-parametrix-A}
There exist constants $c_0>0$ and $h_0>0$ such that for all $0<h<h_0$ and all $|t|\le T_E(h)=c_0\log(1/h)$,
the operator $W(t)$ is an $h$--FIO of order $0$ associated with $\mathcal C_t$. Moreover, in local charts one may
choose \eqref{eq:local-fio-A} so that for each multi-index $\alpha$ there exist constants $C_\alpha,N_\alpha$ with
\begin{equation}\label{eq:amplitude-growth-A}
\big|\partial_{z,w,\vartheta}^\alpha a(t,z,w,\vartheta;h)\big|
\le C_\alpha\,h^{-N_\alpha}\,e^{C_\alpha|t|},
\end{equation}
and the remainder is smoothing to arbitrary order: for every $s\ge 0$ and every $N\ge 0$,
\begin{equation}\label{eq:smoothing-remainder-A}
\sup_{|t|\le T_E(h)}\|R(t)\|_{H^{-s}\to H^{s}}\le C_{N,s}\,h^{N}.
\end{equation}
\end{theorem}

\begin{remark}[Choice of $c_0$]\label{rem:ehrenfest-c0-A}
Fix a symbol class in which the calculus closes with losses $h^{-\varepsilon}$ under propagation. Choose $c_0$
so that $e^{C|t|}\le h^{-\varepsilon}$ on $|t|\le c_0\log(1/h)$ for the relevant derivative-growth constant $C$.
Equivalently, take $c_0<\varepsilon/C$. This is the only place $c_0$ enters.
\end{remark}

\section{Egorov theorem up to Ehrenfest time}\label{sec:ch5-egorov}

\subsection{Heisenberg evolution of observables}\label{subsec:ch5-heisenberg}

Let $\Op_h(a)$ be a properly supported semiclassical pseudodifferential operator of order $0$ with compact
microsupport inside a fixed coordinate patch. Define the conjugated observable
\[
A(t):=W(-t)\,\Op_h(a)\,W(t).
\]

\begin{theorem}[Egorov on Ehrenfest times]\label{thm:egorov-A}
Assume $|t|\le T_E(h)$ with $c_0$ as in Theorem~\ref{thm:logtime-parametrix-A}. Then
\begin{equation}\label{eq:egorov-A}
W(-t)\,\Op_h(a)\,W(t)=\Op_h(a\circ G^t)+h\,\Op_h(r_t)+\mathcal O_{L^2\to L^2}(h^\infty),
\end{equation}
where $r_t\in S^0$ satisfies seminorm bounds
\begin{equation}\label{eq:egorov-remainder-A}
\|\partial^\alpha r_t\|_\infty\le C_\alpha e^{C_\alpha|t|}.
\end{equation}
In particular, on $|t|\le T_E(h)$ the remainder term is $\mathcal O(h^{1-\varepsilon})$ in $L^2$--operator norm
for some $\varepsilon=\varepsilon(c_0)>0$.
\end{theorem}

\begin{proof}[Proof sketch (standard)]
This is the standard Egorov theorem for conjugation by an $h$--FIO associated with the Hamiltonian flow $G^t$;
the only nontrivial bookkeeping is the control of symbol seminorms on log-times, which follows from the
Ehrenfest restriction and the derivative-growth bounds for $G^t$. See \cite[Ch.~11]{Zworski2012} and
\cite{DyatlovZworski2019}.
\end{proof}

\begin{corollary}[Propagation of quadratic expectations]\label{cor:expectations-A}
Let $u\in L^2(M)$ and $|t|\le T_E(h)$. Then
\[
\langle \Op_h(a)\,W(t)u,\ W(t)u\rangle
=\langle \Op_h(a\circ G^t)\,u,\ u\rangle+\mathcal O(h^{1-\varepsilon}),
\]
with $\varepsilon=\varepsilon(c_0)>0$ as in Theorem~\ref{thm:egorov-A}.
\end{corollary}

\section{Stationary phase with Ehrenfest-uniform remainder}\label{sec:ch5-stationary-phase}

\noindent
The composition of \eqref{eq:projector-wave-A} with the parametrix \eqref{eq:local-fio-A} produces oscillatory
integrals whose amplitudes inherit controlled $e^{C|t|}$ growth. We record a stationary phase estimate in a form
stable under such growth on the log-time window.

\begin{lemma}[Stationary phase with explicit remainder]\label{lem:stationary-phase-A}
Let $\phi\in C^\infty(\R^n)$ have a unique nondegenerate critical point $x_0$ in $\supp(a)$, where
$a\in C_c^\infty(\R^n)$ may depend on $h$ with at most polynomial losses in finitely many derivatives. For
\[
I(h):=\int_{\R^n}e^{\frac{i}{h}\phi(x)}a(x;h)\,dx,
\]
one has for each $N\ge 0$ an expansion
\begin{equation}\label{eq:stationary-expansion-A}
I(h)=e^{\frac{i}{h}\phi(x_0)}(2\pi h)^{n/2}e^{\frac{i\pi}{4}\sgn\phi''(x_0)}
\sum_{j=0}^{N-1}h^j c_j\;+\;R_N(h),
\end{equation}
with remainder estimate
\begin{equation}\label{eq:stationary-remainder-A}
|R_N(h)|\le C_N\,h^{n/2+N}\max_{|\alpha|\le 2N}\|\partial^\alpha a(\cdot;h)\|_\infty.
\end{equation}
\end{lemma}

\section{Microlocal structure of the localized projector}\label{sec:ch5-projector-microlocal}

\subsection{Kernel representation and geometric support}\label{subsec:ch5-kernel}

Let $K_{T,\eta}(z,w)$ denote the Schwartz kernel of $P_{T,\eta}$. From \eqref{eq:projector-wave-A},
\begin{equation}\label{eq:kernel-def-A}
K_{T,\eta}(z,w)=\frac{1}{2\pi}\int_{\R} e^{-isT}\,\widehat\chi(\eta s)\,W(s;z,w)\,ds.
\end{equation}
Since $\widehat\chi(\eta s)$ is supported in $|s|\le \eta^{-1}$ and the wave kernel on $\H$ has finite propagation,
$K_{T,\eta}(z,w)$ is effectively supported (on the universal cover) in the region $d(\tilde z,\tilde w)\lesssim \eta^{-1}$.
No global decay statement is used here; only propagation support.

\subsection{Microlocal cutoffs in the compact core}\label{subsec:ch5-cutoffs}

Fix a microlocal cutoff $\Psi=\Op_h(\psi)$ supported in a coordinate patch contained in the compact core
of $M$ (away from deep cusp regions). All pseudodifferential reductions below are stated for $\Psi P_{T,\eta}\Psi$.

\begin{proposition}[Pseudodifferential reduction]\label{prop:psido-reduction-A}
Assume $T\to\infty$ with $h=T^{-1}$ and $\eta$ in \eqref{eq:eta-regime-A}. For $\Psi=\Op_h(\psi)$ as above there exist
symbols $\sigma_{T,\eta},\sigma^{(1)}_{T,\eta}\in S^0$ such that
\begin{equation}\label{eq:psido-expansion-A}
\Psi\,P_{T,\eta}\,\Psi=\Op_h(\sigma_{T,\eta})+h\,\Op_h(\sigma^{(1)}_{T,\eta})+\mathcal O_{L^2\to L^2}(h^\infty).
\end{equation}
The principal symbol is
\begin{equation}\label{eq:principal-symbol-A}
\sigma_{T,\eta}(x,\xi)
=\chi\!\left(\frac{p(x,\xi)-1}{\eta}\right)+\chi\!\left(\frac{p(x,\xi)+1}{\eta}\right),
\end{equation}
and is supported where $|p(x,\xi)-1|\lesssim \eta$.
\end{proposition}

\begin{proof}[Proof sketch (standard)]
Insert the local parametrix for $W(s)$ (Theorem~\ref{thm:logtime-parametrix-A}) into \eqref{eq:kernel-def-A},
compose with $\Psi$ on both sides, and apply stationary phase (Lemma~\ref{lem:stationary-phase-A}) in the relevant
oscillatory variables. This is microlocal functional calculus for $A$ in semiclassical scaling; see
\cite[Ch.~14]{Zworski2012} and \cite[Vol.~IV]{Hormander1985}. Ehrenfest-uniformity is ensured by
\eqref{eq:amplitude-growth-A} and \eqref{eq:ehrenfest-time-A}.
\end{proof}

\subsection{Diagonal scale (local density of states)}\label{subsec:ch5-diagonal}

In the compact core, \eqref{eq:psido-expansion-A} implies that $K_{T,\eta}(z,z)$ has size comparable to $T\eta$
(up to fixed cutoffs), consistent with local Weyl law. Precise leading constants are fixed at trace level below.

\section{Trace asymptotics and Selberg lock}\label{sec:ch5-trace}

\begin{proposition}[Microlocal Weyl law for the window]\label{prop:microlocal-weyl-A}
Let $\Psi=\Op_h(\psi)$ be supported in the compact core. Assume $T\to\infty$ and $\eta$ satisfies \eqref{eq:eta-regime-A}.
Then
\begin{equation}\label{eq:microlocal-weyl-A}
\Tr(\Psi P_{T,\eta}\Psi)
=\frac{T\eta}{2\pi}\,\widehat\chi(0)\int_M \psi_0(z)\,d\mu(z)
\;+\;\mathcal O(T\eta^2)\;+\;\mathcal O(T^{1-\delta}),
\end{equation}
where $\psi_0$ denotes the base (principal) component of $\psi$, and $\delta>0$ depends only on the Ehrenfest constant
$c_0$ and on the fixed symbol class.
\end{proposition}

\begin{proof}[Proof sketch]
By Proposition~\ref{prop:psido-reduction-A}, the trace reduces to the semiclassical trace of a pseudodifferential
operator with principal symbol supported in the shell $|p-1|\lesssim\eta$. The principal term is the phase-space
integral of \eqref{eq:principal-symbol-A}; the identity $\int_\R \chi(u)\,du=\widehat\chi(0)$ follows from
\eqref{eq:chi-def-A} and yields the normalization constant. The term $\mathcal O(T\eta^2)$ comes from expansion
transverse to the energy surface, and $\mathcal O(T^{1-\delta})$ collects the log-time parametrix/Egorov remainders.
See \cite{Hormander1983,Zworski2012,DyatlovZworski2019}.
\end{proof}

\begin{remark}[Selberg normalization lock]\label{rem:selberg-lock-A}
Chapter~6 evaluates the Selberg identity term using the same window $m_{T,\eta}$, obtaining the leading constant
\[
\frac{\vol(M)}{2\pi}\,T\eta\,\widehat\chi(0),
\]
which matches \eqref{eq:microlocal-weyl-A} when $\Psi$ is chosen with $\psi_0\equiv 1$ on the compact core.
This agreement is treated as a normalization constraint and will be enforced in all later trace comparisons.
\end{remark}

\section{Cusps and truncation compatibility}\label{sec:ch5-cusps}

\noindent
Because $M$ may be noncompact, microlocal reductions are asserted only in the compact core. We record an explicit
truncation insertion principle showing that deep cusp contributions can be cut off without affecting compact
microlocal statements at the level required later.

\subsection{Cusp truncation and truncated propagators}\label{subsec:ch5-truncation}

Fix $Y\ge 1$. Let $\kappa_Y\in C^\infty(M)$ be a cutoff such that $\kappa_Y\equiv 1$ on $M(2Y)$ and $\kappa_Y\equiv 0$
for $y>3Y$ in each cusp chart (with a fixed smoothing buffer). Define
\[
W_Y(t):=\kappa_Y W(t)\kappa_Y,\qquad P_{T,\eta}^{(Y)}:=\kappa_Y P_{T,\eta}\kappa_Y.
\]

\begin{lemma}[Truncation does not affect compact microlocalization]\label{lem:truncation-A}
Let $\Psi=\Op_h(\psi)$ be microlocalized in the compact core $M(2Y)$. Then for every $N\ge 0$,
\begin{equation}\label{eq:truncation-negligible-A}
\|\Psi(P_{T,\eta}-P_{T,\eta}^{(Y)})\Psi\|_{L^2\to L^2}\le C_{N,Y}\,h^{N},
\end{equation}
uniformly as $T\to\infty$ with $\eta$ in \eqref{eq:eta-regime-A}.
\end{lemma}

\begin{proof}[Proof sketch]
The difference is supported where $\kappa_Y$ varies (deep cusp) and can contribute to $\Psi(\cdot)\Psi$ only through
wave propagation from/to that region. Using \eqref{eq:projector-wave-A}, only times $|t|\le \eta^{-1}$ appear.
For fixed $Y$ and $\Psi$ supported in $M(2Y)$, these propagation channels are separated by a positive hyperbolic
distance uniformly in $T$, and the resulting operator is smoothing to arbitrary order on the compactly supported
microlocal sector. See \cite{Hejhal1983,Iwaniec2002,Hormander1985}.
\end{proof}

\section{Black-box output for Chapters 6--7}\label{sec:ch5-summary}

\begin{theorem}[Microlocal structure package for $P_{T,\eta}$]\label{thm:projector-structure-A}
Let $P_{T,\eta}$ be defined by \eqref{eq:projector-def-A}--\eqref{eq:projector-wave-A} with $\eta$ satisfying
\eqref{eq:eta-regime-A}$.$ Let $\Psi=\Op_h(\psi)$ be supported in the compact core.
Then:
\begin{itemize}
\item[(i)] (\emph{Log-time parametrix}) $W(t)$ admits an $h$--FIO parametrix on $|t|\le T_E(h)$ with amplitude bounds
\eqref{eq:amplitude-growth-A} and smoothing remainder \eqref{eq:smoothing-remainder-A}.
\item[(ii)] (\emph{Egorov}) Conjugation by $W(t)$ transports observables along $G^t$ on $|t|\le T_E(h)$ with remainder
controlled as in \eqref{eq:egorov-A}.
\item[(iii)] (\emph{Projector as PDO}) $\Psi P_{T,\eta}\Psi$ admits the expansion \eqref{eq:psido-expansion-A} with principal
symbol \eqref{eq:principal-symbol-A}.
\item[(iv)] (\emph{Trace}) The microlocal trace asymptotic \eqref{eq:microlocal-weyl-A} holds, with leading constant locked to
$\widehat\chi(0)$.
\item[(v)] (\emph{Cusps}) Truncation can be inserted without affecting compact microlocal statements in operator norm to
arbitrary order (Lemma~\ref{lem:truncation-A}).
\end{itemize}
\end{theorem}

\subsection*{Audit (Microlocal certificate)}\label{audit:ch5}

\begin{itemize}
\item[(M1)] Object lock: $(M,\Delta,A)$ fixed; window profile $\widehat\chi$ fixed by \eqref{eq:chi-hat-A}.
\item[(M2)] Time scale: all FIO/Egorov statements explicitly restricted to $|t|\le T_E(h)$ \eqref{eq:ehrenfest-time-A}.
\item[(M3)] Projector identity: \eqref{eq:projector-wave-A} is exact; only propagation support is used from it.
\item[(M4)] PDO reduction: \eqref{eq:psido-expansion-A}--\eqref{eq:principal-symbol-A} stated only after compact microlocal cutoff.
\item[(M5)] Trace constant: leading term depends on $\widehat\chi(0)$ and is declared as Selberg-locked (Remark~\ref{rem:selberg-lock-A}).
\item[(M6)] Cusp safety: truncation is an explicit operator insertion with negligible effect on compact microlocalization
\eqref{eq:truncation-negligible-A}.
\end{itemize}
