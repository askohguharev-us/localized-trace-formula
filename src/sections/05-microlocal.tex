\chapter{Microlocal Analysis of the Wave Kernel and a Localized Spectral Projector}
\label{chap:microlocal-wave-projector}

\section*{Overview and standing conventions}
\label{sec:ch5-overview}

\noindent
This chapter develops the microlocal analytic tools required for the localized trace formula.
The central objects are the shifted wave group generated by $\sqrt{\Delta-\tfrac14}$ and the smooth
spectral window $P_{T,\eta}$ concentrating the spectrum near a large frequency $T$ with thickness $\eta$.
We construct a log-time semiclassical parametrix for the wave kernel, prove an Egorov theorem up to
Ehrenfest (logarithmic) times, and derive a microlocal description of the localized projector,
including a trace asymptotic compatible with the Selberg normalization.

\medskip
\noindent\textbf{Geometric setting.}
Let $M=\Gamma\backslash\mathbb{H}$ be a finite-area hyperbolic surface, where
$\Gamma\subset\mathrm{PSL}_2(\mathbb{R})$ is discrete and cofinite. Let $\Delta\ge 0$ denote the
(nonnegative) Laplace--Beltrami operator on $M$.
We use the standard spectral parameterization
\[
  \lambda = \tfrac14 + t^2,\qquad t\in\mathbb{R},
\]
as in Chapter~\ref{sec:spectral-projector-part1}. The discrete spectrum corresponds to
$t=t_j\in[0,\infty)$ with eigenfunctions $\phi_j$, and the continuous spectrum is generated by Eisenstein
series $E_{\mathfrak a}(\cdot,\tfrac12+it)$ with Plancherel measure $dt/(4\pi)$.

\medskip
\noindent\textbf{Semiclassical parameter and Ehrenfest time.}
Fix a large frequency parameter $T\ge 1$ and set
\[
  h := T^{-1}.
\]
Let $G^t$ denote the geodesic flow on $T^*M$.
On a negatively curved surface, derivatives of $G^t$ grow exponentially in $|t|$; accordingly,
microlocal symbol seminorms propagated by the flow grow like $e^{C|t|}$.
We therefore restrict attention to logarithmic (Ehrenfest) times
\begin{equation}\label{eq:ehrenfest-time}
  |t| \le T_E(h) := c_0 \log(1/h),
\end{equation}
where $c_0>0$ is chosen sufficiently small so that the exponential growth in $t$ becomes at most
polynomial in $h^{-1}$. Precise choices are discussed in
Remark~\ref{rem:ehrenfest-c0}.

\medskip
\noindent\textbf{Wave group.}
We work with the unitary group
\begin{equation}\label{eq:wave-group}
  W(t) := e^{it\sqrt{\Delta-\tfrac14}},
\end{equation}
and the even wave propagator
\begin{equation}\label{eq:cos-wave-group}
  U(t) := \cos\!\big(t\sqrt{\Delta-\tfrac14}\big)=\tfrac12\big(W(t)+W(-t)\big).
\end{equation}
Both have Schwartz kernels denoted by $W(t;z,w)$ and $U(t;z,w)$ respectively.
All microlocal statements below apply to $W(t)$ and hence to $U(t)$.

\medskip
\noindent\textbf{Window profile and localized projector.}
Fix an even cutoff $\widehat\chi\in C_c^\infty(\mathbb{R})$ with $\widehat\chi(0)\neq 0$ and
$\mathrm{supp}\,\widehat\chi\subset(-1,1)$.
Define the inverse Fourier transform
\begin{equation}\label{eq:chi-def}
  \chi(u):=\frac{1}{2\pi}\int_{\mathbb{R}} \widehat{\chi}(s)\,e^{isu}\,ds.
\end{equation}
For $(T,\eta)$ with $T\ge 1$ and $\eta\in(0,1]$, define the even spectral multiplier on the $t$--axis
\begin{equation}\label{eq:multiplier}
  m_{T,\eta}(t):=\chi\!\left(\frac{t-T}{\eta}\right)+\chi\!\left(\frac{t+T}{\eta}\right).
\end{equation}
The associated localized spectral projector is defined by the spectral theorem as
\begin{equation}\label{eq:projector-def}
  P_{T,\eta} := m_{T,\eta}\!\left(\sqrt{\Delta-\tfrac14}\right).
\end{equation}
In the principal regime used in later chapters we assume
\begin{equation}\label{eq:eta-regime}
  T^{-\theta}\le \eta \le 1,
  \qquad 0<\theta<\theta_0,
\end{equation}
for a fixed $\theta_0>0$ (depending only on the geometric constants of $M$).

\medskip
\noindent\textbf{Wave representation of $P_{T,\eta}$.}
By Fourier inversion applied to \eqref{eq:multiplier} and the functional calculus for $\sqrt{\Delta-\tfrac14}$,
\begin{equation}\label{eq:projector-wave}
  P_{T,\eta}
  = \frac{1}{2\pi}\int_{\mathbb{R}} e^{-isT}\,\widehat\chi(\eta s)\,W(s)\,ds.
\end{equation}
Since $\widehat\chi(\eta s)$ is supported in $|s|\le \eta^{-1}$ (up to the fixed support of $\widehat\chi$),
only times $|s|\lesssim \eta^{-1}$ enter.

\medskip
\noindent\textbf{References.}
We use standard microlocal analysis and semiclassical dynamics, in particular the Fourier integral operator
structure of wave propagators and Egorov-type theorems; see
H\"ormander~\cite{Hormander1983,Hormander1985}, Duistermaat--Guillemin~\cite{DuistermaatGuillemin1975},
Sogge~\cite{Sogge2014}, Zworski~\cite{Zworski2012}, and Dyatlov--Zworski~\cite{DyatlovZworski2019}.
For cusp analysis and finite-area surfaces, see Hejhal~\cite{Hejhal1983} and Iwaniec~\cite{Iwaniec2002}.

\section{A log-time semiclassical parametrix for the wave kernel}
\label{sec:ch5-parametrix}

\noindent
We construct a semiclassical Fourier integral representation for $W(t)$ valid for
$|t|\le T_E(h)$, with explicit control of symbol seminorms and a smoothing remainder.

\subsection{Local Fourier integral structure}

Let $p(x,\xi)$ denote the principal symbol of $\sqrt{\Delta-\tfrac14}$; on a surface,
$p(x,\xi)=|\xi|_g$ in local coordinates. The Hamiltonian flow of $p$ coincides with the geodesic flow on
$T^*M\setminus 0$; we denote it by $G^t$.

It is classical that $W(t)$ is a Fourier integral operator associated with the canonical graph
\[
  \mathcal{C}_t = \{(x,\xi;y,\eta)\in T^*M\times T^*M:\ (x,\xi)=G^t(y,\eta)\}.
\]
Microlocally away from conjugate points (in particular, in a sufficiently small neighborhood of the diagonal),
the Schwartz kernel of $W(t)$ admits an oscillatory integral representation of standard form
\begin{equation}\label{eq:local-fio}
  W(t;z,w)
  = (2\pi h)^{-1}\int_{\mathbb{R}} e^{\frac{i}{h}\Phi(t,z,w,\theta)}\,a(t,z,w,\theta;h)\,d\theta
  \;+\; R(t;z,w;h).
\end{equation}
Here $\Phi$ is a nondegenerate phase function parametrizing $\mathcal{C}_t$ and
$a$ is a classical symbol (of order $0$) solving transport equations along the flow.
See H\"ormander~\cite[Vol.~IV]{Hormander1985} and Duistermaat--Guillemin~\cite{DuistermaatGuillemin1975}.

\subsection{Symbol growth and Ehrenfest restriction}

On negatively curved manifolds, derivatives of $G^t$ satisfy bounds of the form
$\|\partial^\alpha G^t\|\le C_\alpha e^{C_\alpha |t|}$. Consequently, the propagated symbol seminorms of $a$
also grow like $e^{C|t|}$.
The logarithmic time restriction \eqref{eq:ehrenfest-time} ensures that such factors remain bounded by
a fixed power of $h^{-1}$, and hence the semiclassical calculus remains valid on $|t|\le T_E(h)$.

\begin{theorem}[Log-time semiclassical parametrix]\label{thm:logtime-parametrix}
There exist constants $c_0>0$ and $h_0>0$ such that for all $0<h<h_0$ and all $|t|\le T_E(h)=c_0\log(1/h)$,
the operator $W(t)$ is a semiclassical Fourier integral operator of order $0$ associated with $\mathcal{C}_t$.
Moreover, in local coordinates the kernel admits a representation of the form \eqref{eq:local-fio}
in which the amplitude satisfies, for each multi-index $\alpha$,
\begin{equation}\label{eq:amplitude-growth}
  \big|\partial_{z,w,\theta}^\alpha a(t,z,w,\theta;h)\big|
  \le C_\alpha\,h^{-N_\alpha}\,e^{C_\alpha |t|},
\end{equation}
and the remainder is smoothing to arbitrary order:
for every $s\ge 0$ and every $N\ge 0$,
\begin{equation}\label{eq:smoothing-remainder}
  \sup_{|t|\le T_E(h)}\|R(t)\|_{H^{-s}\to H^{s}} \le C_{N,s}\,h^N.
\end{equation}
\end{theorem}

\begin{proof}
This follows from the standard geometric-optics parametrix construction for hyperbolic equations,
together with the control of derivatives of the flow on Ehrenfest times.
See H\"ormander~\cite[Vol.~IV]{Hormander1985} for the FIO structure and symbol transport,
and Zworski~\cite[Ch.~10--11]{Zworski2012} or Dyatlov--Zworski~\cite{DyatlovZworski2019}
for the log-time bookkeeping in semiclassical dynamics.
\end{proof}

\begin{remark}[Choice of the Ehrenfest constant]\label{rem:ehrenfest-c0}
The constant $c_0$ is chosen so that $e^{C|t|}\le h^{-\varepsilon}$ on $|t|\le c_0\log(1/h)$
for a fixed $\varepsilon>0$ determined by the symbol class used in the Egorov theorem below.
Equivalently, $c_0<\varepsilon/C$, where $C$ bounds the relevant derivative growth of $G^t$.
\end{remark}

\section{Egorov theorem up to logarithmic times}
\label{sec:ch5-egorov}

\noindent
We now quantify the conjugation of semiclassical observables by the wave group, with a remainder controlled on
$|t|\le T_E(h)$.

Let $\Op_h(a)$ denote a properly supported semiclassical pseudodifferential operator of order $0$
with compact microsupport in a fixed coordinate patch.
Define the Heisenberg evolution
\[
  A(t):=W(-t)\,\Op_h(a)\,W(t).
\]

\begin{theorem}[Egorov on Ehrenfest times]\label{thm:egorov}
Assume $|t|\le T_E(h)$ with $c_0$ as in Theorem~\ref{thm:logtime-parametrix}.
Then there exists a symbol $a_t\in S^0$ such that
\begin{equation}\label{eq:egorov}
  W(-t)\,\Op_h(a)\,W(t) = \Op_h(a\circ G^t) + h\,\Op_h(r_t) + \mathcal{O}_{L^2\to L^2}(h^\infty),
\end{equation}
where $r_t\in S^0$ satisfies seminorm bounds of the form
\begin{equation}\label{eq:egorov-remainder}
  \|\partial^\alpha r_t\|_\infty \le C_\alpha e^{C_\alpha |t|}.
\end{equation}
In particular, on $|t|\le T_E(h)$ the remainder is $\mathcal{O}(h^{1-\varepsilon})$ in operator norm for some
$\varepsilon=\varepsilon(c_0)>0$.
\end{theorem}

\begin{proof}
This is the standard semiclassical Egorov theorem for conjugation by an FIO associated with the flow $G^t$,
where symbol-derivative losses are controlled by derivatives of $G^t$. The log-time restriction ensures
the resulting exponential factors are absorbed by a power of $h^{-1}$, preserving the semiclassical expansion.
See Zworski~\cite[Ch.~11]{Zworski2012} and Dyatlov--Zworski~\cite{DyatlovZworski2019}.
\end{proof}

\begin{corollary}[Propagation of expectations]\label{cor:expectations}
Let $u\in L^2(M)$ and $|t|\le T_E(h)$. Then
\[
  \big\langle \Op_h(a)\,W(t)u,\; W(t)u\big\rangle
  = \big\langle \Op_h(a\circ G^t)\,u,\;u\big\rangle + \mathcal{O}(h^{1-\varepsilon}),
\]
with $\varepsilon=\varepsilon(c_0)>0$ as above.
\end{corollary}

\section{Stationary phase with uniform remainder}
\label{sec:ch5-stationary-phase}

\noindent
The wave representation \eqref{eq:projector-wave} and the parametrix \eqref{eq:local-fio} lead to oscillatory
integrals in which the phase and amplitude depend on the propagated flow. We record a stationary phase lemma
with a remainder hierarchy uniform under the derivative-growth bounds on Ehrenfest times.

\begin{lemma}[Stationary phase with explicit remainder]\label{lem:stationary-phase}
Let $\phi\in C^\infty(\mathbb{R}^n)$ have a unique nondegenerate critical point $x_0$ in the support of
$a\in C_c^\infty(\mathbb{R}^n)$. For $h\in(0,1]$ define
\[
  I(h):=\int_{\mathbb{R}^n} e^{\frac{i}{h}\phi(x)}\,a(x;h)\,dx,
\]
where $a(\cdot;h)$ is smooth in $x$ and has at most polynomial dependence on $h^{-1}$ in finitely many derivatives.
Then for each $N\ge 0$ there exist coefficients $c_j=c_j(a,\phi)$ such that
\begin{equation}\label{eq:stationary-expansion}
  I(h)
  = e^{\frac{i}{h}\phi(x_0)} (2\pi h)^{n/2}\,e^{\frac{i\pi}{4}\operatorname{sgn}\phi''(x_0)}
    \sum_{j=0}^{N-1} h^j\,c_j
  \;+\; R_N(h),
\end{equation}
and the remainder satisfies
\begin{equation}\label{eq:stationary-remainder}
  |R_N(h)| \le C_N\,h^{n/2+N}\max_{|\alpha|\le 2N}\|\partial^\alpha a(\cdot;h)\|_\infty.
\end{equation}
\end{lemma}

\begin{proof}
This is a standard stationary phase estimate; see H\"ormander~\cite[Thm.~7.7.5]{Hormander1983}.
Uniformity under controlled parameter dependence follows whenever the required derivatives of $\phi$ and $a$
are bounded on the relevant compact set, up to factors of the form $e^{C|t|}$, which is admissible on
$|t|\le T_E(h)$.
\end{proof}

\section{Microlocal structure of the localized projector}
\label{sec:ch5-projector-microlocal}

\noindent
We now use the wave representation \eqref{eq:projector-wave} and the parametrix for $W(t)$ to obtain a microlocal
description of $P_{T,\eta}$ and of its kernel.

\subsection{Kernel representation}

Let $K_{T,\eta}(z,w)$ denote the Schwartz kernel of $P_{T,\eta}$.
From \eqref{eq:projector-wave},
\begin{equation}\label{eq:kernel-def}
  K_{T,\eta}(z,w)
  = \frac{1}{2\pi}\int_{\mathbb{R}} e^{-isT}\,\widehat\chi(\eta s)\,W(s;z,w)\,ds.
\end{equation}
Finite propagation speed for the wave equation implies that $W(s;z,w)$ is supported on
$d(z,w)\le |s|$ on the universal cover, hence the effective support of $K_{T,\eta}(z,w)$ is constrained by
$|s|\lesssim \eta^{-1}$, i.e.\ $d(z,w)\lesssim \eta^{-1}$ in the region where the parametrix applies.

\subsection{Local pseudodifferential reduction}

Fix a compactly supported microlocal cutoff $\Psi=\Op_h(\psi)$ supported in a coordinate patch in the compact part
of $M$ (i.e.\ away from the deep cusp region introduced in \S\ref{sec:ch5-cusps} below).
Then, modulo negligible smoothing errors, $\Psi P_{T,\eta}\Psi$ is a semiclassical pseudodifferential operator
microlocalized to the energy shell $p(x,\xi)=1$ with thickness $\eta$.

\begin{proposition}[Microlocal reduction to a pseudodifferential operator]\label{prop:psido-reduction}
Let $\Psi=\Op_h(\psi)$ be as above. Then, for $T\to\infty$ with $h=T^{-1}$ and $\eta$ in the regime
\eqref{eq:eta-regime}, there exist symbols $\sigma_{T,\eta},\sigma^{(1)}_{T,\eta}\in S^0$ such that
\begin{equation}\label{eq:psido-expansion}
  \Psi\,P_{T,\eta}\,\Psi
  = \Op_h(\sigma_{T,\eta}) + h\,\Op_h(\sigma^{(1)}_{T,\eta}) + \mathcal{O}_{L^2\to L^2}(h^\infty).
\end{equation}
Moreover, the principal symbol satisfies
\begin{equation}\label{eq:principal-symbol}
  \sigma_{T,\eta}(x,\xi)
  = \chi\!\left(\frac{p(x,\xi)-1}{\eta}\right) + \chi\!\left(\frac{p(x,\xi)+1}{\eta}\right),
\end{equation}
and is supported where $|p(x,\xi)-1|\lesssim \eta$.
\end{proposition}

\begin{proof}
Insert the local FIO representation of $W(s)$ from Theorem~\ref{thm:logtime-parametrix} into \eqref{eq:kernel-def},
and apply stationary phase (Lemma~\ref{lem:stationary-phase}) in the $(s,\theta)$ variables.
This is the standard microlocal functional calculus for elliptic operators; see H\"ormander~\cite{Hormander1985}
or Zworski~\cite[Ch.~14]{Zworski2012}. Uniformity follows from the symbol bounds \eqref{eq:amplitude-growth}
on Ehrenfest times.
\end{proof}

\subsection{Diagonal behavior}

On the diagonal, $K_{T,\eta}(z,z)$ represents a local density of states.
Proposition~\ref{prop:psido-reduction} implies that, microlocally in the compact core,
\[
  K_{T,\eta}(z,z) \asymp T\,\eta,
\]
consistent with the local Weyl law. The precise constant is fixed below at the trace level and matches the
identity term in the Selberg trace formula (Chapter~6).

\section{Trace asymptotics and normalization}
\label{sec:ch5-trace}

\noindent
We now record a microlocal Weyl law for $P_{T,\eta}$ and emphasize the normalization constant
$\widehat\chi(0)$, which later locks the microlocal side to the Selberg identity term.

\begin{proposition}[Microlocal Weyl law for the window]\label{prop:microlocal-weyl}
Let $\Psi=\Op_h(\psi)$ be a compactly supported microlocal cutoff in the compact core of $M$.
Assume $T\to\infty$ and $\eta$ satisfies \eqref{eq:eta-regime}. Then
\begin{equation}\label{eq:microlocal-weyl}
  \operatorname{Tr}(\Psi P_{T,\eta}\Psi)
  = \frac{T\eta}{2\pi}\,\widehat\chi(0)\int_M \psi_0(z)\,d\mu(z)
    \;+\; \mathcal{O}(T\eta^2)\;+\;\mathcal{O}(T^{1-\delta}),
\end{equation}
where $\psi_0$ denotes the principal (base-point) component of $\psi$,
and $\delta>0$ depends on the choice of $c_0$ and on the symbol class used.
\end{proposition}

\begin{proof}
By Proposition~\ref{prop:psido-reduction}, the trace reduces to the semiclassical trace of a pseudodifferential
operator with symbol supported in the shell $|p-1|\lesssim \eta$.
The principal term is the phase-space integral of the principal symbol; on a surface this yields a factor
proportional to $T\eta$.
The identity $\int_{\mathbb{R}}\chi(u)\,du=\widehat\chi(0)$ follows from \eqref{eq:chi-def}, and contributes the
normalization constant.
The error $\mathcal{O}(T\eta^2)$ comes from Taylor expansion of the density in the normal direction to the energy
surface, while $\mathcal{O}(T^{1-\delta})$ collects remainder terms originating in the Ehrenfest-time parametrix
and Egorov expansions.
See H\"ormander~\cite{Hormander1983} and Zworski~\cite{Zworski2012} for trace expansions, and
Dyatlov--Zworski~\cite{DyatlovZworski2019} for the log-time remainder bookkeeping.
\end{proof}

\begin{remark}[Consistency with Selberg normalization]\label{rem:selberg-lock}
Chapter~6 evaluates the Selberg identity term in the form
\[
  I_{T,\eta}
  = \frac{\operatorname{vol}(M)}{4\pi}\int_{\mathbb{R}} t\,\tanh(\pi t)\,m_{T,\eta}(t)\,dt,
\]
and, using that $m_{T,\eta}$ is even and localized near $\pm T$, obtains the same leading constant
$\frac{\operatorname{vol}(M)}{2\pi}\,T\eta\,\widehat\chi(0)$.
This agreement is a required normalization constraint between the microlocal and Selberg sides.
\end{remark}

\section{Cusps and truncation compatibility}
\label{sec:ch5-cusps}

\noindent
Since $M$ has finite area and may be noncompact, we record the truncation devices ensuring that the preceding
microlocal statements remain valid uniformly away from the deep cusp.

\subsection{Cusp model and truncation cutoffs}

A cusp neighborhood is modeled by
\[
  C(Y)=\{x+iy: y>Y\}/\langle x\mapsto x+1\rangle,
\]
with hyperbolic metric $y^{-2}(dx^2+dy^2)$.
Let $M(Y)$ denote the truncated surface obtained by removing the deep cusps $C_{\mathfrak a}(Y)$.
Choose a smooth cutoff $\kappa_Y$ satisfying $\kappa_Y\equiv 1$ on $M(2Y)$ and $\kappa_Y\equiv 0$ for $y>3Y$
in each cusp.
Define truncated operators
\[
  W_Y(t):=\kappa_Y\,W(t)\,\kappa_Y,\qquad P_{T,\eta}^{(Y)}:=\kappa_Y\,P_{T,\eta}\,\kappa_Y.
\]

\subsection{A truncation lemma}

\begin{lemma}[Truncation does not affect compact microlocalization]\label{lem:truncation}
Let $\Psi=\Op_h(\psi)$ be microlocalized in the compact core $M(2Y)$ for a fixed $Y\ge 1$.
Then for every $N\ge 0$,
\[
  \|\Psi(P_{T,\eta}-P_{T,\eta}^{(Y)})\Psi\|_{L^2\to L^2} \le C_{N,Y}\,h^N,
\]
uniformly for $T\to\infty$ with $\eta$ in \eqref{eq:eta-regime}.
\end{lemma}

\begin{proof}
The difference $P_{T,\eta}-P_{T,\eta}^{(Y)}$ is supported by propagation from and to the deep cusp region where
$\kappa_Y$ varies. Finite propagation speed for $W(t)$ combined with the time localization $|t|\lesssim\eta^{-1}$
in \eqref{eq:projector-wave} implies that, when $\Psi$ is supported in $M(2Y)$ with $Y$ fixed,
the corresponding kernels are separated by a positive hyperbolic distance uniformly in $T$.
Standard microlocal arguments then show that the contribution is smoothing to arbitrary order.
See Hejhal~\cite{Hejhal1983} and Iwaniec~\cite{Iwaniec2002} for cusp truncation devices, and
H\"ormander~\cite{Hormander1985} for smoothing estimates for Fourier integral operators under separated supports.
\end{proof}

\section{Summary: microlocal structure of the localized projector}
\label{sec:ch5-summary}

\noindent
For later use in Chapters~6--7, we summarize the outcomes of this chapter as a single black-box statement.

\begin{theorem}[Microlocal structure of $P_{T,\eta}$]\label{thm:projector-structure}
Let $P_{T,\eta}$ be defined by \eqref{eq:projector-def}--\eqref{eq:projector-wave} with $\eta$ satisfying
\eqref{eq:eta-regime}. Let $\Psi=\Op_h(\psi)$ be a microlocal cutoff supported in the compact core.
Then:
\begin{itemize}
  \item[(i)] (\emph{Log-time wave parametrix}) $W(t)$ is an $h$--FIO on $|t|\le T_E(h)$ with symbol bounds
  \eqref{eq:amplitude-growth} and smoothing remainder \eqref{eq:smoothing-remainder}.
  \item[(ii)] (\emph{Egorov}) Conjugation by $W(t)$ transports observables by $G^t$ up to remainders described in
  Theorem~\ref{thm:egorov}.
  \item[(iii)] (\emph{Projector as a PDO}) $\Psi P_{T,\eta}\Psi$ admits the expansion \eqref{eq:psido-expansion}
  with principal symbol \eqref{eq:principal-symbol}.
  \item[(iv)] (\emph{Trace}) The microlocal trace asymptotic \eqref{eq:microlocal-weyl} holds, with leading constant
  $\widehat\chi(0)$ matching the Selberg identity term.
  \item[(v)] (\emph{Cusps}) Truncation may be inserted without affecting compact microlocal statements
  (Lemma~\ref{lem:truncation}).
\end{itemize}
\end{theorem}
