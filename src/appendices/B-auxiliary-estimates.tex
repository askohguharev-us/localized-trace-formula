\section{Purpose and interface with the main text}
\label{sec:B.scope}

This appendix records the analytic bounds that are invoked repeatedly in Chapters~\ref{chap:06-geometric-terms}
through~\ref{chap:08-normalization} and in the applications chapter(s).  The role of these estimates is twofold:

\begin{enumerate}
\item They provide uniform control of the continuous spectrum and the scattering data in the localized window regime
\[
\lambda^{-\theta}\le \eta\le 1,\qquad \lambda\to\infty,
\]
ensuring that continuous and parabolic contributions remain subordinate to the identity main term.

\item They supply a catalogue of ``plug-in'' bounds for oscillatory integral transforms (Bessel/Kuznetsov kernels),
length-spectrum tails, and elementary arithmetic weights that enter the off-diagonal analysis.
\end{enumerate}

All implied constants depend at most on the geometric input $(\Gamma,\text{cusp data})$ and on finitely many seminorms
of fixed cutoffs.  No dependence on $(\lambda,\eta)$ is permitted unless explicitly stated.

\section{Notation and standing assumptions}
\label{sec:B.notation}

Let $M=\Gamma\backslash\mathbb{H}$ be a finite-area hyperbolic surface with cusps.  We write the discrete spectrum of
$\Delta$ as $\{1/4+r_j^2\}_{j\ge 0}$ and parameterize the continuous spectrum by $r\in\mathbb{R}$ via Eisenstein series.
The cusp set is denoted by $\{\mathfrak{a}\}$.

We fix:
\begin{itemize}
\item an even Schwartz cutoff $\chi\in\mathcal{S}(\mathbb{R})$ with $\chi(0)=1$ and its rescaling
$\chi_\eta(u)=\chi(u/\eta)$,
\item a spectral localization scale $\eta$ satisfying $\lambda^{-\theta}\le \eta\le 1$ for some fixed
$0<\theta<\theta_0(\Gamma)$,
\item the Fourier transform conventions of Appendix~\ref{app:G-normalizations}.
\end{itemize}

We write $A\ll B$ and $A=O(B)$ interchangeably, with subscripts indicating parameter dependence.

\section{Scattering determinants and logarithmic derivatives}
\label{sec:B.scattering}

\subsection{Scattering matrices and the determinant}
\label{subsec:B.scattering-matrix}

Let $\Phi(s)=(\varphi_{\mathfrak{a}\mathfrak{b}}(s))$ denote the scattering matrix associated with $\Gamma$ and the
chosen cusp normalization.  It is meromorphic in $s$ and unitary on $\Re(s)=1/2$ away from poles.  We denote the
scattering determinant by
\[
\varphi(s):=\det \Phi(s).
\]
We require bounds for $\frac{\varphi'}{\varphi}(s)$ on the critical line and in short vertical windows.

\begin{proposition}[Polynomial growth of scattering derivatives]
\label{prop:B.scattering-poly}
For $\sigma\in[1/2,1]$ fixed and $|t|\ge 2$, one has
\begin{equation}
\label{eq:B.scattering-poly}
\frac{\varphi'}{\varphi}(\sigma+it)
\ll_{\Gamma,\sigma}
\log(2+|t|)\,(2+|t|)^{A_\Gamma},
\end{equation}
for some exponent $A_\Gamma\ge 0$ depending only on $\Gamma$ and the cusp normalization.
In particular, on the line $\sigma=1/2$ one has
\[
\left|\frac{\varphi'}{\varphi}\!\left(\frac12+it\right)\right|
\ll_{\Gamma}
\log(2+|t|)\,(2+|t|)^{A_\Gamma}.
\]
\end{proposition}

\begin{proof}[Proof sketch]
This is standard: $\varphi(s)$ is expressible as a product of gamma factors and Dirichlet series-type Euler factors
associated with $\Gamma$ (or with its congruence data in arithmetic cases).  Stirling bounds for gamma factors and
Phragm\'en--Lindel\"of control in vertical strips yield polynomial growth for $\log \varphi(s)$ and hence for
$\varphi'/\varphi(s)$.
Precise exponents are irrelevant for the localized remainder hierarchy; only polynomial control is used.
\end{proof}

\subsection{Localized integral bounds}
\label{subsec:B.scattering-local}

The continuous spectrum enters the localized trace formula through expressions of the form
\[
\frac{1}{4\pi}\sum_{\mathfrak{a}}\int_{\mathbb{R}}
\chi_\eta(r-\lambda)\,\frac{\varphi'_{\mathfrak{a}}}{\varphi_{\mathfrak{a}}}\!\left(\frac12+ir\right)\,dr
\quad\text{or}\quad
\int_{\mathbb{R}}\chi_\eta(r-\lambda)\,\frac{\varphi'}{\varphi}\!\left(\frac12+ir\right)\,dr.
\]
The next lemma packages the window-level bound.

\begin{lemma}[Window bound for scattering logarithmic derivatives]
\label{lem:B.scattering-window}
Let $\chi\in\mathcal{S}(\mathbb{R})$ be fixed and even.  Then uniformly for $\lambda\ge 2$ and
$\lambda^{-\theta}\le \eta\le 1$,
\begin{equation}
\label{eq:B.scattering-window}
\int_{\mathbb{R}}
\left|\chi_\eta(r-\lambda)\right|
\left|\frac{\varphi'}{\varphi}\!\left(\frac12+ir\right)\right|\,dr
\ll_{\Gamma,\chi}
\eta\,(\log \lambda)\,\lambda^{A_\Gamma}.
\end{equation}
\end{lemma}

\begin{proof}
By the change of variables $r=\lambda+\eta u$,
\[
\int |\chi_\eta(r-\lambda)|\,\left|\frac{\varphi'}{\varphi}(1/2+ir)\right|\,dr
=
\eta\int |\chi(u)|\,\left|\frac{\varphi'}{\varphi}\!\left(\frac12+i(\lambda+\eta u)\right)\right|\,du.
\]
Apply Proposition~\ref{prop:B.scattering-poly} with $t=\lambda+\eta u$ and use the rapid decay of $\chi$ to bound the
$u$-integral by a constant multiple of $(\log\lambda)\lambda^{A_\Gamma}$.
\end{proof}

\subsection{Maass--Selberg type control}
\label{subsec:B.maass-selberg}

For cusp truncations and the control of Eisenstein contributions, one needs a uniform Maass--Selberg relation.
We record the form used in Chapter~\ref{chap:06-geometric-terms}.

\begin{proposition}[Maass--Selberg relation (uniform form)]
\label{prop:B.maass-selberg}
Let $E_{\mathfrak{a}}(z,1/2+ir)$ denote the Eisenstein series at cusp $\mathfrak{a}$ and spectral parameter $r$.
Let $M^Y$ denote truncation at height $Y$ in all cusps.  Then for $r\in\mathbb{R}$,
\begin{equation}
\label{eq:B.maass-selberg}
\int_{M} \left|M^Y E_{\mathfrak{a}}(z,1/2+ir)\right|^2\,d\vol(z)
=
2\log Y
+
\Re\left(\frac{\varphi'_{\mathfrak{a}}}{\varphi_{\mathfrak{a}}}\!\left(\frac12+ir\right)\right)
+
O_\Gamma(1),
\end{equation}
with an implied constant uniform in $r$ and $Y\ge 2$.
\end{proposition}

\begin{proof}[Bibliographic proof sketch]
This is a classical identity: expand the truncated Eisenstein series in the cusp and integrate termwise; the boundary
contribution yields $2\log Y$ and the scattering term yields the logarithmic derivative.
Uniformity follows from the standard cusp normalization and unitarity of the scattering matrix on $\Re(s)=1/2$.
\end{proof}

\section{Length-spectrum tails and orbital sum bounds}
\label{sec:B.length-spectrum}

Hyperbolic conjugacy classes contribute through sums indexed by primitive closed geodesics $\gamma$ and their
iterations.  In localized regimes, these contributions are controlled by two mechanisms: decay of the Fourier
transform of the cutoff and coarse growth bounds for the length spectrum.

\subsection{A coarse geodesic counting bound}
\label{subsec:B.geodesic-count}

Let $\pi_\Gamma(L)$ denote the number of primitive closed geodesics of length at most $L$ (counted with multiplicity).
We use only the crude exponential upper bound.

\begin{lemma}[Crude length-spectrum growth]
\label{lem:B.geodesic-growth}
There exists $C_\Gamma>0$ such that for all $L\ge 1$,
\begin{equation}
\label{eq:B.geodesic-growth}
\pi_\Gamma(L)\ll_\Gamma e^{C_\Gamma L}.
\end{equation}
\end{lemma}

\begin{proof}
This follows from standard coding of the geodesic flow on $M$ and the fact that the number of conjugacy classes with
translation length $\le L$ grows at most exponentially.  No optimal exponent is needed.
\end{proof}

\subsection{Fourier decay of localized cutoffs}
\label{subsec:B.fourier-decay}

Let $\widehat{\chi}$ denote the Fourier transform of $\chi$ under our convention, and note that
\begin{equation}
\label{eq:B.fourier-rescale}
\widehat{\chi_\eta}(t) = \eta\,\widehat{\chi}(\eta t).
\end{equation}

\begin{lemma}[Rapid decay]
\label{lem:B.rapid-decay}
For every $N\ge 0$ there exists $C_N(\chi)$ such that for all $t\in\mathbb{R}$ and $\eta\in(0,1]$,
\begin{equation}
\label{eq:B.rapid-decay}
|\widehat{\chi_\eta}(t)|
\le
C_N(\chi)\,\eta\,(1+|\eta t|)^{-N}.
\end{equation}
\end{lemma}

\begin{proof}
This is immediate from \eqref{eq:B.fourier-rescale} and the Schwartz decay of $\widehat{\chi}$.
\end{proof}

\subsection{Tail control for hyperbolic contributions}
\label{subsec:B.hyperbolic-tail}

We record a model bound sufficient for truncating the hyperbolic sum at length $L\asymp \eta^{-1}\log\lambda$.

\begin{proposition}[Hyperbolic tail bound (model form)]
\label{prop:B.hyperbolic-tail}
Let $\chi$ be fixed and even.  Consider a hyperbolic trace contribution of the schematic form
\[
\mathcal{H}(\lambda,\eta)
=
\sum_{[\gamma]}\sum_{k\ge 1} w(\gamma,k)\,\widehat{\chi_\eta}(kL(\gamma))\,e^{-i\lambda kL(\gamma)},
\]
where $|w(\gamma,k)|\ll_\Gamma e^{-\alpha kL(\gamma)}$ for some $\alpha>0$.
Then for any $N\ge 0$,
\[
\mathcal{H}(\lambda,\eta)
\ll_{\Gamma,\chi,N}
\eta\sum_{[\gamma]}\sum_{k\ge 1} e^{-\alpha kL(\gamma)}(1+\eta kL(\gamma))^{-N}.
\]
In particular, the contribution of lengths $kL(\gamma)\ge \eta^{-1}(\log\lambda)$ is $O_{\Gamma,\chi,N}(\lambda^{-100})$
upon choosing $N$ large enough.
\end{proposition}

\begin{proof}
Use Lemma~\ref{lem:B.rapid-decay} and sum over $(\gamma,k)$ using Lemma~\ref{lem:B.geodesic-growth} to control the
number of terms at a given length scale.  The exponential damping from $w(\gamma,k)$ guarantees convergence; the
polynomial factor $(1+\eta kL)^{-N}$ provides arbitrarily strong savings beyond the cutoff scale $\eta^{-1}\log\lambda$.
\end{proof}

\section{Bessel/Kuznetsov transforms}
\label{sec:B.bessel}

Localized Kuznetsov formulas introduce integral transforms of the form
\[
\mathcal{B}_{\lambda,\eta}(n,c)
=
\int_0^\infty \chi_\eta(r-\lambda)\,\mathcal{K}(r; n,c)\,dr,
\]
where $\mathcal{K}(r;n,c)$ is a Bessel kernel (typically involving $J_{2ir}(\cdot)$ or $K_{ir}(\cdot)$ depending on
normalization).  The main text requires only generic properties: smoothness in the parameters, rapid decay away from
the resonance region, and power bounds in the transition range.

We state these in an abstract form, suitable for plugging into the off-diagonal analysis.

\subsection{A generic oscillatory transform lemma}
\label{subsec:B.oscillatory-lemma}

\begin{lemma}[Localized oscillatory transform: nonstationary regime]
\label{lem:B.oscillatory-nonstat}
Let $a(r)$ be a smooth amplitude with polynomial growth and let $\phi(r)$ be a smooth phase such that
$|\phi'(r)|\ge \kappa>0$ on $\supp \chi_\eta(\cdot-\lambda)$.
Then for any $N\ge 0$,
\begin{equation}
\label{eq:B.oscillatory-nonstat}
\left|
\int_{\mathbb{R}} \chi_\eta(r-\lambda)\,a(r)\,e^{i\phi(r)}\,dr
\right|
\ll_{N,\chi}
\eta\,\kappa^{-N}\,\sup_{0\le j\le N}\sup_{|r-\lambda|\le 2\eta}\left|a^{(j)}(r)\right|.
\end{equation}
\end{lemma}

\begin{proof}
Integrate by parts $N$ times using the operator $(i\phi'(r))^{-1}\partial_r$ and bound derivatives of $\chi_\eta$ by
$\eta^{-j}$ times Schwartz seminorms of $\chi$.
\end{proof}

\subsection{A stationary phase template}
\label{subsec:B.stationary-template}

\begin{lemma}[Localized stationary phase template]
\label{lem:B.stationary-template}
Let $\phi$ have a unique nondegenerate stationary point $r_0$ in $|r-\lambda|\le 2\eta$ with
$|\phi''(r_0)|\asymp \Phi>0$, and assume $\phi'$ has no other zeros on the window.
Then
\begin{equation}
\label{eq:B.stationary-template}
\int_{\mathbb{R}} \chi_\eta(r-\lambda)\,a(r)\,e^{i\phi(r)}\,dr
=
e^{i\phi(r_0)}\,a(r_0)\,\chi_\eta(r_0-\lambda)\,\Phi^{-1/2}
+
O\!\left(\eta\,\Phi^{-3/2}\,\mathcal{A}_2\right),
\end{equation}
where $\mathcal{A}_2$ depends on finitely many derivatives of $a$ and $\phi$ on the window and on Schwartz seminorms
of $\chi$.
\end{lemma}

\begin{proof}
This is a standard one-dimensional stationary phase expansion localized to a window of size $\eta$.  The error term is
uniform once $\phi''$ is bounded above and below on the window and higher derivatives of $a,\phi$ are controlled.
\end{proof}

\subsection{Application: resonance localization for Kuznetsov kernels}
\label{subsec:B.kuznetsov-resonance}

In the Kuznetsov setting, the phase has the schematic form $\phi(r)\sim r\log(\tfrac{4\pi n}{c})$ and thus the
stationary region corresponds to $c\asymp \tfrac{4\pi n}{\lambda}$.  The preceding templates yield:

\begin{proposition}[Resonance window bound (schematic)]
\label{prop:B.resonance-bound}
Let $\mathcal{B}_{\lambda,\eta}(n,c)$ denote the Bessel transform arising in the localized Kuznetsov formula.
Then for every $N\ge 0$,
\begin{equation}
\label{eq:B.resonance-bound}
\mathcal{B}_{\lambda,\eta}(n,c)
\ll_{\Gamma,\chi,N}
\eta\,(1+\lambda)^{A}\,
\left(1+\left|\log\frac{c\lambda}{n}\right|\right)^{-N},
\end{equation}
for some exponent $A\ge 0$ depending only on the chosen normalization of the kernel.
In particular, $\mathcal{B}_{\lambda,\eta}(n,c)$ is negligible unless $c\lambda/n$ lies in a fixed compact subset of
$\mathbb{R}_{>0}$.
\end{proposition}

\begin{proof}[Proof sketch]
Apply Lemma~\ref{lem:B.oscillatory-nonstat} away from the resonance region, where the phase derivative is bounded from
below.  Near resonance, apply Lemma~\ref{lem:B.stationary-template}.  The polynomial factor $(1+\lambda)^A$ collects
amplitude growth bounds for the Bessel kernel.
\end{proof}

\section{Elementary divisor bounds and smoothing estimates}
\label{sec:B.arithmetic}

The off-diagonal analysis uses only standard arithmetic bounds, recorded here for completeness.

\begin{lemma}[Divisor bounds]
\label{lem:B.divisor}
For every $\varepsilon>0$ one has $\tau(n)\ll_\varepsilon n^\varepsilon$ and
\[
\sum_{d\mid n} d^{-1/2} \ll_\varepsilon n^\varepsilon.
\]
\end{lemma}

\begin{proof}
Standard.
\end{proof}

\begin{lemma}[Smooth partition stability]
\label{lem:B.partition-stability}
Let $\{\psi_\alpha\}$ be a smooth partition of unity subordinate to a uniformly bounded overlap cover of a manifold.
Then for each $k\ge 0$,
\[
\sum_\alpha \|\psi_\alpha\|_{C^k}\ll_k 1,
\]
with implied constant depending only on the cover geometry.  In particular, refining the partition at scale
$\rho^{-1}$ introduces at most polynomial losses in $\rho$ in symbol seminorms.
\end{lemma}

\begin{proof}
This is standard in microlocal analysis: bounded overlap implies uniform control of the local finite sums.
\end{proof}

\section{Consolidated form for the error ledger interface}
\label{sec:B.ledger-interface}

For use in Chapter~\ref{chap:08-normalization}, it is convenient to summarize how the auxiliary bounds enter the
ledger terms.

\subsection{Continuous-spectrum ledger term}
\label{subsec:B.ledger-cont}

Lemma~\ref{lem:B.scattering-window} and Proposition~\ref{prop:B.maass-selberg} imply that the continuous contribution
to localized traces satisfies a bound of the form
\begin{equation}
\label{eq:B.cont-ledger}
\mathcal{E}^{\mathrm{cont}}_\Gamma[h;\chi]
\ll_{\Gamma,\chi,h}
\eta\,(\log\lambda)\,\lambda^{A_\Gamma},
\end{equation}
which is subordinate to the identity main term $\asymp \lambda\eta$ in the regimes used in the main text, and is
absorbed into the global remainder hierarchy.

\subsection{Hyperbolic tail ledger term}
\label{subsec:B.ledger-tail}

Proposition~\ref{prop:B.hyperbolic-tail} yields effective truncation of the hyperbolic orbital sum, with tail bound
faster than any power of $\lambda$ beyond the natural cutoff scale $\eta^{-1}\log\lambda$.  This is the mechanism
behind the tail ledger component $\mathcal{E}^{\mathrm{tail}}_\Gamma[h;\chi]$.

\subsection{Kuznetsov/off-diagonal ledger term}
\label{subsec:B.ledger-kuz}

Proposition~\ref{prop:B.resonance-bound} provides the resonance localization needed for off-diagonal control in the
Kuznetsov applications, and serves as the analytic input behind the off-diagonal ledger contributions in
Blocks~8.2--8.4.
