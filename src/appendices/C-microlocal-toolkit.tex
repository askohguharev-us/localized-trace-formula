\chapter{A microlocal toolkit for localized trace constructions}
\label{app:C-microlocal-toolkit}

\section{Purpose and usage conventions}
\label{sec:C.purpose}

This appendix collects the microlocal tools used implicitly throughout the construction of localized trace
functionals in Chapters~\ref{chap:05-localization}--\ref{chap:08-normalization}.
The emphasis is not on maximal generality but on a stable and reproducible set of estimates sufficient for:

\begin{enumerate}
\item the construction of microlocal projectors adapted to a prescribed phase-space localization;
\item uniform trace class and Hilbert--Schmidt bounds for cutoff resolvents and propagators;
\item symbolic control of quantization changes and refinement errors;
\item local Weyl-type expansions with explicit remainder structure.
\end{enumerate}

We adopt a fixed semiclassical parameter $h\in(0,h_0]$ with $h\to 0$.
The correspondence with the large spectral parameter regime is via $h\asymp \lambda^{-1}$.
All symbol classes and operator bounds are formulated in a way that makes this scaling explicit.

\section{Symbol classes and quantization}
\label{sec:C.symbols}

\subsection{Standard semiclassical symbols}
\label{subsec:C.symbols.standard}

Let $X$ be a smooth manifold of dimension $d$ and let $T^\ast X$ be its cotangent bundle.
For $m\in\mathbb{R}$, we denote by $S^m(T^\ast X)$ the standard semiclassical symbol class:
a family $a(x,\xi;h)$ belongs to $S^m$ if for all multiindices $\alpha,\beta$ there exists $C_{\alpha,\beta}$ such that
\begin{equation}
\label{eq:C.symbol-est}
\left|\partial_x^\alpha\partial_\xi^\beta a(x,\xi;h)\right|
\le
C_{\alpha,\beta}\,\langle\xi\rangle^{m-|\beta|},
\qquad
0<h\le h_0.
\end{equation}
We write $S^{-\infty}=\bigcap_{m\in\mathbb{R}}S^m$.

When a scale parameter $\rho=\rho(h)$ is present, we use the rescaled class $S^m_\rho$ defined by
\begin{equation}
\label{eq:C.symbol-est-rho}
\left|\partial_x^\alpha\partial_\xi^\beta a(x,\xi;h)\right|
\le
C_{\alpha,\beta}\,\rho(h)^{-|\alpha|-|\beta|}\langle\xi\rangle^{m-|\beta|}.
\end{equation}
This is the natural class for microlocal refinement at scale $\rho^{-1}$.

\subsection{Quantization schemes}
\label{subsec:C.quantization}

Let $\Op_h$ denote a fixed quantization scheme, e.g.\ Weyl quantization $\Op_h^w$.
For $a\in S^m$ we write $A=\Op_h(a)$.
We use only the following structural facts.

\begin{proposition}[Quantization stability]
\label{prop:C.quant-stability}
Let $\Op_h$ and $\Op'_h$ be two admissible quantizations on $X$.
Then for $a\in S^m$ one has
\[
\Op_h(a)-\Op'_h(a)=h\,\Op_h(r_a),
\qquad
r_a\in S^{m-1},
\]
with seminorm bounds for $r_a$ controlled by finitely many seminorms of $a$.
\end{proposition}

\begin{proof}
This is standard: any two admissible quantizations differ by an $h$-expansion whose first nontrivial term is of order
$h$ and involves derivatives of $a$.
\end{proof}

\subsection{Composition and commutators}
\label{subsec:C.composition}

We recall the standard symbolic calculus in a form suitable for trace applications.

\begin{proposition}[Composition formula]
\label{prop:C.composition}
Let $a\in S^{m_1}$ and $b\in S^{m_2}$.
Then
\[
\Op_h(a)\Op_h(b)=\Op_h(a\# b),
\]
where $a\# b\in S^{m_1+m_2}$ admits an asymptotic expansion
\[
a\# b \sim \sum_{j\ge 0} h^j c_j(a,b),
\]
with $c_0(a,b)=ab$ and $c_1(a,b)=\frac{1}{i}\{a,b\}/2$ in Weyl quantization.
Moreover, for every $N\ge 1$,
\[
a\# b - \sum_{j=0}^{N-1} h^j c_j(a,b) \in h^N S^{m_1+m_2-N}.
\]
\end{proposition}

\begin{proposition}[Commutator estimate]
\label{prop:C.commutator}
Let $a\in S^{m_1}$ and $b\in S^{m_2}$.
Then
\[
[\Op_h(a),\Op_h(b)]
=
\frac{h}{i}\Op_h(\{a,b\}) + h^2 \Op_h(r_{a,b}),
\qquad
r_{a,b}\in S^{m_1+m_2-2}.
\]
\end{proposition}

\section{Microlocal cutoffs and projectors}
\label{sec:C.cutoffs}

\subsection{Phase-space localization}
\label{subsec:C.phase-local}

Let $\Omega\subset T^\ast X$ be a compact subset.
A microlocal cutoff adapted to $\Omega$ is an operator of the form $\Pi=\Op_h(\pi)$ with $\pi\in S^0$ satisfying
$\pi\equiv 1$ on $\Omega$ and $\supp \pi$ contained in a fixed neighborhood of $\Omega$.

We record the two basic properties used in the main text.

\begin{lemma}[Microlocal idempotence]
\label{lem:C.idempotence}
Let $\Pi=\Op_h(\pi)$ with $\pi\in S^0$ and $\pi^2=\pi$ on a neighborhood of $\Omega$.
Then
\[
\Pi^2-\Pi \in h\,\Psi_h^{-1}(X),
\]
and in particular $\Pi^2=\Pi+O_{L^2\to L^2}(h)$.
\end{lemma}

\begin{proof}
Apply Proposition~\ref{prop:C.composition} with $a=b=\pi$.
\end{proof}

\begin{lemma}[Microlocal support control]
\label{lem:C.support}
If $\Pi_1=\Op_h(\pi_1)$ and $\Pi_2=\Op_h(\pi_2)$ with $\supp \pi_1$ disjoint from $\supp \pi_2$, then
\[
\Pi_1\Pi_2 \in h^\infty \Psi_h^{-\infty}(X).
\]
\end{lemma}

\begin{proof}
Repeated integration by parts in the oscillatory integral representation yields rapid decay in $h$ when the symbols
have disjoint microlocal support.
\end{proof}

\subsection{Refinement partitions and bounded overlap}
\label{subsec:C.partition}

Let $\{\Omega_\alpha\}$ be a finite microlocal cover of a compact region $\Omega\subset T^\ast X$ with bounded overlap.
Let $\{\pi_\alpha\}$ be a subordinate partition of unity, $\sum_\alpha \pi_\alpha=1$ on $\Omega$.

We define microlocal projectors $\Pi_\alpha=\Op_h(\pi_\alpha)$ and the refined decomposition
\[
\Pi_\Omega:=\sum_\alpha \Pi_\alpha.
\]

\begin{proposition}[Refinement stability]
\label{prop:C.refinement-stability}
Assume the overlap multiplicity of the cover $\{\Omega_\alpha\}$ is uniformly bounded by $N_0$.
Then for any $u\in L^2(X)$,
\[
\sum_\alpha \|\Pi_\alpha u\|_{L^2}^2
\ll
\|u\|_{L^2}^2,
\]
with an implied constant depending only on $N_0$ and finitely many symbol seminorms.
\end{proposition}

\begin{proof}
This is a microlocal version of Bessel's inequality, obtained from almost orthogonality using
Lemma~\ref{lem:C.support} and bounded overlap.
\end{proof}

\section{Trace class and Hilbert--Schmidt bounds}
\label{sec:C.trace-class}

Trace bounds are used to justify the localized trace expansions and to quantify the error ledger contributions.

\subsection{Kernel bounds and Schatten norms}
\label{subsec:C.schatten}

Let $A$ be an integral operator with kernel $K_A(x,y)$ on $X$.
Recall the Hilbert--Schmidt bound
\[
\|A\|_{\mathrm{HS}}^2
=
\int_{X\times X} |K_A(x,y)|^2\,dx\,dy.
\]

\begin{lemma}[Semiclassical Hilbert--Schmidt bound]
\label{lem:C.hs}
Let $A=\Op_h(a)$ with $a\in S^m$ compactly supported in $x$.
Then
\[
\|A\|_{\mathrm{HS}}
\ll
h^{-d/2}\,\|a\|_{L^2(T^\ast X)} + O(h^\infty).
\]
\end{lemma}

\begin{proof}
The kernel of $\Op_h(a)$ is given by an oscillatory integral with amplitude $a$.
Squaring and integrating yields the standard $h^{-d}$ scaling and identifies the leading contribution with the
$L^2$-mass of $a$.
\end{proof}

\begin{lemma}[Trace class bound for smoothing operators]
\label{lem:C.trace-class}
If $A\in \Psi_h^{-N}(X)$ with $N>d$, then $A$ is trace class and
\[
\|A\|_{\mathrm{tr}}
\ll
h^{-d}\,\sup_{|\alpha|+|\beta|\le M}\|\partial_x^\alpha\partial_\xi^\beta a\|_{L^\infty},
\]
for some $M=M(d,N)$.
\end{lemma}

\begin{proof}
Standard: trace class follows from integrability of the kernel, and the $h^{-d}$ factor is the phase-space volume
scaling.
\end{proof}

\subsection{Localized resolvents and functional calculus}
\label{subsec:C.resolvent}

Let $P$ be a self-adjoint semiclassical pseudodifferential operator of order $2$.
We use the Helffer--Sj\"ostrand functional calculus to represent $f(P)$.

\begin{proposition}[Helffer--Sj\"ostrand representation]
\label{prop:C.helffer-sjostrand}
Let $f\in C_c^\infty(\mathbb{R})$ and let $\widetilde{f}$ be an almost-analytic extension.
Then
\[
f(P)
=
\frac{1}{\pi}\int_{\mathbb{C}} \bar{\partial}\widetilde{f}(z)\,(z-P)^{-1}\,dL(z),
\]
where $dL$ denotes Lebesgue measure on $\mathbb{C}$.
\end{proposition}

\begin{lemma}[Microlocal resolvent bounds]
\label{lem:C.resolvent-bound}
Let $\Pi=\Op_h(\pi)$ be a microlocal cutoff supported away from the characteristic set of $P-z$ for $z$ in a fixed
compact subset of $\mathbb{C}$.
Then uniformly in $z$ in that set,
\[
\Pi(z-P)^{-1}\Pi \in \Psi_h^{-2}(X),
\qquad
\|\Pi(z-P)^{-1}\Pi\|_{L^2\to L^2}\ll 1.
\]
\end{lemma}

\begin{proof}
Elliptic parametrix construction.
\end{proof}

\section{Egorov theorem and propagation}
\label{sec:C.egorov}

The localized trace construction relies on the stability of microlocal cutoffs under propagation.

\subsection{Semiclassical propagation}
\label{subsec:C.propagation}

Let $U(t)=e^{-itP/h}$ be the unitary propagator generated by $P$.
Let $\Phi^t$ denote the Hamiltonian flow of the principal symbol $p$ of $P$.

\begin{theorem}[Egorov theorem]
\label{thm:C.egorov}
Let $A=\Op_h(a)$ with $a\in S^0$ compactly supported.
Then for $|t|\le t_0$ fixed,
\[
U(-t)\,A\,U(t)
=
\Op_h(a\circ\Phi^t) + h\,\Op_h(r_t),
\qquad
r_t\in S^{-1},
\]
with symbol seminorm bounds uniform in $|t|\le t_0$.
\end{theorem}

\begin{proof}
Standard semiclassical Egorov theorem.
\end{proof}

\subsection{Propagation of microlocal projectors}
\label{subsec:C.projector-propagation}

A key use is to control refinement errors in microlocal decompositions.

\begin{corollary}[Propagation of microlocal cutoffs]
\label{cor:C.propagation-cutoff}
Let $\Pi=\Op_h(\pi)$ with $\pi\in S^0$.
Then for $|t|\le t_0$,
\[
U(-t)\,\Pi\,U(t)
=
\Op_h(\pi\circ\Phi^t) + O_{L^2\to L^2}(h).
\]
\end{corollary}

\section{Local Weyl laws and trace expansions}
\label{sec:C.local-weyl}

\subsection{A standard local Weyl expansion}
\label{subsec:C.local-weyl-expansion}

Let $f\in C_c^\infty(\mathbb{R})$ and let $A=\Op_h(a)$ with $a\in S^0$ compactly supported.
We require the asymptotic expansion for $\Tr(A f(P))$.

\begin{theorem}[Local Weyl law (microlocal form)]
\label{thm:C.local-weyl}
Let $P$ be a self-adjoint semiclassical operator with principal symbol $p$.
Then
\begin{equation}
\label{eq:C.local-weyl}
\Tr\bigl(\Op_h(a)\,f(P)\bigr)
=
(2\pi h)^{-d}
\int_{T^\ast X} a(x,\xi)\,f(p(x,\xi))\,dx\,d\xi
+
O\!\left(h^{1-d}\right),
\end{equation}
where the implied constant depends on finitely many symbol seminorms of $a$ and on finitely many derivatives of $f$.
\end{theorem}

\begin{proof}
Combine the Helffer--Sj\"ostrand formula with a semiclassical parametrix for the resolvent and take traces.
\end{proof}

\subsection{Trace of cutoff propagators}
\label{subsec:C.trace-propagator}

A complementary trace object is the cutoff propagator $\Pi U(t)\Pi$.

\begin{proposition}[Cutoff propagator trace class bound]
\label{prop:C.cutoff-propagator-trace}
Let $\Pi=\Op_h(\pi)$ with $\pi\in C_c^\infty(T^\ast X)$.
Then for each fixed $t$,
\[
\Pi U(t)\Pi
\quad\text{is trace class and}\quad
\|\Pi U(t)\Pi\|_{\mathrm{tr}}
\ll
h^{-d}.
\]
\end{proposition}

\begin{proof}
Microlocal cutoff ensures compact phase-space support.  The kernel of $\Pi U(t)\Pi$ is an oscillatory integral with
compact microlocal support, yielding the $h^{-d}$ scaling.
\end{proof}

\section{Almost orthogonality and refinement error bounds}
\label{sec:C.almost-orthogonality}

Refinement errors arise when a microlocal projector is replaced by a finer partition.

\subsection{A standard almost orthogonality lemma}
\label{subsec:C.almost-orth}

\begin{lemma}[Almost orthogonality]
\label{lem:C.almost-orth}
Let $\{A_\alpha\}$ be a family of operators with symbols supported in a bounded overlap cover of phase space.
Assume that for $\alpha\neq\beta$ the product $A_\alpha^\ast A_\beta$ is $O(h^\infty)$ in operator norm unless
$\Omega_\alpha$ intersects $\Omega_\beta$.
Then
\[
\left\|\sum_\alpha A_\alpha u\right\|_{L^2}^2
\ll
\sum_\alpha \|A_\alpha u\|_{L^2}^2
+
O(h^\infty)\|u\|_{L^2}^2.
\]
\end{lemma}

\begin{proof}
Expand the square and use bounded overlap plus rapid off-diagonal decay.
\end{proof}

\subsection{Refinement error in trace functionals}
\label{subsec:C.refinement-trace}

Let $\Pi=\sum_\alpha \Pi_\alpha$ be a refined partition as in Section~\ref{subsec:C.partition}.
A typical refinement error is the difference between $\Tr(\Pi A)$ and $\sum_\alpha \Tr(\Pi_\alpha A)$.

\begin{proposition}[Trace refinement bound]
\label{prop:C.trace-refinement}
Let $A$ be trace class and let $\Pi=\sum_\alpha \Pi_\alpha$ be a microlocal decomposition with bounded overlap.
Then
\[
\Tr(\Pi A) - \sum_\alpha \Tr(\Pi_\alpha A)
=
O\!\left(\sum_{\alpha\neq\beta}\|\Pi_\alpha A \Pi_\beta\|_{\mathrm{tr}}\right).
\]
Moreover, if $\Pi_\alpha A \Pi_\beta$ is $O(h^\infty)$ in trace norm for $\alpha\neq\beta$, then the refinement error
is $O(h^\infty)$.
\end{proposition}

\begin{proof}
Insert $\Pi=\sum_\alpha \Pi_\alpha$ on both sides of $A$ and expand.  The diagonal terms coincide; the remainder is the
sum of off-diagonal contributions.
\end{proof}

\section{Quantization changes and symbolic remainders}
\label{sec:C.quantization-changes}

Quantization changes are treated as perturbations absorbed into the ledger.

\begin{proposition}[Quantization change error bound]
\label{prop:C.quantization-change}
Let $\Op_h$ and $\Op'_h$ be admissible quantizations and let $a\in S^0$.
Then for any trace class operator $B$ with $\|B\|_{\mathrm{tr}}\ll h^{-d}$,
\[
\Tr\bigl((\Op_h(a)-\Op'_h(a))B\bigr)
=
O(h)\,\|B\|_{\mathrm{tr}}.
\]
\end{proposition}

\begin{proof}
By Proposition~\ref{prop:C.quant-stability},
\[
\Op_h(a)-\Op'_h(a)=h\,\Op_h(r_a),
\]
with $\Op_h(r_a)$ bounded on $L^2$.  Then
\[
\left|\Tr((\Op_h(a)-\Op'_h(a))B)\right|
\le
\|\Op_h(a)-\Op'_h(a)\|_{L^2\to L^2}\,\|B\|_{\mathrm{tr}}
\ll
h\,\|B\|_{\mathrm{tr}}.
\]
\end{proof}

\section{Microlocal smoothing and trace regularization}
\label{sec:C.smoothing}

In localized trace constructions one often inserts smoothing operators to justify trace class manipulations.
We record a canonical smoothing device.

\subsection{A trace regularizer}
\label{subsec:C.regularizer}

Let $\langle P\rangle=(I+P^2)^{1/2}$ and define the regularizer
\[
R_N := \langle P\rangle^{-N},
\qquad N>d.
\]
Then $R_N$ is trace class and improves symbolic orders.

\begin{lemma}[Regularizer bounds]
\label{lem:C.regularizer}
For $N>d$, the operator $R_N$ is trace class and satisfies
\[
\|R_N\|_{\mathrm{tr}}\ll h^{-d}.
\]
Moreover, if $A\in \Psi_h^m$ then $AR_N\in \Psi_h^{m-N}$ and is trace class whenever $m-N<-d$.
\end{lemma}

\begin{proof}
Standard from the functional calculus and semiclassical Weyl asymptotics.
\end{proof}

\section{A canonical microlocal error hierarchy}
\label{sec:C.error-hierarchy}

For the normalization framework, it is convenient to isolate three microlocal error sources:

\begin{enumerate}
\item \emph{Quantization errors}, arising from replacing $\Op_h$ by $\Op'_h$.
\item \emph{Refinement errors}, arising from replacing a projector $\Pi$ by a finer decomposition.
\item \emph{Tail errors}, arising from truncating propagation times or length sums beyond the natural localization scale.
\end{enumerate}

Each of these errors is controlled by an estimate of the form
\[
|\Tr(A)-\Tr(A')|
\le
C\,h^\alpha\,\mathcal{S}(a),
\]
where $\mathcal{S}(a)$ is a finite symbol seminorm and $\alpha>0$ is a structural exponent.

The essential methodological point is that all such errors are \emph{explicitly ledgered} and do not contaminate the
canonical trace object.
