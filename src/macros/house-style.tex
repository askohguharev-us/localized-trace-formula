% House style for the monograph (no theorem declarations here).
% Keep this file free of any \newtheorem / \declaretheorem.
% All theorem-like environments are defined in macros/environments.tex.

% --- Font and spacing tweaks (safe for arXiv/Annals) ---
\setlength{\parskip}{0.25em}
\setlength{\parindent}{15pt}
\linespread{1.04}

% --- Numbering for figures/tables per section ---
\numberwithin{figure}{section}
\numberwithin{table}{section}

% --- Microtype fine tuning ---
\microtypesetup{final,tracking=true,kerning=true,spacing=true,protrusion=true}

% --- URL and link formatting ---
\urlstyle{same}

% --- Cleveref names (English only) ---
\crefname{section}{Section}{Sections}
\Crefname{section}{Section}{Sections}
\crefname{subsection}{Subsection}{Subsections}
\Crefname{subsection}{Subsection}{Subsections}

\Crefname{subsubsection}{Subsection}{Subsections}
\crefname{appendix}{Appendix}{Appendices}

\Crefname{figure}{Figure}{Figures}

\crefname{equation}{Equation}{Equations}
\Crefname{equation}{Equation}{Equations}

\crefname{theorem}{Theorem}{Theorems}
\Crefname{theorem}{Theorem}{Theorems}
\crefname{lemma}{Lemma}{Lemmas}
\Crefname{lemma}{Lemma}{Lemmas}
\crefname{proposition}{Proposition}{Propositions}

\crefname{corollary}{Corollary}{Corollaries}
