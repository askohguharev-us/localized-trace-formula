\ProvidesFile{house-style.tex}[2026/02/17 Diamond safe helpers]

\makeatletter

% --------------------------------------------------------------
% Safe helpers: define only if undefined.
% --------------------------------------------------------------

% ----- Delimiters / inner products -----
\providecommand{\abs}[1]{\left\lvert #1\right\rvert}
\providecommand{\norm}[1]{\left\lVert #1\right\rVert}
\providecommand{\ip}[2]{\left\langle #1,\,#2\right\rangle}
\providecommand{\set}[1]{\left\{\,#1\,\right\}}

% ----- Basic symbols -----
\providecommand{\dd}{\mathrm{d}}
\providecommand{\eps}{\varepsilon}
\providecommand{\OO}{\mathcal{O}}
\providecommand{\oo}{o}

% ----- Number sets -----
\providecommand{\R}{\mathbb{R}}
\providecommand{\C}{\mathbb{C}}
\providecommand{\Z}{\mathbb{Z}}
\providecommand{\N}{\mathbb{N}}
\providecommand{\Hh}{\mathbb{H}}

% --------------------------------------------------------------
% Safe operator declaration helper:
%   \SafeDeclareOp{name}{printed}
% defines \name iff it is undefined.
% --------------------------------------------------------------
\providecommand{\SafeDeclareOp}[2]{%
  \@ifundefined{#1}{%
    \expandafter\DeclareMathOperator\csname #1\endcsname{#2}%
  }{}%
}

% Minimal house ops (fine if also declared in operators.tex)
\SafeDeclareOp{inj}{inj}
\SafeDeclareOp{diam}{diam}

% Quantization names (only if operators.tex not loaded)
\SafeDeclareOp{Op}{Op}
\providecommand{\Opw}{\Op_{h}^{\mathrm{w}}}
\providecommand{\Opl}{\Op_{h}^{\mathrm{left}}}
\providecommand{\Opr}{\Op_{h}^{\mathrm{right}}}

% --------------------------------------------------------------
% Epigraph support (fallback).
% If the 'epigraph' package is loaded, it defines \epigraph itself.
% If not loaded, we provide a minimal, robust definition here.
%
% Usage: \epigraph{<text>}{<author/source>}
% --------------------------------------------------------------
\@ifundefined{epigraph}{%
  \newcommand{\epigraph}[2]{%
    \begin{flushright}
      \itshape #1\par
      \vspace{0.4em}%
      \normalfont\small --- #2\par
    \end{flushright}
    \vspace{0.8em}%
  }%
}{}%

% ----- TODO helper -----
\@ifundefined{TODO}{%
  \newcommand{\TODO}[1]{\textcolor{red!70!black}{\textbf{TODO:}~#1}}%
}{}%

\makeatother
