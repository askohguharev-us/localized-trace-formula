% --------------------------------------------------------------
% Safe operator declaration helper:
%   \SafeDeclareOp{name}{printed}
% defines \name iff it is undefined.
% --------------------------------------------------------------
\providecommand{\SafeDeclareOp}[2]{%
  \@ifundefined{#1}{%
    \expandafter\DeclareMathOperator\csname #1\endcsname{#2}%
  }{}%
}

% Minimal house ops (fine if also declared in operators.tex)
\SafeDeclareOp{inj}{inj}
\SafeDeclareOp{diam}{diam}

% Quantization names (only if operators.tex not loaded)
\SafeDeclareOp{Op}{Op}
\providecommand{\Opw}{\Op_{h}^{\mathrm{w}}}
\providecommand{\Opl}{\Op_{h}^{\mathrm{left}}}
\providecommand{\Opr}{\Op_{h}^{\mathrm{right}}}
