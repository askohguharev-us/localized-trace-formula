% environments.tex — Diamond-Guarded Theorem Environments (CHAPTER SAFE)
%
% Diamond law:
%   - Single authority for theorem-like environments.
%   - Theorem counter resets by chapter (never by section) to avoid ".59"
%     inside \section* / appendix-style blocks.
%   - Lemma/Proposition/Corollary/Definition/Remark/etc share theorem counter.
%   - Safe under multiple inputs; safe under missing packages; safe under
%     cleveref load-order; no hard dependency on main.tex.
% ==================================================================

\ProvidesFile{environments.tex}[2026/02/18 Diamond theorem envs (chapter-safe)]

\makeatletter

% --------------------------------------------------------------
% Hard dependency we actually need
%   - \theoremstyle, \newtheorem are defined by amsthm
%   - RequirePackage is safe here even if loaded earlier.
% --------------------------------------------------------------
\RequirePackage{amsthm}

% --------------------------------------------------------------
% Equation numbering (optional; harmless if already configured)
% --------------------------------------------------------------
\@ifundefined{c@equation}{}{%
  \@ifundefined{numberwithin}{}{%
    \numberwithin{equation}{section}%
  }%
}

% --------------------------------------------------------------
% Diamond-safe constructors
%   - Guard on the environment name macro (e.g. \theorem)
%   - Never redefines if already present.
% --------------------------------------------------------------
\providecommand{\DiamondNewTheorem}[3][]{%
  \@ifundefined{#2}{%
    \ifx\\#1\\%
      \newtheorem{#2}{#3}%
    \else
      \newtheorem{#2}{#3}[#1]%
    \fi
  }{}%
}

\providecommand{\DiamondNewTheoremSibling}[3]{%
  \@ifundefined{#1}{%
    \newtheorem{#1}[#2]{#3}%
  }{}%
}

% --------------------------------------------------------------
% Core family (CHAPTER-BASED counter) — kills ".59"
% --------------------------------------------------------------
\theoremstyle{plain}

% Primary counter: reset by chapter
\DiamondNewTheorem[chapter]{theorem}{Theorem}

% Siblings share theorem counter
\DiamondNewTheoremSibling{lemma}{theorem}{Lemma}
\DiamondNewTheoremSibling{proposition}{theorem}{Proposition}
\DiamondNewTheoremSibling{corollary}{theorem}{Corollary}
\DiamondNewTheoremSibling{hypothesis}{theorem}{Hypothesis}

% --------------------------------------------------------------
% Definition / remark families (share theorem counter)
% --------------------------------------------------------------
\theoremstyle{definition}
\DiamondNewTheoremSibling{definition}{theorem}{Definition}
\DiamondNewTheoremSibling{assumption}{theorem}{Assumption}

\theoremstyle{remark}
\DiamondNewTheoremSibling{remark}{theorem}{Remark}
\DiamondNewTheoremSibling{example}{theorem}{Example}

% --------------------------------------------------------------
% Audit block (remark wrapper, stable + simple)
% --------------------------------------------------------------
\@ifundefined{auditblock}{%
  \newenvironment{auditblock}[1]{%
    \begin{remark}[#1]%
  }{%
    \end{remark}%
  }%
}{}

% --------------------------------------------------------------
% cleveref integration (late binding; load-order safe)
% --------------------------------------------------------------
\AtBeginDocument{%
  \@ifpackageloaded{cleveref}{%
    \crefname{theorem}{theorem}{theorems}%
    \crefname{lemma}{lemma}{lemmas}%
    \crefname{proposition}{proposition}{propositions}%
    \crefname{corollary}{corollary}{corollaries}%
    \crefname{hypothesis}{hypothesis}{hypotheses}%
    \crefname{definition}{definition}{definitions}%
    \crefname{assumption}{assumption}{assumptions}%
    \crefname{remark}{remark}{remarks}%
    \crefname{example}{example}{examples}%
    \crefname{equation}{equation}{equations}%
    \crefname{auditblock}{audit block}{audit blocks}%
    \Crefname{auditblock}{Audit block}{Audit blocks}%
  }{%
    % Minimal fallback if cleveref is absent
    \providecommand{\cref}[1]{\ref{#1}}%
    \providecommand{\Cref}[1]{\ref{#1}}%
  }%
}

\makeatother
