% --- Системный хелпер для безопасного объявления ---
% Теперь он правильно превращает строку 'sgn' в команду '\sgn'
\providecommand{\SafeDeclareOp}[2]{%
  \@ifundefined{#1}{%
    \expandafter\DeclareMathOperator\csname #1\endcsname{#2}%
  }{%
    % Если оператор уже определен, ничего не делаем, чтобы избежать ошибок
  }%
}

% --- Semiclassical / Microlocal Operators ---
% Используем \SafeDeclareOp для единообразия
\SafeDeclareOp{Op}{Op}
\SafeDeclareOp{Opw}{Op_h^{\mathrm{w}}}
\SafeDeclareOp{Opl}{Op_h^{\mathrm{left}}}
\SafeDeclareOp{Opr}{Op_h^{\mathrm{right}}}

% --- Spectral & Geometric Operators ---
\SafeDeclareOp{sgn}{sgn}
\SafeDeclareOp{Tr}{Tr}
\SafeDeclareOp{tr}{tr}
\SafeDeclareOp{supp}{supp}
\SafeDeclareOp{diag}{diag}
\SafeDeclareOp{Res}{Res}
\SafeDeclareOp{dist}{dist}
\SafeDeclareOp{vol}{vol}
\SafeDeclareOp{rank}{rank}
\SafeDeclareOp{WF}{WF}      % Wave Front set
\SafeDeclareOp{Var}{Var}    % Variance
\SafeDeclareOp{inj}{inj}
\SafeDeclareOp{sys}{sys}
\SafeDeclareOp{Spec}{Spec}
\SafeDeclareOp{diam}{diam}
