% --- Системный хелпер для безопасного объявления ---
% Исправлено: теперь он корректно создает команду из строки
\providecommand{\SafeDeclareOp}[2]{%
  \@ifundefined{#1}{%
    \expandafter\DeclareMathOperator\csname #1\endcsname{#2}%
  }{}%
}

\SafeDeclareOp{supp}{supp}

\SafeDeclareOp{Spec}{Spec}
