% --------------------------------------------------------------
% Minimal fallbacks (in case the macro file is loaded early)
% --------------------------------------------------------------
\@ifundefined{operatorname}{%
  \newcommand{\operatorname}[1]{\mathop{\mathrm{#1}}\nolimits}%
}{}%

\@ifundefined{DeclareMathOperator}{%
  % Fallback: define operator via \operatorname
  \newcommand{\DeclareMathOperator}[2]{%
    \expandafter\providecommand\csname\expandafter\@gobble\string#1\endcsname{%
      \operatorname{#2}%
    }%
  }%
}{}%

\@ifundefined{textcolor}{%
  % If xcolor not loaded, ignore coloring.
  \newcommand{\textcolor}[2]{#2}%
}{}%

% --------------------------------------------------------------
% Safe declaration helper:
%   \SafeDeclareMathOperator{Tr}{Tr}
% creates \Tr iff it is undefined
% --------------------------------------------------------------
\providecommand{\SafeDeclareMathOperator}[2]{%
  \@ifundefined{#1}{%
    \expandafter\DeclareMathOperator\csname #1\endcsname{#2}%
  }{}%
}

% --------------------------------------------------------------
% Core operators (roman, with proper spacing)
% --------------------------------------------------------------
\SafeDeclareMathOperator{sgn}{sgn}
\SafeDeclareMathOperator{Tr}{Tr}
\SafeDeclareMathOperator{tr}{tr}
\SafeDeclareMathOperator{supp}{supp}
\SafeDeclareMathOperator{diag}{diag}
\SafeDeclareMathOperator{Res}{Res}
\SafeDeclareMathOperator{dist}{dist}
\SafeDeclareMathOperator{vol}{vol}
\SafeDeclareMathOperator{rank}{rank}
\SafeDeclareMathOperator{WF}{WF}
\SafeDeclareMathOperator{Var}{Var}
\SafeDeclareMathOperator{inj}{inj}
\SafeDeclareMathOperator{sys}{sys}
\SafeDeclareMathOperator{Spec}{Spec}
\SafeDeclareMathOperator{diam}{diam}

% Quantization / semiclassical notation
\SafeDeclareMathOperator{Op}{Op}
% Variants must behave like \Op with indices, not as standalone operator names:
\providecommand{\Opw}{\Op_{h}^{\mathrm{w}}}
\providecommand{\Opl}{\Op_{h}^{\mathrm{left}}}
\providecommand{\Opr}{\Op_{h}^{\mathrm{right}}}

% --------------------------------------------------------------
% Standard Lie groups / arithmetic groups (fixes build crash: \SL)

% Delimiters / norms / pairings / sets

\providecommand{\norm}[1]{\left\lVert #1\right\rVert}
\providecommand{\ip}[2]{\left\langle #1,\,#2\right\rangle}
\providecommand{\set}[1]{\left\{\,#1\,\right\}}
